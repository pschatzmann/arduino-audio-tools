110 // gets the scaled audio data as result
111 void getStepData(float *result, int stride, float maxResult) {
112 for (int j = 0; j < stride; j++) {
113 // determine max value to scale
114 if (data[j] > rfft_max) rfft_max = data[j];
115 }
116 for (int j = 0; j < stride; j++) {
117 result[j] = data[j] / rfft_max * maxResult;
118 // clip
119 if (result[j] > maxResult) {
120 result[j] = maxResult;
121 }
122 if (result[j] < -maxResult) {
123 result[j] = -maxResult;
124 }
125 }
126 // copy data to head
127 for (int j = 0; j < len - stride; j++) {
128 data[j] = data[j + stride];
129 }
130 // clear tail
131 for (int j = len - stride; j < len; j++) {
132 data[j] = 0.0;
133 }
134 }

465 // Add samples to input data p_x - and process them if full
466 template <typename T>
467 void processSamples(const void *data, size_t count) {
468 T *dataT = (T *)data;
469 T sample;
470 for (int j = 0; j < count; j += cfg.channels) {
471 sample = dataT[j + cfg.channel_used];
472 if (writeStrideBuffer((uint8_t *)&sample, sizeof(T))){
473 // process data if buffer is full
474 T* samples = (T*) stride_buffer.data();
475 int sample_count = stride_buffer.size() / sizeof(T);
476 assert(sample_count == cfg.length);
477 for (int j=0; j< sample_count; j++){
478 T out_sample = samples[j];
479 p_driver->setValue(j, windowedSample(out_sample, j));
480 }
481
482 fft<T>();
483
484 // remove stride samples
485 stride_buffer.clearArray(cfg.stride * sizeof(T));
486
487 // validate available data in stride buffer
488 if (cfg.stride == cfg.length) assert(stride_buffer.available()==0);
489
490 }
491 }

556 template <typename T>
557 void writeIFFT(float *data, int len) {
558 for (int j = 0; j < len; j++) {
559 T sample = data[j];
560 T out_data[cfg.channels];
561 for (int ch = 0; ch < cfg.channels; ch++) {
562 out_data[ch] = sample;
563 }
564 int result = rfft_data.writeArray((uint8_t *)out_data, sizeof(out_data));

572 template <int N>
573 void insertSorted(AudioFFTResult (&result)[N], AudioFFTResult tmp) {
574 // find place where we need to insert new record
575 for (int j = 0; j < N; j++) {
576 // insert when biggen then current record
577 if (tmp.magnitude > result[j].magnitude) {
578 // shift existing values right
579 for (int i = N - 2; i >= j; i--) {
580 result[i + 1] = result[i];
581 }
582 // insert new value
583 result[j] = tmp;
584 // stop after we found the correct index
585 break;
586 }
587 }
