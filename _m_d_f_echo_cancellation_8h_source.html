390 void capture(const echo_int16_t* rec, echo_int16_t* out) {
391 ensureInitialized();
392 state->play_buf_started = 1;
393 if (state->play_buf_pos >= state->frame_size) {
394 echoCancellationImpl(state, rec, state->play_buf, out);
395 state->play_buf_pos -= state->frame_size;
396 std::memmove(state->play_buf,
397 &state->play_buf[state->frame_size],
398 state->play_buf_pos * sizeof(echo_int16_t));
399 } else {
400 echoWarning("No playback frame available");
401 if (state->play_buf_pos != 0) {
402 echoWarning("Internal playback buffer corruption");
403 state->play_buf_pos = 0;
404 }
405 for (int i = 0; i < state->frame_size; i++) out[i] = rec[i];
406 }

412 void playback(const echo_int16_t* play) {
413 ensureInitialized();
414 if (!state->play_buf_started) {
415 echoWarning("Discarded first playback frame");
416 return;
417 }
418 if (state->play_buf_pos <= PLAYBACK_DELAY * state->frame_size) {
419 for (int i = 0; i < state->frame_size; i++)
420 state->play_buf[state->play_buf_pos + i] = play[i];
421 state->play_buf_pos += state->frame_size;
422 if (state->play_buf_pos <= (PLAYBACK_DELAY - 1) * state->frame_size) {
423 echoWarning("Auto-filling buffer");
424 for (int i = 0; i < state->frame_size; i++)
425 state->play_buf[state->play_buf_pos + i] = play[i];
426 state->play_buf_pos += state->frame_size;

480 int control(int request, void* ptr) {
481 switch (request) {
482 case ECHO_GET_FRAME_SIZE:
483 (*(int*)ptr) = state->frame_size;
484 break;
485 case ECHO_SET_SAMPLING_RATE:
486 state->sampling_rate = (*(int*)ptr);
487 state->spec_average = state->frame_size / (float)state->sampling_rate;
488 state->beta0 = (2.0f * state->frame_size) / state->sampling_rate;
489 state->beta_max = (.5f * state->frame_size) / state->sampling_rate;
490 if (state->sampling_rate < 12000)
491 state->notch_radius = .9f;
492 else if (state->sampling_rate < 24000)
493 state->notch_radius = .982f;
494 else
495 state->notch_radius = .992f;
496 break;
497 case ECHO_GET_SAMPLING_RATE:
498 (*(int*)ptr) = state->sampling_rate;
499 break;
500 case ECHO_GET_IMPULSE_RESPONSE_SIZE:
501 *((echo_int32_t*)ptr) = state->M * state->frame_size;
502 break;
503 case ECHO_GET_IMPULSE_RESPONSE: {
504 int M = state->M, N = state->window_size, n = state->frame_size;
505 echo_int32_t* filt = (echo_int32_t*)ptr;
506 for (int j = 0; j < M; j++) {
507 echo_ifft(state->fft_table, &state->W[j * N], state->wtmp);
508 for (int i = 0; i < n; i++) filt[j * n + i] = 32767 * state->wtmp[i];
509 }
510 } break;
511 default:
512 return -1;
513 }
514 return 0;
515 }

699 inline void filterDcNotch16(const echo_int16_t* in, Word16 radius,
700 Word16* out, int len, Mem* mem,
701 int stride) {
702 Word16 den2 = radius * radius + .7f * (1 - radius) * (1 - radius);
703 for (int i = 0; i < len; i++) {
704 Word16 vin = in[i * stride];
705 Word32 vout = mem[0] + vin;
706 mem[0] = mem[1] + 2 * (-vin + radius * vout);
707 mem[1] = vin - den2 * vout;
708 out[i] = radius * vout;
709 }
710 }

765 inline void spectralMulAccum(const Word16* X, const Word32* Y,
766 Word16* acc, int N, int M) {
767 for (int i = 0; i < N; i++) acc[i] = 0;
768
769 for (int j = 0; j < M; j++) {
770 acc[0] += X[0] * Y[0];
771 for (int i = 1; i < N - 1; i += 2) {
772 acc[i] += (X[i] * Y[i] - X[i + 1] * Y[i + 1]);
773 acc[i + 1] += (X[i + 1] * Y[i] + X[i] * Y[i + 1]);
774 }
775 acc[N - 1] += X[N - 1] * Y[N - 1];
776 X += N;
777 Y += N;
778 }

790 inline void weightedSpectralMulConj(const Num* w, const Num p,
791 const Word16* X,
792 const Word16* Y, Word32* prod,
793 int N) {
794 // NOTE: the (int32_t) casts this code originally had here were only
795 // correct in true fixed-point mode (X/Y as int16_t, needing widening
796 // before multiplying to avoid 16-bit overflow). With Word16 /
797 // Word32 now float (see the aliases above), those casts instead
798 // truncated every frequency-domain sample to an integer before ever
799 // multiplying it by anything -- e.g. a bin magnitude of 0.03 became 0,
800 // silently zeroing out the filter's gradient. Removed.
801 Num W;
802 W = p * w[0];
803 prod[0] = W * (X[0] * Y[0]);
804 for (int i = 1, j = 1; i < N - 1; i += 2, j++) {
805 W = p * w[j];
806 prod[i] = W * (X[i] * Y[i] + X[i + 1] * Y[i + 1]);
807 prod[i + 1] = W * (-X[i + 1] * Y[i] + X[i] * Y[i + 1]);
808 }
809 W = p * w[N / 2];
810 prod[N - 1] = W * (X[N - 1] * Y[N - 1]);
811 }

821 inline void mdfAdjustProp(const Word32* W, int N, int M, int P,
822 Word16* prop) {
823 Word16 max_sum = 1;
824 Word32 prop_sum = 1;
825
826 for (int i = 0; i < M; i++) {
827 Word32 tmp = 1;
828 for (int p = 0; p < P; p++) {
829 for (int j = 0; j < N; j++) {
830 float val = W[p * N * M + i * N + j];
831 tmp += val * val;
832 }
833 }
834 prop[i] = spxSqrt(tmp);
835 if (prop[i] > max_sum) max_sum = prop[i];
836 }
837
838 for (int i = 0; i < M; i++) {
839 prop[i] += 0.1f * max_sum;
840 prop_sum += prop[i];
841 }
842
843 for (int i = 0; i < M; i++) {
844 prop[i] = 0.99f * prop[i] / prop_sum;
845 }
846 }

856 State* echoStateInitMc(int frame_size, int filter_length, int nb_mic,
857 int nb_speakers) {
858 int N = frame_size * 2;
859 int M = (filter_length + frame_size - 1) / frame_size;
860 int C = nb_mic;
861 int K = nb_speakers;
862
863 State* st = new State();
864 if (!st) return nullptr;
865
866 st->K = K;
867 st->C = C;
868 st->frame_size = frame_size;
869 st->window_size = N;
870 st->M = M;
871 st->cancel_count = 0;
872 st->sum_adapt = 0;
873 st->saturated = 0;
874 st->screwed_up = 0;
875 st->sampling_rate = 8000;
876 st->spec_average = st->frame_size / (float)st->sampling_rate;
877 st->beta0 = (2.0f * st->frame_size) / st->sampling_rate;
878 st->beta_max = (.5f * st->frame_size) / st->sampling_rate;
879 st->leak_estimate = 0;
880
881 // Allocate buffers
882 st->e = echoAlloc<Word16>(C * N);
883 st->x = echoAlloc<Word16>(K * N);
884 st->input = echoAlloc<Word16>(C * st->frame_size);
885 st->y = echoAlloc<Word16>(C * N);
886 st->last_y = echoAlloc<Word16>(C * N);
887 st->Yf = echoAlloc<Word32>(st->frame_size + 1);
888 st->Rf = echoAlloc<Word32>(st->frame_size + 1);
889 st->Xf = echoAlloc<Word32>(st->frame_size + 1);
890 st->Yh = echoAlloc<Word32>(st->frame_size + 1);
891 st->Eh = echoAlloc<Word32>(st->frame_size + 1);
892 st->X = echoAlloc<Word16>(K * (M + 1) * N);
893 st->Y = echoAlloc<Word16>(C * N);
894 st->E = echoAlloc<Word16>(C * N);
895 st->W = echoAlloc<Word32>(C * K * M * N);
896
897#ifdef TWO_PATH
898 st->foreground = echoAlloc<Word16>(M * N * C * K);
899#endif
900
901 st->PHI = echoAlloc<Word32>(N);
902 st->power = echoAlloc<Word32>(frame_size + 1);
903 st->power_1 = echoAlloc<Num>(frame_size + 1);
904 st->window = echoAlloc<Word16>(N);
905 st->prop = echoAlloc<Word16>(M);
906 st->wtmp = echoAlloc<Word16>(N);
907
908 // Initialize window
909 for (int i = 0; i < N; i++)
910 st->window[i] = .5f - .5f * cosf(2 * M_PI * i / N);
911
912 // Initialize power_1
913 for (int i = 0; i <= st->frame_size; i++) st->power_1[i] = 1.0f;
914
915 // Initialize W
916 for (int i = 0; i < N * M * K * C; i++) st->W[i] = 0;
917
918 // Initialize prop
919 {
920 Word32 sum = 0;
921 float decay = expf(-2.4f / M);
922 st->prop[0] = .7f;
923 sum = st->prop[0];
924 for (int i = 1; i < M; i++) {
925 st->prop[i] = st->prop[i - 1] * decay;
926 sum += st->prop[i];
927 }
928 for (int i = M - 1; i >= 0; i--) {
929 st->prop[i] = .8f * st->prop[i] / sum;
930 }
931 }
932
933 st->memX = echoAlloc<Word16>(K);
934 st->memD = echoAlloc<Word16>(C);
935 st->memE = echoAlloc<Word16>(C);
936 st->preemph = .9f;
937
938 if (st->sampling_rate < 12000)
939 st->notch_radius = .9f;
940 else if (st->sampling_rate < 24000)
941 st->notch_radius = .982f;
942 else
943 st->notch_radius = .992f;
944
945 st->notch_mem = echoAlloc<Mem>(2 * C);
946 st->adapted = 0;
947 st->Pey = st->Pyy = 1.0f;
948
949#ifdef TWO_PATH
950 st->Davg1 = st->Davg2 = 0;
951 st->Dvar1 = st->Dvar2 = 0.0f;
952#endif
953
954 st->play_buf =
955 echoAlloc<echo_int16_t>(K * (PLAYBACK_DELAY + 1) * st->frame_size);
956 st->play_buf_pos = PLAYBACK_DELAY * st->frame_size;
957 st->play_buf_started = 0;
958
959 st->fft_table = nullptr;
960
961 return st;
962 }

980 inline void echoCancellationImpl(State* st, const echo_int16_t* in,
981 const echo_int16_t* far_end,
982 echo_int16_t* out) {
983 int N = st->window_size;
984 int M = st->M;
985 int C = st->C;
986 int K = st->K;
987
988 // Wide-dynamic-range tuning constants, as Num (float or PseudoFloat)
989 // so they multiply/compare directly against Num-typed state below.
990 const Num min_leak(0.005f); // Minimum leak estimate for the adaptive filter
991 const Num var1_smooth(0.36f); // Smoothing coefficient, 1st variance estimator
992 const Num var2_smooth(0.7225f); // Smoothing coefficient, 2nd variance estimator
993 const Num var1_update(0.5f); // Update threshold, 1st variance estimator
994 const Num var2_update(0.25f); // Update threshold, 2nd variance estimator
995 const Num var_backtrack(4.0f); // Backtrack threshold for filter reset
996
997 st->cancel_count++;
998 float ss = .35f / M;
999 float ss_1 = 1 - ss;
1000
1001 // Apply notch filter and pre-emphasis to input
1002 for (int chan = 0; chan < C; chan++) {
1003 filterDcNotch16(in + chan, st->notch_radius,
1004 st->input + chan * st->frame_size, st->frame_size,
1005 st->notch_mem + 2 * chan, C);
1006
1007 for (int i = 0; i < st->frame_size; i++) {
1008 Word32 tmp32 =
1009 st->input[chan * st->frame_size + i] - st->preemph * st->memD[chan];
1010 st->memD[chan] = st->input[chan * st->frame_size + i];
1011 st->input[chan * st->frame_size + i] = tmp32;
1012 }
1013 }
1014
1015 // Process far-end signal
1016 for (int speak = 0; speak < K; speak++) {
1017 std::memmove(&st->x[speak * N],
1018 &st->x[speak * N + st->frame_size],
1019 st->frame_size * sizeof(Word16));
1020 for (int i = 0; i < st->frame_size; i++) {
1021 Word32 tmp32 =
1022 far_end[i * K + speak] - st->preemph * st->memX[speak];
1023 st->x[speak * N + i + st->frame_size] = tmp32;
1024 st->memX[speak] = far_end[i * K + speak];
1025 }
1026 }
1027
1028 // Shift memory and compute FFT of far-end
1029 for (int speak = 0; speak < K; speak++) {
1030 for (int j = M - 1; j >= 0; j--) {
1031 std::memmove(&st->X[(j + 1) * N * K + speak * N],
1032 &st->X[j * N * K + speak * N],
1033 N * sizeof(Word16));
1034 }
1035 echo_fft(st->fft_table, st->x + speak * N, &st->X[speak * N]);
1036 }
1037
1038 // Compute power spectrum of far-end
1039 Word32 Sxx = 0;
1040 for (int speak = 0; speak < K; speak++) {
1041 Sxx += mdfInnerProd(st->x + speak * N + st->frame_size,
1042 st->x + speak * N + st->frame_size, st->frame_size);
1043 powerSpectrumAccum(st->X + speak * N, st->Xf, N);
1044 }
1045
1046 // Compute foreground filter output and residual
1047 Word32 Sff = 0;
1048 for (int chan = 0; chan < C; chan++) {
1049#ifdef TWO_PATH
1050 spectralMulAccum(st->X, st->foreground + chan * N * K * M,
1051 st->Y + chan * N, N, M * K);
1052 echo_ifft(st->fft_table, st->Y + chan * N, st->e + chan * N);
1053 for (int i = 0; i < st->frame_size; i++)
1054 st->e[chan * N + i] = st->input[chan * st->frame_size + i] -
1055 st->e[chan * N + i + st->frame_size];
1056 Sff += mdfInnerProd(st->e + chan * N, st->e + chan * N, st->frame_size);
1057#endif
1058 }
1059
1060 // Adjust proportional adaptation
1061 if (st->adapted) mdfAdjustProp(st->W, N, M, C * K, st->prop);
1062
1063 // Compute weight gradient
1064 if (st->saturated == 0) {
1065 for (int chan = 0; chan < C; chan++) {
1066 for (int speak = 0; speak < K; speak++) {
1067 for (int j = M - 1; j >= 0; j--) {
1068 weightedSpectralMulConj(st->power_1, st->prop[j],
1069 &st->X[(j + 1) * N * K + speak * N],
1070 st->E + chan * N, st->PHI, N);
1071 for (int i = 0; i < N; i++)
1072 st->W[chan * N * K * M + j * N * K + speak * N + i] += st->PHI[i];
1073 }
1074 }
1075 }
1076 } else {
1077 st->saturated--;
1078 }
1079
1080 // Update weights (AUMDF)
1081 for (int chan = 0; chan < C; chan++) {
1082 for (int speak = 0; speak < K; speak++) {
1083 for (int j = 0; j < M; j++) {
1084 if (j == 0 || st->cancel_count % (M - 1) == j - 1) {
1085 echo_ifft(
1086 st->fft_table, &st->W[chan * N * K * M + j * N * K + speak * N],
1087 st->wtmp);
1088 for (int i = st->frame_size; i < N; i++) st->wtmp[i] = 0;
1089 echo_fft(
1090 st->fft_table, st->wtmp,
1091 &st->W[chan * N * K * M + j * N * K + speak * N]);
1092 }
1093 }
1094 }
1095 }
1096
1097 // Initialize spectrum buffers
1098 for (int i = 0; i <= st->frame_size; i++)
1099 st->Rf[i] = st->Yf[i] = st->Xf[i] = 0;
1100
1101 Word32 Dbf = 0;
1102 Word32 See = 0;
1103
1104#ifdef TWO_PATH
1105 // Compute background filter output
1106 for (int chan = 0; chan < C; chan++) {
1107 spectralMulAccum(st->X, st->W + chan * N * K * M, st->Y + chan * N, N,
1108 M * K);
1109 echo_ifft(st->fft_table, st->Y + chan * N, st->y + chan * N);
1110 for (int i = 0; i < st->frame_size; i++)
1111 st->e[chan * N + i] = st->e[chan * N + i + st->frame_size] -
1112 st->y[chan * N + i + st->frame_size];
1113 Dbf +=
1114 10 + mdfInnerProd(st->e + chan * N, st->e + chan * N, st->frame_size);
1115 for (int i = 0; i < st->frame_size; i++)
1116 st->e[chan * N + i] = st->input[chan * st->frame_size + i] -
1117 st->y[chan * N + i + st->frame_size];
1118 See += mdfInnerProd(st->e + chan * N, st->e + chan * N, st->frame_size);
1119 }
1120#endif
1121
1122#ifndef TWO_PATH
1123 Sff = See;
1124#endif
1125
1126#ifdef TWO_PATH
1127 // Two-path filter logic
1128 st->Davg1 = .6f * st->Davg1 + .4f * (Sff - See);
1129 st->Davg2 = .85f * st->Davg2 + .15f * (Sff - See);
1130 st->Dvar1 = var1_smooth * st->Dvar1 + (.4f * Sff) * (.4f * Dbf);
1131 st->Dvar2 = var2_smooth * st->Dvar2 + (.15f * Sff) * (.15f * Dbf);
1132
1133 int update_foreground = 0;
1134 if ((Sff - See) * fabsf(Sff - See) > Sff * Dbf)
1135 update_foreground = 1;
1136 else if (st->Davg1 * fabsf((float)st->Davg1) > var1_update * st->Dvar1)
1137 update_foreground = 1;
1138 else if (st->Davg2 * fabsf((float)st->Davg2) > var2_update * st->Dvar2)
1139 update_foreground = 1;
1140
1141 if (update_foreground) {
1142 st->Davg1 = st->Davg2 = 0;
1143 st->Dvar1 = st->Dvar2 = 0.0f;
1144 std::memcpy(st->foreground, st->W, N * M * C * K * sizeof(Word16));
1145 for (int chan = 0; chan < C; chan++)
1146 for (int i = 0; i < st->frame_size; i++)
1147 st->e[chan * N + i + st->frame_size] =
1148 st->window[i + st->frame_size] *
1149 st->e[chan * N + i + st->frame_size] +
1150 st->window[i] * st->y[chan * N + i + st->frame_size];
1151 } else {
1152 int reset_background = 0;
1153 if ((-(Sff - See)) * fabsf((float)(Sff - See)) >
1154 var_backtrack * (Sff * Dbf))
1155 reset_background = 1;
1156 if ((-st->Davg1) * fabsf((float)st->Davg1) > var_backtrack * st->Dvar1)
1157 reset_background = 1;
1158 if ((-st->Davg2) * fabsf((float)st->Davg2) > var_backtrack * st->Dvar2)
1159 reset_background = 1;
1160
1161 if (reset_background) {
1162 std::memcpy(st->W, st->foreground, N * M * C * K * sizeof(Word32));
1163 for (int chan = 0; chan < C; chan++) {
1164 for (int i = 0; i < st->frame_size; i++)
1165 st->y[chan * N + i + st->frame_size] =
1166 st->e[chan * N + i + st->frame_size];
1167 for (int i = 0; i < st->frame_size; i++)
1168 st->e[chan * N + i] = st->input[chan * st->frame_size + i] -
1169 st->y[chan * N + i + st->frame_size];
1170 }
1171 See = Sff;
1172 st->Davg1 = st->Davg2 = 0;
1173 st->Dvar1 = st->Dvar2 = 0.0f;
1174 }
1175 }
1176#endif
1177
1178 Word32 Sey = 0, Syy = 0, Sdd = 0;
1179 for (int chan = 0; chan < C; chan++) {
1180 // Compute output with de-emphasis
1181 for (int i = 0; i < st->frame_size; i++) {
1182 Word32 tmp_out;
1183#ifdef TWO_PATH
1184 tmp_out = st->input[chan * st->frame_size + i] -
1185 st->e[chan * N + i + st->frame_size];
1186#else
1187 tmp_out = st->input[chan * st->frame_size + i] -
1188 st->y[chan * N + i + st->frame_size];
1189#endif
1190 tmp_out = tmp_out + st->preemph * st->memE[chan];
1191 if (in[i * C + chan] <= -32000 || in[i * C + chan] >= 32000) {
1192 if (st->saturated == 0) st->saturated = 1;
1193 }
1194 out[i * C + chan] = WORD2INT(tmp_out);
1195 st->memE[chan] = tmp_out;
1196 }
1197
1198 // Prepare error signal for filter update
1199 for (int i = 0; i < st->frame_size; i++) {
1200 st->e[chan * N + i + st->frame_size] = st->e[chan * N + i];
1201 st->e[chan * N + i] = 0;
1202 }
1203
1204 // Compute correlations
1205 Sey += mdfInnerProd(st->e + chan * N + st->frame_size,
1206 st->y + chan * N + st->frame_size, st->frame_size);
1207 Syy += mdfInnerProd(st->y + chan * N + st->frame_size,
1208 st->y + chan * N + st->frame_size, st->frame_size);
1209 Sdd += mdfInnerProd(st->input + chan * st->frame_size,
1210 st->input + chan * st->frame_size, st->frame_size);
1211
1212 // Convert error to frequency domain
1213 echo_fft(st->fft_table, st->e + chan * N, st->E + chan * N);
1214
1215 for (int i = 0; i < st->frame_size; i++) st->y[i + chan * N] = 0;
1216 echo_fft(st->fft_table, st->y + chan * N, st->Y + chan * N);
1217
1218 // Compute power spectra
1219 powerSpectrumAccum(st->E + chan * N, st->Rf, N);
1220 powerSpectrumAccum(st->Y + chan * N, st->Yf, N);
1221 }
1222
1223 // Sanity checks
1224 if (!(Syy >= 0 && Sxx >= 0 && See >= 0) ||
1225 !(Sff < N * 1e9f && Syy < N * 1e9f && Sxx < N * 1e9f)) {
1226 st->screwed_up += 50;
1227 for (int i = 0; i < st->frame_size * C; i++) out[i] = 0;
1228 } else if (Sff > Sdd + N * 10000.0f) {
1229 st->screwed_up++;
1230 } else {
1231 st->screwed_up = 0;
1232 }
1233
1234 if (st->screwed_up >= 50) {
1235 echoWarning("Echo canceller reset");
1236 reset();
1237 return;
1238 }
1239
1240 if (See < N * 100.0f) See = N * 100.0f;
1241
1242 for (int speak = 0; speak < K; speak++) {
1243 Sxx += mdfInnerProd(st->x + speak * N + st->frame_size,
1244 st->x + speak * N + st->frame_size, st->frame_size);
1245 powerSpectrumAccum(st->X + speak * N, st->Xf, N);
1246 }
1247
1248 // Smooth far-end energy
1249 for (int j = 0; j <= st->frame_size; j++)
1250 st->power[j] = ss_1 * st->power[j] + 1 + ss * st->Xf[j];
1251
1252 // Compute filtered spectra and correlations
1253 Num Pey = 0.0f, Pyy = 0.0f;
1254 for (int j = st->frame_size; j >= 0; j--) {
1255 Num Eh = st->Rf[j] - st->Eh[j];
1256 Num Yh = st->Yf[j] - st->Yh[j];
1257 Pey = Pey + Eh * Yh;
1258 Pyy = Pyy + Yh * Yh;
1259 st->Eh[j] =
1260 (1 - st->spec_average) * st->Eh[j] + st->spec_average * st->Rf[j];
1261 st->Yh[j] =
1262 (1 - st->spec_average) * st->Yh[j] + st->spec_average * st->Yf[j];
1263 }
1264
1265 Pyy = spxSqrt(Pyy);
1266 // Pyy is 0 on the very first frames (no echo spectrum has built up yet),
1267 // which would make Pey/Pyy evaluate to 0/0 = NaN and permanently poison
1268 // st->Pey / leak_estimate (a NaN comparison is always false, so the
1269 // clamps below can never recover from it). Floor it at 1.0f, same
1270 // as the clamp already applied to st->Pyy a few lines down.
1271 if (Pyy < 1.0f) Pyy = 1.0f;
1272 Pey = Pey / Pyy;
1273
1274 // Compute correlation update rate
1275 Word32 tmp32 = st->beta0 * Syy;
1276 if (tmp32 > st->beta_max * See) tmp32 = st->beta_max * See;
1277 Num alpha = tmp32 / See;
1278 Num alpha_1 = 1.0f - alpha;
1279
1280 st->Pey = alpha_1 * st->Pey + alpha * Pey;
1281 st->Pyy = alpha_1 * st->Pyy + alpha * Pyy;
1282 if (st->Pyy < 1.0f) st->Pyy = 1.0f;
1283 if (st->Pey < min_leak * st->Pyy) st->Pey = min_leak * st->Pyy;
1284 if (st->Pey > st->Pyy) st->Pey = st->Pyy;
1285
1286 st->leak_estimate = st->Pey / st->Pyy;
1287 if (st->leak_estimate > 16383)
1288 st->leak_estimate = 32767;
1289 else
1290 st->leak_estimate = st->leak_estimate * 2;
1291
1292 // Compute RER
1293 Word16 RER;
1294 RER = (.0001f * Sxx + 3.f * st->leak_estimate * Syy) / See;
1295 if (RER < Sey * Sey / (1 + See * Syy)) RER = Sey * Sey / (1 + See * Syy);
1296 if (RER > .5f) RER = .5f;
1297
1298 if (!st->adapted && st->sum_adapt > M &&
1299 st->leak_estimate * Syy > .03f * Syy) {
1300 st->adapted = 1;
1301 }
1302
1303 if (st->adapted) {
1304 for (int i = 0; i <= st->frame_size; i++) {
1305 Word32 r = st->leak_estimate * st->Yf[i];
1306 Word32 e = st->Rf[i] + 1;
1307 if (r > .5f * e) r = .5f * e;
1308 r = .7f * r + .3f * (RER * e);
1309 st->power_1[i] = r / (e * (st->power[i] + 10));
1310 }
1311 } else {
1312 Word16 adapt_rate = 0;
1313 if (Sxx > N * 1000.0f) {
1314 tmp32 = .25f * Sxx;
1315 if (tmp32 > .25f * See) tmp32 = .25f * See;
1316 adapt_rate = tmp32 / See;
1317 }
1318 for (int i = 0; i <= st->frame_size; i++)
1319 st->power_1[i] = adapt_rate / (st->power[i] + 10);
1320 st->sum_adapt = st->sum_adapt + adapt_rate;
1321 }
1322
1323 std::memmove(st->last_y,
1324 &st->last_y[st->frame_size],
1325 st->frame_size * sizeof(Word16));
1326 if (st->adapted) {
1327 for (int i = 0; i < st->frame_size; i++)
1328 st->last_y[st->frame_size + i] = in[i] - out[i];
1329 }
1330 }

1378template <typename T>
1379inline void echo_fft(void* table, T* in, T* out) {
1380 auto* st = static_cast<fft_state*>(table);
1381 if (!st || !st->driver) return;
1382
1383 // Set input values
1384 for (int i = 0; i < st->N; i++) {
1385 st->driver->setValue(i, (float)in[i] / (float)st->N);
1386 }
1387
1388 // Perform FFT
1389 st->driver->fft();
1390
1391 // Get output in packed format: out[0]=real[0], out[1]=real[1],
1392 // out[2]=img[1],
1393 // ...
1394 // Note: getValue() only reflects the frequency domain after rfft()
1395 // (inverse); right after fft() (forward), it is still the raw input, so
1396 // DC/Nyquist must come from getBin() too (its imaginary part is always 0
1397 // for those two bins).
1398 FFTBin dc_bin;
1399 st->driver->getBin(0, dc_bin);
1400 out[0] = dc_bin.real; // DC component
1401 for (int i = 1; i < st->N - 1; i += 2) {
1402 int bin = (i + 1) / 2;
1403 FFTBin fft_bin;
1404 st->driver->getBin(bin, fft_bin);
1405 out[i] = fft_bin.real;
1406 out[i + 1] = fft_bin.img;
1407 }
1408 FFTBin nyquist_bin;
1409 st->driver->getBin(st->N / 2, nyquist_bin);
1410 out[st->N - 1] = nyquist_bin.real; // Nyquist
1411}

1424 if (!st || !st->driver || !st->driver->isReverseFFT()) return;

1426 // Set bins from packed format
1427 st->driver->setBin(0, (float)in[0], 0);

1429 int bin = (i + 1) / 2;

1436
1437 // Get output
1438 for (int i = 0; i < st->N; i++) {
1439 out[i] = st->driver->getValue(i);
1440 }
