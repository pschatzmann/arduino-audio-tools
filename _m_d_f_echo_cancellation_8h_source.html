183// Fixed-point type for pseudo-float operations
184struct echo_float_t {
185 int16_t m; // Mantissa
186 int16_t e; // Exponent
187
188 // Constructor for initialization
189 constexpr echo_float_t(int16_t mantissa = 0, int16_t exponent = 0)
190 : m(mantissa), e(exponent) {}
191
192 // Operator overloads for arithmetic
193 inline echo_float_t operator*(const echo_float_t& other) const {
194 return echo_float_mult(*this, other);
195 }
196
197 inline echo_word32_t operator*(int16_t scalar) const {
198 // Result is a regular integer when multiplying float by int
199 int32_t result = ((int32_t)m * scalar);
200 if (e >= 14) {
201 return result << (e - 14);
202 } else {
203 return result >> (14 - e);
204 }
205 }
206
207 inline echo_word32_t operator*(int32_t scalar) const {
208 // Result is a regular integer when multiplying float by int
209 int64_t result = ((int64_t)m * scalar);
210 if (e >= 14) {
211 return result << (e - 14);
212 } else {
213 return result >> (14 - e);
214 }
215 }
216
217 friend inline echo_word32_t operator*(int32_t scalar, const echo_float_t& f) {
218 return f * scalar;
219 }
220};

222// Fixed-point arithmetic helper functions
223inline echo_float_t echo_float_from_double(double x) {
224 echo_float_t r;
225 if (x == 0) {
226 r.m = 0;
227 r.e = 0;
228 return r;
229 }
230 int e = 0;
231 while (x >= 32768) { x *= 0.5; e++; }
232 while (x < 16384) { x *= 2.0; e--; }
233 r.m = (int16_t)x;
234 r.e = (int16_t)e;
235 return r;
236}
237
238inline echo_float_t echo_float_mult(echo_float_t a, echo_float_t b) {
239 echo_float_t r;
240 r.m = (int16_t)((((int32_t)a.m) * b.m) >> 15);
241 r.e = a.e + b.e + 15;
242 if (r.m > 0) {
243 while (r.m < 16384) { r.m <<= 1; r.e--; }
244 }
245 return r;
246}
247
248inline echo_float_t echo_float_mul32_u(float a, float b) {
249 return echo_float_from_double(a * b);
250}
251
252inline echo_float_t echo_float_add(echo_float_t a, echo_float_t b) {
253 if (a.e > b.e) {
254 int shift = a.e - b.e;
255 if (shift > 15) return a;
256 echo_float_t r = {(int16_t)(a.m + (b.m >> shift)), a.e};
257 return r;
258 } else {
259 int shift = b.e - a.e;
260 if (shift > 15) return b;
261 echo_float_t r = {(int16_t)((a.m >> shift) + b.m), b.e};
262 return r;
263 }
264}
265
266inline echo_float_t echo_float_sub(echo_float_t a, echo_float_t b) {
267 if (a.e > b.e) {
268 int shift = a.e - b.e;
269 if (shift > 15) return a;
270 echo_float_t r = {(int16_t)(a.m - (b.m >> shift)), a.e};
271 return r;
272 } else {
273 int shift = b.e - a.e;
274 if (shift > 15) {
275 echo_float_t r = {(int16_t)(-b.m), b.e};
276 return r;
277 }
278 echo_float_t r = {(int16_t)((a.m >> shift) - b.m), b.e};
279 return r;
280 }
281}

283inline bool echo_float_lt(echo_float_t a, echo_float_t b) {
284 if (a.e != b.e) return a.e < b.e;
285 return a.m < b.m;
286}

288inline bool echo_float_gt(echo_float_t a, echo_float_t b) {
289 if (a.e != b.e) return a.e > b.e;
290 return a.m > b.m;
291}

293inline echo_float_t echo_float_divu(echo_float_t a, echo_float_t b) {
294 echo_float_t r;
295 if (b.m == 0) {
296 r.m = 32767;
297 r.e = 15;
298 return r;
299 }
300 r.m = (int16_t)((((int32_t)a.m) << 15) / b.m);
301 r.e = a.e - b.e - 15;
302 if (r.m > 0) {
303 while (r.m < 16384) { r.m <<= 1; r.e--; }
304 }
305 return r;
306}

308inline echo_float_t echo_float_sqrt(echo_float_t x) {
309 double val = x.m * pow(2.0, x.e - 14);
310 return echo_float_from_double(sqrt(val));
311}

313inline int16_t echo_float_extract16(echo_float_t x) {
314 if (x.e > 0) {
315 return x.m << (x.e > 15 ? 15 : x.e);
316 } else {
317 return x.m >> (-x.e > 15 ? 15 : -x.e);
318 }
319}

321inline echo_float_t echo_float_shl(echo_float_t x, int bits) {
322 echo_float_t r = {x.m, (int16_t)(x.e + bits)};
323 return r;
324}

636 void capture(const echo_int16_t* rec, echo_int16_t* out) {
637 ensureInitialized();
638 state->play_buf_started = 1;
639 if (state->play_buf_pos >= state->frame_size) {
640 echoCancellationImpl(state, rec, state->play_buf, out);
641 state->play_buf_pos -= state->frame_size;
642 std::memmove(state->play_buf,
643 &state->play_buf[state->frame_size],
644 state->play_buf_pos * sizeof(echo_int16_t));
645 } else {
646 echoWarning("No playback frame available");
647 if (state->play_buf_pos != 0) {
648 echoWarning("Internal playback buffer corruption");
649 state->play_buf_pos = 0;
650 }
651 for (int i = 0; i < state->frame_size; i++) out[i] = rec[i];
652 }
653 }

658 void playback(const echo_int16_t* play) {
659 ensureInitialized();
660 if (!state->play_buf_started) {
661 echoWarning("Discarded first playback frame");
662 return;
663 }
664 if (state->play_buf_pos <= PLAYBACK_DELAY * state->frame_size) {
665 for (int i = 0; i < state->frame_size; i++)
666 state->play_buf[state->play_buf_pos + i] = play[i];
667 state->play_buf_pos += state->frame_size;
668 if (state->play_buf_pos <= (PLAYBACK_DELAY - 1) * state->frame_size) {
669 echoWarning("Auto-filling buffer");
670 for (int i = 0; i < state->frame_size; i++)
671 state->play_buf[state->play_buf_pos + i] = play[i];
672 state->play_buf_pos += state->frame_size;
673 }
674 } else {
675 echoWarning("Had to discard playback frame");
676 }
677 }

680 void reset() {
681 ensureInitialized();
682 int N = state->window_size;
683 int M = state->M;
684 int C = state->C;
685 int K = state->K;
686
687 state->cancel_count = 0;
688 state->screwed_up = 0;
689
690 for (int i = 0; i < N * M; i++) state->W[i] = 0;
691#ifdef TWO_PATH
692 for (int i = 0; i < N * M; i++) state->foreground[i] = 0;
693#endif
694 for (int i = 0; i < N * (M + 1); i++) state->X[i] = 0;
695 for (int i = 0; i <= state->frame_size; i++) {
696 state->power[i] = 0;
697 state->power_1[i] = FLOAT_ONE;
698 state->Eh[i] = 0;
699 state->Yh[i] = 0;
700 }
701 for (int i = 0; i < state->frame_size; i++) state->last_y[i] = 0;
702 for (int i = 0; i < N * C; i++) state->E[i] = 0;
703 for (int i = 0; i < N * K; i++) state->x[i] = 0;
704 for (int i = 0; i < 2 * C; i++) state->notch_mem[i] = 0;
705 for (int i = 0; i < C; i++) state->memD[i] = state->memE[i] = 0;
706 for (int i = 0; i < K; i++) state->memX[i] = 0;
707
708 state->saturated = 0;
709 state->adapted = 0;
710 state->sum_adapt = 0;
711 state->Pey = state->Pyy = FLOAT_ONE;
712#ifdef TWO_PATH
713 state->Davg1 = state->Davg2 = 0;
714 state->Dvar1 = state->Dvar2 = FLOAT_ZERO;

944 inline void filterDcNotch16(const echo_int16_t* in, echo_word16_t radius,
945 echo_word16_t* out, int len, echo_mem_t* mem,
946 int stride) {
947 echo_word16_t den2 = radius * radius + .7f * (1 - radius) * (1 - radius);
948 for (int i = 0; i < len; i++) {
949 echo_word16_t vin = in[i * stride];
950 echo_word32_t vout = mem[0] + vin;
951 mem[0] = mem[1] + 2 * (-vin + radius * vout);
952 mem[1] = vin - den2 * vout;
953 out[i] = radius * vout;
954 }
955 }

1010 inline void spectralMulAccum(const echo_word16_t* X, const echo_word32_t* Y,
1011 echo_word16_t* acc, int N, int M) {
1012 for (int i = 0; i < N; i++) acc[i] = 0;
1013
1014 for (int j = 0; j < M; j++) {
1015 acc[0] += X[0] * Y[0];
1016 for (int i = 1; i < N - 1; i += 2) {
1017 acc[i] += (X[i] * Y[i] - X[i + 1] * Y[i + 1]);
1018 acc[i + 1] += (X[i + 1] * Y[i] + X[i] * Y[i + 1]);
1019 }
1020 acc[N - 1] += X[N - 1] * Y[N - 1];
1021 X += N;
1022 Y += N;
1023 }

1035 inline void weightedSpectralMulConj(const echo_float_t* w, const echo_float_t p,
1036 const echo_word16_t* X,
1037 const echo_word16_t* Y, echo_word32_t* prod,
1038 int N) {
1039 // NOTE: the (int32_t) casts this code originally had here were only
1040 // correct in true fixed-point mode (X/Y as int16_t, needing widening
1041 // before multiplying to avoid 16-bit overflow). With echo_word16_t /
1042 // echo_word32_t now float (see the aliases above), those casts instead
1043 // truncated every frequency-domain sample to an integer before ever
1044 // multiplying it by anything -- e.g. a bin magnitude of 0.03 became 0,
1045 // silently zeroing out the filter's gradient. Removed.
1046 echo_float_t W;
1047 W = p * w[0];
1048 prod[0] = W * (X[0] * Y[0]);
1049 for (int i = 1, j = 1; i < N - 1; i += 2, j++) {
1050 W = p * w[j];
1051 prod[i] = W * (X[i] * Y[i] + X[i + 1] * Y[i + 1]);
1052 prod[i + 1] = W * (-X[i + 1] * Y[i] + X[i] * Y[i + 1]);
1053 }
1054 W = p * w[N / 2];
1055 prod[N - 1] = W * (X[N - 1] * Y[N - 1]);
1056 }

1066 inline void mdfAdjustProp(const echo_word32_t* W, int N, int M, int P,
1067 echo_word16_t* prop) {
1068 echo_word16_t max_sum = 1;
1069 echo_word32_t prop_sum = 1;
1070
1071 for (int i = 0; i < M; i++) {
1072 echo_word32_t tmp = 1;
1073 for (int p = 0; p < P; p++) {
1074 for (int j = 0; j < N; j++) {
1075 float val = W[p * N * M + i * N + j];
1076 tmp += val * val;
1077 }
1078 }
1079 prop[i] = spxSqrt(tmp);
1080 if (prop[i] > max_sum) max_sum = prop[i];
1081 }
1082
1083 for (int i = 0; i < M; i++) {
1084 prop[i] += 0.1f * max_sum;
1085 prop_sum += prop[i];
1086 }
1087
1088 for (int i = 0; i < M; i++) {
1089 prop[i] = 0.99f * prop[i] / prop_sum;
1090 }
1091 }

1202 st->play_buf_started = 0;
1203
1204 st->fft_table = nullptr;

1225 inline void echoCancellationImpl(EchoState* st, const echo_int16_t* in,
1226 const echo_int16_t* far_end,
1227 echo_int16_t* out) {
1228 int N = st->window_size;
1229 int M = st->M;
1230 int C = st->C;
1231 int K = st->K;
1232
1233 st->cancel_count++;
1234 float ss = .35f / M;
1235 float ss_1 = 1 - ss;
1236
1237 // Apply notch filter and pre-emphasis to input
1238 for (int chan = 0; chan < C; chan++) {
1239 filterDcNotch16(in + chan, st->notch_radius,
1240 st->input + chan * st->frame_size, st->frame_size,
1241 st->notch_mem + 2 * chan, C);
1242
1243 for (int i = 0; i < st->frame_size; i++) {
1244 echo_word32_t tmp32 =
1245 st->input[chan * st->frame_size + i] - st->preemph * st->memD[chan];
1246 st->memD[chan] = st->input[chan * st->frame_size + i];
1247 st->input[chan * st->frame_size + i] = tmp32;
1248 }
1249 }
1250
1251 // Process far-end signal
1252 for (int speak = 0; speak < K; speak++) {
1253 std::memmove(&st->x[speak * N],
1254 &st->x[speak * N + st->frame_size],
1255 st->frame_size * sizeof(echo_word16_t));
1256 for (int i = 0; i < st->frame_size; i++) {
1257 echo_word32_t tmp32 =
1258 far_end[i * K + speak] - st->preemph * st->memX[speak];
1259 st->x[speak * N + i + st->frame_size] = tmp32;
1260 st->memX[speak] = far_end[i * K + speak];
1261 }
1262 }
1263
1264 // Shift memory and compute FFT of far-end
1265 for (int speak = 0; speak < K; speak++) {
1266 for (int j = M - 1; j >= 0; j--) {
1267 std::memmove(&st->X[(j + 1) * N * K + speak * N],
1268 &st->X[j * N * K + speak * N],
1269 N * sizeof(echo_word16_t));
1270 }
1271 echo_fft(st->fft_table, st->x + speak * N, &st->X[speak * N]);
1272 }
1273
1274 // Compute power spectrum of far-end
1275 echo_word32_t Sxx = 0;
1276 for (int speak = 0; speak < K; speak++) {
1277 Sxx += mdfInnerProd(st->x + speak * N + st->frame_size,
1278 st->x + speak * N + st->frame_size, st->frame_size);
1279 powerSpectrumAccum(st->X + speak * N, st->Xf, N);
1280 }
1281
1282 // Compute foreground filter output and residual
1283 echo_word32_t Sff = 0;
1284 for (int chan = 0; chan < C; chan++) {
1285#ifdef TWO_PATH
1286 spectralMulAccum(st->X, st->foreground + chan * N * K * M,
1287 st->Y + chan * N, N, M * K);
1288 echo_ifft(st->fft_table, st->Y + chan * N, st->e + chan * N);
1289 for (int i = 0; i < st->frame_size; i++)
1290 st->e[chan * N + i] = st->input[chan * st->frame_size + i] -
1291 st->e[chan * N + i + st->frame_size];
1292 Sff += mdfInnerProd(st->e + chan * N, st->e + chan * N, st->frame_size);
1293#endif
1294 }
1295
1296 // Adjust proportional adaptation
1297 if (st->adapted) mdfAdjustProp(st->W, N, M, C * K, st->prop);
1298
1299 // Compute weight gradient
1300 if (st->saturated == 0) {
1301 for (int chan = 0; chan < C; chan++) {
1302 for (int speak = 0; speak < K; speak++) {
1303 for (int j = M - 1; j >= 0; j--) {
1304 weightedSpectralMulConj(st->power_1, st->prop[j],
1305 &st->X[(j + 1) * N * K + speak * N],
1306 st->E + chan * N, st->PHI, N);
1307 for (int i = 0; i < N; i++)
1308 st->W[chan * N * K * M + j * N * K + speak * N + i] += st->PHI[i];
1309 }
1310 }
1311 }
1312 } else {
1313 st->saturated--;
1314 }
1315
1316 // Update weights (AUMDF)
1317 for (int chan = 0; chan < C; chan++) {
1318 for (int speak = 0; speak < K; speak++) {
1319 for (int j = 0; j < M; j++) {
1320 if (j == 0 || st->cancel_count % (M - 1) == j - 1) {
1321 echo_ifft(
1322 st->fft_table, &st->W[chan * N * K * M + j * N * K + speak * N],
1323 st->wtmp);
1324 for (int i = st->frame_size; i < N; i++) st->wtmp[i] = 0;
1325 echo_fft(
1326 st->fft_table, st->wtmp,
1327 &st->W[chan * N * K * M + j * N * K + speak * N]);
1328 }
1329 }
1330 }
1331 }
1332
1333 // Initialize spectrum buffers
1334 for (int i = 0; i <= st->frame_size; i++)
1335 st->Rf[i] = st->Yf[i] = st->Xf[i] = 0;
1336
1337 echo_word32_t Dbf = 0;
1338 echo_word32_t See = 0;
1339
1340#ifdef TWO_PATH
1341 // Compute background filter output
1342 for (int chan = 0; chan < C; chan++) {
1343 spectralMulAccum(st->X, st->W + chan * N * K * M, st->Y + chan * N, N,
1344 M * K);
1345 echo_ifft(st->fft_table, st->Y + chan * N, st->y + chan * N);
1346 for (int i = 0; i < st->frame_size; i++)
1347 st->e[chan * N + i] = st->e[chan * N + i + st->frame_size] -
1348 st->y[chan * N + i + st->frame_size];
1349 Dbf +=
1350 10 + mdfInnerProd(st->e + chan * N, st->e + chan * N, st->frame_size);
1351 for (int i = 0; i < st->frame_size; i++)
1352 st->e[chan * N + i] = st->input[chan * st->frame_size + i] -
1353 st->y[chan * N + i + st->frame_size];
1354 See += mdfInnerProd(st->e + chan * N, st->e + chan * N, st->frame_size);
1355 }
1356#endif
1357
1358#ifndef TWO_PATH
1359 Sff = See;
1360#endif
1361
1362#ifdef TWO_PATH
1363 // Two-path filter logic
1364 st->Davg1 = .6f * st->Davg1 + .4f * (Sff - See);
1365 st->Davg2 = .85f * st->Davg2 + .15f * (Sff - See);
1366 st->Dvar1 = FLOAT_ADD(FLOAT_MULT(VAR1_SMOOTH, st->Dvar1),
1367 FLOAT_MUL32U(.4f * Sff, .4f * Dbf));
1368 st->Dvar2 = FLOAT_ADD(FLOAT_MULT(VAR2_SMOOTH, st->Dvar2),
1369 FLOAT_MUL32U(.15f * Sff, .15f * Dbf));
1370
1371 int update_foreground = 0;
1372 if (FLOAT_GT(FLOAT_MUL32U(Sff - See, fabsf(Sff - See)),
1373 FLOAT_MUL32U(Sff, Dbf)))
1374 update_foreground = 1;
1375 else if (FLOAT_GT(FLOAT_MUL32U(st->Davg1, fabsf(st->Davg1)),
1376 FLOAT_MULT(VAR1_UPDATE, st->Dvar1)))
1377 update_foreground = 1;
1378 else if (FLOAT_GT(FLOAT_MUL32U(st->Davg2, fabsf(st->Davg2)),
1379 FLOAT_MULT(VAR2_UPDATE, st->Dvar2)))
1380 update_foreground = 1;
1381
1382 if (update_foreground) {
1383 st->Davg1 = st->Davg2 = 0;
1384 st->Dvar1 = st->Dvar2 = FLOAT_ZERO;
1385 std::memcpy(st->foreground, st->W, N * M * C * K * sizeof(echo_word16_t));
1386 for (int chan = 0; chan < C; chan++)
1387 for (int i = 0; i < st->frame_size; i++)
1388 st->e[chan * N + i + st->frame_size] =
1389 st->window[i + st->frame_size] *
1390 st->e[chan * N + i + st->frame_size] +
1391 st->window[i] * st->y[chan * N + i + st->frame_size];
1392 } else {
1393 int reset_background = 0;
1394 if (FLOAT_GT(FLOAT_MUL32U(-(Sff - See), fabsf(Sff - See)),
1395 FLOAT_MULT(VAR_BACKTRACK, FLOAT_MUL32U(Sff, Dbf))))
1396 reset_background = 1;
1397 if (FLOAT_GT(FLOAT_MUL32U(-st->Davg1, fabsf(st->Davg1)),
1398 FLOAT_MULT(VAR_BACKTRACK, st->Dvar1)))
1399 reset_background = 1;
1400 if (FLOAT_GT(FLOAT_MUL32U(-st->Davg2, fabsf(st->Davg2)),
1401 FLOAT_MULT(VAR_BACKTRACK, st->Dvar2)))
1402 reset_background = 1;
1403
1404 if (reset_background) {
1405 std::memcpy(st->W, st->foreground, N * M * C * K * sizeof(echo_word32_t));
1406 for (int chan = 0; chan < C; chan++) {
1407 for (int i = 0; i < st->frame_size; i++)
1408 st->y[chan * N + i + st->frame_size] =
1409 st->e[chan * N + i + st->frame_size];
1410 for (int i = 0; i < st->frame_size; i++)
1411 st->e[chan * N + i] = st->input[chan * st->frame_size + i] -
1412 st->y[chan * N + i + st->frame_size];
1413 }
1414 See = Sff;
1415 st->Davg1 = st->Davg2 = 0;
1416 st->Dvar1 = st->Dvar2 = FLOAT_ZERO;
1417 }
1418 }
1419#endif
1420
1421 echo_word32_t Sey = 0, Syy = 0, Sdd = 0;
1422 for (int chan = 0; chan < C; chan++) {
1423 // Compute output with de-emphasis
1424 for (int i = 0; i < st->frame_size; i++) {
1425 echo_word32_t tmp_out;
1426#ifdef TWO_PATH
1427 tmp_out = st->input[chan * st->frame_size + i] -
1428 st->e[chan * N + i + st->frame_size];
1429#else
1430 tmp_out = st->input[chan * st->frame_size + i] -
1431 st->y[chan * N + i + st->frame_size];
1432#endif
1433 tmp_out = tmp_out + st->preemph * st->memE[chan];
1434 if (in[i * C + chan] <= -32000 || in[i * C + chan] >= 32000) {
1435 if (st->saturated == 0) st->saturated = 1;
1436 }
1437 out[i * C + chan] = WORD2INT(tmp_out);
1438 st->memE[chan] = tmp_out;
1439 }
1440
1441 // Prepare error signal for filter update
1442 for (int i = 0; i < st->frame_size; i++) {
1443 st->e[chan * N + i + st->frame_size] = st->e[chan * N + i];
1444 st->e[chan * N + i] = 0;
1445 }
1446
1447 // Compute correlations
1448 Sey += mdfInnerProd(st->e + chan * N + st->frame_size,
1449 st->y + chan * N + st->frame_size, st->frame_size);
1450 Syy += mdfInnerProd(st->y + chan * N + st->frame_size,
1451 st->y + chan * N + st->frame_size, st->frame_size);
1452 Sdd += mdfInnerProd(st->input + chan * st->frame_size,
1453 st->input + chan * st->frame_size, st->frame_size);
1454
1455 // Convert error to frequency domain
1456 echo_fft(st->fft_table, st->e + chan * N, st->E + chan * N);
1457
1458 for (int i = 0; i < st->frame_size; i++) st->y[i + chan * N] = 0;
1459 echo_fft(st->fft_table, st->y + chan * N, st->Y + chan * N);
1460
1461 // Compute power spectra
1462 powerSpectrumAccum(st->E + chan * N, st->Rf, N);
1463 powerSpectrumAccum(st->Y + chan * N, st->Yf, N);
1464 }
1465
1466 // Sanity checks
1467 if (!(Syy >= 0 && Sxx >= 0 && See >= 0) ||
1468 !(Sff < N * 1e9f && Syy < N * 1e9f && Sxx < N * 1e9f)) {
1469 st->screwed_up += 50;
1470 for (int i = 0; i < st->frame_size * C; i++) out[i] = 0;
1471 } else if (Sff > Sdd + N * 10000.0f) {
1472 st->screwed_up++;
1473 } else {
1474 st->screwed_up = 0;
1475 }
1476
1477 if (st->screwed_up >= 50) {
1478 echoWarning("Echo canceller reset");
1479 reset();
1480 return;
1481 }
1482
1483 See = MAX32(See, N * 100.0f);
1484
1485 for (int speak = 0; speak < K; speak++) {
1486 Sxx += mdfInnerProd(st->x + speak * N + st->frame_size,
1487 st->x + speak * N + st->frame_size, st->frame_size);
1488 powerSpectrumAccum(st->X + speak * N, st->Xf, N);
1489 }
1490
1491 // Smooth far-end energy
1492 for (int j = 0; j <= st->frame_size; j++)
1493 st->power[j] = ss_1 * st->power[j] + 1 + ss * st->Xf[j];
1494
1495 // Compute filtered spectra and correlations
1496 echo_float_t Pey = FLOAT_ZERO, Pyy = FLOAT_ZERO;
1497 for (int j = st->frame_size; j >= 0; j--) {
1498 echo_float_t Eh = PSEUDOFLOAT(st->Rf[j] - st->Eh[j]);
1499 echo_float_t Yh = PSEUDOFLOAT(st->Yf[j] - st->Yh[j]);
1500 Pey = FLOAT_ADD(Pey, FLOAT_MULT(Eh, Yh));
1501 Pyy = FLOAT_ADD(Pyy, FLOAT_MULT(Yh, Yh));
1502 st->Eh[j] =
1503 (1 - st->spec_average) * st->Eh[j] + st->spec_average * st->Rf[j];
1504 st->Yh[j] =
1505 (1 - st->spec_average) * st->Yh[j] + st->spec_average * st->Yf[j];
1506 }
1507
1508 Pyy = FLOAT_SQRT(Pyy);
1509 // Pyy is 0 on the very first frames (no echo spectrum has built up yet),
1510 // which would make Pey/Pyy evaluate to 0/0 = NaN and permanently poison
1511 // st->Pey / leak_estimate (FLOAT_LT against NaN is always false, so the
1512 // clamps below can never recover from it). Floor it at FLOAT_ONE, same
1513 // as the clamp already applied to st->Pyy a few lines down.
1514 if (FLOAT_LT(Pyy, FLOAT_ONE)) Pyy = FLOAT_ONE;
1515 Pey = FLOAT_DIVU(Pey, Pyy);
1516
1517 // Compute correlation update rate
1518 echo_word32_t tmp32 = st->beta0 * Syy;
1519 if (tmp32 > st->beta_max * See) tmp32 = st->beta_max * See;
1520 echo_float_t alpha = tmp32 / See;
1521 echo_float_t alpha_1 = FLOAT_SUB(FLOAT_ONE, alpha);
1522
1523 st->Pey = FLOAT_ADD(FLOAT_MULT(alpha_1, st->Pey), FLOAT_MULT(alpha, Pey));
1524 st->Pyy = FLOAT_ADD(FLOAT_MULT(alpha_1, st->Pyy), FLOAT_MULT(alpha, Pyy));
1525 if (FLOAT_LT(st->Pyy, FLOAT_ONE)) st->Pyy = FLOAT_ONE;
1526 if (FLOAT_LT(st->Pey, FLOAT_MULT(MIN_LEAK, st->Pyy)))
1527 st->Pey = FLOAT_MULT(MIN_LEAK, st->Pyy);
1528 if (FLOAT_GT(st->Pey, st->Pyy)) st->Pey = st->Pyy;
1529
1530 st->leak_estimate =
1531 FLOAT_EXTRACT16(FLOAT_SHL(FLOAT_DIVU(st->Pey, st->Pyy), 14));
1532 if (st->leak_estimate > 16383)
1533 st->leak_estimate = 32767;
1534 else
1535 st->leak_estimate = st->leak_estimate * 2;
1536
1537 // Compute RER
1538 echo_word16_t RER;
1539 RER = (.0001f * Sxx + 3.f * st->leak_estimate * Syy) / See;
1540 if (RER < Sey * Sey / (1 + See * Syy)) RER = Sey * Sey / (1 + See * Syy);
1541 if (RER > .5f) RER = .5f;
1542
1543 if (!st->adapted && st->sum_adapt > M &&
1544 st->leak_estimate * Syy > .03f * Syy) {
1545 st->adapted = 1;
1546 }
1547
1548 if (st->adapted) {
1549 for (int i = 0; i <= st->frame_size; i++) {
1550 echo_word32_t r = st->leak_estimate * st->Yf[i];
1551 echo_word32_t e = st->Rf[i] + 1;
1552 if (r > .5f * e) r = .5f * e;
1553 r = .7f * r + .3f * (RER * e);
1554 st->power_1[i] = r / (e * (st->power[i] + 10));
1555 }
1556 } else {
1557 echo_word16_t adapt_rate = 0;
1558 if (Sxx > N * 1000.0f) {
1559 tmp32 = .25f * Sxx;
1560 if (tmp32 > .25f * See) tmp32 = .25f * See;
1561 adapt_rate = tmp32 / See;
1562 }
1563 for (int i = 0; i <= st->frame_size; i++)
1564 st->power_1[i] = adapt_rate / (st->power[i] + 10);
1565 st->sum_adapt = st->sum_adapt + adapt_rate;
1566 }
1567
1568 std::memmove(st->last_y,
1569 &st->last_y[st->frame_size],
1570 st->frame_size * sizeof(echo_word16_t));
1571 if (st->adapted) {
1572 for (int i = 0; i < st->frame_size; i++)
1573 st->last_y[st->frame_size + i] = in[i] - out[i];
1574 }
1575 }
1576};

1622inline void echo_fft(void* table, echo_word16_t* in,
1623 echo_word16_t* out) {
1624 auto* st = static_cast<fft_state*>(table);
1625 if (!st || !st->driver) return;
1626
1627 // Set input values
1628 for (int i = 0; i < st->N; i++) {
1629 st->driver->setValue(i, in[i] / (float)st->N);
1630 }
1631
1632 // Perform FFT
1633 st->driver->fft();
1634
1635 // Get output in packed format: out[0]=real[0], out[1]=real[1],
1636 // out[2]=img[1],
1637 // ...
1638 // Note: getValue() only reflects the frequency domain after rfft()
1639 // (inverse); right after fft() (forward), it is still the raw input, so
1640 // DC/Nyquist must come from getBin() too (its imaginary part is always 0
1641 // for those two bins).
1642 FFTBin dc_bin;
1643 st->driver->getBin(0, dc_bin);
1644 out[0] = dc_bin.real; // DC component
1645 for (int i = 1; i < st->N - 1; i += 2) {
1646 int bin = (i + 1) / 2;
1647 FFTBin fft_bin;
1648 st->driver->getBin(bin, fft_bin);
1649 out[i] = fft_bin.real;
1650 out[i + 1] = fft_bin.img;
1651 }
1652 FFTBin nyquist_bin;
1653 st->driver->getBin(st->N / 2, nyquist_bin);
1654 out[st->N - 1] = nyquist_bin.real; // Nyquist
1655}

1664inline void echo_ifft(void* table, echo_word16_t* in,
1665 echo_word16_t* out) {
1666 auto* st = static_cast<fft_state*>(table);
1667 if (!st || !st->driver || !st->driver->isReverseFFT()) return;
1668
1669 // Set bins from packed format
1670 st->driver->setBin(0, in[0], 0);
1671 for (int i = 1; i < st->N - 1; i += 2) {
1672 int bin = (i + 1) / 2;
1673 st->driver->setBin(bin, in[i], in[i + 1]);
1674 }
1675 st->driver->setBin(st->N / 2, in[st->N - 1], 0);
1676
1677 // Perform inverse FFT
1678 st->driver->rfft();
1679
1680 // Get output
1681 for (int i = 0; i < st->N; i++) {
1682 out[i] = st->driver->getValue(i);
1683 }
1684}
