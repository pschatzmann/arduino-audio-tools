28class WindowFunction {
29 public:
30 WindowFunction() = default;

33 virtual void begin(int samples) {

41 inline float factor(int idx) {
42 assert(i_half_samples == i_samples / 2);
43 float result = idx <= i_half_samples ? factor_internal(idx)
44 : factor_internal(i_samples - idx - 1);
45 return result > 1.0f ? 1.0f : result;
46 }
47
49 inline int samples() { return i_samples; }
50
51 virtual const char* name() = 0;
52
53 protected:
54 float samples_minus_1 = 0.0f;
55 int i_samples = 0;
56 int i_half_samples = 0;
57 const float twoPi = 6.28318531f;
58 const float fourPi = 12.56637061f;
59 const float sixPi = 18.84955593f;
60
61 // virtual function provide implementation in subclass
62 virtual float factor_internal(int idx) = 0;
63
64 // the ratio idx / samples -1
65 inline float ratio(int idx) {
66 return (static_cast<float>(idx)) / samples_minus_1;
67 }

77class BufferedWindow : public WindowFunction {

79 BufferedWindow(WindowFunction* wf) { p_wf = wf; }
80
81 const char* name() override {
82 static char buffer[80] = "Buffered ";
83 strncpy(buffer + 9, p_wf->name(), 69);
84 return buffer;
85 }
86
87 virtual void begin(int samples) override {
88 // process only if there is a change
89 WindowFunction::begin(samples);
90 if (p_wf->samples() != samples) {
91 p_wf->begin(samples);
92 int to_be_size = i_half_samples + 1;
93 if (buffer.size() != to_be_size) {
94 buffer.resize(to_be_size);
95 for (int j = 0; j <= i_half_samples; j++) {
96 buffer[j] = p_wf->factor(j);
97 }
98 }
99 }
100 }
101
102 protected:
103 WindowFunction* p_wf = nullptr;
104 Vector<float> buffer{0};
105
106 float factor_internal(int idx) override {
107 if (idx < 0 || idx > i_half_samples) return 0.0;
108 return buffer[idx];
109 }
