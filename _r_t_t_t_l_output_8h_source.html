210 void play_note(float freq, int msec, int midi = -1) {
211 // invoke the optional callback first
212 LOGI("play_note: freq=%.2f Hz, msec=%d, midi=%d", freq, msec, midi);
213 if (noteCallback) noteCallback(freq, msec, midi);
214 if (p_print == nullptr || p_generator == nullptr) return;
215 p_generator->setFrequency(freq);
216 AudioInfo info = audioInfo();
217 if (!info) return;
218 int frames = (int)((uint64_t)info.sample_rate * (uint64_t)msec / 1000);
219 int frame_size = (info.channels * info.bits_per_sample) / 8;
220 int open = frames * frame_size;
221 uint8_t buffer[1024];
222 while (open > 0) {
223 int toCopy = std::min(open, (int)sizeof(buffer));
224 p_generator->readBytes(buffer, toCopy);
225 p_print->write(buffer, toCopy);
226 open -= toCopy;
227 delay(1);
228 }

260 void parse_defaults() {

290 void parse_notes() {
291 // Ensure we start reading after the defaults
292 // section
293 if (m_actual == ':') next_char();
294
295 while (m_actual != 0) {
296 // skip separators
297 while (m_actual == ' ' || m_actual == ',') {
298 if (next_char() == 0) return;
299 }
300
301 if (m_actual == 0) break;
302
303 // optional duration (number)
304 int duration = m_duration;
305 if (isdigit(m_actual)) {
306 duration = parse_num();
307 if (m_actual == 0) break;
308 }
309
310 // note letter or rest
311 char name = m_actual;
312 if (name == 'p' || name == 'r') { // pause/rest
313 next_char();
314 double mult = 1.0;
315 if (m_actual == '.') {
316 mult = 1.5;
317 next_char();
318 }
319 int msec = (int)((m_msec_semi / duration) * mult);
320 play_note(0.0f, msec, -1);
321 continue;
322 }
323
324 // map letter to MusicalNotes enum
325 MusicalNotes::MusicalNotesEnum noteEnum = MusicalNotes::C;
326 switch (name) {
327 case 'c':
328 noteEnum = MusicalNotes::C;
329 break;
330 case 'd':
331 noteEnum = MusicalNotes::D;
332 break;
333 case 'e':
334 noteEnum = MusicalNotes::E;
335 break;
336 case 'f':
337 noteEnum = MusicalNotes::F;
338 break;
339 case 'g':
340 noteEnum = MusicalNotes::G;
341 break;
342 case 'a':
343 noteEnum = MusicalNotes::A;
344 break;
345 case 'b':
346 case 'h':
347 noteEnum = MusicalNotes::B;
348 break;
349 default:
350 // unknown symbol -> skip
351 next_char();
352 continue;
353 }
354
355 // advance to next char to inspect
356 // accidental/octave/dot
357 if (next_char() == 0) break;
358
359 // accidental
360 int semitone = (int)noteEnum;
361 if (m_actual == '#') {
362 semitone++;
363 if (semitone > 11) {
364 semitone = 0;
365 m_octave += 1;
366 }
367 noteEnum = (MusicalNotes::MusicalNotesEnum)semitone;
368 if (next_char() == 0) break;
369 }
370
371 // octave
372 int octave = m_octave;
373 if (m_actual >= '0' && m_actual <= '9') {
374 octave = m_actual - '0';
375 if (next_char() == 0) {
376 // compute and play
377 float freq = m_notes.frequency(noteEnum, (uint8_t)octave);
378 int msec = (int)((m_msec_semi / duration) * 1.0);
379 int midi = m_notes.frequencyToMidiNote(freq);
380 play_note(freq, msec, midi);
381 break;
382 }
383 }
384
385 // dotted duration
386 double mult_duration = 1.0;
387 if (m_actual == '.') {
388 mult_duration = 1.5;
389 next_char();
390 }
391
392 float freq = m_notes.frequency(noteEnum, (uint8_t)octave);
393 int msec = (int)((m_msec_semi / duration) * mult_duration);
394 int midi = m_notes.frequencyToMidiNote(freq);
395 play_note(freq, msec, midi);
396 }
397 }
