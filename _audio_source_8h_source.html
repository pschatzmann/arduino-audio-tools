178class NamePrinter : public Print {
179 public:
180 NamePrinter(PathNamesRegistry& dataSource, const char* prefix = nullptr)
181 : dataSource(dataSource), prefix(prefix) {}
182 void setPrefix(const char* prefix) { this->prefix = prefix; }
183 size_t write(uint8_t ch) override {
184 if (ch == '\n' || ch == '\r') {
185 // End of line - process the accumulated line
186 if (line_buffer.length() > 0) {
187 line_buffer.trim();
188 if (prefix != nullptr) {
189 // Prepend prefix if set
190 Str name{prefix};
191 name.add("/");
192 name.add(line_buffer.c_str());
193 LOGD("adding '%s'", name.c_str());
194 dataSource.addName(name.c_str());
195 } else {
196 // Add line as is
197 LOGD("adding '%s'", line_buffer.c_str());
198 dataSource.addName(line_buffer.c_str());
199 }
200 line_buffer.clear(false);
201 }
202 return 1;
203 } else {
204 // Accumulate characters
205 line_buffer.add((char)ch);
206 return 1;
207 }
208 }
209
210 size_t write(const uint8_t* buffer, size_t size) override {
211 for (size_t i = 0; i < size; i++) {
212 write(buffer[i]);
213 }
214 return size;
215 }
216
218 void flush() override {
219 if (line_buffer.length() > 0) {
220 dataSource.addName(line_buffer.c_str());
221 line_buffer.clear();
222 }
223 }
224
225 private:
226 PathNamesRegistry& dataSource;
227 Str line_buffer{200}; // Buffer to accumulate characters for each line
228 const char* prefix = nullptr;
229};

267 current_index = files.size() - 1;

280 return nullptr;
281 }
282
283 current_index = index;
284 Str fullPath = getFullPath(index);
285 LOGI("selectStream: %d -> %s", index, fullPath.c_str());
286
287 if (nameToStreamCallback) {
288 current_stream = nameToStreamCallback(fullPath.c_str(), getCurrentFile());
289 return current_stream;
290 }
291
292 LOGE("No file to stream callback set!");
293 return nullptr;

297 virtual FileType* selectStream(const char* path) override {
298 TRACED();
299 if (path == nullptr) return nullptr;
300
301 int idx = indexOf(path);
302 if (idx >= 0) {
303 return selectStream(idx);
304 }
305
306 LOGE("File not found: %s", path);
307 return nullptr;
308 }
309
311 virtual int index() override { return current_index; }
312
314 int indexOf(const char* path) {
315 if (path == nullptr) return -1;
316
317 // Find the file by path
318 for (int i = 0; i < (int)files.size(); i++) {
319 Str fullPath = getFullPath(i);
320 if (fullPath.equals(path)) {
321 return i;
322 }
323 }
324 return -1;
325 }
326
328 void addName(const char* nameWithPath) override {
329 TRACED();
330 if (nameWithPath == nullptr) return;
331 LOGI("addName: '%s'", nameWithPath);
332
333 // Split path and name
334 StrView nameWithPathStr(nameWithPath);
335 int lastSlashPos = nameWithPathStr.lastIndexOf("/");
336 int len = nameWithPathStr.length();
337
338 Str pathStr;
339 Str nameStr;
340 if (lastSlashPos < 0) {
341 // No path, just name
342 pathStr.set("");
343 nameStr.set(nameWithPath);
344 } else {
345 // Split into path and name
346 pathStr.substring(nameWithPath, 0, lastSlashPos);
347 nameStr.substring(nameWithPath, lastSlashPos + 1, len);
348 }
349
350 // Find or add path to registry
351 int pathIndex = findOrAddPath(pathStr.c_str());
352
353 // Create file entry with path index
354 FileEntry entry{pathIndex, nameStr.c_str()};
355 files.push_back(entry);

416 int findOrAddPath(const char* path) {
417 if (path == nullptr) path = "";
418
419 // Search for existing path
420 for (int i = 0; i < (int)path_registry.size(); i++) {
421 if (strcmp(path_registry[i].c_str(), path) == 0) {
422 return i; // Found existing path
423 }
424 }
425
426 // Path not found, add new one
427 Str newPath;
428 newPath.set(path);
429 path_registry.push_back(newPath);
430 return path_registry.size() - 1;
431 }
432
434 FileEntry& getFileEntry(int index) const { return files[index]; }
435
437 Str getFullPath(int index) {
438 if (index < 0 || index >= (int)files.size()) {
439 return Str();
440 }
441
442 FileEntry& entry = getFileEntry(index);
443
444 // Get path
445 Str result;
446 if (entry.path_index >= 0 && entry.path_index < (int)path_registry.size()) {
447 result = path_registry[entry.path_index].c_str();
448 }
449 // delimite with /
450 if (!result.endsWith("/") && !entry.name.startsWith("/")) {
451 result.add("/");
452 }
453 // add name
454 result.add(entry.name.c_str());
455
456 return result;
457 }
458
460 const char* getPath(int pathIndex) {
461 if (pathIndex >= 0 && pathIndex < (int)path_registry.size()) {
462 return path_registry[pathIndex].c_str();
463 }
464 return "";
465 }
