240 TRACED();

243 }
244
246 virtual FileType* nextStream(int offset) override {
247 TRACED();
248 if (files.empty()) return nullptr;
249
250 current_index += offset;
251 // Wrap around if out of bounds
252 if (current_index < 0) {
253 current_index = files.size() - 1;
254 } else if (current_index >= (int)files.size()) {
255 current_index = 0;
256 }
257
258 return selectStream(current_index);

262 virtual FileType* selectStream(int index) override {
263 TRACED();
264 if (files.empty() || index < 0 || index >= (int)files.size()) {
265 LOGE("Invalid index: %d (size: %d)", index, files.size());
266 return nullptr;
267 }
268
269 current_index = index;
270 Str fullPath = getFullPath(index);
271 LOGI("selectStream: %d -> %s", index, fullPath.c_str());
272
273 if (nameToStreamCallback) {
274 current_stream = nameToStreamCallback(fullPath.c_str(), getCurrentFile());
275 return current_stream;
276 }
277
278 LOGE("No file to stream callback set!");
279 return nullptr;
280 }
281
283 virtual FileType* selectStream(const char* path) override {
284 TRACED();
285 if (path == nullptr) return nullptr;
286
287 int idx = indexOf(path);
288 if (idx >= 0) {
289 return selectStream(idx);
290 }
291
292 LOGE("File not found: %s", path);
293 return nullptr;
294 }
295
297 virtual int index() override { return current_index; }
298
300 int indexOf(const char* path) {
301 if (path == nullptr) return -1;
302
303 // Find the file by path
304 for (int i = 0; i < (int)files.size(); i++) {
305 Str fullPath = getFullPath(i);
306 if (fullPath.equals(path)) {
307 return i;
308 }
309 }
310 return -1;
311 }
312
314 void addName(const char* nameWithPath) override {
315 TRACED();
316 if (nameWithPath == nullptr) return;
317
318 // Split path and name
319 StrView nameWithPathStr(nameWithPath);
320 int lastSlashPos = nameWithPathStr.lastIndexOf("/");
321 int len = nameWithPathStr.length();
322
323 Str pathStr;
324 Str nameStr;
325 if (lastSlashPos < 0) {
326 // No path, just name
327 pathStr.set("");
328 nameStr.set(nameWithPath);
329 } else {
330 // Split into path and name
331 pathStr.substring(nameWithPath, 0, lastSlashPos);
332 nameStr.substring(nameWithPath, lastSlashPos + 1, len);
333 }
334
335 // Find or add path to registry
336 int pathIndex = findOrAddPath(pathStr.c_str());
337
338 // Create file entry with path index
339 FileEntry entry{pathIndex, nameStr.c_str()};
340 files.push_back(entry);
341
342 LOGI("Added file: %s (total: %d, path_index: %d)", nameWithPath,
343 files.size(), pathIndex);
344 }

394 protected:
395 Vector<FileEntry> files; // List of all files
396 Vector<Str> path_registry; // Shared registry of unique paths
397 int current_index = 0;
398 FileType* current_stream = nullptr;
399 FileToStreamCallback nameToStreamCallback = nullptr;
400 Str current_path; // Cache for toStr() method
401
403 int findOrAddPath(const char* path) {
404 if (path == nullptr) path = "";
405
406 // Search for existing path
407 for (int i = 0; i < (int)path_registry.size(); i++) {
408 if (strcmp(path_registry[i].c_str(), path) == 0) {
409 return i; // Found existing path
410 }
411 }
412
413 // Path not found, add new one
414 Str newPath;
415 newPath.set(path);
416 path_registry.push_back(newPath);
417 return path_registry.size() - 1;
418 }
419
421 FileEntry& getFileEntry(int index) const { return files[index]; }
422
424 Str getFullPath(int index) {
425 if (index < 0 || index >= (int)files.size()) {
426 return Str();
427 }
428
429 FileEntry& entry = getFileEntry(index);
430
431 // Get path
432 Str result;
433 if (entry.path_index >= 0 && entry.path_index < (int)path_registry.size()) {
434 result = path_registry[entry.path_index].c_str();
435 }
436 // delimite with /
437 if (!result.endsWith("/") && !entry.name.startsWith("/")) {
438 result.add("/");
439 }
440 // add name
441 result.add(entry.name.c_str());
442
443 return result;
444 }
445
447 const char* getPath(int pathIndex) {
448 if (pathIndex >= 0 && pathIndex < (int)path_registry.size()) {
449 return path_registry[pathIndex].c_str();
450 }
451 return "";
452 }
