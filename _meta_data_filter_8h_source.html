50 size_t write(const uint8_t *data, size_t len) override {
51 LOGI("write: %u", (unsigned)len);
52 // prevent npe
53 if ((p_out == nullptr && p_writer == nullptr) || (data == nullptr) ||
54 (len == 0))
55 return 0;
56
57 // an ID3v2 header can only ever be located at the very start of the
58 // stream, so it is only ever checked once, on the first bytes seen
59 if (total_pos == 0) checkID3v2Header(data, len);
60
61 // Bytes that turn out not to belong to the leading ID3v2 tag are held
62 // back in a small trailing ring buffer: this way a byte is only ever
63 // forwarded once we know it is not part of a trailing ID3v1/TAG+ tag
64 // (which is only recognizable once we reach the real end of stream).
65 // This avoids treating a coincidental "TAG"/"ID3" byte sequence inside
66 // the actual audio data as metadata.
67 out_buf.resize(len);
68 out_buf.reset();
69 for (size_t j = 0; j < len; j++) {
70 total_pos++;
71 if (id3v2_skip_remaining > 0) {
72 id3v2_skip_remaining--;
73 continue;
74 }
75 if (tail.isFull()) {
76 uint8_t old = 0;
77 tail.read(old);
78 out_buf.write(old);
79 }
80 tail.write(data[j]);
81 }
82
83 size_t to_write = out_buf.available();
84 if (to_write > 0) {
85 LOGI("output: %u", (unsigned)to_write);
86 writeOut(out_buf.data(), to_write);
87 } else {
88 LOGI("output ignored");
89 }
90
91 return len;
92 }

128 void checkID3v2Header(const uint8_t *data, size_t len) {
129 if (len < sizeof(ID3v2Header)) return;
130 if (data[0] == 'I' && data[1] == 'D' && data[2] == '3') {
131 ID3v2Header hdr;
132 memcpy(&hdr, data, sizeof(hdr));
133 id3v2_skip_remaining =
134 (int)sizeof(ID3v2Header) + (int)calcSizeID3v2(hdr.size);
135 LOGI("ignoring ID3v2 header at stream start, len: %d",
136 id3v2_skip_remaining);
137 }
138 }

143 void flushTail() {
144 int n = tail.available();
145 if (n <= 0) return;
146 uint8_t buf[kMaxTailLen];
147 tail.readArray(buf, n);
148
149 int drop_from = n;
150 if (n >= kID3v1Len && buf[n - kID3v1Len] == 'T' &&
151 buf[n - kID3v1Len + 1] == 'A' && buf[n - kID3v1Len + 2] == 'G') {
152 drop_from = n - kID3v1Len;
153 if (drop_from >= kID3v1EnhancedLen) {
154 int e = drop_from - kID3v1EnhancedLen;
155 if (buf[e] == 'T' && buf[e + 1] == 'A' && buf[e + 2] == 'G' &&
156 buf[e + 3] == '+') {
157 drop_from = e;
158 }
159 }
160 }
161
162 if (drop_from > 0) {
163 LOGI("output: %d", drop_from);
164 writeOut(buf, drop_from);
165 }
166 if (drop_from < n) {
167 LOGI("ignoring trailing tag, len: %d", n - drop_from);
168 }
169 }
170
171 // calculate the synch save size for ID3v2
172 uint32_t calcSizeID3v2(uint8_t chars[4]) {
173 uint32_t byte0 = chars[0];
174 uint32_t byte1 = chars[1];
175 uint32_t byte2 = chars[2];
176 uint32_t byte3 = chars[3];
177 return byte0 << 21 | byte1 << 14 | byte2 << 7 | byte3;
178 }
