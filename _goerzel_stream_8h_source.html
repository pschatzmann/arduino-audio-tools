61 void begin(const GoertzelConfig& config) {
62 this->config = config;
63 this->sample_count = 0;
64
65 // Calculate the coefficient k and omega
66 float omega = (2.0f * M_PI * config.target_frequency) / config.sample_rate;
67 coeff = 2.0f * cos(omega);
68
69 // Reset state variables
70 reset();
71 }

78 bool processSample(float sample) {
79 // Goertzel algorithm core computation
80 float s0 = sample + coeff * s1 - s2;
81 s2 = s1;
82 s1 = s0;
83
84 sample_count++;
85
86 // Check if we've processed a complete block
87 if (sample_count >= config.block_size) {
88 // Calculate magnitude squared
89 float real = s1 - s2 * cos(2.0f * M_PI * config.target_frequency /
90 config.sample_rate);
91 float imag =
92 s2 * sin(2.0f * M_PI * config.target_frequency / config.sample_rate);
93 magnitude_squared = real * real + imag * imag;
94 magnitude = sqrt(magnitude_squared);
95
96 // Reset for next block
97 reset();
98 return true;
99 }
100
101 return false;
102 }

115 float getMagnitudeSquared() const { return magnitude_squared; }

331 void processSamplesOfType(const uint8_t* data, size_t data_len,
332 int channels) {
333 const T* samples = reinterpret_cast<const T*>(data);
334 size_t num_samples = data_len / sizeof(T);
335
336 for (size_t i = 0; i < num_samples; i++) {
337 float normalized = NumberConverter::toFloatT<T>(samples[i]);
338 int channel = i % channels;

348 void processSamples(const uint8_t* data, size_t data_len) {
349 int channels = single_config.channels;

351 switch (single_config.bits_per_sample) {
352 case 8:
353 processSamplesOfType<uint8_t>(data, data_len, channels);

357 break;
358 case 24:

361 case 32:

363 break;
