67 void begin(const GoertzelConfig& config) {
68 this->config = config;
69 this->sample_count = 0;
70
71 // Calculate the coefficient k and omega
72 float omega = (2.0f * M_PI * config.target_frequency) / config.sample_rate;
73 coeff = 2.0f * cos(omega);
74
75 // Reset state variables
76 reset();
77 }
78
84 bool processSample(float sample) {
85 // Goertzel algorithm core computation
86 float s0 = sample + coeff * s1 - s2;
87 s2 = s1;
88 s1 = s0;
89
90 sample_count++;
91
92 // Check if we've processed a complete block
93 if (sample_count >= config.block_size) {
94 // Calculate magnitude squared
95 float real = s1 - s2 * cos(2.0f * M_PI * config.target_frequency /
96 config.sample_rate);
97 float imag =
98 s2 * sin(2.0f * M_PI * config.target_frequency / config.sample_rate);
99 magnitude_squared = real * real + imag * imag;
100 magnitude = sqrt(magnitude_squared);
101
102 // Reset for next block
103 reset();
104 return true;
105 }
106
107 return false;
108 }
109

115
121 float getMagnitudeSquared() const { return magnitude_squared; }
122
128 bool isDetected(float threshold) const { return magnitude > threshold; }

135 s2 = 0.0f;
136 sample_count = 0;
137 magnitude = 0.0f;
138 magnitude_squared = 0.0f;
139 }

415 template <typename T>
416 void processSamplesOfType(const uint8_t* data, size_t data_len,
417 int channels) {
418 const T* samples = reinterpret_cast<const T*>(data);
419 size_t num_samples = data_len / sizeof(T);
420
421 for (size_t i = 0; i < num_samples; i++) {
422 // Convert to normalized float and apply volume scaling
423 float normalized = clip(NumberConverter::toFloatT<T>(samples[i]) * single_config.volume);
424 // Distribute samples to channels in interleaved format
425 int channel = i % channels;
426 if (detectors[channel].processSample(normalized)) {
427 checkDetection(channel);
428 }
429 }
430 }

465 void processSamples(const uint8_t* data, size_t data_len) {
466 int channels = single_config.channels;
467
468 switch (single_config.bits_per_sample) {
469 case 8:
470 processSamplesOfType<uint8_t>(data, data_len, channels);
471 break;
472 case 16:
473 processSamplesOfType<int16_t>(data, data_len, channels);
474 break;
475 case 24:
476 processSamplesOfType<int24_t>(data, data_len, channels);
477 break;
478 case 32:
479 processSamplesOfType<int32_t>(data, data_len, channels);
480 break;
481 default:
482 LOGE("Unsupported bits_per_sample: %d", single_config.bits_per_sample);
483 break;
