645 size_t write(const uint8_t* data, size_t len) {
646 if (!is_started_) return 0;
647 serviceTinyUSB();
648
649 // disregard data if the host has not opened the capture device (alt=0)
650 if (!isStreamingActiveTx()) return len;
651
652 // update the volume
653 if (config_.volume_active) processVolume((uint8_t*)data, len);
654
655 // Write all data, retrying if the buffer is full. On single-core
656 // platforms (RP2040), serviceTinyUSB() drains the buffer by running
657 // tud_task() → xfer_cb. On dual-core (ESP32), the USB task drains
658 // independently and the SynchronizedNBufferRTOS blocks internally.
659 size_t written = 0;
660 while (written < len) {
661 int n = bufferTx().writeArray(data + written, len - written);
662 written += n;
663 if (written < len) {
664 serviceTinyUSB(); // drain buffer to make space
665 if (!isStreamingActiveTx()) break; // host stopped
666 }
667 }
668 return written;
669 }

1105 // Called by audiod_sof_isr() at the feedback interval.
1106 // Computes the current feedback value then claims the EP and sends it.
1107 void tud_audio_feedback_interval_isr(uint8_t func_id,
1108 uint32_t /*frame_count*/,
1109 uint8_t frame_shift) {
1110 audiod_function_t* audio = &audiod_fct_[func_id];
1111
1112 switch (audio->feedback.compute_method) {
1113 case AUDIO_FEEDBACK_METHOD_FIFO_COUNT: {
1114 // In linear-buffer mode, audio data flows lin_buf_out → bufferRx(),
1115 // completely bypassing ep_out_ff. Reading ep_out_ff would always
1116 // return 0, driving the feedback to min_value and causing the host
1117 // to gradually reduce its send rate until the buffer drains (audible
1118 // periodic drop every 5-10 s). Use the platform RX ring buffer level.
1119 uint32_t ff_count = isUseLinearBufferRx()
1120 ? (uint32_t)bufferRx().available()
1121 : tu_fifo_count(&audio->ep_out_ff);
1122 // Exponential weighted average keeps the level estimate stable
1123 audio->feedback.compute.fifo_count.fifo_lvl_avg =
1124 audio->feedback.compute.fifo_count.fifo_lvl_avg -
1125 (audio->feedback.compute.fifo_count.fifo_lvl_avg >> 8) +
1126 ((uint32_t)ff_count << 8);
1127 uint32_t avg = audio->feedback.compute.fifo_count.fifo_lvl_avg >> 8;
1128 uint32_t thr = audio->feedback.compute.fifo_count.fifo_lvl_thr;
1129 uint32_t nom = audio->feedback.compute.fifo_count.nom_value;
1130 if (avg > thr) {
1131 audio->feedback.value =
1132 nom + (uint32_t)audio->feedback.compute.fifo_count.rate_const[0] *
1133 (avg - thr);
1134 } else {
1135 uint32_t drop =
1136 (uint32_t)audio->feedback.compute.fifo_count.rate_const[1] *
1137 (thr - avg);
1138 audio->feedback.value =
1139 (nom > drop) ? nom - drop : audio->feedback.min_value;
1140 }
1141 audio->feedback.value =
1142 TU_MIN(TU_MAX(audio->feedback.value, audio->feedback.min_value),
1143 audio->feedback.max_value);
1144 } break;
1145
1146 case AUDIO_FEEDBACK_METHOD_FREQUENCY_POWER_OF_2:
1147 audio->feedback.value = 1UL << audio->feedback.compute.power_of_2;
1148 break;
1149
1150 case AUDIO_FEEDBACK_METHOD_FREQUENCY_FLOAT:
1151 audio->feedback.value =
1152 (uint32_t)(audio->feedback.compute.float_const *
1153 (float)(1UL << (16u - (frame_shift - 1u))));
1154 break;
1155
1156 case AUDIO_FEEDBACK_METHOD_FREQUENCY_FIXED: {
1157 uint32_t frame_div =
1158 (TUSB_SPEED_FULL == tud_speed_get()) ? 1000u : 8000u;
1159 audio->feedback.value =
1160 (audio->feedback.compute.fixed.sample_freq << 16) / frame_div;
1161 } break;
1162
1163 default:
1164 break;
1165 }
1166
1167 if (usbd_edpt_claim(audio->rhport, audio->ep_fb)) {
1168 audiod_fb_send(audio);
1169 }
1170 }

1175 alloc_mutex();

1177 // Initialize control buffers
1178 for (uint8_t i = 0; i < getAudioCount(); i++) {
1179 audiod_function_t* audio = &audiod_fct_[i];
1180 // Initialize control buffers
1181 int size = getCtrlBufSz(i);

1183 audio->ctrl_buf_sz = size;

1185 audio->alt_setting.resize(descr_builder.audioFunctionsCount());

1188 if (isEpInEnabled()) {

1190 uint16_t max_pkt = descr_builder.calcPacketSizeForRate(192000);
1191 audio->lin_buf_in.resize(max_pkt);
1192 }

1194 // The FIFO is needed by TinyUSB internals even in linear buffer mode.
1195 if (isEpOutEnabled()) {
1196 audio->ep_out_sw_buf.resize(getEpOutSwBufSz(i));
1197 TUSB_FIFO_CONFIG(&audio->ep_out_ff, audio->ep_out_sw_buf.data(),
1198 getEpOutSwBufSz(i), true);
1199 if (isFifoMutexEnabled()) {
1200 tu_fifo_config_mutex(&audio->ep_out_ff, NULL,
1201 osal_mutex_create(&ep_out_ff_mutex_rd_[i]));
1202 }
1203 uint16_t max_pkt = descr_builder.calcPacketSizeForRate(192000);
1204 audio->lin_buf_out.resize(max_pkt);
1205 }
1206 if (isFeedbackEpEnabled()) {
1207 audio->fb_buf.resize(1); // one uint32_t = 4 bytes of feedback data
1208 }
1209 }

1240 uint16_t audiod_open(uint8_t rhport, tusb_desc_interface_t const* itf_desc,
1241 uint16_t max_len) {
1242 (void)max_len;
1243 TU_VERIFY(TUSB_CLASS_AUDIO == itf_desc->bInterfaceClass &&
1244 AUDIO_SUBCLASS_CONTROL == itf_desc->bInterfaceSubClass);
1245 TU_VERIFY(itf_desc->bInterfaceProtocol == AUDIO_INT_PROTOCOL_CODE_V2);
1246 TU_ASSERT(itf_desc->bNumEndpoints <= 1);
1247 if (itf_desc->bNumEndpoints == 1) {
1248 TU_ASSERT(isInterruptEpEnabled());
1249 }
1250 TU_VERIFY(itf_desc->bAlternateSetting == 0);
1251 uint8_t i;
1252 for (i = 0; i < getAudioCount(); i++) {
1253 if (!audiod_fct_[i].p_desc) {
1254 audiod_fct_[i].p_desc = (uint8_t const*)itf_desc;
1255 audiod_fct_[i].rhport = rhport;
1256 audiod_fct_[i].desc_length = getDescLen(i);
1257 // audiod_reset() zeroes ctrl_buf_sz via memset — restore it so
1258 // tud_control_xfer() receives the correct buffer length.
1259 audiod_fct_[i].ctrl_buf_sz = getCtrlBufSz(i);
1260 if (isEpInEnabled() || isEpOutEnabled() || isFeedbackEpEnabled()) {
1261 uint8_t ep_in = 0, ep_out = 0, ep_fb = 0;
1262 uint16_t ep_in_size = 0, ep_out_size = 0;
1263 tusb_desc_endpoint_t const* desc_ep_out = nullptr;
1264 uint8_t const* p_desc = audiod_fct_[i].p_desc;
1265 uint8_t const* p_desc_end =
1266 p_desc + audiod_fct_[i].desc_length - TUD_AUDIO_DESC_IAD_LEN;
1267 while (p_desc_end - p_desc > 0) {
1268 if (tu_desc_type(p_desc) == TUSB_DESC_ENDPOINT) {
1269 tusb_desc_endpoint_t const* desc_ep =
1270 (tusb_desc_endpoint_t const*)p_desc;
1271 if (desc_ep->bmAttributes.xfer == TUSB_XFER_ISOCHRONOUS) {
1272 if (isFeedbackEpEnabled() && desc_ep->bmAttributes.usage == 1) {
1273 ep_fb = desc_ep->bEndpointAddress;
1274 }
1275 if (desc_ep->bmAttributes.usage == 0) {
1276 if (isEpInEnabled() &&
1277 tu_edpt_dir(desc_ep->bEndpointAddress) == TUSB_DIR_IN) {
1278 ep_in = desc_ep->bEndpointAddress;
1279 ep_in_size =
1280 TU_MAX(tu_edpt_packet_size(desc_ep), ep_in_size);
1281 } else if (isEpOutEnabled() &&
1282 tu_edpt_dir(desc_ep->bEndpointAddress) ==
1283 TUSB_DIR_OUT) {
1284 ep_out = desc_ep->bEndpointAddress;
1285 ep_out_size =
1286 TU_MAX(tu_edpt_packet_size(desc_ep), ep_out_size);
1287 desc_ep_out = desc_ep;
1288 }
1289 }
1290 }
1291 }
1292 p_desc = tu_desc_next(p_desc);
1293 }
1294 if (isEpInEnabled() && ep_in) {
1295 bool alloc_ok = usbd_edpt_iso_alloc(rhport, ep_in, ep_in_size);
1296 LOGD("iso_alloc IN ep=0x%02x sz=%u: %s", ep_in, ep_in_size,
1297 alloc_ok ? "OK" : "FAIL");
1298 }
1299 if (isEpOutEnabled() && ep_out) {
1300 bool alloc_ok = usbd_edpt_iso_alloc(rhport, ep_out, ep_out_size);
1301 LOGD("iso_alloc OUT ep=0x%02x sz=%u: %s", ep_out, ep_out_size,
1302 alloc_ok ? "OK" : "FAIL");
1303#ifdef TUP_DCD_EDPT_ISO_ALLOC
1304 // Pre-activate during enumeration (no isochronous traffic).
1305 // Cannot be done in SET_INTERFACE because iso_activate blocks
1306 // on ESP32's DWC2 when the host is already sending.
1307 // release clears the busy flag so XFER can arm later.
1308 if (desc_ep_out) {
1309 usbd_edpt_iso_activate(rhport, desc_ep_out);
1310 usbd_edpt_release(rhport, ep_out);
1311 LOGD("iso_activate+release OUT: done");
1312 }
1313#endif
1314 }
1315 if (isFeedbackEpEnabled() && ep_fb) {
1316 usbd_edpt_iso_alloc(rhport, ep_fb, 4);
1317 }
1318 }
1319 // Scan for bclock_id_tx (clock entity referenced by the USB-streaming
1320 // terminal) and interval_tx. Runs in TX, RX, and RXTX mode so that
1321 // clock-validity/frequency GET requests always succeed.
1322 // TX/RXTX: Output Terminal type=USB_STREAMING → bCSourceID at [8]
1323 // RX: Input Terminal type=USB_STREAMING → bCSourceID at [7]
1324 // interval_tx is only meaningful for the ISO IN endpoint (TX/RXTX).
1325 if (isEpInEnabled() || isEpOutEnabled()) {
1326 uint8_t const* p_desc = audiod_fct_[i].p_desc;
1327 uint8_t const* p_desc_end =
1328 p_desc + audiod_fct_[i].desc_length - TUD_AUDIO_DESC_IAD_LEN;
1329 while (p_desc_end - p_desc > 0) {
1330 if (tu_desc_type(p_desc) == TUSB_DESC_ENDPOINT) {
1331 if (isEpInEnabled()) {
1332 tusb_desc_endpoint_t const* desc_ep =
1333 (tusb_desc_endpoint_t const*)p_desc;
1334 if (desc_ep->bmAttributes.xfer == TUSB_XFER_ISOCHRONOUS &&
1335 desc_ep->bmAttributes.usage == 0 &&
1336 tu_edpt_dir(desc_ep->bEndpointAddress) == TUSB_DIR_IN) {
1337 audiod_fct_[i].interval_tx = desc_ep->bInterval;
1338 }
1339 }
1340 } else if (tu_desc_type(p_desc) == TUSB_DESC_CS_INTERFACE) {
1341 if (tu_desc_subtype(p_desc) ==
1342 AUDIO_CS_AC_INTERFACE_OUTPUT_TERMINAL) {
1343 if (tu_unaligned_read16(p_desc + 4) ==
1344 AUDIO_TERM_TYPE_USB_STREAMING) {
1345 audiod_fct_[i].bclock_id_tx = p_desc[8]; // OT bCSourceID
1346 }
1347 } else if (tu_desc_subtype(p_desc) ==
1348 AUDIO_CS_AC_INTERFACE_INPUT_TERMINAL) {
1349 if (tu_unaligned_read16(p_desc + 4) ==
1350 AUDIO_TERM_TYPE_USB_STREAMING) {
1351 audiod_fct_[i].bclock_id_tx = p_desc[7]; // IT bCSourceID
1352 }
1353 }
1354 }
1355 p_desc = tu_desc_next(p_desc);
1356 }
1357 }
1358
1359 if (isInterruptEpEnabled()) {
1360 uint8_t const* p_desc = audiod_fct_[i].p_desc;
1361 uint8_t const* p_desc_end =
1362 p_desc + audiod_fct_[i].desc_length - TUD_AUDIO_DESC_IAD_LEN;
1363 while (p_desc_end - p_desc > 0) {
1364 if (tu_desc_type(p_desc) == TUSB_DESC_ENDPOINT) {
1365 tusb_desc_endpoint_t const* desc_ep =
1366 (tusb_desc_endpoint_t const*)p_desc;
1367 uint8_t const ep_addr = desc_ep->bEndpointAddress;
1368 if (tu_edpt_dir(ep_addr) == TUSB_DIR_IN &&
1369 desc_ep->bmAttributes.xfer == TUSB_XFER_INTERRUPT) {
1370 if (usbd_edpt_open(audiod_fct_[i].rhport, desc_ep)) {
1371 audiod_fct_[i].ep_int = ep_addr;
1372 } else {
1373 LOGE(" UAC2: interrupt EP 0x%02x open failed", ep_addr);
1374
1375 }
1376 }
1377 }
1378 p_desc = tu_desc_next(p_desc);
1379 }
1380 }
1381 audiod_fct_[i].mounted = true;
1382 break;
1383 }
1384 }
1385 TU_ASSERT(i < getAudioCount());
1386 uint16_t drv_len = audiod_fct_[i].desc_length - TUD_AUDIO_DESC_IAD_LEN;
1387 return drv_len;
1388 }

1399 // Invoked when class request DATA stage is finished.
1400 // return false to stall control EP (e.g Host send non-sense DATA)
1401 bool audiod_control_complete(uint8_t rhport,
1402 tusb_control_request_t const* p_request) {
1403 // Handle audio class specific set requests
1404 if (p_request->bmRequestType_bit.type == TUSB_REQ_TYPE_CLASS &&
1405 p_request->bmRequestType_bit.direction == TUSB_DIR_OUT) {
1406 uint8_t func_id;
1407
1408 switch (p_request->bmRequestType_bit.recipient) {
1409 case TUSB_REQ_RCPT_INTERFACE: {
1410 uint8_t itf = TU_U16_LOW(p_request->wIndex);
1411 uint8_t entityID = TU_U16_HIGH(p_request->wIndex);
1412
1413 if (entityID != 0) {
1414 func_id = 0;
1415 uint8_t ctrlSel = TU_U16_HIGH(p_request->wValue);
1416 uint8_t* cb = audiod_fct_[func_id].ctrl_buf.data();
1417
1418 // ── Clock Source SET_CUR (sample rate) ──────────────
1419 if (entityID == USBAudio2DescriptorBuilder::ENTITY_CLOCK &&
1420 ctrlSel == AUDIO_CS_CTRL_SAM_FREQ &&
1421 p_request->bRequest == AUDIO_CS_REQ_CUR) {
1422 setSampleRate(tu_unaligned_read32(cb));
1423 }
1424
1425 // ── Feature Unit SET_CUR (mute / volume) ────────────
1426 if (isFeatureUnit(entityID) &&
1427 p_request->bRequest == AUDIO_CS_REQ_CUR) {
1428 uint8_t channel = TU_U16_LOW(p_request->wValue);
1429 if (ctrlSel == AUDIO_FU_CTRL_MUTE) {
1430 setMute(cb[0] != 0, channel);
1431 } else if (ctrlSel == AUDIO_FU_CTRL_VOLUME) {
1432 int16_t v;
1433 memcpy(&v, cb, 2);
1434 setVolume(uac2ToFloat(v), channel);
1435 }
1436 }
1437
1438 // Invoke callback
1439 if (tud_audio_set_req_entity_cb_) {
1440 return tud_audio_set_req_entity_cb_(this, rhport, p_request, cb);
1441 }
1442 } else {
1443 // Find index of audio driver structure and verify interface really
1444 // exists
1445 TU_VERIFY(audiod_verify_itf_exists(itf, &func_id));
1446
1447 // Invoke callback
1448 if (tud_audio_set_req_itf_cb_) {
1449 return tud_audio_set_req_itf_cb_(
1450 this, rhport, p_request,
1451 audiod_fct_[func_id].ctrl_buf.data());
1452 }
1453 }
1454 } break;
1455
1456 case TUSB_REQ_RCPT_ENDPOINT: {
1457 uint8_t ep = TU_U16_LOW(p_request->wIndex);
1458
1459 // Check if entity is present and get corresponding driver index
1460 TU_VERIFY(audiod_verify_ep_exists(ep, &func_id));
1461
1462 // Invoke callback
1463 if (tud_audio_set_req_ep_cb_) {
1464 return tud_audio_set_req_ep_cb_(
1465 this, rhport, p_request, audiod_fct_[func_id].ctrl_buf.data());
1466 }
1467 } break;
1468 // Unknown/Unsupported recipient
1469 default:
1470 TU_BREAKPOINT();
1471 return false;
1472 }
1473 }
1474 return true;
1475 }

1478 bool audiod_xfer_cb(uint8_t rhport, uint8_t ep_addr, xfer_result_t result,
1479 uint32_t xferred_bytes) {
1480 (void)result;
1481 (void)xferred_bytes;
1482 for (uint8_t func_id = 0; func_id < getAudioCount(); func_id++) {
1483 audiod_function_t* audio = &audiod_fct_[func_id];
1484 if (isInterruptEpEnabled() && audio->ep_int == ep_addr) {
1485 if (int_done_cb_) int_done_cb_(this, rhport);
1486 return true;
1487 }
1488 if (isEpInEnabled() && audio->ep_in == ep_addr &&
1489 audio->alt_setting.size() != 0) {
1490 xfer_cb_tx_count_ = xfer_cb_tx_count_ + 1;
1491 if (tx_done_cb_) tx_done_cb_(this, rhport, audio);
1492
1493 uint16_t frame_bytes = isEpInFlowControlEnabled()
1494 ? audiod_tx_packet_size_fc(audio)
1495 : audio->ep_in_sz;
1496 if (frame_bytes > audio->ep_in_sz) frame_bytes = audio->ep_in_sz;
1497 tx_frame_bytes_last_ = frame_bytes;
1498 tx_xferred_last_ = xferred_bytes;
1499
1500 // Drain platform buffer into lin_buf_in, zero-pad, send via DMA.
1501 {
1502 uint8_t* dst = audio->lin_buf_in.data();
1503 uint16_t n = (uint16_t)bufferTx().readArray(dst, frame_bytes);
1504 tx_fifo_read_total_ += n;
1505 if (n < frame_bytes) memset(dst + n, 0, frame_bytes - n);
1506 (void)TUSB_EDPT_XFER(rhport, audio->ep_in, dst, frame_bytes);
1507 }
1508 return true;
1509 }
1510 if (isEpOutEnabled() && audio->ep_out == ep_addr) {
1511 xfer_cb_rx_count_ = xfer_cb_rx_count_ + 1;
1512 rx_total_bytes_ += xferred_bytes;
1513 if (isUseLinearBufferRx()) {
1514 // Copy DMA-received data into the platform buffer, re-arm DMA.
1515 if (xferred_bytes > 0)
1516 bufferRx().writeArray(audio->lin_buf_out.data(),
1517 (int)xferred_bytes);
1518 if (rx_done_cb_)
1519 rx_done_cb_(this, rhport, audio, (uint16_t)xferred_bytes);
1520 (void)TUSB_EDPT_XFER(rhport, audio->ep_out, audio->lin_buf_out.data(),
1521 audio->ep_out_sz);
1522 } else {
1523 // FIFO mode: data is already in ep_out_ff from DMA, copy to buffer
1524 if (xferred_bytes > 0) {
1525 uint8_t tmp[768];
1526 uint16_t n =
1527 tu_fifo_read_n(&audio->ep_out_ff, tmp, (uint16_t)xferred_bytes);
1528 if (n > 0) bufferRx().writeArray(tmp, n);
1529 }
1530 if (rx_done_cb_)
1531 rx_done_cb_(this, rhport, audio, (uint16_t)xferred_bytes);
1532 (void)TUSB_EDPT_XFER_FIFO(rhport, audio->ep_out, &audio->ep_out_ff,
1533 audio->ep_out_sz);
1534 }
1535 return true;
1536 }
1537 if (isFeedbackEpEnabled() && audio->ep_fb == ep_addr) {
1538 // SOF ISR owns re-sending; just notify the application.
1539 if (fb_done_cb_) fb_done_cb_(this, func_id);
1540 return true;
1541 }
1542 }
1543 return false;
1544 }

1566 // ── Clock Source GET handler ────────────────────────────────────────────
1567 bool handleClockSourceGet(uint8_t rhport,
1568 tusb_control_request_t const* p_request,
1569 uint8_t* cb) {
1570 uint8_t ctrlSel = TU_U16_HIGH(p_request->wValue);
1571 if (ctrlSel == AUDIO_CS_CTRL_CLK_VALID &&
1572 p_request->bRequest == AUDIO_CS_REQ_CUR) {
1573 cb[0] = 1;
1574 return tud_control_xfer(rhport, p_request, cb, 1);
1575 }
1576 if (ctrlSel == AUDIO_CS_CTRL_SAM_FREQ) {
1577 uint32_t rate = (uint32_t)config_.sample_rate;
1578 if (p_request->bRequest == AUDIO_CS_REQ_CUR) {
1579 memcpy(cb, &rate, 4);
1580 return tud_control_xfer(rhport, p_request, cb, 4);
1581 }
1582 if (p_request->bRequest == AUDIO_CS_REQ_RANGE) {
1583 if (config_.enable_multi_sample_rate) {
1584 // List all supported discrete rates
1585 uint16_t cnt = kNumSupportedSampleRates;
1586 memcpy(cb, &cnt, 2);
1587 for (uint8_t i = 0; i < kNumSupportedSampleRates; i++) {
1588 uint32_t r = kSupportedSampleRates[i];
1589 uint32_t z = 0;
1590 memcpy(cb + 2 + i * 12, &r, 4);
1591 memcpy(cb + 2 + i * 12 + 4, &r, 4);
1592 memcpy(cb + 2 + i * 12 + 8, &z, 4);
1593 }
1594 return tud_control_xfer(rhport, p_request, cb,
1595 (uint16_t)(2 + kNumSupportedSampleRates * 12));
1596 } else {
1597 // Single fixed rate from config
1598 uint16_t cnt = 1;
1599 uint32_t z = 0;
1600 memcpy(cb, &cnt, 2);
1601 memcpy(cb + 2, &rate, 4); // dMIN
1602 memcpy(cb + 6, &rate, 4); // dMAX
1603 memcpy(cb + 10, &z, 4); // dRES = 0 (fixed)
1604 return tud_control_xfer(rhport, p_request, cb, 14);
1605 }
1606 }
1607 }
1608 return false;
1609 }

1611 // ── Feature Unit GET handler ──────────────────────────────────────────
1612 bool handleFeatureUnitGet(uint8_t rhport,
1613 tusb_control_request_t const* p_request,
1614 uint8_t* cb) {

1616 uint8_t channel = TU_U16_LOW(p_request->wValue);
1617 if (ctrlSel == AUDIO_FU_CTRL_MUTE &&
1618 p_request->bRequest == AUDIO_CS_REQ_CUR) {
1619 cb[0] = isMute(channel) ? 1 : 0;
1620 return tud_control_xfer(rhport, p_request, cb, 1);
1621 }
1622 if (ctrlSel == AUDIO_FU_CTRL_VOLUME) {
1623 if (p_request->bRequest == AUDIO_CS_REQ_CUR) {
1624 int16_t v = floatToUac2(volume(channel));
1625 memcpy(cb, &v, 2);
1626 return tud_control_xfer(rhport, p_request, cb, 2);

1628 if (p_request->bRequest == AUDIO_CS_REQ_RANGE) {
1629 uint16_t cnt = 1;
1630 int16_t vmin = -25600, vmax = 0, vres = 256;
1631 memcpy(cb + 0, &cnt, 2);
1632 memcpy(cb + 2, &vmin, 2);
1633 memcpy(cb + 4, &vmax, 2);
1634 memcpy(cb + 6, &vres, 2);
1635 return tud_control_xfer(rhport, p_request, cb, 8);
1636 }

1643 tusb_control_request_t const* p_request,
1644 uint8_t entityID) {

1732 return false;
1733 }
1734 }

1739 // Verify an entity with the given ID exists and returns also the
1740 // corresponding driver index
1741 bool audiod_verify_entity_exists(uint8_t itf, uint8_t entityID,
1742 uint8_t* func_id) {
1743 uint8_t i;
1744 for (i = 0; i < getAudioCount(); i++) {
1745 // Look for the correct driver by checking if the unique standard AC
1746 // interface number fits
1747 if (audiod_fct_[i].p_desc &&
1748 ((tusb_desc_interface_t const*)audiod_fct_[i].p_desc)
1749 ->bInterfaceNumber == itf) {
1750 // Get pointers after class specific AC descriptors and end of AC
1751 // descriptors - entities are defined in between
1752 uint8_t const* p_desc =
1753 tu_desc_next(audiod_fct_[i].p_desc); // Points to CS AC descriptor
1754 uint8_t const* p_desc_end =
1755 ((audio_desc_cs_ac_interface_t const*)p_desc)->wTotalLength +
1756 p_desc;
1757 p_desc = tu_desc_next(p_desc); // Get past CS AC descriptor
1758
1759 // Condition modified from p_desc < p_desc_end to prevent gcc>=12
1760 // strict-overflow warning
1761 while (p_desc_end - p_desc > 0) {
1762 if (p_desc[3] == entityID) // Entity IDs are always at offset 3
1763 {
1764 *func_id = i;
1765 return true;
1766 }
1767 p_desc = tu_desc_next(p_desc);
1768 }
1769 }
1770 }
1771 return false;

1826 void audiod_parse_flow_control_params(audiod_function_t* audio,
1827 uint8_t const* p_desc) {
1828 // Seed the TX sample rate from the configured AudioInfo so packet-size
1829 // calculation works even when the host never issues a SET_CUR(SAM_FREQ)
1830 // request (typical for single-frequency clock sources). The host may still
1831 // override this later via audiod_control_complete().
1832 if (audio->sample_rate_tx == 0)
1833 audio->sample_rate_tx = (uint32_t)config_.sample_rate;
1834
1835 p_desc = tu_desc_next(p_desc); // Exclude standard AS interface descriptor
1836 // of current alternate interface descriptor
1837
1838 // Look for a Class-Specific AS Interface Descriptor(4.9.2) to verify format
1839 // type and format and also to get number of physical channels
1840 if (tu_desc_type(p_desc) == TUSB_DESC_CS_INTERFACE &&
1841 tu_desc_subtype(p_desc) == AUDIO_CS_AS_INTERFACE_AS_GENERAL) {
1842 audio->n_channels_tx =
1843 ((audio_desc_cs_as_interface_t const*)p_desc)->bNrChannels;
1844 audio->format_type_tx =
1845 (audio_format_type_t)(((audio_desc_cs_as_interface_t const*)p_desc)
1846 ->bFormatType);
1847 // Look for a Type I Format Type Descriptor(2.3.1.6 - Audio Formats)
1848 p_desc = tu_desc_next(p_desc);
1849 if (tu_desc_type(p_desc) == TUSB_DESC_CS_INTERFACE &&
1850 tu_desc_subtype(p_desc) == AUDIO_CS_AS_INTERFACE_FORMAT_TYPE &&
1851 ((audio_desc_type_I_format_t const*)p_desc)->bFormatType ==
1852 AUDIO_FORMAT_TYPE_I) {
1853 audio->n_bytes_per_sample_tx =
1854 ((audio_desc_type_I_format_t const*)p_desc)->bSubslotSize;
1855 }
1856 }
1857
1858 // Fallback from config if descriptor parsing missed any field.
1859 // The descriptor struct layout may differ across TinyUSB versions.
1860 if (audio->n_channels_tx == 0) audio->n_channels_tx = config_.channels;
1861 if (audio->n_bytes_per_sample_tx == 0)
1862 audio->n_bytes_per_sample_tx = config_.bits_per_sample / 8;
1863 if (audio->format_type_tx == 0) audio->format_type_tx = AUDIO_FORMAT_TYPE_I;
1864 }

1866 // This helper function finds for a given audio function and AS interface
1867 // number the index of the attached driver structure, the index of the
1868 // interface in the audio function
1869 // (e.g. the std. AS interface with interface number 15 is the first AS
1870 // interface for the given audio function and thus gets index zero), and
1871 // finally a pointer to the std. AS interface, where the pointer always points
1872 // to the first alternate setting i.e. alternate interface zero.
1873 bool audiod_get_AS_interface_index(uint8_t itf, audiod_function_t* audio,
1874 uint8_t* idxItf,
1875 uint8_t const** pp_desc_int) {
1876 if (audio->p_desc) {
1877 // Get pointer at end
1878 uint8_t const* p_desc_end =
1879 audio->p_desc + audio->desc_length - TUD_AUDIO_DESC_IAD_LEN;
1880
1881 // Advance past AC descriptors
1882 uint8_t const* p_desc = tu_desc_next(audio->p_desc);
1883 p_desc += ((audio_desc_cs_ac_interface_t const*)p_desc)->wTotalLength;
1884
1885 uint8_t tmp = 0;
1886 // Condition modified from p_desc < p_desc_end to prevent gcc>=12
1887 // strict-overflow warning
1888 while (p_desc_end - p_desc > 0) {
1889 // We assume the number of alternate settings is increasing thus we
1890 // return the index of alternate setting zero!
1891 if (tu_desc_type(p_desc) == TUSB_DESC_INTERFACE &&
1892 ((tusb_desc_interface_t const*)p_desc)->bAlternateSetting == 0) {
1893 if (((tusb_desc_interface_t const*)p_desc)->bInterfaceNumber == itf) {
1894 *idxItf = tmp;
1895 *pp_desc_int = p_desc;
1896 return true;
1897 }
1898 // Increase index, bytes read, and pointer
1899 tmp++;
1900 }
1901 p_desc = tu_desc_next(p_desc);
1902 }
1903 }
1904 return false;
1905 }

2306 bool audiod_calc_tx_packet_sz(audiod_function_t* audio) {
2307 TU_VERIFY(audio->format_type_tx == AUDIO_FORMAT_TYPE_I);
2308 TU_VERIFY(audio->n_channels_tx);
2309 TU_VERIFY(audio->n_bytes_per_sample_tx);
2310 TU_VERIFY(audio->interval_tx);
2311 TU_VERIFY(audio->sample_rate_tx);
2312
2313 // Restart the fractional accumulator for this streaming session.
2314 audio->tx_sample_acc = 0;
2315
2316 const uint8_t interval = (tud_speed_get() == TUSB_SPEED_FULL)
2317 ? audio->interval_tx
2318 : 1 << (audio->interval_tx - 1);
2319
2320 const uint16_t sample_normimal =
2321 (uint16_t)(audio->sample_rate_tx * interval /
2322 ((tud_speed_get() == TUSB_SPEED_FULL) ? 1000 : 8000));
2323 const uint16_t sample_reminder =
2324 (uint16_t)(audio->sample_rate_tx * interval %
2325 ((tud_speed_get() == TUSB_SPEED_FULL) ? 1000 : 8000));
2326
2327 const uint16_t packet_sz_tx_min =
2328 (uint16_t)((sample_normimal - 1) * audio->n_channels_tx *
2329 audio->n_bytes_per_sample_tx);
2330 const uint16_t packet_sz_tx_norm =
2331 (uint16_t)(sample_normimal * audio->n_channels_tx *
2332 audio->n_bytes_per_sample_tx);
2333 const uint16_t packet_sz_tx_max =
2334 (uint16_t)((sample_normimal + 1) * audio->n_channels_tx *
2335 audio->n_bytes_per_sample_tx);
2336
2337 // Endpoint size must larger than packet size
2338 TU_ASSERT(packet_sz_tx_max <= audio->ep_in_sz);
2339
2340 // Frmt20.pdf 2.3.1.1 USB Packets
2341 if (sample_reminder) {
2342 // All virtual frame packets must either contain INT(nav) audio slots
2343 // (small VFP) or INT(nav)+1 (large VFP) audio slots
2344 audio->packet_sz_tx[0] = packet_sz_tx_norm;
2345 audio->packet_sz_tx[1] = packet_sz_tx_norm;
2346 audio->packet_sz_tx[2] = packet_sz_tx_max;
2347 } else {
2348 // In the case where nav = INT(nav), ni may vary between INT(nav)-1 (small
2349 // VFP), INT(nav) (medium VFP) and INT(nav)+1 (large VFP).
2350 audio->packet_sz_tx[0] = packet_sz_tx_min;
2351 audio->packet_sz_tx[1] = packet_sz_tx_norm;
2352 audio->packet_sz_tx[2] = packet_sz_tx_max;
2353 }
2354
2355 return true;
2356 }
2357
2358 // Number of audio bytes to transmit in the current (micro)frame when IN
2359 // flow control is enabled. A fractional accumulator distributes the
2360 // sub-frame sample remainder over successive frames so the long-term average
2361 // matches the configured sample rate (e.g. alternating 176/180 bytes for
2362 // 44100 Hz stereo 16-bit, averaging 176.4 bytes = 44.1 samples per frame).
2363 uint16_t audiod_tx_packet_size_fc(audiod_function_t* audio) {
2364 if (audio->sample_rate_tx == 0 || audio->n_channels_tx == 0 ||
2365 audio->n_bytes_per_sample_tx == 0) {
2366 // Not enough info to size precisely: fall back to the max packet.
2367 return audio->ep_in_sz;
2368 }
2369 const uint32_t denom = (tud_speed_get() == TUSB_SPEED_FULL) ? 1000u : 8000u;
2370 const uint8_t iv = audio->interval_tx ? audio->interval_tx : 1;
2371 const uint32_t interval =
2372 (tud_speed_get() == TUSB_SPEED_FULL) ? iv : (1u << (iv - 1));
2373
2374 audio->tx_sample_acc += audio->sample_rate_tx * interval;
2375 const uint32_t samples = audio->tx_sample_acc / denom;
2376 audio->tx_sample_acc -= samples * denom;
2377
2378 uint32_t bytes =
2379 samples * audio->n_channels_tx * audio->n_bytes_per_sample_tx;
2380 if (bytes > audio->ep_in_sz) bytes = audio->ep_in_sz;
2381 return (uint16_t)bytes;
2382 }
