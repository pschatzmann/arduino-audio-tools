44 int streams = 1;
45 int coupled_streams = 1;
46 opus_channel_mapping_enum_t default_channel_mapping = OPUS_CHANNEL_MAPPING_COMBINED;
47 unsigned char mapping[255] = {};

49 void setupChannelMapping() {
50 switch (default_channel_mapping) {
51 case OPUS_CHANNEL_MAPPING_SEPARATE:
52 setupSeparateChannelsMapping();
53 break;
54 case OPUS_CHANNEL_MAPPING_COMBINED:
55 setupDefaultMapping();
56 break;
57 default:
58 break;
59 }
60 }
61
62 void setupDefaultMapping() {
63 memset(mapping, 0, sizeof(mapping));
64 if (channels == 1) {
65 streams = 1;
66 coupled_streams = 0;
67 mapping[0] = 0;
68 } else if (channels == 2) {
69 streams = 1;
70 coupled_streams = 1;
71 mapping[0] = 0;
72 mapping[1] = 1;
73 } else {
74 streams = channels;
75 coupled_streams = 0;
76 for (int i = 0; i < channels; i++) {
77 mapping[i] = i;
78 }
79 }
80 }
81
82 void setupSeparateChannelsMapping() {
83 memset(mapping, 0, sizeof(mapping));
84 streams = channels;
85 coupled_streams = 0;
86 for (int i = 0; i < channels; i++) {
87 mapping[i] = i;
88 }
89 }

206 size_t write(const uint8_t* data, size_t len) override {
207 if (!active || p_print == nullptr) return 0;
208 LOGD("OpusMultiStreamAudioDecoder::write: %d", (int)len);
209 int in_band_forward_error_correction = 0;
210 int frame_count = cfg.max_buffer_size / cfg.channels / sizeof(opus_int16);
211 int out_samples = opus_multistream_decode(
212 dec, (uint8_t*)data, len, (opus_int16*)outbuf.data(), frame_count,
213 in_band_forward_error_correction);
214 if (out_samples < 0) {
215 LOGW("opus_multistream_decode: %s", opus_strerror(out_samples));
216 } else if (out_samples > 0) {
217 int out_bytes = out_samples * cfg.channels * sizeof(int16_t);
218 LOGD("opus_multistream_decode: %d", out_bytes);
219 int open = out_bytes;
220 int processed = 0;
221 while (open > 0) {
222 int to_write = std::min(open, cfg.max_buffer_write_size);
223 int written = p_print->write(outbuf.data() + processed, to_write);
224 open -= written;
225 processed += written;
226 }
227 }
228 return len;
229 }
