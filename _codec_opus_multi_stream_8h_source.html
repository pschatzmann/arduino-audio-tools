36 void setupDefaultMapping() {

163 size_t write(const uint8_t *data, size_t len) override {
164 if (!active || p_print == nullptr) return 0;
165 LOGD("OpusMultiStreamAudioDecoder::write: %d", (int)len);
166 int in_band_forward_error_correction = 0;
167 int frame_count = cfg.max_buffer_size / cfg.channels / sizeof(opus_int16);
168 int out_samples = opus_multistream_decode(
169 dec, (uint8_t *)data, len, (opus_int16 *)outbuf.data(), frame_count,
170 in_band_forward_error_correction);
171 if (out_samples < 0) {
172 LOGW("opus_multistream_decode: %s", opus_strerror(out_samples));
173 } else if (out_samples > 0) {
174 int out_bytes = out_samples * cfg.channels * sizeof(int16_t);
175 LOGD("opus_multistream_decode: %d", out_bytes);
176 int open = out_bytes;
177 int processed = 0;
178 while (open > 0) {
179 int to_write = std::min(open, cfg.max_buffer_write_size);
180 int written = p_print->write(outbuf.data() + processed, to_write);
181 open -= written;
182 processed += written;
183 }
184 }
185 return len;
186 }
