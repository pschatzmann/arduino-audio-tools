123 TaskHandle_t usb_task_handle_ = nullptr;
124
125 // Task body: processes all pending USB events then sleeps for one FreeRTOS
126 // tick (1 ms at the default 1 kHz tick rate). vTaskDelay(1) — not
127 // taskYIELD() — is essential: taskYIELD() at max priority returns
128 // immediately when no equal-or-higher-priority task is ready, so it never
129 // yields to the lower-priority loop() task and starves the audio copier.
130 // One tick matches the USB full-speed 1 ms frame period and costs no audio
131 // quality: tud_task() drains the entire event queue each call, so there is
132 // no pending work left by the time we sleep.
133 static void usbTaskBody(void* param) {
134 (void)param;
135 while (true) {
136 tud_task();
137 vTaskDelay(1);
138 }
139 }
140
141 void startUsbTask() {
142 if (usb_task_handle_ != nullptr) return;
143 xTaskCreate(usbTaskBody, "tud",
144 (configSTACK_DEPTH_TYPE)(config_.usb_task_stack_size / sizeof(StackType_t)),
145 nullptr, (UBaseType_t)config_.usb_task_priority,
146 &usb_task_handle_);
147 if (usb_task_handle_ != nullptr) {
148 // Suppress serviceTinyUSB() in write()/readBytes(): concurrent tud_task()
149 // calls corrupt TinyUSB's internal event queue (not re-entrant).
150 usb_task_active_ = true;
151 // A dedicated task now drains buffer_tx_: blocking writes are safe.
152 buffer_tx_.setWriteMaxWait((TickType_t)config_.usb_task_write_wait_ms);
153 }
154 }
155
156 void stopUsbTask() {
157 if (usb_task_handle_ == nullptr) return;
158 usb_task_active_ = false;
159 vTaskDelete(usb_task_handle_);
160 usb_task_handle_ = nullptr;
161 buffer_tx_.setWriteMaxWait(0); // revert to non-blocking if task is gone
162 }
