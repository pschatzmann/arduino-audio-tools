34 prefix_pos_ = 0;
35 original_stream_ = &original_stream;

37
39 virtual int available() override {
40 int remaining_prefix = prefix_len_ - prefix_pos_;
41 if (original_stream_ != nullptr) {
42 return remaining_prefix + original_stream_->available();
43 }
44 return remaining_prefix;
45 }
46
48 virtual int read() override {
49 if (prefix_pos_ < prefix_len_) {
50 return prefix_data_[prefix_pos_++];
51 }
52 if (original_stream_ != nullptr) {
53 return original_stream_->read();
54 }
55 return -1;
56 }
57
59 virtual int peek() override {
60 if (prefix_pos_ < prefix_len_) {
61 return prefix_data_[prefix_pos_];
62 }
63 if (original_stream_ != nullptr) {
64 return original_stream_->peek();
65 }
66 return -1;
67 }
68
70 virtual size_t readBytes(uint8_t* buffer, size_t length) override {
71 size_t bytes_read = 0;
72
73 // First, read from prefix data
74 if (prefix_pos_ < prefix_len_) {
75 size_t prefix_available = prefix_len_ - prefix_pos_;
76 size_t prefix_to_read = (length < prefix_available) ? length : prefix_available;
77 memcpy(buffer, prefix_data_ + prefix_pos_, prefix_to_read);
78 prefix_pos_ += prefix_to_read;
79 bytes_read += prefix_to_read;
80 buffer += prefix_to_read;
81 length -= prefix_to_read;
82 }
83
84 // Then read from original stream if more data is needed
85 if (length > 0 && original_stream_ != nullptr) {
86 bytes_read += original_stream_->readBytes(buffer, length);
87 }
88
89 return bytes_read;
90 }
91
93 virtual size_t write(uint8_t) override { return 0; }
94 virtual size_t write(const uint8_t*, size_t) override { return 0; }
95 virtual void flush() override {}
96
97 private:
98 const uint8_t* prefix_data_;
99 size_t prefix_len_;
100 size_t prefix_pos_;
101 Stream* original_stream_;
102};

630 bool selectDecoder(const char* mime) {
631 TRACEI();
632 bool result = false;
633
634 // Guard against null MIME type - cannot proceed without valid MIME
635 if (mime == nullptr) {
636 LOGE("mime is null");
637 return false;
638 }
639
640 // Optimization: Check if the requested MIME type is already active
641 // This avoids unnecessary decoder switching when the same format is detected
642 if (StrView(mime).equals(actual_decoder.mime)) {
643 is_first = false; // Mark initialization as complete
644 return true; // Already using the correct decoder
645 }
646
647 // Clean shutdown of currently active decoder before switching
648 // This ensures proper resource cleanup and state reset
649 if (actual_decoder.decoder != nullptr) {
650 actual_decoder.decoder->end();
651 actual_decoder.is_open = false; // Mark as inactive
652 }
653
654 // Search through all registered decoders to find one that handles this MIME type
655 selected_mime = nullptr; // Clear previous selection
656 for (int j = 0; j < decoders.size(); j++) {
657 DecoderInfo info = decoders[j];
658
659 // Check if this decoder supports the detected MIME type
660 if (StrView(info.mime).equals(mime)) {
661 LOGI("Using Decoder %s for %s", toStr(info.mime), toStr(mime));
662
663 // Switch to the matching decoder
664 actual_decoder = info;
665
666 // Configure the decoder's output stream to match our output
667 // This ensures decoded audio data flows to the correct destination
668 if (p_print != nullptr) {
669 actual_decoder.decoder->setOutput(*p_print);
670 }
671
672 // Initialize the selected decoder and mark it as active
673 LOGI("available: %d", p_data_source->available());
674 assert(p_data_source != nullptr);
675 actual_decoder.decoder->setInput(*p_data_source);
676 actual_decoder.decoder->clearNotifyAudioChange();
677 actual_decoder.decoder->addNotifyAudioChange(*this);
678 if (actual_decoder.decoder->begin()) {
679 actual_decoder.is_open = true;
680 LOGI("StreamingDecoder %s started", toStr(actual_decoder.mime));
681 } else {
682 // Decoder failed to start - this is a critical error
683 LOGE("Failed to start StreamingDecoder %s", toStr(actual_decoder.mime));
684 return false;
685 }
686
687 // Successfully found and initialized a decoder
688 result = true;
689 selected_mime = mime; // Store the MIME type that was selected
690 break; // Stop searching once we find a match
691 }
692 }
693
694 // Mark initialization phase as complete regardless of success/failure
695 is_first = false;
696 return result; // true if decoder was found and started, false otherwise
697 }

858 bool selectDecoder() {
859 // Only perform MIME detection and decoder selection if no decoder is active yet
860 // This prevents re-detection on subsequent calls during the same stream
861 if (actual_decoder.decoder == nullptr) {
862 const char* mime = nullptr;
863 p_data_source = nullptr;
864
865 // Two methods for MIME type determination: external source or auto-detection
866 if (p_mime_source != nullptr) {
867 // Option 1: Use externally provided MIME source (e.g., from HTTP headers)
868 // This is more efficient as it avoids reading and analyzing stream data
869 mime = p_mime_source->mime();
870 LOGI("mime from source: %s", toStr(mime));
871 assert(p_input != nullptr);
872 p_data_source = p_input;
873 } else {
874 // Option 2: Auto-detect MIME type by analyzing stream content
875 assert(p_input != nullptr);
876
877 // Read a sample of data for MIME detection
878 // Allocate buffer for MIME detection sample (160 bytes is sufficient
879 // for most audio format headers to be identified)
880 detection_buffer.resize(160);
881 size_t bytesRead = p_input->readBytes(detection_buffer.data(), detection_buffer.size());
882
883 // If no data is available, we cannot proceed with detection
884 if (bytesRead == 0) return false;
885
886 // Feed the sample data to the MIME detector for format analysis
887 // The detector examines file headers, magic numbers, etc.
888 mime_detector.write(detection_buffer.data(), bytesRead);
889 mime = mime_detector.mime();
890 LOGI("mime from detector: %s", toStr(mime));
891
892 // Create a prefix stream that combines the detection data with the original stream
893 // This ensures the decoder receives the complete stream including the bytes used for detection
894 prefix_stream.setData(detection_buffer.data(), bytesRead, *p_input);
895 p_data_source = &prefix_stream;
896 }
897
898 // Process the detected/provided MIME type
899 if (mime != nullptr) {
900 // Delegate to the overloaded selectDecoder(mime) method to find
901 // and initialize the appropriate decoder for this MIME type
902 if (!selectDecoder(mime)) {
903 LOGE("The decoder could not be selected for %s", toStr(mime));
904 return false; // No registered decoder can handle this format
905 }
906 } else {
907 // MIME detection failed - format is unknown or unsupported
908 LOGE("Could not determine mime type");
909 return false;
910 }
911 } else {
912 LOGI("Decoder already selected: %s", toStr(actual_decoder.mime));
913 assert(p_input != nullptr);
914 actual_decoder.decoder->setInput(*p_input);
915 }
916
917 // Success: either decoder was already selected or selection completed successfully
918 return true;
919 }
