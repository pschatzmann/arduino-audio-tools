556 int available() override {
557 int result = 0;
558 if (has_buffer && buffer_pos < buffer_size) {
559 result += buffer_size - buffer_pos;
560 }
561 if (p_stream != nullptr) {
562 result += p_stream->available();
563 }
564 return result;
565 }
566
574 int read() override {
575 if (has_buffer && buffer_pos < buffer_size) {
576 return buffer_data[buffer_pos++];
577 }
578 if (p_stream != nullptr) {
579 return p_stream->read();
580 }
581 return -1;
582 }
583
593 size_t readBytes(uint8_t* data, size_t len) override {
594 size_t total_read = 0;
595
596 // First read from buffer if available
597 if (has_buffer && buffer_pos < buffer_size) {
598 size_t buffer_available = buffer_size - buffer_pos;
599 size_t from_buffer = min(len, buffer_available);
600 memcpy(data, buffer_data + buffer_pos, from_buffer);
601 buffer_pos += from_buffer;
602 total_read += from_buffer;
603 data += from_buffer;
604 len -= from_buffer;
605
606 // If we consumed all buffer data, mark as done
607 if (buffer_pos >= buffer_size) {
608 has_buffer = false;
609 }
610 }
611
612 // Then read remaining from stream
613 if (len > 0 && p_stream != nullptr) {
614 total_read += p_stream->readBytes(data, len);
615 }
616
617 return total_read;
618 }
619
625 size_t write(uint8_t) override { return 0; } // Not used for input
626
632 int peek() override {
633 if (has_buffer && buffer_pos < buffer_size) {
634 return buffer_data[buffer_pos];
635 }
636 if (p_stream != nullptr) {
637 return p_stream->peek();
638 }
639 return -1;
640 }

642 protected:

644 uint8_t* buffer_data = nullptr;
645 size_t buffer_size = 0;

691 bool selectDecoder(const char* mime) {
692 bool result = false;
693 if (mime == nullptr) return false;
694
695 // Do nothing if no change
696 if (StrView(mime).equals(actual_decoder.mime)) {
697 is_first = false;
698 return true;
699 }
700
701 // Close actual decoder if different
702 if (actual_decoder.decoder != nullptr) {
703 actual_decoder.decoder->end();
704 }
705
706 // Find the corresponding decoder
707 selected_mime = nullptr;
708 for (int j = 0; j < decoders.size(); j++) {
709 DecoderInfo info = decoders[j];
710 if (StrView(info.mime).equals(mime)) {
711 LOGI("Using StreamingDecoder for %s (%s)", info.mime, mime);
712 actual_decoder = info;
713
714 // Set up output for the selected decoder
715 if (p_print != nullptr) {
716 actual_decoder.decoder->setOutput(*p_print);
717 }
718
719 // Note: Input will be set up later with buffered stream
720
721 // Start the decoder
722 if (actual_decoder.decoder->begin()) {
723 actual_decoder.is_open = true;
724 LOGI("StreamingDecoder %s started", actual_decoder.mime);
725 } else {
726 LOGE("Failed to start StreamingDecoder %s", actual_decoder.mime);
727 return false;
728 }
729
730 result = true;
731 selected_mime = mime;
732 break;
733 }
734 }
735 is_first = false;
736 return result;
737 }
