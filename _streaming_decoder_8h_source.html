568
574 int available() override {
575 int result = 0;
576 if (has_buffer && buffer_pos < buffer_size) {
577 result += buffer_size - buffer_pos;
578 }
579 if (p_stream != nullptr) {
580 result += p_stream->available();
581 }
582 return result;
583 }
584
592 int read() override {
593 if (has_buffer && buffer_pos < buffer_size) {
594 return buffer_data[buffer_pos++];
595 }
596 if (p_stream != nullptr) {
597 return p_stream->read();
598 }
599 return -1;
600 }
601
611 size_t readBytes(uint8_t* data, size_t len) override {
612 size_t total_read = 0;
613
614 // First read from buffer if available
615 if (has_buffer && buffer_pos < buffer_size) {
616 size_t buffer_available = buffer_size - buffer_pos;
617 size_t from_buffer = min(len, buffer_available);
618 memcpy(data, buffer_data + buffer_pos, from_buffer);
619 buffer_pos += from_buffer;
620 total_read += from_buffer;
621 data += from_buffer;
622 len -= from_buffer;
623
624 // If we consumed all buffer data, mark as done
625 if (buffer_pos >= buffer_size) {
626 has_buffer = false;
627 }
628 }
629
630 // Then read remaining from stream
631 if (len > 0 && p_stream != nullptr) {
632 total_read += p_stream->readBytes(data, len);
633 }
634
635 return total_read;
636 }
637
643 size_t write(uint8_t) override { return 0; } // Not used for input
644
650 int peek() override {
651 if (has_buffer && buffer_pos < buffer_size) {
652 return buffer_data[buffer_pos];
653 }
654 if (p_stream != nullptr) {
655 return p_stream->peek();
656 }
657 return -1;
658 }
659
660 protected:
661 Stream* p_stream = nullptr;
662 uint8_t* buffer_data = nullptr;
663 size_t buffer_size = 0;
664 size_t buffer_pos = 0;
665 bool has_buffer = false;

687 DecoderInfo(const char* mime, StreamingDecoder* decoder) {

691 } actual_decoder;
692
693 Vector<DecoderInfo> decoders{0};
694 MimeDetector mime_detector;
695 BufferedPrefixStream buffered_stream;
696 Vector<uint8_t> detection_buffer{0};

698 const char* selected_mime = nullptr;

709 bool selectDecoder(const char* mime) {
710 bool result = false;
711 if (mime == nullptr) return false;
712
713 // Do nothing if no change
714 if (StrView(mime).equals(actual_decoder.mime)) {
715 is_first = false;
716 return true;
717 }
718
719 // Close actual decoder if different
720 if (actual_decoder.decoder != nullptr) {
721 actual_decoder.decoder->end();
722 }
723
724 // Find the corresponding decoder
725 selected_mime = nullptr;
726 for (int j = 0; j < decoders.size(); j++) {
727 DecoderInfo info = decoders[j];
728 if (StrView(info.mime).equals(mime)) {
729 LOGI("Using StreamingDecoder for %s (%s)", info.mime, mime);
730 actual_decoder = info;
731
732 // Set up output for the selected decoder
733 if (p_print != nullptr) {
734 actual_decoder.decoder->setOutput(*p_print);
735 }
736
737 // Note: Input will be set up later with buffered stream
738
739 // Start the decoder
740 if (actual_decoder.decoder->begin()) {
741 actual_decoder.is_open = true;
742 LOGI("StreamingDecoder %s started", actual_decoder.mime);
743 } else {
744 LOGE("Failed to start StreamingDecoder %s", actual_decoder.mime);
745 return false;
746 }
747
748 result = true;
749 selected_mime = mime;
750 break;
751 }
752 }
753 is_first = false;
754 return result;
755 }

765 if (p_input == nullptr) return 0;
766 return p_input->readBytes(data, len);
