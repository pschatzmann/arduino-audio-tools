485 virtual bool copy() override {
486 if (p_input == nullptr) return false;
487
488 // Automatically select decoder if not already selected
489 if (is_first) {
490 if (actual_decoder.decoder == nullptr) {
491 // Read some data to determine the mime type
492 detection_buffer.resize(80);
493 size_t bytesRead =
494 readBytes(detection_buffer.data(), detection_buffer.size());
495 if (bytesRead == 0) return false;
496
497 // Make data available again
498 buffered_stream.setBuffer(detection_buffer.data(), bytesRead);
499
500 // Use the mime detector to determine format
501 mime_detector.write(detection_buffer.data(), bytesRead);
502 const char* mime = mime_detector.mime();
503
504 if (mime != nullptr) {

506 // Select the decoder based on the determined mime type
507 if (!selectDecoder(mime)) {

514 }

518 buffered_stream.setStream(*p_input);
519
520 // Redirect the decoder to use the buffered stream
521 if (actual_decoder.decoder != nullptr) {
522 actual_decoder.decoder->setInput(buffered_stream);
523 }
524
525 is_first = false;
526 }
527
528 // Check if we have a decoder
529 if (actual_decoder.decoder == nullptr) return false;
530
531 // Use the selected decoder to process data
532 return actual_decoder.decoder->copy();

546 bool selectDecoder(const char* mime) {
547 bool result = false;
548 if (mime == nullptr) return false;
549
550 // Do nothing if no change
551 if (StrView(mime).equals(actual_decoder.mime)) {
552 is_first = false;
553 return true;
554 }
555
556 // Close actual decoder if different
557 if (actual_decoder.decoder != nullptr) {
558 actual_decoder.decoder->end();
559 }
560
561 // Find the corresponding decoder
562 selected_mime = nullptr;
563 for (int j = 0; j < decoders.size(); j++) {
564 DecoderInfo info = decoders[j];
565 if (StrView(info.mime).equals(mime)) {
566 LOGI("Using StreamingDecoder for %s (%s)", info.mime, mime);
567 actual_decoder = info;
568
569 // Set up output for the selected decoder
570 if (p_print != nullptr) {
571 actual_decoder.decoder->setOutput(*p_print);
572 }
573
574 // Note: Input will be set up later with buffered stream
575
576 // Start the decoder
577 if (actual_decoder.decoder->begin()) {
578 actual_decoder.is_open = true;
579 LOGI("StreamingDecoder %s started", actual_decoder.mime);
580 } else {
581 LOGE("Failed to start StreamingDecoder %s", actual_decoder.mime);
582 return false;
583 }
584
585 result = true;
586 selected_mime = mime;
587 break;
588 }
589 }
590 is_first = false;
591 return result;
592 }

723 size_t readBytes(uint8_t* data, size_t len) override {
724 size_t total_read = 0;
725
726 // First read from buffer if available
727 if (has_buffer && buffer_pos < buffer_size) {
728 size_t buffer_available = buffer_size - buffer_pos;
729 size_t from_buffer = min(len, buffer_available);
730 memcpy(data, buffer_data + buffer_pos, from_buffer);
731 buffer_pos += from_buffer;
732 total_read += from_buffer;
733 data += from_buffer;
734 len -= from_buffer;
735
736 // If we consumed all buffer data, mark as done
737 if (buffer_pos >= buffer_size) {
738 has_buffer = false;
739 }
740 }
741
742 // Then read remaining from stream
743 if (len > 0 && p_stream != nullptr) {
744 total_read += p_stream->readBytes(data, len);
745 }
746
747 return total_read;
748 }
749
755 size_t write(uint8_t) override { return 0; } // Not used for input
756
762 int peek() override {
763 if (has_buffer && buffer_pos < buffer_size) {
764 return buffer_data[buffer_pos];
765 }
766 if (p_stream != nullptr) {
767 return p_stream->peek();
768 }
769 return -1;
770 }
