615 int available() override {
616 int result = 0;
617 if (has_buffer && buffer_pos < buffer_size) {
618 result += buffer_size - buffer_pos;
619 }

621 result += p_stream->available();
622 }

633 int read() override {
634 if (has_buffer && buffer_pos < buffer_size) {
635 return buffer_data[buffer_pos++];
636 }
637 if (p_stream != nullptr) {
638 return p_stream->read();
639 }
640 return -1;
641 }
642
652 size_t readBytes(uint8_t* data, size_t len) override {
653 size_t total_read = 0;
654
655 // First read from buffer if available
656 if (has_buffer && buffer_pos < buffer_size) {
657 size_t buffer_available = buffer_size - buffer_pos;
658 size_t from_buffer = min(len, buffer_available);
659 memcpy(data, buffer_data + buffer_pos, from_buffer);
660 buffer_pos += from_buffer;
661 total_read += from_buffer;
662 data += from_buffer;
663 len -= from_buffer;
664
665 // If we consumed all buffer data, mark as done
666 if (buffer_pos >= buffer_size) {
667 has_buffer = false;
668 }
669 }
670
671 // Then read remaining from stream
672 if (len > 0 && p_stream != nullptr) {
673 total_read += p_stream->readBytes(data, len);
674 }
675
676 return total_read;
677 }
678
684 size_t write(uint8_t) override { return 0; } // Not used for input
685
691 int peek() override {
692 if (has_buffer && buffer_pos < buffer_size) {
693 return buffer_data[buffer_pos];
694 }
695 if (p_stream != nullptr) {
696 return p_stream->peek();
697 }
698 return -1;
699 }

703 uint8_t* buffer_data = nullptr;

751 bool selectDecoder(const char* mime) {
752 bool result = false;
753 if (mime == nullptr) return false;
754
755 // Do nothing if no change
756 if (StrView(mime).equals(actual_decoder.mime)) {
757 is_first = false;
758 return true;
759 }
760
761 // Close actual decoder if different
762 if (actual_decoder.decoder != nullptr) {
763 actual_decoder.decoder->end();
764 }
765
766 // Find the corresponding decoder
767 selected_mime = nullptr;
768 for (int j = 0; j < decoders.size(); j++) {
769 DecoderInfo info = decoders[j];
770 if (StrView(info.mime).equals(mime)) {
771 LOGI("Using StreamingDecoder for %s (%s)", info.mime, mime);
772 actual_decoder = info;
773
774 // Set up output for the selected decoder
775 if (p_print != nullptr) {
776 actual_decoder.decoder->setOutput(*p_print);
777 }
778
779 // Note: Input will be set up later with buffered stream
780
781 // Start the decoder
782 if (actual_decoder.decoder->begin()) {
783 actual_decoder.is_open = true;
784 LOGI("StreamingDecoder %s started", actual_decoder.mime);
785 } else {
786 LOGE("Failed to start StreamingDecoder %s", actual_decoder.mime);
787 return false;
788 }
789
790 result = true;
791 selected_mime = mime;
792 break;
793 }
794 }
795 is_first = false;
796 return result;
797 }

807 if (p_input == nullptr) return 0;
