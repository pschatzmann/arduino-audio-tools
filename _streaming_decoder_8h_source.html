523 bool selectDecoder(const char* mime) {
524 bool result = false;
525
526 // Guard against null MIME type - cannot proceed without valid MIME
527 if (mime == nullptr) return false;
528
529 // Optimization: Check if the requested MIME type is already active
530 // This avoids unnecessary decoder switching when the same format is detected
531 if (StrView(mime).equals(actual_decoder.mime)) {
532 is_first = false; // Mark initialization as complete
533 return true; // Already using the correct decoder
534 }
535
536 // Clean shutdown of currently active decoder before switching
537 // This ensures proper resource cleanup and state reset
538 if (actual_decoder.decoder != nullptr) {
539 actual_decoder.decoder->end();
540 }
541
542 // Search through all registered decoders to find one that handles this MIME type
543 selected_mime = nullptr; // Clear previous selection
544 for (int j = 0; j < decoders.size(); j++) {
545 DecoderInfo info = decoders[j];
546
547 // Check if this decoder supports the detected MIME type
548 if (StrView(info.mime).equals(mime)) {
549 LOGI("Using StreamingDecoder for %s (%s)", info.mime, mime);
550
551 // Switch to the matching decoder
552 actual_decoder = info;
553
554 // Configure the decoder's output stream to match our output
555 // This ensures decoded audio data flows to the correct destination
556 if (p_print != nullptr) {
557 actual_decoder.decoder->setOutput(*p_print);
558 }
559
560 // Note: Input stream will be configured later with the buffered stream
561 // that preserves the data used for MIME detection
562
563 // Initialize the selected decoder and mark it as active
564 if (actual_decoder.decoder->begin()) {
565 actual_decoder.is_open = true;
566 LOGI("StreamingDecoder %s started", actual_decoder.mime);
567 } else {
568 // Decoder failed to start - this is a critical error
569 LOGE("Failed to start StreamingDecoder %s", actual_decoder.mime);
570 return false;
571 }
572
573 // Successfully found and initialized a decoder
574 result = true;
575 selected_mime = mime; // Store the MIME type that was selected
576 break; // Stop searching once we find a match
577 }
578 }
579
580 // Mark initialization phase as complete regardless of success/failure
581 is_first = false;
582 return result; // true if decoder was found and started, false otherwise
583 }

739 bool selectDecoder() {
740 // Only perform MIME detection and decoder selection if no decoder is active yet
741 // This prevents re-detection on subsequent calls during the same stream
742 if (actual_decoder.decoder == nullptr) {
743 const char* mime = nullptr;
744
745 // Two methods for MIME type determination: external source or auto-detection
746 if (p_mime_source) {
747 // Option 1: Use externally provided MIME source (e.g., from HTTP headers)
748 // This is more efficient as it avoids reading and analyzing stream data
749 mime = p_mime_source->mime();
750 LOGI("mime from source: %s", mime);
751 } else {
752 // Option 2: Auto-detect MIME type by analyzing stream content
753 // This requires reading a sample of data to identify the format
754
755 // Allocate buffer for MIME detection sample (80 bytes is typically sufficient
756 // for most audio format headers to be identified)
757 detection_buffer.resize(160);
758 size_t bytesRead = buffered_stream.peekBytes(detection_buffer.data(), detection_buffer.size());
759
760 // If no data is available, we cannot proceed with detection
761 if (bytesRead == 0) return false;
762
763 // Feed the sample data to the MIME detector for format analysis
764 // The detector examines file headers, magic numbers, etc.
765 mime_detector.write(detection_buffer.data(), bytesRead);
766 mime = mime_detector.mime();
767 LOGI("mime from detector: %s", mime);
768 }
769
770 // Process the detected/provided MIME type
771 if (mime != nullptr) {
772 // Delegate to the overloaded selectDecoder(mime) method to find
773 // and initialize the appropriate decoder for this MIME type
774 if (!selectDecoder(mime)) {
775 LOGE("The decoder could not be found for %s", mime);
776 return false; // No registered decoder can handle this format
777 }
778 } else {
779 // MIME detection failed - format is unknown or unsupported
780 LOGE("Could not determine mime type");
781 return false;
782 }
783 }
784
785 // Success: either decoder was already selected or selection completed successfully
786 return true;
787 }
