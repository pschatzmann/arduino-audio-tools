167 bool selectDecoder(const char* mime) {
168 bool result = false;
169 if (mime == nullptr) return false;
170 // do nothing if no change
171 if (StrView(mime).equals(actual_decoder.mime)) {
172 is_first = false;
173 return true;
174 }
175 // close actual decoder
176 if (actual_decoder.decoder != this) end();
177
178 // find the corresponding decoder
179 selected_mime = nullptr;
180 for (int j = 0; j < decoders.size(); j++) {
181 DecoderInfo info = decoders[j];
182 if (StrView(info.mime).equals(mime)) {
183 LOGI("Using decoder for %s (%s)", info.mime, mime);
184 actual_decoder = info;
185 // define output if it has not been defined
186 if (p_print != nullptr && actual_decoder.decoder != this
187 && actual_decoder.decoder->getOutput() == nullptr) {
188 actual_decoder.decoder->setOutput(*p_print);
189 }
190 if (!*actual_decoder.decoder) {
191 actual_decoder.decoder->begin();
192 LOGI("Decoder %s started", actual_decoder.mime);
193 }
194 result = true;
195 selected_mime = mime;
196 break;
197 }
198 }
199 is_first = false;
200 return result;
201 }

225 size_t write(const uint8_t* data, size_t len) override {
226 if (is_first) {
227 const char* mime = nullptr;
228 if (p_mime_source != nullptr) {
229 // get content type from http header
230 mime = p_mime_source->mime();
231 if (mime) LOGI("mime from http request: %s", mime);
232 }
233 if (mime == nullptr) {
234 // use the mime detector
235 mime_detector.write((uint8_t*)data, len);
236 mime = mime_detector.mime();
237 if (mime) LOGI("mime from mime_detector: %s", mime);
238 }
239 if (mime != nullptr) {
240 // select the decoder based on the detemined mime type
241 if (!selectDecoder(mime)) {
242 LOGE("The decoder could not be found for %s", mime);
243 actual_decoder.decoder = &nop;
244 actual_decoder.is_open = true;
245 }
246 }
247 is_first = false;
248 }
249 // check if we have a decoder
250 if (actual_decoder.decoder == nullptr) return 0;
251 // decode the data
252 return actual_decoder.decoder->write(data, len);
253 }
