181 bool selectDecoder(const char* mime) {
182 bool result = false;
183 if (mime == nullptr) return false;
184 // do nothing if no change
185 if (StrView(mime).equals(actual_decoder.mime)) {
186 is_first = false;
187 return true;
188 }
189 // close actual decoder
190 if (actual_decoder.decoder != this) end();
191
192 // find the corresponding decoder
193 selected_mime = nullptr;
194 for (int j = 0; j < decoders.size(); j++) {
195 DecoderInfo info = decoders[j];
196 if (StrView(info.mime).equals(mime)) {
197 LOGI("Using decoder for %s (%s)", info.mime, mime);
198 actual_decoder = info;
199 // define output if it has not been defined
200 if (p_print != nullptr && actual_decoder.decoder != this
201 && actual_decoder.decoder->getOutput() == nullptr) {
202 actual_decoder.decoder->setOutput(*p_print);
203 }
204 if (!*actual_decoder.decoder) {
205 actual_decoder.decoder->begin();
206 LOGI("Decoder %s started", actual_decoder.mime);
207 }
208 result = true;
209 selected_mime = mime;
210 break;
211 }
212 }
213 is_first = false;
214 return result;
215 }

239 size_t write(const uint8_t* data, size_t len) override {
240 if (is_first) {
241 const char* mime = nullptr;
242 if (p_mime_source != nullptr) {
243 // get content type from http header
244 mime = p_mime_source->mime();
245 if (mime) LOGI("mime from http request: %s", mime);
246 }
247 if (mime == nullptr) {
248 // use the mime detector
249 mime_detector.write((uint8_t*)data, len);
250 mime = mime_detector.mime();
251 if (mime) LOGI("mime from mime_detector: %s", mime);
252 }
253 if (mime != nullptr) {
254 // select the decoder based on the detemined mime type
255 if (!selectDecoder(mime)) {
256 LOGE("The decoder could not be found for %s", mime);
257 actual_decoder.decoder = &nop;
258 actual_decoder.is_open = true;
259 }
260 }
261 is_first = false;
262 }
263 // check if we have a decoder
264 if (actual_decoder.decoder == nullptr) return 0;
265 // decode the data
266 return actual_decoder.decoder->write(data, len);
267 }
