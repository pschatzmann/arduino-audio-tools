218 SampleT process(SampleT sample) override {
219 if (activeKernel == nullptr) {
220 LOGE("Kernel not set!");
221 return sample; // Pass-through if no kernel set
222 }
223
224 xHistory[idxHist] = sample;
225
226 // Use AccT to prevent overflow and/or allow float accumulation
227 AccT acc = (AccT)0;
228 int idx = idxHist;
229
230 for (int n = 0; n < NUM_TAPS; n++) {
231 // Coefficients are Q15 int16_t
232 acc += (AccT)xHistory[idx] * (AccT)activeKernel[n];
233 if (--idx < 0) idx = NUM_TAPS - 1;
234 }
235
236 if (++idxHist >= NUM_TAPS) idxHist = 0;
237
238 // Convert back from Q15 and saturate
239 return fromQ15(acc);
240 }
241
242 protected:
243 SampleT xHistory[NUM_TAPS] = {(SampleT)0};
244 int idxHist = 0;
245 volatile int16_t* activeKernel = nullptr; // Pointer to active kernel

247 static inline SampleT fromQ15(AccT acc) {
248 // Default path: integer SampleT (current behavior)
249 if constexpr (std::numeric_limits<SampleT>::is_integer) {
250 // Shift back from Q15
251 if constexpr (std::numeric_limits<AccT>::is_integer) {
252 acc >>= 15;
253 } else {
254 acc = acc / (AccT)(1 << 15);
255 }
256
257 // Saturation to SampleT range
258 const AccT hi = (AccT)std::numeric_limits<SampleT>::max();
259 const AccT lo = (AccT)std::numeric_limits<SampleT>::min();
260 if (acc > hi) acc = hi;
261 if (acc < lo) acc = lo;
262 return (SampleT)acc;
263 } else {
264 // Floating output sample: scale Q15 back to approximately [-1..1]
265 return (SampleT)(acc / (AccT)(1 << 15));
266 }
267 }
268 };

330 // Setup Center Frequencies Logarithmically spaced between 20Hz and Nyquist
331 void setupFrequencies(int sampleRate) {
332 if (NUM_BANDS <= 0) return;
333 float fMin = log10f(20.0f);
334 float fMax = log10f(sampleRate / 2.0f); // Nyquist frequency
335 if (NUM_BANDS == 1) {
336 centerFreqs[0] = powf(10.0f, (fMin + fMax) * 0.5f);
337 LOGD("Only one band: center frequency set to %.2f Hz", centerFreqs[0]);
338 return;
339 }
340 float step = (fMax - fMin) / (float)(NUM_BANDS - 1);
341 for (int i = 0; i < NUM_BANDS; i++) {
342 centerFreqs[i] = powf(10.0f, fMin + step * (float)i);
343 LOGD("Band %d: center frequency = %.2f Hz", i, centerFreqs[i]);
344 }
345 }

347 // Pre-calculate Blackman window coefficients (performance optimization)
348 void preCalculateWindow() {
349 const float N_minus_1 = (float)(NUM_TAPS - 1);
350 for (int n = 0; n < NUM_TAPS; n++) {
351 windowCoeffs[n] = 0.42f - 0.5f * cosf(2.0f * PI * n / N_minus_1) +
352 0.08f * cosf(4.0f * PI * n / N_minus_1);
353 }
354 }

356 // Initialize a kernel to pass-through (identity filter)
357 void initializeKernel(volatile int16_t* kernel) {
358 const int M = (NUM_TAPS - 1) / 2;
359 for (int i = 0; i < NUM_TAPS; i++) {
360 if (i == M) {
361 kernel[i] = (int16_t)Q15_SCALE; // Unity gain at center tap
362 } else {
363 kernel[i] = 0;
364 }
365 }

368 bool updateFIRKernel() {
369 if (currentSampleRate <= 0) {
370 LOGE("Invalid sample rate: %d", currentSampleRate);
371 return false; // Not initialized yet
372 }
373
374 // Prevent concurrent updates (e.g., from multiple tasks/threads).
375 // We keep the critical section tiny: only the check/set of the flag.
376 enterCritical();
377 if (isUpdating) {
378 exitCritical();
379 return false;
380 }
381 isUpdating = true;
382 exitCritical();
383
384 // Transactional gain update: copy user-updated pendingGains into gains.
385 // Keep this critical section short to avoid blocking audio processing.
386 enterCritical();
387 memcpy(gains, pendingGains, sizeof(gains));
388 exitCritical();
389
390 memset(tempFloat, 0, sizeof(tempFloat));
391
392 const int M = (NUM_TAPS - 1) / 2;
393 const float sampleRateFloat = (float)currentSampleRate;
394
395 // Base impulse (Pass-through)
396 tempFloat[M] = 1.0f;
397
398 for (int i = 0; i < NUM_BANDS; i++) {
399 // Skip bands with 0dB gain to save precision
400 if (fabs(gains[i]) < 0.1f) continue;
401
402 // Calculate Linear Gain Delta
403 // If gain is +6dB (2.0x), we add (2.0 - 1.0) = +1.0 to the impulse
404 float linGain = powf(10.0f, gains[i] / 20.0f) - 1.0f;
405
406 // Use actual sample rate, not hardcoded 44100
407 float fL = (centerFreqs[i] * 0.707f) /
408 sampleRateFloat; // Lower Edge (-3dB point)
409 float fH = (centerFreqs[i] * 1.414f) /
410 sampleRateFloat; // Upper Edge (+3dB point)
411
412 // Clamp to valid normalized frequency range [0, 0.5] (Nyquist)
413 if (fL < 0.0f) fL = 0.0f;
414 if (fH < 0.0f) fH = 0.0f;
415 if (fL > 0.5f) fL = 0.5f;
416 if (fH > 0.5f) fH = 0.5f;
417 if (fH <= fL) continue;
418
419 for (int n = 0; n < NUM_TAPS; n++) {
420 float nM = (float)(n - M);
421
422 // Use pre-calculated window coefficients
423 float window = windowCoeffs[n];
424
425 // Bandpass filter: highpass - lowpass
426 float bp = (2.0f * fH * sinc(2.0f * fH * nM)) -
427 (2.0f * fL * sinc(2.0f * fL * nM));
428
429 // Add the weighted bandpass to the master kernel
430 tempFloat[n] += bp * window * linGain;
431 }
432 }
433
434 // Update the inactive kernel (no interruption needed for writes)
435 for (int i = 0; i < NUM_TAPS; i++) {
436 // DIRECT CONVERSION (No Auto-Normalization)
437 // This ensures +6dB actually outputs louder voltage
438 int32_t q = (int32_t)(tempFloat[i] * Q15_SCALE);
439
440 // Hard Clip the Kernel Coefficients to prevent wrap-around
441 if (q > 32767) q = 32767;
442 if (q < -32768) q = -32768;
443 updateKernel[i] = (int16_t)q;
444 }
445
446 // Atomically swap the kernel pointers
447 // This is the only operation that needs to be atomic
448 enterCritical();
449 volatile int16_t* temp = activeKernel;
450 activeKernel = updateKernel;
451 updateKernel = temp;
452
453 // Update filter references to new active kernel
454 for (auto& fir : fir_vector) {
455 fir.setKernel(activeKernel);
456 }
457 exitCritical();
458
459 // Release re-entrancy guard
460 enterCritical();
461 isUpdating = false;
462 gainsDirty = false;
463 exitCritical();
464
465 LOGI("FIR kernel updated with new gains for %d bands /%d taps.", NUM_BANDS,
466 NUM_TAPS);
467 return true;
468 }
