153  i2s_std_slot_config_t getSlotConfig(I2SConfigESP32V1 &cfg) {
154  TRACED();
155  i2s_std_slot_config_t result;
156  switch (cfg.i2s_format) {
157  case I2S_LEFT_JUSTIFIED_FORMAT:
158  case I2S_MSB_FORMAT:
159  result = I2S_STD_MSB_SLOT_DEFAULT_CONFIG(
160  (i2s_data_bit_width_t)cfg.bits_per_sample,
161  (i2s_slot_mode_t)cfg.channels);
162  case I2S_PCM:
163  result = I2S_STD_PCM_SLOT_DEFAULT_CONFIG(
164  (i2s_data_bit_width_t)cfg.bits_per_sample,
165  (i2s_slot_mode_t)cfg.channels);
166  default:
167  result = I2S_STD_PHILIPS_SLOT_DEFAULT_CONFIG(
168  (i2s_data_bit_width_t)cfg.bits_per_sample,
169  (i2s_slot_mode_t)cfg.channels);
170  }
171 
172  // Update slot_mask if only one channel
173  if (cfg.channels == 1) {
174  switch (cfg.channel_format) {
175  case I2SChannelSelect::Left:
176  result.slot_mask = I2S_STD_SLOT_LEFT;
177  break;
178  case I2SChannelSelect::Right:
179  result.slot_mask = I2S_STD_SLOT_RIGHT;
180  break;
181  case I2SChannelSelect::Stereo:
182  LOGW("Invalid channel_format: %d", cfg.channel_format);
183  break;
184  }
185  }
186 
187  return result;
188  }

208  bool startChannels(I2SConfigESP32V1 &cfg, i2s_chan_handle_t &tx_chan,
209  i2s_chan_handle_t &rx_chan, int txPin, int rxPin) {
210  TRACED();
211  LOGI("tx: %d, rx: %d", txPin, rxPin);
212  i2s_std_config_t std_cfg = {
213  .clk_cfg = getClockConfig(cfg),
214  .slot_cfg = getSlotConfig(cfg),
215  .gpio_cfg =
216  {
217  .mclk = (gpio_num_t)cfg.pin_mck,
218  .bclk = (gpio_num_t)cfg.pin_bck,
219  .ws = (gpio_num_t)cfg.pin_ws,
220  .dout = (gpio_num_t)txPin,
221  .din = (gpio_num_t)rxPin,
222  .invert_flags =
223  {
224  .mclk_inv = false,
225  .bclk_inv = false,
226  .ws_inv = false,
227  },
228  },
229  };
230 
231  if (cfg.rx_tx_mode == RXTX_MODE || cfg.rx_tx_mode == TX_MODE) {
232  if (i2s_channel_init_std_mode(tx_chan, &std_cfg) != ESP_OK) {
233  LOGE("i2s_channel_init_std_mode %s", "tx");
234  return false;
235  }
236  if (i2s_channel_enable(tx_chan) != ESP_OK) {
237  LOGE("i2s_channel_enable %s", "tx");
238  return false;
239  }
240  }
241 
242  if (cfg.rx_tx_mode == RXTX_MODE || cfg.rx_tx_mode == RX_MODE) {
243  if (i2s_channel_init_std_mode(rx_chan, &std_cfg) != ESP_OK) {
244  LOGE("i2s_channel_init_std_mode %s", "rx");
245  return false;
246  }
247  if (i2s_channel_enable(rx_chan) != ESP_OK) {
248  LOGE("i2s_channel_enable %s", "rx");
249  return false;
250  }
251  }
252 
253  LOGD("%s - %s", __func__, "started");
254  return true;
