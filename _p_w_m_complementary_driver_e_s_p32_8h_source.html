124 virtual void setupPWM() {
125 // frequency is driven by selected resolution
126 if (audio_config.pwm_frequency == 0) {
127 audio_config.pwm_frequency = frequency(audio_config.resolution) * 1000;
128 }
129 if (audio_config.channels > maxChannels()) {
130 LOGE("Only %d complementary channels supported", maxChannels());
131 audio_config.channels = maxChannels();
132 }
133 bool has_pairs = audio_config.pins().size() >= (size_t)(audio_config.channels * 2);
134 if (!has_pairs) {
135 LOGW("Expected %d pins for %d complementary channels, got %d - assuming consecutive pin+1 as low-side", audio_config.channels*2, audio_config.channels, audio_config.pins().size());
136 }
137 pins.resize(audio_config.channels);
138 for (int j = 0; j < audio_config.channels; j++) {
139 pins[j].unit = (mcpwm_unit_t)(j / 3);
140 pins[j].timer = (mcpwm_timer_t)(j % 3);
141 if (pins[j].unit > MCPWM_UNIT_1) { LOGE("Too many channels for MCPWM: %d", j); break; }
142 if (has_pairs) {
143 pins[j].gpio_high = audio_config.pins()[j*2];
144 pins[j].gpio_low = audio_config.pins()[j*2 + 1];
145 } else {
146 pins[j].gpio_high = audio_config.pins()[j];
147 pins[j].gpio_low = pins[j].gpio_high + 1;
148 }
149 mcpwm_io_signals_t sigA = (mcpwm_io_signals_t)(MCPWM0A + (pins[j].timer * 2));
150 mcpwm_io_signals_t sigB = (mcpwm_io_signals_t)(MCPWM0B + (pins[j].timer * 2));
151 esp_err_t err = mcpwm_gpio_init(pins[j].unit, sigA, pins[j].gpio_high);
152 if (err != ESP_OK) LOGE("mcpwm_gpio_init high error=%d", (int)err);
153 err = mcpwm_gpio_init(pins[j].unit, sigB, pins[j].gpio_low);
154 if (err != ESP_OK) LOGE("mcpwm_gpio_init low error=%d", (int)err);
155 mcpwm_config_t cfg; cfg.frequency = audio_config.pwm_frequency; cfg.cmpr_a = 0; cfg.cmpr_b = 0; cfg.counter_mode = MCPWM_UP_COUNTER; cfg.duty_mode = MCPWM_DUTY_MODE_0;
156 err = mcpwm_init(pins[j].unit, pins[j].timer, &cfg);
157 if (err != ESP_OK) LOGE("mcpwm_init error=%d", (int)err);
158 if (audio_config.dead_time_us > 0) {
159 uint32_t dead_ticks = audio_config.dead_time_us * 80u; // 80MHz APB
160 uint32_t period_ticks = 80000000UL / audio_config.pwm_frequency;
161 if (dead_ticks * 2 >= period_ticks) dead_ticks = period_ticks / 4;
162 if (dead_ticks > 0) {
163 err = mcpwm_deadtime_enable(pins[j].unit, pins[j].timer, MCPWM_ACTIVE_HIGH_COMPLIMENT_MODE, dead_ticks, dead_ticks);
164 if (err != ESP_OK) LOGE("deadtime_enable error=%d", (int)err);
165 }
166 }
167 LOGI("Complementary PWM ch=%d unit=%d timer=%d high=%d low=%d freq=%u dead_us=%u", j,(int)pins[j].unit,(int)pins[j].timer,pins[j].gpio_high,pins[j].gpio_low,(unsigned)audio_config.pwm_frequency,(unsigned)audio_config.dead_time_us);
168 }

185 virtual void pwmWrite(int channel, int value) {
186 if (channel < 0 || channel >= pins.size()) return;
187 int duty = (int)((int64_t)value * 100 / maxOutputValue());
188 if (duty < 0) duty = 0; else if (duty > 100) duty = 100;
189 mcpwm_set_duty(pins[channel].unit, pins[channel].timer, MCPWM_OPR_A, duty);
190 mcpwm_set_duty_type(pins[channel].unit, pins[channel].timer, MCPWM_OPR_A, MCPWM_DUTY_MODE_0);
191 if (audio_config.dead_time_us == 0) {
192 // software complementary: invert
193 mcpwm_set_duty(pins[channel].unit, pins[channel].timer, MCPWM_OPR_B, 100 - duty);
194 mcpwm_set_duty_type(pins[channel].unit, pins[channel].timer, MCPWM_OPR_B, MCPWM_DUTY_MODE_0);
195 }
