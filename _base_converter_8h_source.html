105template <typename T>
106class ConverterAutoCenterT : public BaseConverter {
107 public:
108 ConverterAutoCenterT(int channels = 2, bool isDynamic = false) {
109 this->channels = channels;
110 this->is_dynamic = isDynamic;
111 }
112
113 size_t convert(uint8_t(*src), size_t byte_count) override {
114 size_t size = byte_count / channels / sizeof(T);
115 T *sample = (T *)src;
116 setup((T *)src, size);
117 // convert data
118 if (is_setup) {
119 if (!is_dynamic) {
120 for (size_t j = 0; j < size; j++) {
121 for (int ch = 0; ch < channels; ch++) {
122 sample[(j * channels) + ch] =
123 sample[(j * channels) + ch] - offset_to[ch];
124 }
125 }
126 } else {
127 for (size_t j = 0; j < size; j++) {
128 for (int ch = 0; ch < channels; ch++) {
129 sample[(j * channels) + ch] = sample[(j * channels) + ch] -
130 offset_from[ch] +
131 (offset_step[ch] * size);
132 }
133 }
134 }
135 }
136 return byte_count;
137 }
138
139 protected:
140 Vector<float> offset_from{0};
141 Vector<float> offset_to{0};
142 Vector<float> offset_step{0};
143 Vector<float> total{0};
144 float left = 0.0;
145 float right = 0.0;
146 bool is_setup = false;
147 bool is_dynamic;
148 int channels;
149
150 void setup(T *src, size_t size) {
151 if (size == 0) return;
152 if (!is_setup || is_dynamic) {
153 if (offset_from.size() == 0) {
154 offset_from.resize(channels);
155 offset_to.resize(channels);
156 offset_step.resize(channels);
157 total.resize(channels);
158 }
159
160 // save last offset
161 for (int ch = 0; ch < channels; ch++) {
162 offset_from[ch] = offset_to[ch];
163 total[ch] = 0.0;
164 }
165
166 // calculate new offset
167 for (size_t j = 0; j < size; j++) {
168 for (int ch = 0; ch < channels; ch++) {
169 total[ch] += src[(j * channels) + ch];
170 }
171 }
172 for (int ch = 0; ch < channels; ch++) {
173 offset_to[ch] = total[ch] / size;
174 offset_step[ch] = (offset_to[ch] - offset_from[ch]) / size;
175 }
176 is_setup = true;
177 }
178 }

423 int frame_count = size / (sizeof(T) * from_channels);

432 *result++ = *source++;
433 result_size += sizeof(T);
434 }
435 // commbined last channels
436 T total = (int16_t)0;
437 for (int j = to_channels - 1; j < from_channels; j++) {
438 total += *source++ / reduceDiv;
439 }
440 *result++ = total;
441 result_size += sizeof(T);
442 }
443 return result_size;
444 }

638template <typename T>
639class BinT : public BaseConverter {
640 public:
641 BinT() = default;
642 BinT(int binSize, int channels, bool average) {
643 setChannels(channels);
644 setBinSize(binSize);
645 setAverage(average);
646 this->partialBinSize = 0;
647 this->partialBin = new T[channels]();
648 }
649
650 ~BinT() {
651 delete[] this->partialBin;
652 }
653
654 void setChannels(int channels) { this->channels = channels; }
655 void setBinSize(int binSize) { this->binSize = binSize; }
656 void setAverage(bool average) { this->average = average; }
657
658 size_t convert(uint8_t *src, size_t size) { return convert(src, src, size); }
659
660 size_t convert(uint8_t *target, uint8_t *src, size_t size) {
661 // The binning takes the following into account
662 // 1) if size is too small it will add up data to partialBin and return 0 size
663 // 2) if there is sufficient data to fill Bins but there is partial data remaining it will be added to the partial Bin
664 // 3) if there was previous partial Bin it will be filled with the new data
665 // 4) if there is insufficient new data to fill the partial Bin it will fill the partial Bin with the new data
666
667 if (size % (sizeof(T) * channels) > 0) {
668 LOGE("Buffer size %d is not a multiple of the number of channels %d", (int)size, channels);
669 return 0;
670 }
671
672 int sample_count = size / (sizeof(T) * channels); // new available samples in each channel
673 int total_samples = partialBinSize + sample_count; // total samples available for each channel including previous number of sample in partial bin
674 int bin_count = total_samples / binSize; // number of bins we can make
675 int remaining_samples = total_samples % binSize; // remaining samples after binning
676 T *p_target = (T *)target;
677 T *p_source = (T *)src;
678 size_t result_size = 0;
679
680 // Allocate sum for each channel with appropriate type
681 typename AppropriateSumType<T>::type sums[channels];
682 int current_sample = 0; // current sample index
683
684 // Is there a partial bin from the previous call?
685 // ----
686 if (partialBinSize > 0) {
687
688 int samples_needed = binSize - partialBinSize;
689 bool have_enough_samples = (samples_needed < sample_count);
690 int samples_to_bin = have_enough_samples ? samples_needed : sample_count;
691
692 for (int ch = 0; ch < channels; ch++) {
693 sums[ch] = partialBin[ch];
694 }
695
696 for (int j = 0; j < samples_to_bin; j++) {
697 for (int ch = 0; ch < channels; ch++) {
698 sums[ch] += p_source[current_sample * channels + ch];
699 }
700 current_sample++;
701 }
702
703 if (have_enough_samples) {
704 // Store the completed bin
705 if (average) {
706 for (int ch = 0; ch < channels; ch++) {
707 p_target[result_size / sizeof(T)] = static_cast<T>(sums[ch] / binSize);
708 result_size += sizeof(T);
709 }
710 } else {
711 for (int ch = 0; ch < channels; ch++) {
712 p_target[result_size / sizeof(T)] = static_cast<T>(sums[ch]);
713 result_size += sizeof(T);
714 }
715 }
716 partialBinSize = 0;
717 } else {
718 // Not enough samples to complete the bin, update partialBin
719 for (int ch = 0; ch < channels; ch++) {
720 partialBin[ch] = sums[ch];
721 }
722 partialBinSize += current_sample;
723
724 LOGD("bin %d: %d of %d remaining %d samples, %d > %d bytes", binSize, current_sample, total_samples, partialBinSize, (int)size, (int)result_size);
725 return result_size;
726 }
727 }
728
729 // Fill bins
730 // ----
731 for (int i = 0; i < bin_count; i++) {
732 for (int ch = 0; ch < channels; ch++) {
733 sums[ch] = p_source[current_sample * channels + ch]; // Initialize sums with first value in the input buffer
734 }
735
736 for (int j = 1; j < binSize; j++) {
737 for (int ch = 0; ch < channels; ch++) {
738 sums[ch] += p_source[(current_sample + j) * channels + ch];
739 }
740 }
741 current_sample += binSize;
742
743 // Store the bin result
744 if (average) {
745 for (int ch = 0; ch < channels; ch++) {
746 p_target[result_size / sizeof(T)] = static_cast<T>(sums[ch] / binSize);
747 result_size += sizeof(T);
748 }
749 } else {
750 for (int ch = 0; ch < channels; ch++) {
751 p_target[result_size / sizeof(T)] = static_cast<T>(sums[ch]);
752 result_size += sizeof(T);
753 }
754 }
755 }
756
757 // Store the remaining samples in the partial bin
758 // ----
759 for (int i = 0; i < remaining_samples; i++) {
760 for (int ch = 0; ch < channels; ch++) {
761 partialBin[ch] += p_source[(current_sample + i) * channels + ch];
762 }
763 }
764 partialBinSize = remaining_samples;
765
766 LOGD("bin %d: %d of %d remaining %d samples, %d > %d bytes", binSize, current_sample, total_samples, partialBinSize, (int) size, (int) result_size);
767
768 return result_size;
769 }
770
771 protected:
772 int channels = 2;
773 int binSize = 1;
774 bool average = true;
775 T *partialBin;
776 int partialBinSize;
777};

1042 // The binning works the same as in the BinT class
1043 // Here we add subtraction before we store the bins
1044

1046 LOGE("Buffer size needs to be multiple of channels")
1047 return 0;
1048 }
1049
1050 int sample_count = size / (sizeof(T) * channels); // new available samples in each channel
1051 int total_samples = partialBinSize + sample_count; // total samples available for each channel including previous number of sample in partial bin
1052 int bin_count = total_samples / binSize; // number of bins we can make
1053 int remaining_samples = total_samples % binSize; // remaining samples after binning
1054 T *p_target = (T *)target;
1055 T *p_source = (T *)src;
1056 size_t result_size = 0;
1057
1058 // Allocate sum for each channel with appropriate type
1059 typename AppropriateSumType<T>::type sums[channels];
1060 int current_sample = 0; // current sample index
1061
1062 // Is there a partial bin from the previous call?
1063 // ----
1064 if (partialBinSize > 0) {
1065
1066 // LOGD("Deal with partial bins");
1067
1068 int samples_needed = binSize - partialBinSize;
1069 bool have_enough_samples = (samples_needed < sample_count);
1070 int samples_to_bin = have_enough_samples ? samples_needed : sample_count;
1071
1072 // initialize
1073 for (int ch = 0; ch < channels; ch++) {
1074 sums[ch] = partialBin[ch];
1075 }
1076
1077 // continue binning
1078 for (int j = 0; j < samples_to_bin; j++) {
1079 for (int ch = 0; ch < channels; ch++) {
1080 sums[ch] += p_source[current_sample * channels + ch];
1081 }
1082 current_sample++;
1083 }
1084
1085 // store the bin results or update the partial bin
1086 if (have_enough_samples) {
1087 // Subtract two channels and store the completed bin
1088 if (average) {
1089 for (int ch = 0; ch < channels; ch+=2) {
1090 p_target[result_size / sizeof(T)] = static_cast<T>((sums[ch] - sums[ch+1]) / binSize);
1091 result_size += sizeof(T);
1092 }
1093 } else {
1094 for (int ch = 0; ch < channels; ch+=2) {
1095 p_target[result_size / sizeof(T)] = static_cast<T>((sums[ch] - sums[ch+1]));
1096 result_size += sizeof(T);
1097 }
1098 }
1099 partialBinSize = 0;
1100 // LOGD("Partial bins are empty");
1101
1102 } else {
1103 // Not enough samples to complete the bin, update partialBin
1104 for (int ch = 0; ch < channels; ch++) {
1105 partialBin[ch] = sums[ch];
1106 }
1107 partialBinSize += current_sample;
1108 LOGD("bin & channel subtract %d: %d of %d remaining %d samples, %d > %d bytes", binSize, current_sample, total_samples, partialBinSize,(int) size,(int) result_size);
1109 // LOGD("Partial bins were updated");
1110 return result_size;
1111 }
1112 }
1113
1114 // Fill bins
1115 // ----
1116 // LOGD("Fillin bins");
1117 for (int i = 0; i < bin_count; i++) {
1118
1119 // LOGD("Current sample %d", current_sample);
1120
1121 for (int ch = 0; ch < channels; ch++) {
1122 sums[ch] = p_source[current_sample * channels + ch]; // Initialize sums with first value in the input buffer
1123 }
1124
1125 for (int j = 1; j < binSize; j++) {
1126 for (int ch = 0; ch < channels; ch++) {
1127 sums[ch] += p_source[(current_sample + j) * channels + ch];
1128 }
1129 }
1130 current_sample += binSize;
1131
1132 // Finish binning, then subtact two channel and store the result
1133 if (average) {
1134 for (int ch = 0; ch < channels; ch+=2) {
1135 p_target[result_size / sizeof(T)] = static_cast<T>((sums[ch]-sums[ch+1]) / binSize);
1136 result_size += sizeof(T);
1137 }
1138 } else {
1139 for (int ch = 0; ch < channels; ch+=2) {
1140 p_target[result_size / sizeof(T)] = static_cast<T>((sums[ch]-sums[ch+1]));
1141 result_size += sizeof(T);
1142 }
1143 }
1144 }
1145
1146 // Store the remaining samples in the partial bin
1147 // ----
1148 // LOGD("Updating partial bins");
1149 for (int i = 0; i < remaining_samples; i++) {
1150 for (int ch = 0; ch < channels; ch++) {
1151 partialBin[ch] += p_source[(current_sample + i) * channels + ch];
1152 }
1153 }
1154 partialBinSize = remaining_samples;
1155
1156 LOGD("bin & channel subtract %d: %d of %d remaining %d samples, %d > %d bytes", binSize, current_sample, total_samples, partialBinSize, (int)size, (int) result_size);
1157
1158 return result_size;
1159 }

1794template <typename T, size_t Cn, size_t Cx, size_t S>
1795class CopyChannels : public BaseConverter {

1800 T *chan = (T *)src;
1801 size_t samples = (size / Cn) / sizeof(T);
1802 for (size_t s = 0; s < samples; s++) {
1803 chan[s * Cn + Cx] = (Cx < Cn) ? chan[s * Cn + Cx] << S : 0;
1804
1805 for (size_t c = 0; c < Cn; c++) {
1806 if (c != Cx) {
1807 chan[s * Cn + c] = chan[s * Cn + Cx];
1808 }
1809 }

1817 if (now - _prev_ms > 1000) {
1818 _prev_ms = now;
1819 LOGI("CopyChannels samples: %u, amplitude: %d", _counter, _max_val);
1820 _max_val = 0;
1821 }
1822 }
1823 return samples * Cn * sizeof(T);
1824 }
