167 getFileAtIndex(dir, pos, count, result);

173
175 void getFileAtIndex(AudioFile dir, size_t pos, size_t &idx, AudioFile &result) {
176 LOGD("%s: %d", LOG_METHOD, idx);
177 AudioFile file;
178 if (idx>pos) return;
179
180 while (file.openNext(&dir, O_READ)) {
181 if (idx>pos) return;
182 if (!file.isHidden()) {
183 if (!file.isDir()) {
184 if (isValidAudioFile(file)){
185 idx++;
186 if (idx-1==pos){
187 LOGI("-> get: '%s'", file_name);
188 result = file;
189 result.getName(file_name, MAX_FILE_LEN);
190 //return;
191 }
192 }
193 } else {
194 getFileAtIndex(file, pos, idx, result);
195 }
196
197 // close all files except result
198 char file_name_act[MAX_FILE_LEN];
199 file.getName(file_name_act, MAX_FILE_LEN);
200 //LOGI("-> %s <-> %s", file_name_act, file_name);
201 if (!Str(file_name_act).equals(file_name)){
202 file.getName(file_name, MAX_FILE_LEN);
203 file.close();
204 LOGI("-> close: '%s'", file_name);
205 } else {
206 return;
207 }
208 }
209 }
