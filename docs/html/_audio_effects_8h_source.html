258 size_t readBytes(uint8_t *buffer, size_t length) override {
259 if (!active || p_io==nullptr)return 0;
260 int frames = length / sizeof(T) / info.channels;
261 size_t result_size = 0;
262
263 if (p_io->available()<(sizeof(T)*info.channels)){
264 return 0;
265 }
266
267 for (int count=0;count<frames;count++){
268 // determine sample by combining all channels in frame
269 T result_sample = 0;
270 for (int ch=0;ch<info.channels;ch++){
271 T sample;
272 p_io->readBytes((uint8_t*)&sample, sizeof(T));
273 result_sample += sample / info.channels;
274 }
275
276 // apply effects
277 for (int j=0; j<size(); j++){
278 result_sample = effects[j]->process(result_sample);
279 }
280
281 // write result multiplying channels
282 T* p_buffer = ((T*)buffer)+(count*info.channels);
283 for (int ch=0;ch<info.channels;ch++){
284 p_buffer[ch] = result_sample;
285 result_size += sizeof(T);
286 }
287 }
288 return result_size;
289 }

295 size_t write(const uint8_t *buffer, size_t length) override {
296 if (!active)return 0;
297 // length must be multple of channels
298 assert(length % (sizeof(T)*info.channels)==0);
299 int frames = length / sizeof(T) / info.channels;
300 size_t result_size = 0;
301
302 // process all samples
303 for (int j=0;j<frames;j++){
304 // calculate sample for frame
305 T* p_buffer = ((T*)buffer) + (j*info.channels);
306 T sample=0;
307 for (int ch=0;ch<info.channels;ch++){
308 sample += p_buffer[ch] / info.channels;
309 result_size += sizeof(T);
310 }
311
312 // apply effects
313 for (int j=0; j<size(); j++){
314 sample = effects[j]->process(sample);
315 }
316
317 // wite result channel times to output defined in constructor
318 for (int ch=0;ch<info.channels;ch++){
319 if (p_io!=nullptr){
320 p_io->write((uint8_t*)&sample, sizeof(T));
321
322 } else if (p_print!=nullptr){
323 p_print->write((uint8_t*)&sample, sizeof(T));
324 }
325 }
326 }
327 return result_size;
328 }
