15 // Output I2S data to built-in DAC, no matter the data format is 16bit or 32 bit, the DAC module will only take the 8bits from MSB
16 static int16_t convert8DAC(int value, int value_bits_per_sample){
17  // -> convert to positive
18  int16_t result = (value * maxValue(8) / maxValue(value_bits_per_sample)) + maxValue(8) / 2;
19  return result;
20 }
