164 void f32_to_s16(float *in, int16_t *out, int num_samples) {
165 int i = 0;
166 for(; i < num_samples; i++){
167 float sample = in[i] * 32768.0f;
168 if (sample >= 32766.5f)
169 out[i] = (int16_t) 32767;
170 else if (sample <= -32767.5f)
171 out[i] = (int16_t)-32768;
172 else {
173 int16_t s = (int16_t)(sample + .5f);
174 s -= (s < 0); /* away from zero, to be compliant */
175 out[i] = s;
