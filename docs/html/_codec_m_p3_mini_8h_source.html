170  void f32_to_s16(float *in, int16_t *out, int num_samples) {
171  int i = 0;
172  for(; i < num_samples; i++){
173  float sample = in[i] * 32768.0f;
174  if (sample >= 32766.5f)
175  out[i] = (int16_t) 32767;
176  else if (sample <= -32767.5f)
177  out[i] = (int16_t)-32768;
178  else {
179  int16_t s = (int16_t)(sample + .5f);
180  s -= (s < 0); /* away from zero, to be compliant */
