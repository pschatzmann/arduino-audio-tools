38class ChartT {
39public:
40 void setup(std::string fName, int channels, int channel) {
41 this->fname = fName;

47 const std::string chartData() {
48 str.clear();
49 str.str("");
50 // reset buffer;
51 if (channel<channels){
52 ifstream is;
53 is.open(fname, is.binary);
54 is.seekg(wav_header_size, is.beg);
55 std::list<int16_t> audioList;
56 T buffer[channels];
57 size_t rec_size = channels*sizeof(T);
58 while(is.read((char *)buffer, rec_size)){
59 audioList.push_back(transform(buffer[channel]));
60 }
61 string str_size = std::to_string(audioList.size());
62 str << "<style>div.x-svg {width: "<< str_size <<"px; }</style>";
63 str << "<div class='x-svg'><svg viewBox='0 0 "<< str_size << " 100'> <polyline fill='none' stroke='blue' stroke-width='1' points ='";
64 // copy data from input stream
65 size_t idx = 0;
66 for(int16_t sample: audioList){
67 str << idx++ << "," << sample << " ";
68 }
69 str << "'/></svg></div>";
70 } else {
71 str << "<p>Channel " << channel << " of " << channels << " does not exist!</p>";
72 }
73 return str.str();
74 }
75
76protected:
77 std::stringstream str;
78 std::string fname;
79 const int wav_header_size = 44;
80 int channels=1;
81 int channel=0;
82
83 int transform(int x){
84 int result = x / 1000; // scale -32 to 32
85 result += 60; // shift down
86 return result;
87 }
