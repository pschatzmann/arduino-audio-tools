99 virtual size_t write(const void *input, size_t length) {
100 if (!active) return 0;

102
103 uint8_t *p_ptr8 = (uint8_t *)input;
104
105 for (int j = 0; j < length; j++) {
106 input_buffer[input_pos++] = p_ptr8[j];
107 if (input_pos >= input_buffer.size()) {
108 if (lc3_decode(lc3_decoder, input_buffer.data(), input_buffer.size(),
109 pcm_format, (int16_t *)output_buffer.data(), 1) != 0) {
110 LOGE("lc3_decode");

112
