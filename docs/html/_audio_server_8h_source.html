203 void processClient() {
204 //LOGD("processClient");
205 if (client_obj) { // if you get a client,
206 LOGI("New Client."); // print a message out the serial port
207 String currentLine = ""; // make a String to hold incoming data from the client
208 while (client_obj.connected()) { // loop while the client's connected
209 if (client_obj.available()) { // if there's bytes to read from the client,
210 char c = client_obj.read(); // read a byte, then
211 if (c == '\n') { // if the byte is a newline character
212
213 // if the current line is blank, you got two newline characters in a row.
214 // that's the end of the client HTTP request, so send a response:
215 if (currentLine.length() == 0){
216 sendReplyHeader();
217 sendReplyContent();
218 // break out of the while loop:
219 break;
220 } else { // if you got a newline, then clear currentLine:
221 currentLine = "";
222 }
223 }
224 else if (c != '\r') { // if you got anything else but a carriage return character,
225 currentLine += c; // add it to the end of the currentLine
226 }
227 }
228 }
