134 static void mp3dec_skip_id3v1(const uint8_t *buf, size_t *pbuf_size)
135 {
136  size_t buf_size = *pbuf_size;
137 #ifndef MINIMP3_NOSKIP_ID3V1
138  if (buf_size >= 128 && !memcmp(buf + buf_size - 128, "TAG", 3))
139  {
140  buf_size -= 128;
141  if (buf_size >= 227 && !memcmp(buf + buf_size - 227, "TAG+", 4))
142  buf_size -= 227;
143  }
144 #endif
145 #ifndef MINIMP3_NOSKIP_APEV2
146  if (buf_size > 32 && !memcmp(buf + buf_size - 32, "APETAGEX", 8))
147  {
148  buf_size -= 32;
149  const uint8_t *tag = buf + buf_size + 8 + 4;
150  uint32_t tag_size = (uint32_t)(tag[3] << 24) | (tag[2] << 16) | (tag[1] << 8) | tag[0];
151  if (buf_size >= tag_size)
152  buf_size -= tag_size;
153  }
154 #endif
155  *pbuf_size = buf_size;
156 }

190 static int mp3dec_check_vbrtag(const uint8_t *frame, int frame_size, uint32_t *frames, int *delay, int *padding)
191 {
192  static const char g_xing_tag[4] = { 'X', 'i', 'n', 'g' };
193  static const char g_info_tag[4] = { 'I', 'n', 'f', 'o' };
194 #define FRAMES_FLAG 1
195 #define BYTES_FLAG 2
196 #define TOC_FLAG 4
197 #define VBR_SCALE_FLAG 8
198  /* Side info offsets after header:
199  / Mono Stereo
200  / MPEG1 17 32
201  / MPEG2 & 2.5 9 17*/
202  bs_t bs[1];
203  L3_gr_info_t gr_info[4];
204  bs_init(bs, frame + HDR_SIZE, frame_size - HDR_SIZE);
205  if (HDR_IS_CRC(frame))
206  get_bits(bs, 16);
207  if (L3_read_side_info(bs, gr_info, frame) < 0)
208  return 0; /* side info corrupted */
209 
210  const uint8_t *tag = frame + HDR_SIZE + bs->pos/8;
211  if (memcmp(g_xing_tag, tag, 4) && memcmp(g_info_tag, tag, 4))
212  return 0;
213  int flags = tag[7];
214  if (!((flags & FRAMES_FLAG)))
215  return -1;
216  tag += 8;
217  *frames = (uint32_t)(tag[0] << 24) | (tag[1] << 16) | (tag[2] << 8) | tag[3];
218  tag += 4;
219  if (flags & BYTES_FLAG)
220  tag += 4;
221  if (flags & TOC_FLAG)
222  tag += 100;
223  if (flags & VBR_SCALE_FLAG)
224  tag += 4;
225  *delay = *padding = 0;
226  if (*tag)
227  { /* extension, LAME, Lavc, etc. Should be the same structure. */
228  tag += 21;
229  if (tag - frame + 14 >= frame_size)
230  return 0;
231  *delay = ((tag[0] << 4) | (tag[1] >> 4)) + (528 + 1);
232  *padding = (((tag[1] & 0xF) << 8) | tag[2]) - (528 + 1);
233  }
234  return 1;
235 }

242 int mp3dec_detect_cb(mp3dec_io_t *io, uint8_t *buf, size_t buf_size)
243 {
244  if (!buf || (size_t)-1 == buf_size || (io && buf_size < MINIMP3_BUF_SIZE))
245  return MP3D_E_PARAM;
246  size_t filled = buf_size;
247  if (io)
248  {
249  if (io->seek(0, io->seek_data))
250  return MP3D_E_IOERROR;
251  filled = io->read(buf, MINIMP3_ID3_DETECT_SIZE, io->read_data);
252  if (filled > MINIMP3_ID3_DETECT_SIZE)
253  return MP3D_E_IOERROR;
254  }
255  if (filled < MINIMP3_ID3_DETECT_SIZE)
256  return MP3D_E_USER; /* too small, can't be mp3/mpa */
257  if (mp3dec_skip_id3v2(buf, filled))
258  return 0; /* id3v2 tag is enough evidence */
259  if (io)
260  {
261  size_t readed = io->read(buf + MINIMP3_ID3_DETECT_SIZE, buf_size - MINIMP3_ID3_DETECT_SIZE, io->read_data);
262  if (readed > (buf_size - MINIMP3_ID3_DETECT_SIZE))
263  return MP3D_E_IOERROR;
264  filled += readed;
265  if (filled < MINIMP3_BUF_SIZE)
266  mp3dec_skip_id3v1(buf, &filled);
267  } else
268  {
269  mp3dec_skip_id3v1(buf, &filled);
270  if (filled > MINIMP3_BUF_SIZE)
271  filled = MINIMP3_BUF_SIZE;
272  }
273  int free_format_bytes, frame_size;
274  mp3d_find_frame(buf, filled, &free_format_bytes, &frame_size);
275  if (frame_size)
276  return 0; /* MAX_FRAME_SYNC_MATCHES consecutive frames found */
277  return MP3D_E_USER;
278 }

285 int mp3dec_load_cb(mp3dec_t *dec, mp3dec_io_t *io, uint8_t *buf, size_t buf_size, mp3dec_file_info_t *info, MP3D_PROGRESS_CB progress_cb, void *user_data)
286 {
287  if (!dec || !buf || !info || (size_t)-1 == buf_size || (io && buf_size < MINIMP3_BUF_SIZE))
288  return MP3D_E_PARAM;
289  uint64_t detected_samples = 0;
290  size_t orig_buf_size = buf_size;
291  int to_skip = 0;
292  mp3dec_frame_info_t frame_info;
293  memset(info, 0, sizeof(*info));
294  memset(&frame_info, 0, sizeof(frame_info));
295 
296  /* skip id3 */
297  size_t filled = 0, consumed = 0;
298  int eof = 0, ret = 0;
299  if (io)
300  {
301  if (io->seek(0, io->seek_data))
302  return MP3D_E_IOERROR;
303  filled = io->read(buf, MINIMP3_ID3_DETECT_SIZE, io->read_data);
304  if (filled > MINIMP3_ID3_DETECT_SIZE)
305  return MP3D_E_IOERROR;
306  if (MINIMP3_ID3_DETECT_SIZE != filled)
307  return 0;
308  size_t id3v2size = mp3dec_skip_id3v2(buf, filled);
309  if (id3v2size)
310  {
311  if (io->seek(id3v2size, io->seek_data))
312  return MP3D_E_IOERROR;
313  filled = io->read(buf, buf_size, io->read_data);
314  if (filled > buf_size)
315  return MP3D_E_IOERROR;
316  } else
317  {
318  size_t readed = io->read(buf + MINIMP3_ID3_DETECT_SIZE, buf_size - MINIMP3_ID3_DETECT_SIZE, io->read_data);
319  if (readed > (buf_size - MINIMP3_ID3_DETECT_SIZE))
320  return MP3D_E_IOERROR;
321  filled += readed;
322  }
323  if (filled < MINIMP3_BUF_SIZE)
324  mp3dec_skip_id3v1(buf, &filled);
325  } else
326  {
327  mp3dec_skip_id3((const uint8_t **)&buf, &buf_size);
328  if (!buf_size)
329  return 0;
330  }
331  /* try to make allocation size assumption by first frame or vbr tag */
332  mp3dec_init(dec);
333  int samples;
334  do
335  {
336  uint32_t frames;
337  int i, delay, padding, free_format_bytes = 0, frame_size = 0;
338  const uint8_t *hdr;
339  if (io)
340  {
341  if (!eof && filled - consumed < MINIMP3_BUF_SIZE)
342  { /* keep minimum 10 consecutive mp3 frames (~16KB) worst case */
343  memmove(buf, buf + consumed, filled - consumed);
344  filled -= consumed;
345  consumed = 0;
346  size_t readed = io->read(buf + filled, buf_size - filled, io->read_data);
347  if (readed > (buf_size - filled))
348  return MP3D_E_IOERROR;
349  if (readed != (buf_size - filled))
350  eof = 1;
351  filled += readed;
352  if (eof)
353  mp3dec_skip_id3v1(buf, &filled);
354  }
355  i = mp3d_find_frame(buf + consumed, filled - consumed, &free_format_bytes, &frame_size);
356  consumed += i;
357  hdr = buf + consumed;
358  } else
359  {
360  i = mp3d_find_frame(buf, buf_size, &free_format_bytes, &frame_size);
361  buf += i;
362  buf_size -= i;
363  hdr = buf;
364  }
365  if (i && !frame_size)
366  continue;
367  if (!frame_size)
368  return 0;
369  frame_info.channels = HDR_IS_MONO(hdr) ? 1 : 2;
370  frame_info.hz = hdr_sample_rate_hz(hdr);
371  frame_info.layer = 4 - HDR_GET_LAYER(hdr);
372  frame_info.bitrate_kbps = hdr_bitrate_kbps(hdr);
373  frame_info.frame_bytes = frame_size;
374  samples = hdr_frame_samples(hdr)*frame_info.channels;
375  if (3 != frame_info.layer)
376  break;
377  int ret = mp3dec_check_vbrtag(hdr, frame_size, &frames, &delay, &padding);
378  if (ret > 0)
379  {
380  padding *= frame_info.channels;
381  to_skip = delay*frame_info.channels;
382  detected_samples = samples*(uint64_t)frames;
383  if (detected_samples >= (uint64_t)to_skip)
384  detected_samples -= to_skip;
385  if (padding > 0 && detected_samples >= (uint64_t)padding)
386  detected_samples -= padding;
387  if (!detected_samples)
388  return 0;
389  }
390  if (ret)
391  {
392  if (io)
393  {
394  consumed += frame_size;
395  } else
396  {
397  buf += frame_size;
398  buf_size -= frame_size;
399  }
400  }
401  break;
402  } while(1);
403  size_t allocated = MINIMP3_MAX_SAMPLES_PER_FRAME*sizeof(mp3d_sample_t);
404  if (detected_samples)
405  allocated += detected_samples*sizeof(mp3d_sample_t);
406  else
407  allocated += (buf_size/frame_info.frame_bytes)*samples*sizeof(mp3d_sample_t);
408  info->buffer = (mp3d_sample_t*)malloc(allocated);
409  if (!info->buffer)
410  return MP3D_E_MEMORY;
411  /* save info */
412  info->channels = frame_info.channels;
413  info->hz = frame_info.hz;
414  info->layer = frame_info.layer;
415  /* decode all frames */
416  size_t avg_bitrate_kbps = 0, frames = 0;
417  do
418  {
419  if ((allocated - info->samples*sizeof(mp3d_sample_t)) < MINIMP3_MAX_SAMPLES_PER_FRAME*sizeof(mp3d_sample_t))
420  {
421  allocated *= 2;
422  mp3d_sample_t *alloc_buf = (mp3d_sample_t*)realloc(info->buffer, allocated);
423  if (!alloc_buf)
424  return MP3D_E_MEMORY;
425  info->buffer = alloc_buf;
426  }
427  if (io)
428  {
429  if (!eof && filled - consumed < MINIMP3_BUF_SIZE)
430  { /* keep minimum 10 consecutive mp3 frames (~16KB) worst case */
431  memmove(buf, buf + consumed, filled - consumed);
432  filled -= consumed;
433  consumed = 0;
434  size_t readed = io->read(buf + filled, buf_size - filled, io->read_data);
435  if (readed != (buf_size - filled))
436  eof = 1;
437  filled += readed;
438  if (eof)
439  mp3dec_skip_id3v1(buf, &filled);
440  }
441  samples = mp3dec_decode_frame(dec, buf + consumed, filled - consumed, info->buffer + info->samples, &frame_info);
442  consumed += frame_info.frame_bytes;
443  } else
444  {
445  samples = mp3dec_decode_frame(dec, buf, MINIMP3_MIN(buf_size, (size_t)INT_MAX), info->buffer + info->samples, &frame_info);
446  buf += frame_info.frame_bytes;
447  buf_size -= frame_info.frame_bytes;
448  }
449  if (samples)
450  {
451  if (info->hz != frame_info.hz || info->layer != frame_info.layer)
452  {
453  ret = MP3D_E_DECODE;
454  break;
455  }
456  if (info->channels && info->channels != frame_info.channels)
457  {
458 #ifdef MINIMP3_ALLOW_MONO_STEREO_TRANSITION
459  info->channels = 0; /* mark file with mono-stereo transition */
460 #else
461  ret = MP3D_E_DECODE;
462  break;
463 #endif
464  }
465  samples *= frame_info.channels;
466  if (to_skip)
467  {
468  size_t skip = MINIMP3_MIN(samples, to_skip);
469  to_skip -= skip;
470  samples -= skip;
471  memmove(info->buffer, info->buffer + skip, samples*sizeof(mp3d_sample_t));
472  }
473  info->samples += samples;
474  avg_bitrate_kbps += frame_info.bitrate_kbps;
475  frames++;
476  if (progress_cb)
477  {
478  ret = progress_cb(user_data, orig_buf_size, orig_buf_size - buf_size, &frame_info);
479  if (ret)
480  break;
481  }
482  }
483  } while (frame_info.frame_bytes);
484  if (detected_samples && info->samples > detected_samples)
485  info->samples = detected_samples; /* cut padding */
486  /* reallocate to normal buffer size */
487  if (allocated != info->samples*sizeof(mp3d_sample_t))
488  {
489  mp3d_sample_t *alloc_buf = (mp3d_sample_t*)realloc(info->buffer, info->samples*sizeof(mp3d_sample_t));
490  if (!alloc_buf && info->samples)
491  return MP3D_E_MEMORY;
492  info->buffer = alloc_buf;
493  }
494  if (frames)
495  info->avg_bitrate_kbps = avg_bitrate_kbps/frames;
496  return ret;
497 }

499 int mp3dec_iterate_buf(const uint8_t *buf, size_t buf_size, MP3D_ITERATE_CB callback, void *user_data)
500 {
501  const uint8_t *orig_buf = buf;
502  if (!buf || (size_t)-1 == buf_size || !callback)
503  return MP3D_E_PARAM;
504  /* skip id3 */
505  mp3dec_skip_id3(&buf, &buf_size);
506  if (!buf_size)
507  return 0;
508  mp3dec_frame_info_t frame_info;
509  memset(&frame_info, 0, sizeof(frame_info));
510  do
511  {
512  int free_format_bytes = 0, frame_size = 0, ret;
513  int i = mp3d_find_frame(buf, buf_size, &free_format_bytes, &frame_size);
514  buf += i;

516  if (i && !frame_size)
517  continue;
518  if (!frame_size)
519  break;

524  frame_info.bitrate_kbps = hdr_bitrate_kbps(hdr);
525  frame_info.frame_bytes = frame_size;
526 
527  if (callback)
528  {
529  if ((ret = callback(user_data, hdr, frame_size, free_format_bytes, buf_size, hdr - orig_buf, &frame_info)))
530  return ret;
531  }
532  buf += frame_size;
533  buf_size -= frame_size;
534  } while (1);

538 int mp3dec_iterate_cb(mp3dec_io_t *io, uint8_t *buf, size_t buf_size, MP3D_ITERATE_CB callback, void *user_data)
539 {
540  if (!io || !buf || (size_t)-1 == buf_size || buf_size < MINIMP3_BUF_SIZE || !callback)
541  return MP3D_E_PARAM;
542  size_t filled = io->read(buf, MINIMP3_ID3_DETECT_SIZE, io->read_data), consumed = 0;
543  uint64_t readed = 0;
544  mp3dec_frame_info_t frame_info;
545  int eof = 0;
546  memset(&frame_info, 0, sizeof(frame_info));
547  if (filled > MINIMP3_ID3_DETECT_SIZE)
548  return MP3D_E_IOERROR;
549  if (MINIMP3_ID3_DETECT_SIZE != filled)
550  return 0;
551  size_t id3v2size = mp3dec_skip_id3v2(buf, filled);
552  if (id3v2size)
553  {
554  if (io->seek(id3v2size, io->seek_data))
555  return MP3D_E_IOERROR;
556  filled = io->read(buf, buf_size, io->read_data);
557  if (filled > buf_size)
558  return MP3D_E_IOERROR;
559  readed += id3v2size;
560  } else
561  {
562  size_t readed = io->read(buf + MINIMP3_ID3_DETECT_SIZE, buf_size - MINIMP3_ID3_DETECT_SIZE, io->read_data);
563  if (readed > (buf_size - MINIMP3_ID3_DETECT_SIZE))
564  return MP3D_E_IOERROR;
565  filled += readed;
566  }
567  if (filled < MINIMP3_BUF_SIZE)
568  mp3dec_skip_id3v1(buf, &filled);
569  do
570  {
571  int free_format_bytes = 0, frame_size = 0, ret;
572  int i = mp3d_find_frame(buf + consumed, filled - consumed, &free_format_bytes, &frame_size);
573  if (i && !frame_size)
574  {
575  consumed += i;
576  continue;
577  }
578  if (!frame_size)
579  break;
580  const uint8_t *hdr = buf + consumed + i;
581  frame_info.channels = HDR_IS_MONO(hdr) ? 1 : 2;
582  frame_info.hz = hdr_sample_rate_hz(hdr);
583  frame_info.layer = 4 - HDR_GET_LAYER(hdr);
584  frame_info.bitrate_kbps = hdr_bitrate_kbps(hdr);
585  frame_info.frame_bytes = frame_size;
586 
587  readed += i;
588  if (callback)
589  {
590  if ((ret = callback(user_data, hdr, frame_size, free_format_bytes, filled - consumed, readed, &frame_info)))
591  return ret;
592  }
593  readed += frame_size;
594  consumed += i + frame_size;
595  if (!eof && filled - consumed < MINIMP3_BUF_SIZE)
596  { /* keep minimum 10 consecutive mp3 frames (~16KB) worst case */
597  memmove(buf, buf + consumed, filled - consumed);
598  filled -= consumed;
599  consumed = 0;
600  size_t readed = io->read(buf + filled, buf_size - filled, io->read_data);
601  if (readed > (buf_size - filled))
602  return MP3D_E_IOERROR;
603  if (readed != (buf_size - filled))
604  eof = 1;
605  filled += readed;
606  if (eof)
607  mp3dec_skip_id3v1(buf, &filled);
608  }
609  } while (1);
610  return 0;
611 }

613 static int mp3dec_load_index(void *user_data, const uint8_t *frame, int frame_size, int free_format_bytes, size_t buf_size, uint64_t offset, mp3dec_frame_info_t *info)
614 {
615  mp3dec_frame_t *idx_frame;
616  mp3dec_ex_t *dec = (mp3dec_ex_t *)user_data;
617  if (!dec->index.frames && !dec->start_offset)
618  { /* detect VBR tag and try to avoid full scan */
619  uint32_t frames;
620  int delay, padding;
621  dec->info = *info;
622  dec->start_offset = dec->offset = offset;
623  dec->end_offset = offset + buf_size;
624  dec->free_format_bytes = free_format_bytes; /* should not change */
625  if (3 == dec->info.layer)
626  {
627  int ret = mp3dec_check_vbrtag(frame, frame_size, &frames, &delay, &padding);
628  if (ret)
629  dec->start_offset = dec->offset = offset + frame_size;
630  if (ret > 0)
631  {
632  padding *= info->channels;
633  dec->start_delay = dec->to_skip = delay*info->channels;
634  dec->samples = hdr_frame_samples(frame)*info->channels*(uint64_t)frames;
635  if (dec->samples >= (uint64_t)dec->start_delay)
636  dec->samples -= dec->start_delay;
637  if (padding > 0 && dec->samples >= (uint64_t)padding)
638  dec->samples -= padding;
639  dec->detected_samples = dec->samples;
640  dec->vbr_tag_found = 1;
641  return MP3D_E_USER;
642  } else if (ret < 0)
643  return 0;
644  }
645  }
646  if (dec->flags & MP3D_DO_NOT_SCAN)
647  return MP3D_E_USER;
648  if (dec->index.num_frames + 1 > dec->index.capacity)
649  {
650  if (!dec->index.capacity)
651  dec->index.capacity = 4096;
652  else
653  dec->index.capacity *= 2;
654  mp3dec_frame_t *alloc_buf = (mp3dec_frame_t *)realloc((void*)dec->index.frames, sizeof(mp3dec_frame_t)*dec->index.capacity);
655  if (!alloc_buf)
656  return MP3D_E_MEMORY;
657  dec->index.frames = alloc_buf;
658  }
659  idx_frame = &dec->index.frames[dec->index.num_frames++];
660  idx_frame->offset = offset;
661  idx_frame->sample = dec->samples;
662  if (!dec->buffer_samples && dec->index.num_frames < 256)
663  { /* for some cutted mp3 frames, bit-reservoir not filled and decoding can't be started from first frames */
664  /* try to decode up to 255 first frames till samples starts to decode */
665  dec->buffer_samples = mp3dec_decode_frame(&dec->mp3d, frame, MINIMP3_MIN(buf_size, (size_t)INT_MAX), dec->buffer, info);
666  dec->samples += dec->buffer_samples*info->channels;
667  } else
668  dec->samples += hdr_frame_samples(frame)*info->channels;
669  return 0;
670 }

715 int mp3dec_ex_seek(mp3dec_ex_t *dec, uint64_t position)
716 {
717  size_t i;
718  if (!dec)
719  return MP3D_E_PARAM;
720  if (!(dec->flags & MP3D_SEEK_TO_SAMPLE))
721  {
722  if (dec->io)
723  {
724  dec->offset = position;
725  } else
726  {
727  dec->offset = MINIMP3_MIN(position, dec->file.size);
728  }
729  dec->cur_sample = 0;
730  goto do_exit;
731  }
732  dec->cur_sample = position;
733  position += dec->start_delay;
734  if (0 == position)
735  { /* optimize seek to zero, no index needed */
736 seek_zero:
737  dec->offset = dec->start_offset;
738  dec->to_skip = 0;
739  goto do_exit;
740  }
741  if (!dec->indexes_built)
742  { /* no index created yet (vbr tag used to calculate track length or MP3D_DO_NOT_SCAN open flag used) */
743  dec->indexes_built = 1;
744  dec->samples = 0;
745  dec->buffer_samples = 0;
746  if (dec->io)
747  {
748  if (dec->io->seek(dec->start_offset, dec->io->seek_data))
749  return MP3D_E_IOERROR;
750  int ret = mp3dec_iterate_cb(dec->io, (uint8_t *)dec->file.buffer, dec->file.size, mp3dec_load_index, dec);
751  if (ret && MP3D_E_USER != ret)
752  return ret;
753  } else
754  {
755  int ret = mp3dec_iterate_buf(dec->file.buffer + dec->start_offset, dec->file.size - dec->start_offset, mp3dec_load_index, dec);
756  if (ret && MP3D_E_USER != ret)
757  return ret;
758  }
759  for (i = 0; i < dec->index.num_frames; i++)
760  dec->index.frames[i].offset += dec->start_offset;
761  dec->samples = dec->detected_samples;
762  }
763  if (!dec->index.frames)
764  goto seek_zero; /* no frames in file - seek to zero */
765 #ifdef MINIMP3_SEEK_IDX_LINEAR_SEARCH
766  for (i = 0; i < dec->index.num_frames; i++)
767  {
768  if (dec->index.frames[i].sample >= position)
769  break;
770  }
771 #else
772  i = mp3dec_idx_binary_search(&dec->index, position);
773 #endif
774  if (i)
775  {
776  int to_fill_bytes = 511;
777  int skip_frames = MINIMP3_PREDECODE_FRAMES
778 #ifdef MINIMP3_SEEK_IDX_LINEAR_SEARCH
779  + ((dec->index.frames[i].sample == position) ? 0 : 1)
780 #endif
781  ;
782  i -= MINIMP3_MIN(i, (size_t)skip_frames);
783  if (3 == dec->info.layer)
784  {
785  while (i && to_fill_bytes)
786  { /* make sure bit-reservoir is filled when we start decoding */
787  bs_t bs[1];
788  L3_gr_info_t gr_info[4];
789  int frame_bytes, frame_size;
790  const uint8_t *hdr;
791  if (dec->io)
792  {
793  hdr = dec->file.buffer;
794  if (dec->io->seek(dec->index.frames[i - 1].offset, dec->io->seek_data))
795  return MP3D_E_IOERROR;
796  size_t readed = dec->io->read((uint8_t *)hdr, HDR_SIZE, dec->io->read_data);
797  if (readed != HDR_SIZE)
798  return MP3D_E_IOERROR;
799  frame_size = hdr_frame_bytes(hdr, dec->free_format_bytes) + hdr_padding(hdr);
800  readed = dec->io->read((uint8_t *)hdr + HDR_SIZE, frame_size - HDR_SIZE, dec->io->read_data);
801  if (readed != (size_t)(frame_size - HDR_SIZE))
802  return MP3D_E_IOERROR;
803  bs_init(bs, hdr + HDR_SIZE, frame_size - HDR_SIZE);
804  } else
805  {
806  hdr = dec->file.buffer + dec->index.frames[i - 1].offset;
807  frame_size = hdr_frame_bytes(hdr, dec->free_format_bytes) + hdr_padding(hdr);
808  bs_init(bs, hdr + HDR_SIZE, frame_size - HDR_SIZE);
809  }
810  if (HDR_IS_CRC(hdr))
811  get_bits(bs, 16);
812  i--;
813  if (L3_read_side_info(bs, gr_info, hdr) < 0)
814  break; /* frame not decodable, we can start from here */
815  frame_bytes = (bs->limit - bs->pos)/8;
816  to_fill_bytes -= MINIMP3_MIN(to_fill_bytes, frame_bytes);
817  }
818  }
819  }
820  dec->offset = dec->index.frames[i].offset;
821  dec->to_skip = position - dec->index.frames[i].sample;
822  while ((i + 1) < dec->index.num_frames && !dec->index.frames[i].sample && !dec->index.frames[i + 1].sample)
823  { /* skip not decodable first frames */
824  const uint8_t *hdr;
825  if (dec->io)
826  {
827  hdr = dec->file.buffer;
828  if (dec->io->seek(dec->index.frames[i].offset, dec->io->seek_data))
829  return MP3D_E_IOERROR;
830  size_t readed = dec->io->read((uint8_t *)hdr, HDR_SIZE, dec->io->read_data);
831  if (readed != HDR_SIZE)
832  return MP3D_E_IOERROR;
833  } else
834  hdr = dec->file.buffer + dec->index.frames[i].offset;
835  dec->to_skip += hdr_frame_samples(hdr)*dec->info.channels;
836  i++;
837  }
838 do_exit:
839  if (dec->io)
840  {
841  if (dec->io->seek(dec->offset, dec->io->seek_data))
842  return MP3D_E_IOERROR;
843  }
844  dec->buffer_samples = 0;
845  dec->buffer_consumed = 0;
846  dec->input_consumed = 0;
847  dec->input_filled = 0;
848  dec->last_error = 0;
849  mp3dec_init(&dec->mp3d);
850  return 0;
851 }

853 size_t mp3dec_ex_read_frame(mp3dec_ex_t *dec, mp3d_sample_t **buf, mp3dec_frame_info_t *frame_info, size_t max_samples)
854 {
855  if (!dec || !buf || !frame_info)
856  {
857  if (dec)
858  dec->last_error = MP3D_E_PARAM;
859  return 0;
860  }
861  if (dec->detected_samples && dec->cur_sample >= dec->detected_samples)
862  return 0; /* at end of stream */
863  if (dec->last_error)
864  return 0; /* error eof state, seek can reset it */
865  *buf = NULL;
866  uint64_t end_offset = dec->end_offset ? dec->end_offset : dec->file.size;
867  int eof = 0;
868  while (dec->buffer_consumed == dec->buffer_samples)
869  {
870  const uint8_t *dec_buf;
871  if (dec->io)
872  {
873  if (!eof && (dec->input_filled - dec->input_consumed) < MINIMP3_BUF_SIZE)
874  { /* keep minimum 10 consecutive mp3 frames (~16KB) worst case */
875  memmove((uint8_t*)dec->file.buffer, (uint8_t*)dec->file.buffer + dec->input_consumed, dec->input_filled - dec->input_consumed);
876  dec->input_filled -= dec->input_consumed;
877  dec->input_consumed = 0;
878  size_t readed = dec->io->read((uint8_t*)dec->file.buffer + dec->input_filled, dec->file.size - dec->input_filled, dec->io->read_data);
879  if (readed > (dec->file.size - dec->input_filled))
880  {
881  dec->last_error = MP3D_E_IOERROR;
882  readed = 0;
883  }
884  if (readed != (dec->file.size - dec->input_filled))
885  eof = 1;
886  dec->input_filled += readed;
887  if (eof)
888  mp3dec_skip_id3v1((uint8_t*)dec->file.buffer, &dec->input_filled);
889  }
890  dec_buf = dec->file.buffer + dec->input_consumed;
891  if (!(dec->input_filled - dec->input_consumed))
892  return 0;
893  dec->buffer_samples = mp3dec_decode_frame(&dec->mp3d, dec_buf, dec->input_filled - dec->input_consumed, dec->buffer, frame_info);
894  dec->input_consumed += frame_info->frame_bytes;
895  } else
896  {
897  dec_buf = dec->file.buffer + dec->offset;
898  uint64_t buf_size = end_offset - dec->offset;
899  if (!buf_size)
900  return 0;
901  dec->buffer_samples = mp3dec_decode_frame(&dec->mp3d, dec_buf, MINIMP3_MIN(buf_size, (uint64_t)INT_MAX), dec->buffer, frame_info);
902  }
903  dec->buffer_consumed = 0;
904  if (dec->info.hz != frame_info->hz || dec->info.layer != frame_info->layer)
905  {
906 return_e_decode:
907  dec->last_error = MP3D_E_DECODE;
908  return 0;
909  }
910  if (dec->buffer_samples)
911  {
912  dec->buffer_samples *= frame_info->channels;
913  if (dec->to_skip)
914  {
915  size_t skip = MINIMP3_MIN(dec->buffer_samples, dec->to_skip);
916  dec->buffer_consumed += skip;
917  dec->to_skip -= skip;
918  }
919  if (
920 #ifdef MINIMP3_ALLOW_MONO_STEREO_TRANSITION
921  !(dec->flags & MP3D_ALLOW_MONO_STEREO_TRANSITION) &&
922 #endif
923  dec->buffer_consumed != dec->buffer_samples && dec->info.channels != frame_info->channels)
924  {
925  goto return_e_decode;
926  }
927  } else if (dec->to_skip)
928  { /* In mp3 decoding not always can start decode from any frame because of bit reservoir,
929  count skip samples for such frames */
930  int frame_samples = hdr_frame_samples(dec_buf)*frame_info->channels;
931  dec->to_skip -= MINIMP3_MIN(frame_samples, dec->to_skip);
932  }
933  dec->offset += frame_info->frame_bytes;
934  }
935  size_t out_samples = MINIMP3_MIN((size_t)(dec->buffer_samples - dec->buffer_consumed), max_samples);
936  if (dec->detected_samples)
937  { /* count decoded samples to properly cut padding */
938  if (dec->cur_sample + out_samples >= dec->detected_samples)
939  out_samples = dec->detected_samples - dec->cur_sample;
940  }
941  dec->cur_sample += out_samples;
942  *buf = dec->buffer + dec->buffer_consumed;
943  dec->buffer_consumed += out_samples;
944  return out_samples;
945 }
