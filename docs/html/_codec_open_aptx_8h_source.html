55  int ret = 0;
56  if (is_first_write) {

63  processed =
64  aptx_decode_sync(ctx, (const uint8_t*)input_buffer, length, output_buffer,
65  sizeof(output_buffer), &written, &synced, &dropped);
66 
67  /* Check all possible states of synced, syncing and dropped status */
68  if (!synced) {
69  if (!syncing) {
70  LOGE("aptX decoding failed, synchronizing");
71  syncing = 1;
72  ret = 1;
73  }
74  if (dropped) {
75  LOGE("aptX synchronization successful, dropped %lu byte%s",
76  (unsigned long)dropped, (dropped != 1) ? "s" : "");
77  syncing = 0;
78  ret = 1;
79  }
80  if (!syncing) {
81  LOGE("aptX decoding failed, synchronizing");
82  syncing = 1;
83  ret = 1;
84  }
85  } else {
86  if (dropped) {
87  if (!syncing) LOGE("aptX decoding failed, synchronizing");
88  LOGE("aptX synchronization successful, dropped %lu byte%s",
89  (unsigned long)dropped, (dropped != 1) ? "s" : "");
90  syncing = 0;
91  ret = 1;
92  } else if (syncing) {
93  LOGE("aptX synchronization successful" );
94  syncing = 0;
95  ret = 1;
96  }
97  }
98 
99  /* If we have not decoded all supplied samples then decoding unrecoverable
100  * failed */
101  if (processed != length) {
102  LOGE("aptX decoding failed");
103  ret = 1;
104  }
105 
106  if (written > 0) {

110  }

131  bool checkPrefix(const void *input_buffer, size_t length) {
132  bool result = true;
133  if (length >= 4 && memcmp(input_buffer, "\x4b\xbf\x4b\xbf", 4) == 0) {
134  if (is_hd) {
135  LOGE("aptX audio stream (not aptX HD)");
136  result = false;
137  }
138  } else if (length >= 6 &&
139  memcmp(input_buffer, "\x73\xbe\xff\x73\xbe\xff", 6) == 0) {
140  if (!is_hd) {
141  LOGE("aptX HD audio stream");
142  result = false;
143  }
144  } else {
145  if (length >= 4 && memcmp(input_buffer, "\x6b\xbf\x6b\xbf", 4) == 0) {
146  LOGE("standard aptX audio stream - not supported");
147  result = false;
148  } else {
149  LOGE("No aptX nor aptX HD audio stream");
150  result = false;
151  }
152  }
