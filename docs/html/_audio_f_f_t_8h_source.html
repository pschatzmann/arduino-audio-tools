286  void processSamples(const void *data, size_t samples) {
287  T *dataT = (T*) data;
288  T sample;
289  float sample_windowed;
290  for (int j=0; j<samples; j+=cfg.channels){
291  sample = dataT[j+cfg.channel_used];
292  p_driver->setValue(current_pos, windowedSample(sample));
293  writeStrideBuffer((uint8_t*)&sample, sizeof(T));
294  if (++current_pos>=cfg.length){

296  fft<T>();

303  stride_buffer.readArray((uint8_t*)&sample, sizeof(T));
