78  audio_hal_iface_samples_t toSampleRate() {
79  TRACED();
80  const static int ia[] = {8000, 11025, 16000, 22050,
81  24000, 32000, 44100, 48000};
82  const static audio_hal_iface_samples_t oa[] = {
83  AUDIO_HAL_08K_SAMPLES, AUDIO_HAL_11K_SAMPLES, AUDIO_HAL_16K_SAMPLES,
84  AUDIO_HAL_22K_SAMPLES, AUDIO_HAL_24K_SAMPLES, AUDIO_HAL_32K_SAMPLES,
85  AUDIO_HAL_44K_SAMPLES, AUDIO_HAL_48K_SAMPLES};
86  int diff = 99999;
87  int result = 0;
88  for (int j = 0; j < 8; j++) {
89  if (ia[j] == sample_rate) {
90  LOGD("-> %d",ia[j])
91  return oa[j];
92  } else {
93  int new_diff = abs(oa[j] - sample_rate);
94  if (new_diff < diff) {
95  result = j;
96  diff = new_diff;
97  }
98  }
99  }
100  LOGE("Sample Rate not supported: %d - using %d", sample_rate, ia[result]);
101  return oa[result];
102  }

242  virtual size_t readBytes(uint8_t *data, size_t length) override {
243  if (cfg.channels == 2) {
244  return kit.read(data, length);
245  } else if (cfg.channels==1) {
246  // convert 2 channels of int16_t to 1
247  int16_t temp[length];
248  int len_res = kit.read(temp, length*2);
249  int res_count = len_res / 2;
250  int16_t *out = (int16_t*) data;
251  for (int j=0; j<res_count; j+=2){
252  int32_t total = temp[j];
253  total+=temp[j+1];
254  total = total/2;
255  *out = total;
256  out++;
257  }
258  return res_count;
259  }
260  LOGE("Unsuported number of channels : %d", cfg.channels);
261  return 0;

266  virtual void setAudioInfo(AudioBaseInfo info) {
267  TRACEI();
268 
269  if (cfg.sample_rate != info.sample_rate
270  && cfg.bits_per_sample == info.bits_per_sample

284  cfg.bits_per_sample = info.bits_per_sample;
285  cfg.channels = info.channels;
286  cfg.logInfo();
287 
288  // Stop first

290  kit.end();
291  }

590  AudioActions::ActiveLogic getActionLogic(int pin){
591  input_key_service_info_t input_key_info[] = INPUT_KEY_DEFAULT_INFO();
592  int size = sizeof(input_key_info) / sizeof(input_key_info[0]);
593  for (int j=0; j<size; j++){
594  if (pin == input_key_info[j].act_id){
595  switch(input_key_info[j].type){
596  case PERIPH_ID_ADC_BTN:
597  LOGD("getActionLogic for pin %d -> %d", pin, AudioActions::ActiveHigh);
598  return AudioActions::ActiveHigh;
599  case PERIPH_ID_BUTTON:
600  LOGD("getActionLogic for pin %d -> %d", pin, AudioActions::ActiveLow);
601  return AudioActions::ActiveLow;
602  case PERIPH_ID_TOUCH:
603  LOGD("getActionLogic for pin %d -> %d", pin, AudioActions::ActiveTouch);
604  return AudioActions::ActiveTouch;
605  }
606  }
607  }
608  LOGW("Undefined ActionLogic for pin: %d ",pin);
609  return AudioActions::ActiveLow;

613  void setupActions() {

623  LOGD("actionStartStop")
624  addAction(kit.pinInputMode(), actionStartStop);
625  } else {
626  LOGW("Mode Button ignored because of conflict: %d ",kit.pinInputMode());

628

632  actions.add(kit.pinHeadphoneDetect(), actionHeadphoneDetection, AudioActions::ActiveChange);
633  } else {
634  LOGW("Headphone detection ignored because of conflict: %d ",kit.pinHeadphoneDetect());
