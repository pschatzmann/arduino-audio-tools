165 virtual int process(MethodID action, Url &url, const char* mime, const char *data, int len=-1){
166 if (client_ptr==nullptr){
167 LOGE("The client has not been defined");
168 return -1;
169 }
170 if (!this->connected()){
171 LOGI("process connecting to host %s port %d", url.host(), url.port());
172 bool is_connected = connect(url.host(), url.port());
173 if (!is_connected){
174 LOGE("Connect failed");
175 return -1;
176 }
177 } else {
178 LOGI("process is already connected");
179 }
180
181#ifdef ESP32
182 LOGI("Free heap: %u", ESP.getFreeHeap());
183#endif
184
185 host_name = url.host();
186 request_header.setValues(action, url.path());
187 if (len>0 && data!=nullptr){
188 len = strlen(data);
189 request_header.put(CONTENT_LENGTH, len);
190 }
191 request_header.put(HOST_C, host_name);
192 request_header.put(CONNECTION, connection);
193 request_header.put(USER_AGENT, agent);
194 request_header.put(ACCEPT_ENCODING, accept_encoding);
195 request_header.put(ACCEPT, accept);
196 request_header.put(CONTENT_TYPE, mime);
197
198 request_header.write(*client_ptr);
199
200 if (len>0){
201 LOGI("Writing data: %d bytes", len);
202 client_ptr->write((const uint8_t*)data,len);
203 LOGD("%s",data);
204 }
205
206 LOGI("Request written ... waiting for reply")
207 client_ptr->flush();
208 reply_header.read(*client_ptr);
209
210 // if we use chunked tranfer we need to read the first chunked length
211 if (reply_header.isChunked()){
212 chunk_reader.open(*client_ptr);
213 };
214
215 // wait for data
216 return reply_header.statusCode();
217 }
