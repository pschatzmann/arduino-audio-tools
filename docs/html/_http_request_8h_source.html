187  is_ready = false;
188  if (client_ptr==nullptr){

194  int is_connected = connect(url.host(), url.port(), clientTimeout);
195  if (is_connected!=1){
196  LOGE("Connect failed");
197  return -1;
198  }
199  } else {
200  LOGI("process is already connected");
201  }
202 
203 #ifdef ESP32

205 #endif
206 
207  host_name = url.host();
208  request_header.setValues(action, url.path());
209  if (len>0 && data!=nullptr){
210  len = strlen(data);
211  request_header.put(CONTENT_LENGTH, len);
212  }
213  request_header.put(HOST_C, host_name);
214  request_header.put(CONNECTION, connection);
215  request_header.put(USER_AGENT, agent);
216  request_header.put(ACCEPT_ENCODING, accept_encoding);
217  request_header.put(ACCEPT, accept);
218  request_header.put(CONTENT_TYPE, mime);
219  request_header.write(*client_ptr);
220 
221  if (len>0){
222  LOGI("Writing data: %d bytes", len);
223  client_ptr->write((const uint8_t*)data,len);
224  LOGD("%s",data);
225  }

228  client_ptr->flush();

231  // if we use chunked tranfer we need to read the first chunked length

234  };
