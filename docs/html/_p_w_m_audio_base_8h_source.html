306 virtual int nextValue() {
307 int result = 0;
308 switch(audio_config.bits_per_sample ){
309 case 8: {
310 int16_t value = buffer->read();
311 if (value<0){
312 LOGE("Could not read full data");
313 value = 0;
314 }
315 result = map(value, -NumberConverter::maxValue(8), NumberConverter::maxValue(8), 0, maxOutputValue());
316 break;
317 }
318 case 16: {
319 int16_t value;
320 if (buffer->readArray((uint8_t*)&value,2)!=2){
321 LOGE("Could not read full data");
322 }
323 result = map(value, -NumberConverter::maxValue(16), NumberConverter::maxValue(16), 0, maxOutputValue());
324 break;
325 }
326 case 24: {
327 int24_t value;
328 if (buffer->readArray((uint8_t*)&value,3)!=3){
329 LOGE("Could not read full data");
330 }
331 result = map((int32_t)value, -NumberConverter::maxValue(24), NumberConverter::maxValue(24), 0, maxOutputValue());
332 break;
333 }
334 case 32: {
335 int32_t value;
336 if (buffer->readArray((uint8_t*)&value,4)!=4){
337 LOGE("Could not read full data");
338 }
339 result = map(value, -NumberConverter::maxValue(32), NumberConverter::maxValue(32), 0, maxOutputValue());
340 break;
341 }
342 }
