323  virtual int nextValue() {
324  int result = 0;
325  switch(audio_config.bits_per_sample ){
326  case 8: {
327  int16_t value = buffer->read();
328  if (value<0){
329  LOGE("Could not read full data");
330  value = 0;
331  }
332  result = map(value, -NumberConverter::maxValue(8), NumberConverter::maxValue(8), 0, maxOutputValue());
333  break;
334  }
335  case 16: {
336  int16_t value;
337  if (buffer->readArray((uint8_t*)&value,2)!=2){
338  LOGE("Could not read full data");
339  }
340  result = map(value, -NumberConverter::maxValue(16), NumberConverter::maxValue(16), 0, maxOutputValue());
341  break;
342  }
343  case 24: {
344  int24_t value;

346  LOGE("Could not read full data");

349  break;
