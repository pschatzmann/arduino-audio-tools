99template <typename T>
100class IIR : public Filter<T> {
101 public:
102 template <size_t B, size_t A>
103 IIR(const T (&b)[B], const T (&_a)[A], T factor=1.0) : factor(factor), lenB(B), lenA(A - 1) {
104 x = new T[lenB]();
105 y = new T[lenA]();
106 coeff_b = new T[2 * lenB - 1];
107 coeff_a = new T[2 * lenA - 1];
108 T a0 = _a[0];
109 const T *a = &_a[1];
110 for (uint16_t i = 0; i < 2 * lenB - 1; i++) {
111 coeff_b[i] = b[(2 * lenB - 1 - i) % lenB] / a0;
112 }
113 for (uint16_t i = 0; i < 2 * lenA - 1; i++) {
114 coeff_a[i] = a[(2 * lenA - 2 - i) % lenA] / a0;
115 }
116 }
117
118 ~IIR() {
119 delete[] x;
120 delete[] y;
121 delete[] coeff_a;
122 delete[] coeff_b;
123 }
124
125 T process(T value) {
126 x[i_b] = value;
127 T b_terms = 0;
128 T *b_shift = &coeff_b[lenB - i_b - 1];
129
130 T a_terms = 0;
131 T *a_shift = &coeff_a[lenA - i_a - 1];
132
133 for (uint8_t i = 0; i < lenB; i++) {
134 b_terms += x[i] * b_shift[i];
135 }
136 for (uint8_t i = 0; i < lenA; i++) {
137 a_terms += y[i] * a_shift[i];
138 }
139
140 T filtered = b_terms - a_terms;
141 y[i_a] = filtered;
142 i_b++;
143 if (i_b == lenB) i_b = 0;
144 i_a++;
145 if (i_a == lenA) i_a = 0;
146
147#ifdef USE_TYPETRAITS
148 if (!(std::is_same<T, float>::value || std::is_same<T, double>::value)) {
149 filtered = filtered / factor;
150 }
151#else
152 if (factor!=1.0) {
153 filtered = filtered / factor;
154 }
155#endif
156 return filtered;
157 }
158
159 private:
160 T factor;
161 const uint8_t lenB, lenA;
162 uint8_t i_b = 0, i_a = 0;
163 T *x;
164 T *y;
165 T *coeff_b;
166 T *coeff_a;
167};
