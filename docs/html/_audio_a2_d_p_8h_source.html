133 void begin(A2DPConfig cfg){
134 this->config = cfg;
135 LOGI("Connecting to %s",cfg.name);
136 if (a2dp_buffer==nullptr){
137 a2dp_buffer = new RingBuffer<uint8_t>(cfg.bufferSize);
138 }
139
140 switch (cfg.mode){
141 case TX_MODE:
142 LOGI("Starting a2dp_source...");
143 source(); // allocate object
144 a2dp_source->set_auto_reconnect(cfg.auto_reconnect);
145 a2dp_source->set_volume(volume * 100);
146 a2dp_source->set_on_connection_state_changed(a2dpStateCallback, this);
147 a2dp_source->start((char*)cfg.name, a2dp_stream_source_sound_data);
148 while(!a2dp_source->is_connected()){
149 LOGD("waiting for connection");
150 delay(1000);
151 }
152 LOGI("a2dp_source is connected...");
153 notifyBaseInfo(44100);
154 //is_a2dp_active = true;
155 break;
156
157 case RX_MODE:
158 LOGI("Starting a2dp_sink...");
159 sink(); // allocate object
160 a2dp_sink->set_auto_reconnect(cfg.auto_reconnect);
161 a2dp_sink->set_stream_reader(&a2dp_stream_sink_sound_data, false);
162 a2dp_sink->set_volume(volume * 100);
163 a2dp_sink->set_on_connection_state_changed(a2dpStateCallback, this);

194 virtual size_t write(const uint8_t* data, size_t len) {
195 if (a2dp_buffer==nullptr) return 0;
196 LOGD("%s: %zu", LOG_METHOD, len);
197
198 // blocking write - we wait for space in buffer
199 bool isBufferFull = true;
200 while(isBufferFull){
201 lockSemaphore(true);
202 isBufferFull = a2dp_buffer->availableForWrite()<len;
203 lockSemaphore(false);
204
205 // we wait until the buffer is full
206 if (isBufferFull){
207 if (!is_a2dp_active){
208 is_a2dp_active = true;
209 LOGW("is_a2dp_active -> true with %d bytes", a2dp_buffer->available());
210 }
211 delay(100);
212 LOGD("Waiting for free buffer space - available: %d", a2dp_buffer->available());
213 }
214 }
215
216 // write to buffer
217 lockSemaphore(true);
218 size_t result = a2dp_buffer->writeArray(data, len);
219 lockSemaphore(false);
220 LOGD("write %d -> %d", len, result);
221 return result;
222 }

320 // callback used by A2DP to provide the a2dp_source sound data
321 static int32_t a2dp_stream_source_sound_data(Frame* data, int32_t len) {
322 if (a2dp_buffer==nullptr) return 0;
323 int32_t result_len = 0;
324
325 bool isAvailable = false;
326 bool isLocked = false;
327 if (is_a2dp_active) {
328 if (A2DPStream_self->lockSemaphore(true)) {
329 isAvailable = a2dp_buffer->available()>0;
330 LOGD("buffer: %d, free %d",a2dp_buffer->available(), a2dp_buffer->availableForWrite() )
331 isLocked = true;
332 } else {
333 LOGE("lock failed");
334 }
335 }
336
337 // at first call we start with some empty data
338 if (isAvailable){
339 // the data in the file must be in int16 with 2 channels
340 size_t result_len_bytes = a2dp_buffer->readArray((uint8_t*)data, len*sizeof(Frame));
341 A2DPStream_self->lockSemaphore(false);
342
343 // result is in number of frames
344 result_len = result_len_bytes / sizeof(Frame);
345 } else {
346 if (isLocked) A2DPStream_self->lockSemaphore(false);
347
348 // prevent underflow on first call
349 switch (A2DPStream_self->config.noData) {
350 case A2DPSilence:
351 memset(data, 0, len*sizeof(Frame));
352 break;
353 case A2DPWhoosh:
354 for (int j=0;j<len;j++){
355 data[j].channel2 = data[j].channel1 = (rand() % 50) - 25;
356 }
357 break;
358 }
359 result_len = len;
360 delay(3);
361
362 // Priority: 22 on core 0
363 // LOGI("Priority: %d on core %d", uxTaskPriorityGet(NULL), xPortGetCoreID());
364
365 }
366
367 LOGD("a2dp_stream_source_sound_data: %d -> %d (%s)", len, result_len, isAvailable?"+":"-");
368 return result_len;
369 }
