49// buffer which is used to exchange data
50RingBuffer<uint8_t> a2dp_buffer(A2DP_BUFFER_SIZE*A2DP_BUFFER_COUNT);
51// flag to indicated that we are ready to process data
52volatile bool is_a2dp_active = false;

54// callback used by A2DP to provide the a2dp_source sound data
55int32_t a2dp_stream_source_sound_data(Frame* data, int32_t len) {
56 LOGD("a2dp_stream_source_sound_data: %d", len);
57 size_t result_len = 0;
58 // at first call we start with some empty data
59 if (!is_a2dp_active){
60 LOGI("setting is_a2dp_active to true");
61 // prevent underflow on first call
62 for(int j=0;j<len;j++){
63 a2dp_buffer.write(0);
64 }
65 is_a2dp_active = true;
66 }
67 // the data in the file must be in int16 with 2 channels
68 size_t result_len_bytes = a2dp_buffer.readArray((uint8_t*)data, len*sizeof(Frame));
69
70 // result is in number of frames
71 result_len = result_len_bytes / sizeof(Frame);
72
73 // Log result
74 if (AudioLogger::instance().level()==AudioLogger::Debug){
75 LOGD("a2dp_stream_source_sound_data %d -> %d", len, result_len);
76 } else {
77 if (result_len<len){
78 LOGW("a2dp_stream_source_sound_data underflow: %d -> %d ", len, result_len);
79 }
80 }
81
82 // allow some other task
83 //yield();
84
85 return result_len;
86}

139 void begin(RxTxMode mode, const char* name){
140 this->mode = mode;

142
143 switch (mode){
144 case TX_MODE:
145 LOGI("Starting a2dp_source...");
146 source(); // allocate object

151 LOGI("a2dp_source is connected...");
152
153 //is_a2dp_active done in callback to prevent a buffer overflow

158 sink(); // allocate object

161 while(!a2dp_sink->isConnected()){

163 }
164 LOGI("a2dp_sink is connected...");

167 }
