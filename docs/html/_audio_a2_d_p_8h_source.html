202  if (a2dp_buffer==nullptr) return 0;

328  static int32_t a2dp_stream_source_sound_data(Frame* data, int32_t len) {
329  if (a2dp_buffer==nullptr) return 0;
330  int32_t result_len = 0;
331 
332  bool isAvailable = false;
333  bool isLocked = false;
334  if (is_a2dp_active) {
335  if (A2DPStream_self->lockSemaphore(true)) {
336  isAvailable = a2dp_buffer->available()>0;
337  LOGD("buffer: %d, free %d",a2dp_buffer->available(), a2dp_buffer->availableForWrite() )
338  isLocked = true;
339  } else {
340  LOGE("lock failed");
341  }
342  }
343 
344  // at first call we start with some empty data
345  if (isAvailable){
346  // the data in the file must be in int16 with 2 channels
347  size_t result_len_bytes = a2dp_buffer->readArray((uint8_t*)data, len*sizeof(Frame));
348  A2DPStream_self->lockSemaphore(false);
349 
350  // result is in number of frames
351  result_len = result_len_bytes / sizeof(Frame);
352  } else {
353  if (isLocked) A2DPStream_self->lockSemaphore(false);
354 
355  // prevent underflow on first call
356  switch (A2DPStream_self->config.noData) {
357  case A2DPSilence:
358  memset(data, 0, len*sizeof(Frame));
359  break;
360  case A2DPWhoosh:
361  for (int j=0;j<len;j++){
362  data[j].channel2 = data[j].channel1 = (rand() % 50) - 25;
363  }
364  break;
365  }
366  result_len = len;
367  delay(3);
368 
369  // Priority: 22 on core 0
370  // LOGI("Priority: %d on core %d", uxTaskPriorityGet(NULL), xPortGetCoreID());
371 
372  }
373 
374  LOGD("a2dp_stream_source_sound_data: %d -> %d (%s)", len, result_len, isAvailable?"+":"-");
