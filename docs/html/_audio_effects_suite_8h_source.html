218 bool setInterpTable() {
219 effectsuite_t *polynomial_normaliser = new effectsuite_t[order];
220 if (!polynomial_normaliser) {
221 return false;
222 }
223 std::fill(polynomial_normaliser, polynomial_normaliser + order, 1);
224 effectsuite_t *alphas = new effectsuite_t[res];
225 if (!alphas) {
226 return false;
227 }
228
229 for (int i = 0; i < res; i++) {
230 alphas[i] = (i / float(res)) - 0.5;
231 }
232
233 effectsuite_t *anchors = new effectsuite_t[order];
234
235 if ((order % 2) == 0) {
236 for (int i = 0; i < order; i++) {
237 anchors[i] = -(effectsuite_t(order) - 1) * 0.5 + effectsuite_t(i);
238 std::fill(interpTable[i], interpTable[i] + res, 1);
239 }
240 } else {
241 for (int i = 0; i < order; i++) {
242 anchors[i] = (-(effectsuite_t(order)) * 0.5) + effectsuite_t(i);
243 }
244 }
245
246 // loop for every value of alpha
247 for (int q = 0; q < res; q++) {
248 // loop for sub polynomial
249 for (int j = 0; j < order; j++) {
250 // loop for each point in subpoly
251 for (int m = 0; m < order; m++) {
252 if (m != j) {
253 if (q == 0) {
254 polynomial_normaliser[j] =
255 polynomial_normaliser[j] * (anchors[j] - anchors[m]);
256 }
257 interpTable[j][q] *= (alphas[q] - anchors[m]);
258 }
259 }
260 interpTable[j][q] /= polynomial_normaliser[j];
261 }
262 }
263
264 delete[] polynomial_normaliser;
265 delete[] alphas;
266 delete[] anchors;
267 return true;

316 effectsuite_t getSplineOut(effectsuite_t bufferIndex, int freq) {
317 if (freq < 1) {
318 freq = 1;
319 }
320 const int n0 = floor(bufferIndex);
321 const int n1 = (n0 + freq) % sampleRate;
322 const int n2 = (n0 + (2 * freq)) % sampleRate;
323 const effectsuite_t alpha = bufferIndex - n0;
324 const effectsuite_t a = waveTable[n1];
325 const effectsuite_t c = ((3 * (waveTable[n2] - waveTable[n1])) -
326 (3 * (waveTable[n1] - waveTable[n0]))) *
327 .25;
328 const effectsuite_t b = (waveTable[n2] - waveTable[n1]) - ((2 * c)) / 3;
329 const effectsuite_t d = (-c) / 3;
330 return a + (b * alpha) + (c * alpha * alpha) + (d * alpha * alpha * alpha);
331 }

666 bool setChebyICoefficients(effectsuite_t cutFreq, bool shelfType, effectsuite_t ripple) {
667 // NOTE: coefficient buffers must be cleared as are additive in the
668 // following code
669 std::fill(firCoefficients, firCoefficients + 22, 0);
670 std::fill(iirCoefficients, iirCoefficients + 22, 0);
671
672 effectsuite_t poles = (effectsuite_t)filterOrder - 1;
673 int order = (int)poles;
674
675 firCoefficients[2] = 1;
676 iirCoefficients[2] = 1;
677
678 effectsuite_t Es, Vx, Kx;
679 if (ripple != 0) {
680 Es = sqrt(pow(1 / (1 - ripple), 2) - 1);
681 Vx = (1 / poles) * log(1 / Es + sqrt(1 / (pow(Es, 2)) + 1));
682 Kx = (1 / poles) * log(1 / Es + sqrt(1 / (pow(Es, 2)) - 1));
683 Kx = cosh(Kx);
684 } else {
685 Vx = 1;
686 Kx = 1;
687 }
688
689 const effectsuite_t T = 2 * tan(.5);
690 const effectsuite_t W = 2 * pi * cutFreq;
691
692 effectsuite_t K;
693
694 if (shelfType == 0)
695 {
696 K = sin(.5 - W / 2) / sin(.5 + W / 2);
697 } else
698 {
699
700 K = -cos(.5 + W / 2) / cos(W / 2 - .5);
701 }
702
704 for (int i = 0; i < (order / 2); i++) {
706 const effectsuite_t alpha = pi / (2 * poles) + (i - 1) * (pi / poles);
707
708 effectsuite_t Rp, Ip;
709 if (ripple != 0) {
710 Rp = -cos(alpha) * sinh(Vx) / Kx;
711 Ip = sin(alpha) * cosh(Vx) / Kx;
712 } else {
713 Rp = -cos(alpha);
714 Ip = sin(alpha);
715 }
716
717 const effectsuite_t M = pow(Rp, 2) + pow(Ip, 2);
718 const effectsuite_t D = 4 - 4 * Rp * T + M * T;
719
720 const effectsuite_t X0 = (pow(T, 2)) / D;
721 const effectsuite_t X1 = (2 * pow(T, 2)) / D;
722 const effectsuite_t X2 = X0;
723
724 const effectsuite_t Y1 = (8 - (2 * M * pow(T, 2))) / D;
725 const effectsuite_t Y2 = (-4 - 4 * Rp * T - M * T) / D;
726
727 // renamed and inverted from original algorithm
728 const effectsuite_t _D1 = 1 / (1 + Y1 * K - Y2 * pow(K, 2));
729
730 const effectsuite_t _A0 = (X0 - X1 * K + X2 * pow(K, 2)) * _D1;
731 effectsuite_t _A1 = (-2 * X0 * K + X1 + X1 * pow(K, 2) - 2 * X2 * K) * _D1;
732 const effectsuite_t _A2 = (X0 * pow(K, 2) - X1 * K + X2) * _D1;
733
734 effectsuite_t _B1 = (2 * K + Y1 + Y1 * pow(K, 2) - 2 * Y2 * K) * _D1;
735 const effectsuite_t B2 = (-(pow(K, 2)) - Y1 * K + Y2) * _D1;
736
737 if (shelfType == 1) {
738 _A1 = -_A1;
739 _B1 = -_B1;
740 }
741
742 for (int j = 0; j < 22; j++) {
743 firTemp[j] = firCoefficients[j];
744 iirTemp[j] = iirCoefficients[j];
745 }
746 for (int j = 2; j < 22; j++) {
747 firCoefficients[j] =
748 _A0 * firTemp[j] + _A1 * firTemp[j - 1] + _A2 * firTemp[j - 2];
749 iirCoefficients[j] =
750 iirTemp[j] - _B1 * iirTemp[j - 1] - B2 * iirTemp[j - 2];
751 }
752 }
753
754 iirCoefficients[2] = 0;
755 for (int j = 0; j < filterOrder; j++) {
756 firCoefficients[j] = firCoefficients[j + 2];
757 iirCoefficients[j] = -iirCoefficients[j + 2];
758 }
760 effectsuite_t SA = 0;
761 effectsuite_t SB = 0;
762 if (shelfType == 0) {
763 for (int j = 0; j < filterOrder; j++) {
764 SA += firCoefficients[j];
765 SB += iirCoefficients[j];
766 }
767 } else {
768 for (int j = 0; j < order; j++) {
769 SA += firCoefficients[j] * pow(-1, j);
770 SB += iirCoefficients[j] * pow(-1, j);
771 }
772 }
773
774 const effectsuite_t gain = SA / (1 - SB);
775 for (int j = 0; j < filterOrder; j++) {
776 firCoefficients[j] /= gain;
777 }
778
779 return true;
780 }
