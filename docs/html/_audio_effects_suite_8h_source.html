227 bool setInterpTable() {
228 effectsuite_t *polynomial_normaliser = new effectsuite_t[order];
229 if (!polynomial_normaliser) {
230 return false;
231 }
232 std::fill(polynomial_normaliser, polynomial_normaliser + order, 1);
233 effectsuite_t *alphas = new effectsuite_t[res];
234 if (!alphas) {
235 return false;
236 }
237
238 for (int i = 0; i < res; i++) {
239 alphas[i] = (i / float(res)) - 0.5;
240 }
241
242 effectsuite_t *anchors = new effectsuite_t[order];
243
244 if ((order % 2) == 0) {
245 for (int i = 0; i < order; i++) {
246 anchors[i] = -(effectsuite_t(order) - 1) * 0.5 + effectsuite_t(i);
247 std::fill(interpTable[i], interpTable[i] + res, 1);
248 }
249 } else {
250 for (int i = 0; i < order; i++) {
251 anchors[i] = (-(effectsuite_t(order)) * 0.5) + effectsuite_t(i);
252 }
253 }
254
255 // loop for every value of alpha
256 for (int q = 0; q < res; q++) {
257 // loop for sub polynomial
258 for (int j = 0; j < order; j++) {
259 // loop for each point in subpoly
260 for (int m = 0; m < order; m++) {
261 if (m != j) {
262 if (q == 0) {
263 polynomial_normaliser[j] =
264 polynomial_normaliser[j] * (anchors[j] - anchors[m]);
265 }
266 interpTable[j][q] *= (alphas[q] - anchors[m]);
267 }
268 }
269 interpTable[j][q] /= polynomial_normaliser[j];
270 }
271 }
272
273 delete[] polynomial_normaliser;
274 delete[] alphas;
275 delete[] anchors;
276 return true;
277 }

326 effectsuite_t getSplineOut(effectsuite_t bufferIndex, int freq) {
327 if (freq < 1) {
328 freq = 1;
329 }
330 const int n0 = floor(bufferIndex);
331 const int n1 = (n0 + freq) % sampleRate;
332 const int n2 = (n0 + (2 * freq)) % sampleRate;
333 const effectsuite_t alpha = bufferIndex - n0;
334 const effectsuite_t a = waveTable[n1];
335 const effectsuite_t c = ((3 * (waveTable[n2] - waveTable[n1])) -
336 (3 * (waveTable[n1] - waveTable[n0]))) *
337 .25;
338 const effectsuite_t b = (waveTable[n2] - waveTable[n1]) - ((2 * c)) / 3;
339 const effectsuite_t d = (-c) / 3;
340 return a + (b * alpha) + (c * alpha * alpha) + (d * alpha * alpha * alpha);
341 }

705 bool setChebyICoefficients(effectsuite_t cutFreq, bool shelfType, effectsuite_t ripple) {
706 // NOTE: coefficient buffers must be cleared as are additive in the
707 // following code
708 std::fill(firCoefficients, firCoefficients + 22, 0);
709 std::fill(iirCoefficients, iirCoefficients + 22, 0);
710
711 effectsuite_t poles = (effectsuite_t)filterOrder - 1;
712 int order = (int)poles;
713
714 firCoefficients[2] = 1;
715 iirCoefficients[2] = 1;
716
717 effectsuite_t Es, Vx, Kx;
718 if (ripple != 0) {
719 Es = sqrt(pow(1 / (1 - ripple), 2) - 1);
720 Vx = (1 / poles) * log(1 / Es + sqrt(1 / (pow(Es, 2)) + 1));
721 Kx = (1 / poles) * log(1 / Es + sqrt(1 / (pow(Es, 2)) - 1));
722 Kx = cosh(Kx);
723 } else {
724 Vx = 1;
725 Kx = 1;
726 }
727
728 const effectsuite_t T = 2 * tan(.5);
729 const effectsuite_t W = 2 * pi * cutFreq;
730
731 effectsuite_t K;
732
733 if (shelfType == 0)
734 {
735 K = sin(.5 - W / 2) / sin(.5 + W / 2);
736 } else
737 {
738
739 K = -cos(.5 + W / 2) / cos(W / 2 - .5);
740 }
741
743 for (int i = 0; i < (order / 2); i++) {
745 const effectsuite_t alpha = pi / (2 * poles) + (i - 1) * (pi / poles);
746
747 effectsuite_t Rp, Ip;
748 if (ripple != 0) {
749 Rp = -cos(alpha) * sinh(Vx) / Kx;
750 Ip = sin(alpha) * cosh(Vx) / Kx;
751 } else {
752 Rp = -cos(alpha);
753 Ip = sin(alpha);
754 }
755
756 const effectsuite_t M = pow(Rp, 2) + pow(Ip, 2);
757 const effectsuite_t D = 4 - 4 * Rp * T + M * T;
758
759 const effectsuite_t X0 = (pow(T, 2)) / D;
760 const effectsuite_t X1 = (2 * pow(T, 2)) / D;
761 const effectsuite_t X2 = X0;
762
763 const effectsuite_t Y1 = (8 - (2 * M * pow(T, 2))) / D;
764 const effectsuite_t Y2 = (-4 - 4 * Rp * T - M * T) / D;
765
766 // renamed and inverted from original algorithm
767 const effectsuite_t _D1 = 1 / (1 + Y1 * K - Y2 * pow(K, 2));
768
769 const effectsuite_t _A0 = (X0 - X1 * K + X2 * pow(K, 2)) * _D1;
770 effectsuite_t _A1 = (-2 * X0 * K + X1 + X1 * pow(K, 2) - 2 * X2 * K) * _D1;
771 const effectsuite_t _A2 = (X0 * pow(K, 2) - X1 * K + X2) * _D1;
772
773 effectsuite_t _B1 = (2 * K + Y1 + Y1 * pow(K, 2) - 2 * Y2 * K) * _D1;
774 const effectsuite_t B2 = (-(pow(K, 2)) - Y1 * K + Y2) * _D1;
775
776 if (shelfType == 1) {
777 _A1 = -_A1;
778 _B1 = -_B1;
779 }
780
781 for (int j = 0; j < 22; j++) {
782 firTemp[j] = firCoefficients[j];
783 iirTemp[j] = iirCoefficients[j];
784 }
785 for (int j = 2; j < 22; j++) {
786 firCoefficients[j] =
787 _A0 * firTemp[j] + _A1 * firTemp[j - 1] + _A2 * firTemp[j - 2];
788 iirCoefficients[j] =
789 iirTemp[j] - _B1 * iirTemp[j - 1] - B2 * iirTemp[j - 2];
790 }
791 }
792
793 iirCoefficients[2] = 0;
794 for (int j = 0; j < filterOrder; j++) {
795 firCoefficients[j] = firCoefficients[j + 2];
796 iirCoefficients[j] = -iirCoefficients[j + 2];
797 }
799 effectsuite_t SA = 0;
800 effectsuite_t SB = 0;
801 if (shelfType == 0) {
802 for (int j = 0; j < filterOrder; j++) {
803 SA += firCoefficients[j];
804 SB += iirCoefficients[j];
805 }
806 } else {
807 for (int j = 0; j < order; j++) {
808 SA += firCoefficients[j] * pow(-1, j);
809 SB += iirCoefficients[j] * pow(-1, j);
810 }
811 }
812
813 const effectsuite_t gain = SA / (1 - SB);
814 for (int j = 0; j < filterOrder; j++) {
815 firCoefficients[j] /= gain;
816 }
817
818 return true;
819 }

938 effectsuite_t rms(effectsuite_t sample) {
939 rmsBuffer[rmsBufferIndex] = sample;
940 effectsuite_t rmsValue = 0;
941 for (int j = 0; j < rmsBufferIndex; j++) {
942 int i = ((rmsBufferIndex - j) + rmsWindowSize) % rmsWindowSize;
943 rmsValue += rmsBuffer[i] * rmsBuffer[i];
944 }
945
946 // printf("samp: %e\tsum: %e\n", sample, rmsValue);
947 rmsValue /= rmsWindowSize;
