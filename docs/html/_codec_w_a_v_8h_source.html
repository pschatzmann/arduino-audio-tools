48 void begin(){
49 LOGI("WAVHeader::begin: %u",(unsigned) len);
50 this->data_pos = 0l;
51 memset((void*)&headerInfo, 0, sizeof(WAVAudioInfo));
52 while (!eof()) {
53 uint32_t tag, tag2, length;
54 tag = read_tag();
55 if (eof())
56 break;
57 length = read_int32();
58 if (!length || length >= 0x7fff0000) {
59 headerInfo.is_streamed = true;
60 length = ~0;
61 }
62 if (tag != TAG('R', 'I', 'F', 'F') || length < 4) {
63 seek(length, SEEK_CUR);
64 continue;
65 }
66 tag2 = read_tag();
67 length -= 4;
68 if (tag2 != TAG('W', 'A', 'V', 'E')) {
69 seek(length, SEEK_CUR);
70 continue;
71 }
72 // RIFF chunk found, iterate through it
73 while (length >= 8) {
74 uint32_t subtag, sublength;
75 subtag = read_tag();
76 if (eof())
77 break;
78 sublength = read_int32();
79 length -= 8;
80 if (length < sublength)
81 break;
82 if (subtag == TAG('f', 'm', 't', ' ')) {
83 if (sublength < 16) {
84 // Insufficient data for 'fmt '
85 break;
86 }
87 headerInfo.format = read_int16();
88 headerInfo.channels = read_int16();
89 headerInfo.sample_rate = read_int32();
90 headerInfo.byte_rate = read_int32();
91 headerInfo.block_align = read_int16();
92 headerInfo.bits_per_sample = read_int16();
93 if (headerInfo.format == 0xfffe) {
94 if (sublength < 28) {
95 // Insufficient data for waveformatex
96 break;
97 }
98 skip(8);
99 headerInfo.format = read_int32();
100 skip(sublength - 28);
101 } else {
102 skip(sublength - 16);
103 }
104 headerInfo.is_valid = true;
105 } else if (subtag == TAG('d', 'a', 't', 'a')) {
106 sound_pos = tell();
107 headerInfo.data_length = sublength;
108 if (!headerInfo.data_length || headerInfo.is_streamed) {
109 headerInfo.is_streamed = true;
110 logInfo();
111 return;
112 }
113 seek(sublength, SEEK_CUR);
114 } else {
115 skip(sublength);
116 }
117 length -= sublength;
118 }
119 if (length > 0) {
120 // Bad chunk?
121 seek(length, SEEK_CUR);
122 }
123 }
124 logInfo();
125 len = 0;
126 }

129 void end(){
130 len = 0;
131 }
