637 FormatConverterStream(Print &out, AudioBaseInfo &infoOut, AudioBaseInfo &infoIn){
638 this->p_out = &out;
639 p_info_out = &infoOut;
640 p_info_in = &infoIn;
641
642 convert.source_bits_per_sample = infoIn.bits_per_sample;
643 convert.target_bits_per_sample = infoOut.bits_per_sample;
644
645 if (p_info_out->bits_per_sample == p_info_in->bits_per_sample
646 && p_info_out->channels == p_info_in->channels ){
647 no_conversion = true;
648 } else if (p_info_out->channels == p_info_in->channels){
649 // copy 1:1
650 convert.multiply = 1;
651 convert.step = 1;
652 } else if (p_info_in->channels == 1 && p_info_out->channels>1){
653 // generate multiple output channels per frame
654 convert.multiply = p_info_out->channels;
655 convert.step = 1;
656 } else if (p_info_out->channels == 1){
657 convert.multiply = 1;
658 convert.step = p_info_in->channels; // just output 1 channel per frame
659 } else {
