106 size_t bytes_read = 0;

108 if (len>0){
109 bytes_to_read = min(len, static_cast<size_t>(buffer_size));
110 size_t samples = bytes_to_read / sizeof(T);
111 bytes_to_read = samples * sizeof(T);
112 // don't overflow buffer
113 if (to_write>0){
114 bytes_to_read = min((int)bytes_to_read, to_write);

237 // blocking write - until everything is processed
238 size_t write(size_t len, size_t &delayCount ){
239 if (!buffer || len==0) return 0;
240 size_t total = 0;
241 long open = len;
242 int retry = 0;
243 while(open>0){
244 size_t written = to->write((const uint8_t*)buffer.data()+total, open);
245 total += written;
246 open -= written;
247 delayCount++;
248
249 if (retry++ > retryLimit){
250 LOGE("write to target has failed! (%ld bytes)", open);
251 break;
252 }
253
254 if (retry>1) {
255 delay(5);
256 LOGI("try write - %d (open %ld bytes) ",retry, open);
257 }
258
259 CHECK_MEMORY();
260 }
261 return total;
262 }
