90 inline size_t copy(){
91 TRACED();
92 // if not initialized we do nothing
93 if (from==nullptr || to==nullptr) return 0;
94
95 // E.g. if we try to write to a server we might not have any output destination yet
96 int to_write = to->availableForWrite();
97 if (check_available_for_write && to_write==0){
98 delay(500);
99 return 0;
100 }
101
102
103 size_t result = 0;
104 size_t delayCount = 0;
105 size_t len = available();
106 size_t bytes_to_read = buffer_size;
107 size_t bytes_read = 0;
108
109 if (len>0){
110 bytes_to_read = min(len, static_cast<size_t>(buffer_size));
111 size_t samples = bytes_to_read / sizeof(T);
112 bytes_to_read = samples * sizeof(T);
113 // don't overflow buffer
114 if (to_write>0){
115 bytes_to_read = min((int)bytes_to_read, to_write);
116 }
117
118 // get the data now
119 bytes_read = 0;
120 if (bytes_to_read>0){
121 bytes_read = from->readBytes((uint8_t*)buffer, bytes_to_read);
122 }
123
124 // determine mime
125 notifyMime(buffer, bytes_to_read);
126 is_first = false;
127
128 // write data
129 result = write(bytes_read, delayCount);
130
131 // callback with unconverted data
132 if (onWrite!=nullptr) onWrite(onWriteObj, buffer, result);
133
134 #ifndef COPY_LOG_OFF
135 LOGI("StreamCopy::copy %u -> %u -> %u bytes - in %u hops", (unsigned int)bytes_to_read,(unsigned int) bytes_read, (unsigned int)result, (unsigned int)delayCount);
136 #endif
137
138 CHECK_MEMORY();
139 } else {
140 // give the processor some time
141 delay(delay_on_no_data);
142 }
143 return result;
144 }

197 TRACED();
198 size_t result = 0;
199 int retry = 0;
200
201 if (from==nullptr || to == nullptr)
202 return result;
203
204 // copy while source has data available
205 int count=0;
206 while (true){
207 count = copy();
208 result += count;

210 // wait for more data

276 // blocking write - until everything is processed
277 size_t write(size_t len, size_t &delayCount ){
278 if (buffer==nullptr || len==0) return 0;
279 size_t total = 0;
280 long open = len;
281 int retry = 0;
282 while(open>0){
283 size_t written = to->write((const uint8_t*)buffer+total, open);
284 total += written;
285 open -= written;
286 delayCount++;
287
288 if (retry++ > retryLimit){
289 LOGE("write to target has failed! (%ld bytes)", open);
290 break;
291 }
292
293 if (retry>1) {
294 delay(5);
295 LOGI("try write - %d (open %ld bytes) ",retry, open);
296 }
297
298 CHECK_MEMORY();
299 }
300 return total;
301 }
