254 int16_t readSample() override {
255 float total = 0;
256 uint16_t count = 0;
257 // calculate sum of all channels
258 for (int j=0;j<channels.size();j++){
259 if (channels[j]->isActive()){
260 count++;
261 total += channels[j]->readSample();
262 }
263 }
264 // prevent divide by zero
265 int16_t result = 0;
266 if (count>0){
267 result = 0.9 * total / count;
268 }
269 return result;

330 AbstractSynthesizerChannel *getNoteChannel(int note){
331 LOGI("getNoteChannel: %d", note);
332 for (int j=0;j<channels.size();j++){

340 // gets a free channel
341 AbstractSynthesizerChannel *getFreeChannel(){
342 LOGI("getFreeChannel");
343 for (int j=0;j<channels.size();j++){
344 if (!channels[j]->isActive()){
345 return channels[j];
346 }
347 }
348 LOGI("No free channel found: Adding new channel");
349 AbstractSynthesizerChannel* ch = defaultChannel->clone();
350 channels.push_back(ch);
351 return ch;
352 }
