196  // calculates a single sample using the indicated state
197  float sample(EQSTATE &es, float sample) {
198  // Locals
199  float l,m,h; // Low / Mid / High - Sample Values
200  // Filter #1 (lowpass)
201  es.f1p0 += (es.lf * (sample - es.f1p0)) + vsa;
202  es.f1p1 += (es.lf * (es.f1p0 - es.f1p1));
203  es.f1p2 += (es.lf * (es.f1p1 - es.f1p2));
204  es.f1p3 += (es.lf * (es.f1p2 - es.f1p3));
205 
206  l = es.f1p3;
207 
208  // Filter #2 (highpass)
209  es.f2p0 += (es.hf * (sample - es.f2p0)) + vsa;
210  es.f2p1 += (es.hf * (es.f2p0 - es.f2p1));
211  es.f2p2 += (es.hf * (es.f2p1 - es.f2p2));
212  es.f2p3 += (es.hf * (es.f2p2 - es.f2p3));
213 
214  h = es.sdm3 - es.f2p3;
215  // Calculate midrange (signal - (low + high))
216  m = es.sdm3 - (h + l);
217  // Scale, Combine and store
218  l *= p_cfg->gain_low;
219  m *= p_cfg->gain_medium;
220  h *= p_cfg->gain_high;
221 
222  // Shuffle history buffer
223  es.sdm3 = es.sdm2;
224  es.sdm2 = es.sdm1;
225  es.sdm1 = sample;
226 
227  // Return result
228  return(l + m + h);
229  }
