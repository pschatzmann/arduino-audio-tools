458 size_t write(int idx, const uint8_t *buffer_c, size_t bytes) {
459 LOGD("write idx %d: %d",stream_idx, bytes);
460 size_t result = 0;
461 RingBuffer<T>* p_buffer = idx<output_count ? buffers[idx] : nullptr;
462 assert(p_buffer!=nullptr);
463 size_t samples = bytes/sizeof(T);
464 if (p_buffer->availableForWrite()<samples){
465 LOGW("Available Buffer too small %d: requested: %d -> increase the buffer size", p_buffer->availableForWrite(), samples);
466 } else {
467 result = p_buffer->writeArray((T*)buffer_c, samples) * sizeof(T);
468 }

483 void flushMixer() {
484 LOGD("flush");
485 bool result = false;
486
487 // determine ringbuffer with mininum available data
488 size_t samples = size_bytes/sizeof(T);
489 for (int j=0;j<output_count;j++){
490 samples = min(samples, (size_t) buffers[j]->available());
491 }
492
493 if (samples>0){
494 result = true;
495 // mix data from ringbuffers to output
496 output.resize(samples);
497 memset(output.data(),0, samples*sizeof(T));
498 for (int j=0;j<output_count;j++) {
499 float weight = weights[j];
500 // sum up input samples to result samples
501 for (int i=0;i<samples; i++){
502 output[i] += weight * buffers[j]->read() / total_weights;
503 }
504 }
505
506 // write output
507 LOGD("write to final out: %d", samples*sizeof(T));
508 p_final_output->write((uint8_t*)output.data(), samples*sizeof(T));
509 }
510 stream_idx = 0;
511 return;
512 }

535 // allocate ringbuffers for each output
536 for (int j=0;j<output_count;j++){
537 if (buffers[j]!=nullptr){
538 delete buffers[j];
539 }
540#ifdef ESP32
541 if (memory_type==PS_RAM && ESP.getFreePsram()>=size_bytes){
542 p_memory = ps_malloc(size_bytes);
543 LOGI("Buffer %d allocated %d bytes in PS_RAM",j, size_bytes);
544 } else {
545 p_memory = malloc(size_bytes);
546 LOGI("Buffer %d allocated %d bytes in RAM",j, size_bytes);
547 }
548 if (p_memory!=nullptr){
549 buffers[j] = new(p_memory) RingBuffer<T>(size_bytes/sizeof(T));
550 } else {
551 LOGE("Not enough memory to allocate %d bytes",size_bytes);
552 }
553#else
554 buffers[j] = new RingBuffer<T>(size_bytes/sizeof(T));
555#endif
556 }
557 }

560 // allocate ringbuffers for each output
561 for (int j=0;j<output_count;j++){
562 if (buffers[j]!=nullptr){
563 delete buffers[j];
564#ifdef ESP32
565 if (p_memory!=nullptr){
566 free(p_memory);
567 }
568#endif

596 void setAudioInfo(AudioBaseInfo info){
597 this->info = info;
598 if (info.channels){
599 volumes = new float[info.channels];
600 volumes_tmp = new float[info.channels];
601 }
602 }
603
604 size_t write(const uint8_t *buffer, size_t size){
605 for (int j=0;j<info.channels;j++){
606 volumes_tmp[j]=0;
607 }
608 switch(info.bits_per_sample){
609 case 16: {
610 int16_t *buffer16 = (int16_t*)buffer;
611 int samples16 = size/2;
612 for (int j=0;j<samples16;j++){
613 float tmp = static_cast<float>(abs(buffer16[j]));
614 updateVolume(tmp,j);
615 }
616 commit();
617 } break;
618 case 32: {
619 int32_t *buffer32 = (int32_t*)buffer;
620 int samples32 = size/4;
621 for (int j=0;j<samples32;j++){
622 float tmp = static_cast<float>(abs(buffer32[j]))/samples32;
623 updateVolume(tmp,j);
624 }
625 commit();
626 }break;
627
628 default:
629 LOGE("Unsupported bits_per_sample: %d", info.bits_per_sample);
630 f_volume = 0;
631 break;
632 }
633 return size;
634 }
635
637 float volume() {
638 return f_volume;
639 }
640
642 float volume(int channel) {
643 return channel<info.channels ? volumes[channel]:0.0;
644 }
645
646 protected:
647 AudioBaseInfo info;
648 float f_volume_tmp = 0;
649 float f_volume = 0;
650 float *volumes=nullptr;
651 float *volumes_tmp=nullptr;
652
653 void updateVolume(float tmp, int j) {
654 if (tmp>f_volume){
655 f_volume_tmp = tmp;
656 }
657 if (volumes_tmp!=nullptr && tmp>volumes_tmp[j%info.channels]){
658 volumes_tmp[j%info.channels] = tmp;
659 }
660 }
661
662 void commit(){
663 f_volume = f_volume_tmp;
664 for (int j=0;j<info.channels;j++){
665 volumes[j] = volumes_tmp[j];
666 }
