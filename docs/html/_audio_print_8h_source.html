473 size_t write(int idx, const uint8_t *buffer_c, size_t bytes) {
474 LOGD("write idx %d: %d",stream_idx, bytes);
475 size_t result = 0;
476 RingBuffer<T>* p_buffer = idx<output_count ? buffers[idx] : nullptr;
477 assert(p_buffer!=nullptr);
478 size_t samples = bytes/sizeof(T);
479 if (p_buffer->availableForWrite()<samples){
480 LOGW("Available Buffer too small %d: requested: %d -> increase the buffer size", p_buffer->availableForWrite(), samples);
481 } else {
482 result = p_buffer->writeArray((T*)buffer_c, samples) * sizeof(T);
483 }
484 return result;

493 if (p_buffer == nullptr) return 0;
494 return p_buffer->availableForWrite();

498 void flushMixer() {
499 LOGD("flush");
500 bool result = false;
501
502 // determine ringbuffer with mininum available data
503 size_t samples = size_bytes/sizeof(T);
504 for (int j=0;j<output_count;j++){
505 samples = min(samples, (size_t) buffers[j]->available());
506 }
507
508 if (samples>0){
509 result = true;
510 // mix data from ringbuffers to output
511 output.resize(samples);
512 memset(output.data(),0, samples*sizeof(T));
513 for (int j=0;j<output_count;j++) {
514 float weight = weights[j];
515 // sum up input samples to result samples
516 for (int i=0;i<samples; i++){
517 output[i] += weight * buffers[j]->read() / total_weights;
518 }
519 }
520
521 // write output
522 LOGD("write to final out: %d", samples*sizeof(T));
523 p_final_output->write((uint8_t*)output.data(), samples*sizeof(T));
524 }
525 stream_idx = 0;
526 return;
527 }

531 Vector<T> output{0};
532 Vector<float> weights{0};
533 Print *p_final_output=nullptr;
534 float total_weights = 0.0;
535 bool is_active = false;

547 }

549 void allocate_buffers() {
550 // allocate ringbuffers for each output
551 for (int j=0;j<output_count;j++){
552 if (buffers[j]!=nullptr){
553 delete buffers[j];
554 }
555#ifdef ESP32
556 if (memory_type==PS_RAM && ESP.getFreePsram()>=size_bytes){
557 p_memory = ps_malloc(size_bytes);
558 LOGI("Buffer %d allocated %d bytes in PS_RAM",j, size_bytes);
559 } else {
560 p_memory = malloc(size_bytes);
561 LOGI("Buffer %d allocated %d bytes in RAM",j, size_bytes);
562 }
563 if (p_memory!=nullptr){
564 buffers[j] = new(p_memory) RingBuffer<T>(size_bytes/sizeof(T));
565 } else {
566 LOGE("Not enough memory to allocate %d bytes",size_bytes);
567 }
568#else
569 buffers[j] = new RingBuffer<T>(size_bytes/sizeof(T));
570#endif
571 }
572 }
573

575 // allocate ringbuffers for each output
576 for (int j=0;j<output_count;j++){
577 if (buffers[j]!=nullptr){
578 delete buffers[j];
579#ifdef ESP32
580 if (p_memory!=nullptr){
581 free(p_memory);
582 }
583#endif
584 buffers[j] = nullptr;

614 if (info.channels){
615 volumes = new float[info.channels];
616 volumes_tmp = new float[info.channels];
617 }
618 }
619
620 size_t write(const uint8_t *buffer, size_t size){
621 for (int j=0;j<info.channels;j++){
622 volumes_tmp[j]=0;
623 }
624 switch(info.bits_per_sample){
625 case 16: {
626 int16_t *buffer16 = (int16_t*)buffer;
627 int samples16 = size/2;
628 for (int j=0;j<samples16;j++){
629 float tmp = static_cast<float>(abs(buffer16[j]));
630 updateVolume(tmp,j);
631 }
632 commit();
633 } break;
634 case 32: {
635 int32_t *buffer32 = (int32_t*)buffer;
636 int samples32 = size/4;
637 for (int j=0;j<samples32;j++){
638 float tmp = static_cast<float>(abs(buffer32[j]))/samples32;
639 updateVolume(tmp,j);
640 }
641 commit();
642 }break;
643
644 default:
645 LOGE("Unsupported bits_per_sample: %d", info.bits_per_sample);
646 f_volume = 0;
647 break;
648 }
649 return size;
650 }
651
653 float volume() {
654 return f_volume;
655 }
656
658 float volume(int channel) {
659 return channel<info.channels ? volumes[channel]:0.0;
660 }
661
662 protected:
663 AudioBaseInfo info;
664 float f_volume_tmp = 0;
665 float f_volume = 0;
666 float *volumes=nullptr;
667 float *volumes_tmp=nullptr;
668
669 void updateVolume(float tmp, int j) {
670 if (tmp>f_volume){
671 f_volume_tmp = tmp;
672 }
673 if (volumes_tmp!=nullptr && tmp>volumes_tmp[j%info.channels]){
674 volumes_tmp[j%info.channels] = tmp;
675 }
676 }
677
678 void commit(){
679 f_volume = f_volume_tmp;
680 for (int j=0;j<info.channels;j++){
681 volumes[j] = volumes_tmp[j];
682 }
