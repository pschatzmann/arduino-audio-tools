140 size_t write(Print *p_out, const uint8_t* buffer, size_t bytes, size_t &written ) {
141 T* data = (T*)buffer;
142 int samples = bytes / sizeof(T);
143 size_t frames = samples / info.channels;
144 written = 0;
145
146 // avoid noise if audio does not start with 0
147 if (is_first){
148 is_first = false;
149 setupLastSamples(data, 0);
150 }
151 // process all samples
152 while (idx<frames){
153 for (int ch=0;ch<info.channels;ch++){
154 T result = getValue(data, idx, ch);
155 written += p_out->write((uint8_t*)&result, sizeof(T));
156 }
157 idx+=step_size;
158 }
159 // save last samples
160 setupLastSamples(data, frames-1);
161 idx -= frames;
162 return bytes;
163 }

165 // interpolate value
166 inline T getValue(T *data,float frame_idx, int channel){
167 int frame_idx1 = frame_idx;
168 int frame_idx0 = frame_idx1-1;
169 T val0 = lookup(data, frame_idx0, channel);
170 T val1 = lookup(data, frame_idx1, channel);
171 T diff = val1 - val0;
172 T result;
173 if (diff!=0){
174 float delta = frame_idx - frame_idx0;
175 T diffEffective = delta / diff;
176 result = val0+diffEffective;
177 } else {
178 result = val0;
179 }
180 return result;
181 }

184 T lookup(T *data, int frame, int channel){
