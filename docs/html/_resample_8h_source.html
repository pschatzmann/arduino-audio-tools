252  size_t upsample(T *from, T* to, int sample_count, int channels, int factor ){
253  int frame_count = sample_count/channels;
254  size_t result = 0;
255  int pos = 0;
256  // we start with the last saved frame
257  for (int16_t frame_pos=-1; frame_pos<frame_count-1; frame_pos++){
258  for (int8_t ch=0; ch<channels; ch++){
259  T actual_data = *p_data(frame_pos, ch, from);
260  T next_data = *p_data(frame_pos+1, ch, from);
261  float diff = (next_data - actual_data) / factor;
262  pos =(frame_pos+1)*factor;
263  *p_data(pos, ch, to) = actual_data;
264  result++;
265  for (int16_t f=1;f<factor;f++){
266  pos = ((frame_pos+1)*factor)+f;
267  T tmp = actual_data + (diff*f);
268  *p_data(pos, ch, to) = tmp;
269  result++;
270  }
271  }
272  }
273  // save last frame_pos data
274  pos = frame_count*factor;
275  for (int8_t ch=0; ch<channels; ch++){
276  T tmp = *p_data(frame_count-1, ch, from);
277  last_end[ch] = tmp;
278  }
279  return pos*channels;
280  }
