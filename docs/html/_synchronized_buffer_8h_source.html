145template <typename T> class SynchronizedBuffer : public BaseBuffer<T> {
146public:
147 SynchronizedBuffer(BaseBuffer<T> &buffer, Mutex &mutex) {
148 p_buffer = &buffer;
149 p_mutex = &mutex;
150 }
151
152 // reads a single value
153 T read() override {
154 TRACED();
155 LockGuard guard(p_mutex);
156 return p_buffer->read();
157 }
158
159 // reads multiple values
160 int readArray(T data[], int len) {
161 TRACED();
162 LockGuard guard(p_mutex);
163 int lenResult = MIN(len, available());
164 for (int j = 0; j < lenResult; j++) {
165 data[j] = p_buffer->read();
166 }
167 return lenResult;
168 }
169
170 int writeArray(const T data[], int len) {
171 LOGD("%s: %d", LOG_METHOD, len);
172 LockGuard guard(p_mutex);
173 int result = 0;
174 for (int j = 0; j < len; j++) {
175 if (p_buffer->write(data[j]) == 0) {
176 break;
177 }
178 result = j + 1;
179 }
180 return result;
181 }
182
183 // peeks the actual entry from the buffer
184 T peek() override {
185 TRACED();
186 LockGuard guard(p_mutex);
187 return p_buffer->peek();
188 }
189
190 // checks if the buffer is full
191 bool isFull() override { return p_buffer->isFull(); }
192
193 bool isEmpty() { return available() == 0; }
194
195 // write add an entry to the buffer
196 bool write(T data) override {
197 TRACED();
198 LockGuard guard(p_mutex);
199 return p_buffer->write(data);
200 }
201
202 // clears the buffer
203 void reset() override {
204 TRACED();
205 LockGuard guard(p_mutex);
206 p_buffer->reset();
207 }
208
209 // provides the number of entries that are available to read
210 int available() override {
211 TRACED();
212 return p_buffer->available();
213 }
214
215 // provides the number of entries that are available to write
216 int availableForWrite() override {
217 TRACED();
218 LockGuard guard(p_mutex);
219 return p_buffer->availableForWrite();
220 }
221
222 // returns the address of the start of the physical read buffer
223 T *address() override {
224 TRACED();
225 return p_buffer->address();
226 }
227
228protected:
229 BaseBuffer<T> *p_buffer = nullptr;
230 Mutex *p_mutex = nullptr;
231};
