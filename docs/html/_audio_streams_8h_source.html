193 size_t result = 0;

374 bool ok;
375 do{
376 ok = audio_list.pop_front(p_node);
377 if (ok){
378 delete p_node;
379 }

398 DataNode *p_node = new DataNode((void*)buffer, size);
399 if (p_node->data!=nullptr){
400 alloc_failed = false;
401 total_available += size;
402 audio_list.push_back(p_node);
403
404 // setup interator to point to first record
405 if (it == audio_list.end()){
406 it = audio_list.begin();
407 }
408
409 return size;
410 }

412 return 0;

420 if (it == audio_list.end()){

422 if (it == audio_list.end()) {
423 return 0;
424 }
425 }

427 }

429 virtual size_t readBytes(uint8_t *buffer, size_t length) override {
430 // provide unprocessed data
431 if (temp_audio.available()>0){
432 return temp_audio.readArray(buffer, length);
433 }
434
435 // We have no more data
436 if (it==audio_list.end()){
437 if (is_loop){
438 rewind();
439 } else {
440 // stop the processing
441 return 0;
442 }
443 }
444
445 // provide data from next node
446 DataNode *p_node = *it;
447 int result_len = min(length, (size_t) p_node->len);
448 memcpy(buffer, p_node->data, result_len);
449 // save unprocessed data to temp buffer
450 if (p_node->len>length){
451 uint8_t *start = p_node->data+result_len;
452 int uprocessed_len = p_node->len - length;
453 temp_audio.writeArray(start, uprocessed_len);
454 }
455 //move to next pos
456 ++it;
457 return result_len;
458 }

468 void postProcessSmoothTransition(int channels, float factor = 0.01, int remove=0){
469 if (remove>0){
470 for (int j=0;j<remove;j++){
471 DataNode* node = nullptr;
472 audio_list.pop_front(node);
473 if (node!=nullptr) delete node;
474 node = nullptr;
475 audio_list.pop_back(node);
476 if (node!=nullptr) delete node;
477 }
478 }
479
480 // Remove popping noise
481 SmoothTransition<T> clean_start(channels, true, false, factor);
482 auto first = *list().begin();
483 if (first!=nullptr){
484 clean_start.convert(first->data,first->len);
485 }
486
487 SmoothTransition<T> clean_end(channels, false, true, factor);
488 auto last = * (--(list().end()));
489 if (last!=nullptr){
490 clean_end.convert(last->data,last->len);
491 }
492 }
