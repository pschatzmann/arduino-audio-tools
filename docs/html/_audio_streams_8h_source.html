738 // relevant only if use_timer == true
739 void TimerCallbackAudioStream::timerCallback(void *obj) {
740  TimerCallbackAudioStream *src = (TimerCallbackAudioStream *)obj;
741  if (src != nullptr) {
742  // LOGD("%s: %s", LOG_METHOD, src->cfg.rx_tx_mode==RX_MODE ?
743  // "RX_MODE":"TX_MODE");
744  if (src->cfg.rx_tx_mode == RX_MODE) {
745  // input
746  uint16_t available_bytes = src->frameCallback(src->frame, src->frameSize);
747  uint16_t buffer_available = src->buffer->availableForWrite();
748  if (buffer_available < available_bytes) {
749  // if buffer is full make space
750  uint16_t to_clear = available_bytes - buffer_available;
751  uint8_t tmp[to_clear];
752  src->buffer->readArray(tmp, to_clear);
753  }
754  if (src->buffer->writeArray(src->frame, available_bytes) !=
755  available_bytes) {
756  // LOGE(UNDERFLOW_MSG);
757  assert(false);
758  }
759  } else {
760  // output
761  if (src->buffer != nullptr && src->frame != nullptr &&
762  src->frameSize > 0) {
763  uint16_t available_bytes =
764  src->buffer->readArray(src->frame, src->frameSize);
765  if (available_bytes !=
766  src->frameCallback(src->frame, available_bytes)) {
767  LOGE(UNDERFLOW_MSG);
768  }
769  }
770  }
771  src->measureSampleRate();
772  }
