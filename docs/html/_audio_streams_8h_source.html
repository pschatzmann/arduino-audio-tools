148 virtual size_t write(uint8_t byte) override {

159 size_t result = 0;
160 for (size_t j = 0; j < size; j++) {
161 if (!write(buffer[j])) {
162 break;
163 }
164 result = j;
165 }
166 return result;
167 }
168
169 virtual int available() override { return write_pos - read_pos; }
170
171 virtual int read() override {
172 int result = peek();
173 if (result >= 0) {
174 read_pos++;
175 }
176 return result;
177 }
178
179 virtual size_t readBytes(uint8_t *buffer, size_t length) override {
180 size_t count = 0;
181 while (count < length) {
182 int c = read();
183 if (c < 0) break;
184 *buffer++ = (char)c;
185 count++;
186 }
187 return count;
188 }
189
190 virtual int peek() override {
191 int result = -1;
192 if (available() > 0) {
193 result = buffer[read_pos];
194 }
195 return result;
196 }
197

199
200 virtual void end() override {
201 read_pos = 0;
202 }
203
204 virtual void clear(bool reset = false) {
205 write_pos = 0;
206 read_pos = 0;
207 if (reset) {
208 // we clear the buffer data
209 memset(buffer, 0, buffer_size);
210 }
211 }

882template<typename T>
883class Mixer : public AudioStreamX {
884 public:
885 Mixer() = default;
886
888 void add(Stream &in, float weight=1.0){
889 streams.push_back(&in);
890 weights.push_back(weight);
891 total_weights += weight;
892 }
893
895 void setWeight(int channel, float weight){
896 if (channel<size()){
897 weights[channel] = weight;
898 float total = 0;
899 for (int j=0;j<weights.size();j++){
900 total += weights[j];
901 }
902 total_weights = total;
903 } else {
904 LOGE("Invalid channel %d - max is %d", channel, size()-1);
905 }
906 }
907
909 void end() {
910 streams.clear();
911 weights.clear();
912 total_weights = 0.0;
913 }
914
916 int size() {
917 return streams.size();
918 }
919
921 size_t readBytes(uint8_t* data, size_t len) override {
922 LOGD("readBytes: %d",len);
923 int sample_count = len / sizeof(T);
924 T* samples_result = (T*)data;
925 memset(data,0, len); // clear data so that we can add values
926 buffer.resize(len); // usually does only something the first time
927 T* buffer_in = (T*)&buffer[0];
928 for (int j=0;j<size();j++){
929 float weight = weights[j];
930 LOGD("adding stream %d with len %d with weight %f",j, len, weight);
931 memset(&buffer[0],0, len); // if no data is read we have an array of 0
932 streams[j]->readBytes(&buffer[0], len);
933 for (int i=0;i<sample_count; i++){
934 samples_result[i] += weight * buffer_in[i] / total_weights;
935 }
936 }
937 return len;
938 }
939
941 int available() override {
942 int result = 0;
943 for (int j=0;j<size();j++){
944 result = streams[j]->available();
945 if (result>0)
946 break;
947 }
948 return result;
949 }
950
951 protected:
952 Vector<Stream*> streams{10};
953 Vector<uint8_t> buffer{DEFAULT_BUFFER_SIZE};
954 Vector<float> weights{10};
955 float total_weights = 0.0;
956
957};

1204void TimerCallbackAudioStream::timerCallback(void *obj) {
1205 TimerCallbackAudioStream *src = (TimerCallbackAudioStream *)obj;
1206 if (src != nullptr) {
1207 // LOGD("%s: %s", LOG_METHOD, src->cfg.rx_tx_mode==RX_MODE ?
1208 // "RX_MODE":"TX_MODE");
1209 if (src->cfg.rx_tx_mode == RX_MODE) {
1210 // input
1211 uint16_t available_bytes = src->frameCallback(src->frame, src->frameSize);
1212 uint16_t buffer_available = src->buffer->availableForWrite();
1213 if (buffer_available < available_bytes) {
1214 // if buffer is full make space
1215 uint16_t to_clear = available_bytes - buffer_available;
1216 uint8_t tmp[to_clear];
1217 src->buffer->readArray(tmp, to_clear);
1218 }
1219 if (src->buffer->writeArray(src->frame, available_bytes) !=
1220 available_bytes) {
1221 // LOGE(UNDERFLOW_MSG);
1222 assert(false);
1223 }
1224 } else {
1225 // output
1226 if (src->buffer != nullptr && src->frame != nullptr &&
1227 src->frameSize > 0) {
1228 uint16_t available_bytes =
1229 src->buffer->readArray(src->frame, src->frameSize);
1230 if (available_bytes !=
1231 src->frameCallback(src->frame, available_bytes)) {
1232 LOGE(UNDERFLOW_MSG);
1233 }
1234 }
1235 }
1236 src->measureSampleRate();
1237 }
