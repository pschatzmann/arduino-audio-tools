269 size_t write(const void *in_ptr, size_t in_size) {

292 void encodeByte(uint8_t data) {
293 // add byte to frame
294 frame[frame_pos++] = data;
295
296 // if frame is complete -> encode
297 if (frame_pos >= frame.size()) {
298 encodeFrame();
299 frame_pos = 0;
300 }

303 void encodeFrame() {
304 if (frame.size() > 0) {
305 // allocate temp buffer on stack
306 int packet_len = OPUS_ENC_MAX_BUFFER_SIZE > 0 ? OPUS_ENC_MAX_BUFFER_SIZE : 512;
307 uint8_t packet[packet_len];
308
309 int frames = frame.size() / cfg.channels / sizeof(int16_t);
310 LOGD("opus_encode - frame_size: %d", frames);
311 int len = opus_encode(enc, (opus_int16 *)frame.data(), frames,
312 packet, packet_len);
313 if (len < 0) {
314 LOGE("opus_encode: %s", opus_strerror(len));
315 } else if (len > 0) {
316 int eff = p_print->write(packet, len);
317 if (eff!=len){
318 LOGE("encodeFrame data lost");
319 }
320 }
321 }

325 int getFrameSizeSamples(int sampling_rate) {
326 switch (cfg.frame_sizes_ms_x2) {
327 case OPUS_FRAMESIZE_2_5_MS:
328 return sampling_rate / 400;
329 case OPUS_FRAMESIZE_5_MS:
330 return sampling_rate / 200;
331 case OPUS_FRAMESIZE_10_MS:
332 return sampling_rate / 100;
333 case OPUS_FRAMESIZE_20_MS:
334 return sampling_rate / 50;
335 case OPUS_FRAMESIZE_40_MS:
336 return sampling_rate / 25;
337 case OPUS_FRAMESIZE_60_MS:
338 return 3 * sampling_rate / 50;
339 case OPUS_FRAMESIZE_80_MS:
340 return 4 * sampling_rate / 50;
341 case OPUS_FRAMESIZE_100_MS:
342 return 5 * sampling_rate / 50;
343 case OPUS_FRAMESIZE_120_MS:
344 return 6 * sampling_rate / 50;
345 }
346 return sampling_rate / 100;
