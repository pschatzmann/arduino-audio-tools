196  } else if (out_samples > 0) {

311  void encodeFrame() {
312  if (frame.size() > 0) {
313  // allocate temp buffer on stack
314  int packet_len = OPUS_ENC_MAX_BUFFER_SIZE > 0 ? OPUS_ENC_MAX_BUFFER_SIZE : 512;
315  uint8_t packet[packet_len];
316 
317  int frames = frame.size() / cfg.channels / sizeof(int16_t);
318  LOGD("opus_encode - frame_size: %d", frames);
319  int len = opus_encode(enc, (opus_int16 *)frame.data(), frames,
320  packet, packet_len);
321  if (len < 0) {
322  LOGE("opus_encode: %s", opus_strerror(len));
323  } else if (len > 0) {
324  int eff = p_print->write(packet, len);
325  if (eff!=len){
326  LOGE("encodeFrame data lost");
327  }
328  }
329  }
