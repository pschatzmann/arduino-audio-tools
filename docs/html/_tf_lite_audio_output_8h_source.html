126 // Partial implementation of std::dequeue, just providing the functionality
127 // that's needed to keep a record of previous neural network results over a
128 // short time period, so they can be averaged together to produce a more
129 // accurate overall prediction. This doesn't use any dynamic memory allocation
130 // so it's a better fit for microcontroller applications, but this does mean
131 // there are hard limits on the number of results it can store.
132 template <int N>
133 class TfLiteResultsQueue {
134  public:
135  TfLiteResultsQueue() : front_index_(0), size_(0) {}
136 
137  // Data structure that holds an inference result, and the time when it
138  // was recorded.
139  struct Result {
140  Result() : time_(0), scores() {}
141  Result(int32_t time, int8_t* input_scores) : time_(time) {
142  for (int i = 0; i < N; ++i) {
143  scores[i] = input_scores[i];
144  }
145  }
146  int32_t time_;
147  int8_t scores[N];
148  };
149 
150  int size() { return size_; }
151  bool empty() { return size_ == 0; }
152  Result& front() { return results_[front_index_]; }
153  Result& back() {
154  int back_index = front_index_ + (size_ - 1);
155  if (back_index >= kMaxResults) {
156  back_index -= kMaxResults;
157  }
158  return results_[back_index];
159  }
160 
161  void push_back(const Result& entry) {
162  if (size() >= kMaxResults) {
163  LOGE("Couldn't push_back latest result, too many already!");
164  return;
165  }
166  size_ += 1;
167  back() = entry;
168  }
169 
170  Result pop_front() {
171  if (size() <= 0) {
172  LOGE("Couldn't pop_front result, none present!");
173  return Result();
174  }
175  Result result = front();
176  front_index_ += 1;
177  if (front_index_ >= kMaxResults) {
178  front_index_ = 0;
179  }
180  size_ -= 1;
181  return result;
182  }
183 
184  // Most of the functions are duplicates of dequeue containers, but this
185  // is a helper that makes it easy to iterate through the contents of the
186  // queue.
187  Result& from_front(int offset) {
188  if ((offset < 0) || (offset >= size_)) {
189  LOGE("Attempt to read beyond the end of the queue!");
190  offset = size_ - 1;
191  }
192  int index = front_index_ + offset;
193  if (index >= kMaxResults) {
194  index -= kMaxResults;
195  }
196  return results_[index];
197  }
198 
199  private:
200  static constexpr int kMaxResults = 50;
201  Result results_[kMaxResults];
202 
203  int front_index_;
204  int size_;
205 };

239 class TfLiteRecognizeCommands : public TfLiteAbstractRecognizeCommands<N> {
240  public:
241  // labels should be a list of the strings associated with each one-hot score.
242  // The window duration controls the smoothing. Longer durations will give a
243  // higher confidence that the results are correct, but may miss some commands.
244  // The detection threshold has a similar effect, with high values increasing
245  // the precision at the cost of recall. The minimum count controls how many
246  // results need to be in the averaging window before it's seen as a reliable
247  // average. This prevents erroneous results when the averaging window is
248  // initially being populated for example. The suppression argument disables
249  // further recognitions for a set time after one has been triggered, which can
250  // help reduce spurious recognitions.
251 
252  TfLiteRecognizeCommands() {
253  previous_top_label_ = "silence";
254  previous_top_label_time_ = std::numeric_limits<int32_t>::min();
255  kCategoryCount = N;
256  }
257 
259  bool begin(TfLiteConfig cfg) override {
260  this->cfg = cfg;
261  if (kCategoryCount == 0) {
262  LOGE("kCategoryCount must not be 0");
263  return false;
264  }
265  if (cfg.labels == nullptr) {
266  LOGE("config.labels not defined");
267  return false;
268  }
269  started = true;
270  return true;

273  // Call this with the results of running a model on sample data.
274  virtual TfLiteStatus processLatestResults(const TfLiteTensor* latest_results,
275  const int32_t current_time_ms,
276  const char** found_command,
277  uint8_t* score,
278  bool* is_new_command) override {
279  LOGD(LOG_METHOD);
280  if (!started) {
281  LOGE("TfLiteRecognizeCommands not started");
282  return kTfLiteError;
283  }
284  if ((latest_results->dims->size != 2) ||
285  (latest_results->dims->data[0] != 1) ||
286  (latest_results->dims->data[1] != kCategoryCount)) {
287  LOGE(
288  "The results for recognition should contain %d "
289  "elements, but there are "
290  "%d in an %d-dimensional shape",
291  kCategoryCount, (int)latest_results->dims->data[1],
292  (int)latest_results->dims->size);
293  return kTfLiteError;
294  }
295 
296  if (latest_results->type != kTfLiteInt8) {
297  LOGE("The results for recognition should be int8 elements, but are %d",
298  (int)latest_results->type);
299  return kTfLiteError;
300  }
301 
302  if ((!previous_results_.empty()) &&
303  (current_time_ms < previous_results_.front().time_)) {
304  LOGE("Results must be in increasing time order: timestamp %d < %d",
305  (int)current_time_ms, (int)previous_results_.front().time_);
306  return kTfLiteError;
307  }
308 
309  // Add the latest results to the head of the queue.
310  previous_results_.push_back({current_time_ms, latest_results->data.int8});
311 
312  // Prune any earlier results that are too old for the averaging window.
313  const int64_t time_limit = current_time_ms - cfg.average_window_duration_ms;
314  while ((!previous_results_.empty()) &&
315  previous_results_.front().time_ < time_limit) {
316  previous_results_.pop_front();
317  }
318 
319  // If there are too few results, assume the result will be unreliable
320  // and bail.
321  const int64_t how_many_results = previous_results_.size();
322  const int64_t earliest_time = previous_results_.front().time_;
323  const int64_t samples_duration = current_time_ms - earliest_time;
324  if ((how_many_results < cfg.minimum_count) ||
325  (samples_duration < (cfg.average_window_duration_ms / 4))) {
326  *found_command = previous_top_label_;
327  *score = 0;
328  *is_new_command = false;
329  return kTfLiteOk;
330  }
331 
332  // Calculate the average score across all the results in the window.
333  int32_t average_scores[kCategoryCount];
334  for (int offset = 0; offset < previous_results_.size(); ++offset) {
335  auto previous_result = previous_results_.from_front(offset);
336  const int8_t* scores = previous_result.scores;
337  for (int i = 0; i < kCategoryCount; ++i) {
338  if (offset == 0) {
339  average_scores[i] = scores[i] + 128;
340  } else {
341  average_scores[i] += scores[i] + 128;
342  }
343  }
344  }
345  for (int i = 0; i < kCategoryCount; ++i) {
346  average_scores[i] /= how_many_results;
347  }
348 
349  // Find the current highest scoring category.
350  int current_top_index = 0;
351  int32_t current_top_score = 0;
352  for (int i = 0; i < kCategoryCount; ++i) {
353  if (average_scores[i] > current_top_score) {
354  current_top_score = average_scores[i];
355  current_top_index = i;
356  }
357  }
358  const char* current_top_label = cfg.labels[current_top_index];
359 
360  // If we've recently had another label trigger, assume one that occurs
361  // too soon afterwards is a bad result.
362  int64_t time_since_last_top;
363  if ((previous_top_label_ == cfg.labels[0]) ||
364  (previous_top_label_time_ == std::numeric_limits<int32_t>::min())) {
365  time_since_last_top = std::numeric_limits<int32_t>::max();
366  } else {
367  time_since_last_top = current_time_ms - previous_top_label_time_;
368  }
369  if ((current_top_score > cfg.detection_threshold) &&
370  ((current_top_label != previous_top_label_) ||
371  (time_since_last_top > cfg.suppression_ms))) {
372  previous_top_label_ = current_top_label;
373  previous_top_label_time_ = current_time_ms;
374  *is_new_command = true;
375  } else {
376  *is_new_command = false;
377  }
378  *found_command = current_top_label;
379  *score = current_top_score;
380 
381  return kTfLiteOk;
382  }
383 
384  protected:
385  // Configuration
386  TfLiteConfig cfg;
387  int kCategoryCount;
388  bool started = false;
389 
390  // Working variables
391  TfLiteResultsQueue<N> previous_results_;
392  const char* previous_top_label_;
393  int32_t previous_top_label_time_;
394 };

577  &g_micro_features_state, frontend_input, input_size, num_samples_read);

580  if (output_size != frontend_output.size) {
581  LOGE("output_size=%d, frontend_output.size=%d", output_size,

585  // printf("input_size: %d, num_samples_read: %d,output_size: %d,

591  // LOGE("audio_samples_size=%d vs num_samples_read=%d", input_size,
592  // *num_samples_read);
593  // }
594 
595  for (size_t i = 0; i < frontend_output.size; ++i) {
596  // These scaling values are derived from those used in input_data.py in
597  // the training pipeline. The feature pipeline outputs 16-bit signed
598  // integers in roughly a 0 to 670 range. In training, these are then
599  // arbitrarily divided by 25.6 to get float values in the rough range of
600  // 0.0 to 26.0. This scaling is performed for historical reasons, to match
601  // up with the output of other feature generators. The process is then
602  // further complicated when we quantize the model. This means we have to
603  // scale the 0.0 to 26.0 real values to the -128 to 127 signed integer
604  // numbers. All this means that to get matching values from our integer
605  // feature output into the tensor input, we have to perform: input =
606  // (((feature / 25.6) / 26.0) * 256) - 128 To simplify this and perform it
607  // in 32-bit integer math, we rearrange to: input = (feature * 256) /
608  // (25.6 * 26.0) - 128
609  constexpr int32_t value_scale = 256;
610  constexpr int32_t value_div =
611  static_cast<int32_t>((25.6f * 26.0f) + 0.5f);
612  int32_t value =
613  ((frontend_output.values[i] * value_scale) + (value_div / 2)) /
614  value_div;
615  value -= 128;
616  if (value < -128) {
617  value = -128;
618  }
619  if (value > 127) {
620  value = 127;
621  }
622  output[i] = value;
623  }
624 
625  return kTfLiteOk;
626  }

692  LOGE("AllocateTensors() failed");
693  return false;
694  }

697  LOGI("Get Input");

699  if ((model_input->dims->size != 2) || (model_input->dims->data[0] != 1) ||
700  (model_input->dims->data[1] !=
701  (cfg.kFeatureSliceCount * cfg.kFeatureSliceSize)) ||
702  (model_input->type != kTfLiteInt8)) {

728  }

742  total_slice_count = 0;
743  }
744  }
