77  template <typename T> void convertFrames(T *data, int frames, int channels) {
78  float delta = 1.0 / frames;

92  template <typename T>
93  void fadeOut(T *data, int frames, int channels, float delta) {
94  for (int j = 0; j < frames; j++) {
95  for (int ch = 0; ch < channels; ch++) {
96  data[j * channels + ch] = data[j * channels + ch] * volume;
97  if (volume > 0) {
98  volume -= delta;
99  if (volume < 0) {
100  volume = 0;
101  }
102  }
103  }
104  }
