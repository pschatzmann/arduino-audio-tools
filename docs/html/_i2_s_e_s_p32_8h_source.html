100  TRACEE();

102  } else if (cfg.channels==1){
103  // I2S has always 2 channels. We support to reduce it to 1
104  uint8_t temp[size_bytes*2];
105  if (i2s_read(i2s_num, temp, size_bytes*2, &result, portMAX_DELAY)!=ESP_OK){
106  TRACEE();

108  // convert to 1 channel
109  switch(cfg.bits_per_sample){
110  case 16: {

114  // case 24: {

224  TRACEE();
