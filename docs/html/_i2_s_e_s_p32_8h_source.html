144 static size_t writeExpandChannel(i2s_port_t i2s_num, const int bits_per_sample, const void *src, size_t size_bytes){
145 size_t result = 0;
146 int j;
147 switch(bits_per_sample){
148
149 case 8:
150 for (j=0;j<size_bytes;j++){
151 int8_t frame[2];
152 int8_t *data = (int8_t *)src;
153 frame[0]=data[j];
154 frame[1]=data[j];
155 size_t result_call = 0;
156 if (i2s_write(i2s_num, frame, sizeof(int8_t)*2, &result_call, portMAX_DELAY)!=ESP_OK){
157 LOGE(LOG_METHOD);
158 } else {
159 result += result_call;
160 }
161 }
162 break;
163
164 case 16:
165 for (j=0;j<size_bytes/2;j++){
166 int16_t frame[2];
167 int16_t *data = (int16_t*)src;
168 frame[0]=data[j];
169 frame[1]=data[j];
170 size_t result_call = 0;
171 if (i2s_write(i2s_num, frame, sizeof(int16_t)*2, &result_call, portMAX_DELAY)!=ESP_OK){
172 LOGE(LOG_METHOD);
173 } else {
174 result += result_call;
175 }
176 }
177 break;
178
179 case 24:
180 for (j=0;j<size_bytes/4;j++){
181 int24_t frame[2];
182 int24_t *data = (int24_t*) src;
183 frame[0]=data[j];
184 frame[1]=data[j];
185 size_t result_call = 0;

187 LOGE(LOG_METHOD);

196 int32_t frame[2];

198 frame[0]=data[j];

202 LOGE(LOG_METHOD);

213 i2s_comm_format_t toCommFormat(I2SMode mode){
214 switch(mode){
215 case I2S_PHILIPS_MODE:
216 case I2S_STD_MODE:
217 case I2S_LSB_MODE:
218 case I2S_RIGHT_JUSTIFIED_MODE:
219 return (i2s_comm_format_t) I2S_COMM_FORMAT_STAND_I2S;
220 case I2S_MSB_MODE:
221 case I2S_LEFT_JUSTIFIED_MODE:
222 return (i2s_comm_format_t) I2S_COMM_FORMAT_STAND_MSB;
223 default:
224 LOGE("unsupported mode");
225 return (i2s_comm_format_t) I2S_COMM_FORMAT_STAND_I2S;
226 }

229 // determines the i2s_mode_t
230 i2s_mode_t toMode(I2SConfig &cfg) {
231 i2s_mode_t mode;
232 if (cfg.is_digital){
233 int i2s_mode = cfg.is_master ? I2S_MODE_MASTER : I2S_MODE_SLAVE;
234 int rx_tx = cfg.rx_tx_mode == TX_MODE ? I2S_MODE_TX : I2S_MODE_RX;
235 mode = (i2s_mode_t) (i2s_mode | rx_tx);
236 } else {
237 mode = (i2s_mode_t) (cfg.rx_tx_mode ? I2S_MODE_DAC_BUILT_IN : I2S_MODE_ADC_BUILT_IN);
238 }
239 return mode;
