184 static size_t writeExpandChannel(i2s_port_t i2s_num, const int bits_per_sample, const void *src, size_t size_bytes){
185 size_t result = 0;
186 int j;
187 switch(bits_per_sample){
188
189 case 8:
190 for (j=0;j<size_bytes;j++){
191 int8_t frame[2];
192 int8_t *data = (int8_t *)src;
193 frame[0]=data[j];
194 frame[1]=data[j];
195 size_t result_call = 0;
196 if (i2s_write(i2s_num, frame, sizeof(int8_t)*2, &result_call, portMAX_DELAY)!=ESP_OK){
197 LOGE(LOG_METHOD);
198 } else {
199 result += result_call;
200 }
201 }
202 break;
203
204 case 16:
205 for (j=0;j<size_bytes/2;j++){
206 int16_t frame[2];
207 int16_t *data = (int16_t*)src;
208 frame[0]=data[j];
209 frame[1]=data[j];
210 size_t result_call = 0;
211 if (i2s_write(i2s_num, frame, sizeof(int16_t)*2, &result_call, portMAX_DELAY)!=ESP_OK){
212 LOGE(LOG_METHOD);
213 } else {
214 result += result_call;
215 }
216 }
217 break;
218
219 case 24:
220 for (j=0;j<size_bytes/4;j++){
221 int24_t frame[2];
222 int24_t *data = (int24_t*) src;
223 frame[0]=data[j];
224 frame[1]=data[j];
225 size_t result_call = 0;
226 if (i2s_write(i2s_num, frame, sizeof(int24_t)*2, &result_call, portMAX_DELAY)!=ESP_OK){
227 LOGE(LOG_METHOD);
228 } else {
229 result += result_call;
230 }
231 }
232 break;
233
234 case 32:
235 for (j=0;j<size_bytes/4;j++){
236 int32_t frame[2];
237 int32_t *data = (int32_t*) src;
238 frame[0]=data[j];
239 frame[1]=data[j];
240 size_t result_call = 0;
241 if (i2s_write(i2s_num, frame, sizeof(int32_t)*2, &result_call, portMAX_DELAY)!=ESP_OK){
242 LOGE(LOG_METHOD);
243 } else {
244 result += result_call;
245 }
246 }
247 break;
248 }
249 return result;

283 i2s_mode_t toMode(I2SConfig &cfg) {
284 i2s_mode_t mode;
285 if (cfg.is_digital){
286 int i2s_format = cfg.is_master ? I2S_MODE_MASTER : I2S_MODE_SLAVE;
287 int i2s_rx_tx = 0;
288 switch(cfg.rx_tx_mode){
289 case TX_MODE:
290 i2s_rx_tx = I2S_MODE_TX;
291 break;
292 case RX_MODE:
293 i2s_rx_tx = I2S_MODE_RX;
294 break;
295 case RXTX_MODE:
296 i2s_rx_tx = I2S_MODE_RX | I2S_MODE_TX;
297 break;
298 default:
299 LOGE("Undefined rx_tx_mode: %d", cfg.rx_tx_mode);
300 }
301 mode = (i2s_mode_t) (i2s_format | i2s_rx_tx);
302 } else {
303 mode = (i2s_mode_t) (cfg.rx_tx_mode ? I2S_MODE_DAC_BUILT_IN : I2S_MODE_ADC_BUILT_IN);
304 }
305 return mode;
306 }
