119 size_t write(const void *data, size_t len) {
120 parsed.clear();
121 // parse data and collect relevant positions
122 char *pt8 = (char *)data;
123 for (int j = 0; j < len - 3; j++) {
124 if (strncmp("CFG", pt8 + j, 3)==0) {
125 parsed.push_back(pt8 + j);
126 } else if (strncmp("DAT", pt8 + j, 3)==0) {
127 parsed.push_back(pt8 + j);
128 }
129 }
130 // add end
131 parsed.push_back(pt8 + len);
132
133 // process data
134 if (parsed.size() > 1) {
135 for (int j = 0; j < parsed.size()-1; j++) {
136 char *start = parsed[j];
137 char *end = parsed[j + 1];
138 if (strncmp(start, "CFG", 3)==0) {
139 memmove((void*)&cfg,(void*) start, sizeof(cfg));
140 if (p_inform != nullptr) {
141 p_inform->setAudioInfo(cfg.info);
142 }
143 } else if (strncmp (start, "DAT", 3)==0) {
144 int len = end - start;
145 p_codec->write(start + 3, len - 3);
146 }
147 }
148 } else {
149 // we have only an end entry -> write all data
150 p_codec->write(data, len);
151 }
152 return len;
153 }
