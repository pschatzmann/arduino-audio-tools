131 bool begin(SPDIFConfig cfg) {
132 TRACED();
133 // Some validations to make sure that the config is valid
134 if (!(cfg.channels == 1 || cfg.channels == 2)) {
135 LOGE("Unsupported number of channels: %d", cfg.channels);
136 return false;
137 }
138 if (cfg.bits_per_sample != 16) {
139 LOGE("Unsupported bits per sample: %d - must be 16!",
140 cfg.bits_per_sample);
141 return false;
142 }
143
144 if (i2sOn) {
145 i2s.end();
146 }
147
148 // initialize S/PDIF buffer
149 spdif_buf_init();
150 spdif_ptr = spdif_buf;
151
152 // Setup I2S
153 int sample_rate = cfg.sample_rate * BMC_BITS_FACTOR;
154 int bclk = sample_rate * I2S_BITS_PER_SAMPLE * I2S_CHANNELS;
155 int mclk = (I2S_BUG_MAGIC / bclk) * bclk; // use mclk for avoiding I2S bug
156
157 I2SConfig i2s_cfg;
158 i2s_cfg.sample_rate = sample_rate;
159 i2s_cfg.channels = cfg.channels;
160 i2s_cfg.i2s_format = I2S_STD_FORMAT;
161 i2s_cfg.bits_per_sample = I2S_BITS_PER_SAMPLE;
162 i2s_cfg.pin_ws = -1;
163 i2s_cfg.pin_bck = -1;
164 i2s_cfg.pin_data = cfg.pin_data;
165#ifdef ESP32
166 i2s_cfg.use_apll = true;
167 i2s_cfg.fixed_mclk = mclk;
168#endif
169 i2s.begin(i2s_cfg);
170 i2sOn = true;

200 size_t write(const uint8_t *src, size_t size) {
201 if (!i2sOn) return 0;
202 const uint8_t *p = src;
203 size_t result = 0;
204
205 while (p < (uint8_t *)src + size) {
206 // convert PCM 16bit data to BMC 32bit pulse pattern
207 if (cfg.channels == 2) {
208 *(spdif_ptr + 1) =
209 (uint32_t)(((bmc_tab[*p] << 16) ^ bmc_tab[*(p + 1)]) << 1) >> 1;
210 p += 2;
211 } else {
212 // must be one channels -> use the same value for both
213 *(spdif_ptr + 1) =
214 (uint32_t)(((bmc_tab[*p] << 16) ^ bmc_tab[*(p)]) << 1) >> 1;
215 p++;
216 }
217 result += 2;
218 spdif_ptr += 2; // advance to next audio data
219
220 if (spdif_ptr >= &spdif_buf[SPDIF_BUF_ARRAY_SIZE]) {
221 // set block start preamble
222 ((uint8_t *)spdif_buf)[SYNC_OFFSET] ^= SYNC_FLIP;
223
224 i2s.write((uint8_t *)spdif_buf, sizeof(spdif_buf));
225 spdif_ptr = spdif_buf;
226 }
227 }
228
229 return result;
230 }
