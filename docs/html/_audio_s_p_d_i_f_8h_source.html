201  size_t write(const uint8_t *src, size_t size) {
202  if (!i2sOn) return 0;
203  const uint8_t *p = src;
204  size_t result = 0;
205 
206  while (p < (uint8_t *)src + size) {
207  // convert PCM 16bit data to BMC 32bit pulse pattern
208  if (cfg.channels == 2) {
209  *(spdif_ptr + 1) =
210  (uint32_t)(((bmc_tab[*p] << 16) ^ bmc_tab[*(p + 1)]) << 1) >> 1;
211  p += 2;
212  } else {
213  // must be one channels -> use the same value for both
214  *(spdif_ptr + 1) =
215  (uint32_t)(((bmc_tab[*p] << 16) ^ bmc_tab[*(p)]) << 1) >> 1;
216  p++;
217  }
218  result += 2;
219  spdif_ptr += 2; // advance to next audio data
220 
221  if (spdif_ptr >= &spdif_buf[SPDIF_BUF_ARRAY_SIZE]) {
222  // set block start preamble
223  ((uint8_t *)spdif_buf)[SYNC_OFFSET] ^= SYNC_FLIP;
224 
225  i2s.write((uint8_t *)spdif_buf, sizeof(spdif_buf));
226  spdif_ptr = spdif_buf;
227  }
228  }
229 
230  return result;
231  }
