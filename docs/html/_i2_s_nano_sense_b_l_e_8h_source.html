179 TRACED();
180 // Enable MCK generator if in master mode
181 if (cfg.is_master){
182 NRF_I2S->CONFIG.MCKEN = (I2S_CONFIG_MCKEN_MCKEN_Enabled << I2S_CONFIG_MCKEN_MCKEN_Pos);

185 // find closest frequency for requested sample_rate
186 float freq_requested = cfg.sample_rate * cfg.bits_per_sample ;
187 float selected_freq = 0;
188 for (auto freq : freq_table) {
189 for (auto div : ratio_table) {
190 float freq_value = freq.freq * 1000000 / div.ratio;
191 if (abs(freq_value-freq_requested) < abs(selected_freq-freq_requested)){
192 // MCKFREQ
193 NRF_I2S->CONFIG.MCKFREQ = freq.id << I2S_CONFIG_MCKFREQ_MCKFREQ_Pos;
194 // Ratio
195 NRF_I2S->CONFIG.RATIO = div.id << I2S_CONFIG_RATIO_RATIO_Pos;
196 selected_freq = freq_value;
197 }
198 }
199 LOGI("frequency requested %f vs %f", freq_requested, selected_freq);
