242  effect_t process(effect_t input) {
243  if (!active()) return input;
244 
245  // limit value to max 100% and calculate factors
246  float tremolo_depth = p_percent > 100 ? 1.0 : 0.01 * p_percent;
247  float signal_depth = (100.0 - p_percent) / 100.0;
248 
249  float tremolo_factor = tremolo_depth / rate_count_half;
250  int32_t out = (signal_depth * input) + (tremolo_factor * count * input);
251 
252  //saw tooth shaped counter
253  count+=inc;
254  if (count>=rate_count_half){
255  inc = -1;
256  } else if (count<=0) {
257  inc = +1;
258  }
259 
260  return clip(out);
261  }

265  }
