212class Tremolo : public AudioEffect {
213 public:
215 Tremolo(int16_t duration_ms=2000, uint8_t depthPercent=50, uint32_t sampleRate=44100) {
216 this->duration_ms = duration_ms;
217 this->sampleRate = sampleRate;
218 this->p_percent = depthPercent;
219 int32_t rate_count = sampleRate * duration_ms / 1000;
220 rate_count_half = rate_count / 2;
221 }
222
223 Tremolo(const Tremolo &copy) = default;
224
225 void setDuration(int16_t ms){
226 int32_t rate_count = sampleRate * ms / 1000;
227 rate_count_half = rate_count / 2;
228 }
229
230 int16_t duration(){
231 return duration_ms;
232 }
233
234 void setDepth(uint8_t percent){
235 p_percent = percent;
236 }
237
238 uint8_t depth() {
239 return p_percent;
240 }
241
242 effect_t process(effect_t input) {
243 if (!active()) return input;
244
245 // limit value to max 100% and calculate factors
246 float tremolo_depth = p_percent > 100 ? 1.0 : 0.01 * p_percent;
247 float signal_depth = (100.0 - p_percent) / 100.0;
248
249 float tremolo_factor = tremolo_depth / rate_count_half;
250 int32_t out = (signal_depth * input) + (tremolo_factor * count * input);
251
252 //saw tooth shaped counter
253 count+=inc;
254 if (count>=rate_count_half){
255 inc = -1;
256 } else if (count<=0) {
257 inc = +1;
258 }
259
260 return clip(out);
261 }
262
263 Tremolo* clone() {
264 return new Tremolo(*this);
265 }
266
267 protected:
268 int16_t duration_ms;
269 uint32_t sampleRate;
270 int32_t count = 0;
271 int16_t inc = 1;
272 int32_t rate_count_half; // number of samples for on raise and fall
273 uint8_t p_percent;
274
