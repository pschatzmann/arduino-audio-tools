255 static uint32_t get_bits(bs_t *bs, int n)
256 {
257  uint32_t next, cache = 0, s = bs->pos & 7;
258  int shl = n + s;
259  const uint8_t *p = bs->buf + (bs->pos >> 3);
260  if ((bs->pos += n) > bs->limit)
261  return 0;
262  next = *p++ & (255 >> s);
263  while ((shl -= 8) > 0)
264  {
265  cache |= next << shl;
266  next = *p++;
267  }
268  return cache | (next >> -shl);
269 }

441 static int L12_dequantize_granule(float *grbuf, bs_t *bs, L12_scale_info *sci, int group_size)
442 {
443  int i, j, k, choff = 576;
444  for (j = 0; j < 4; j++)
445  {
446  float *dst = grbuf + group_size*j;
447  for (i = 0; i < 2*sci->total_bands; i++)
448  {
449  int ba = sci->bitalloc[i];
450  if (ba != 0)
451  {
452  if (ba < 17)
453  {
454  int half = (1 << (ba - 1)) - 1;
455  for (k = 0; k < group_size; k++)
456  {
457  dst[k] = (float)((int)get_bits(bs, ba) - half);
458  }
459  } else
460  {
461  unsigned mod = (2 << (ba - 17)) + 1; /* 3, 5, 9 */
462  unsigned code = get_bits(bs, mod + 2 - (mod >> 3)); /* 5, 7, 10 */
463  for (k = 0; k < group_size; k++, code /= mod)
464  {
465  dst[k] = (float)((int)(code % mod - mod/2));
466  }
467  }
468  }
469  dst += choff;
470  choff = 18 - choff;
471  }
472  }
473  return group_size*4;
474 }
