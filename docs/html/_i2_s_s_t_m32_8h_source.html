112  size_t readBytes(void *dest, size_t size_bytes){
113  TRACED();
114  if (cfg.channels == 2){
115  return p_rx_buffer->readArray((uint8_t*)dest, size_bytes);
116  } else {
117  // combine two channels to 1: so request double the amout
118  int req_bytes = size_bytes*2;
119  uint8_t tmp[req_bytes];
120  int16_t *tmp_16 = (int16_t*) tmp;
121  int eff_bytes = p_rx_buffer->readArray((uint8_t*)tmp, req_bytes);
122  // add 2 channels together
123  int16_t *dest_16 = (int16_t *)dest;
124  int16_t eff_samples = eff_bytes / 2;
125  int16_t idx = 0;
126  for (int j=0;j<eff_samples;j+=2){
127  dest_16[idx++] = static_cast<float>(tmp_16[j])+tmp_16[j+1] / 2.0;
128  }

228  size_t writeBytesExt(const void *src, size_t size_bytes){
229  size_t result = 0;
230  if (cfg.channels == 2){
231  result = p_tx_buffer->writeArray((uint8_t*)src, size_bytes);
232  } else {

244  } else {

247  }
248  }
