171  bool checkPrefix(const void *input_buffer, size_t length) {
172  bool result = true;
173  if (length >= 4 && memcmp(input_buffer, "\x4b\xbf\x4b\xbf", 4) == 0) {
174  if (is_hd) {
175  LOGE("aptX audio stream (not aptX HD)");
176  result = false;
177  }
178  } else if (length >= 6 &&
179  memcmp(input_buffer, "\x73\xbe\xff\x73\xbe\xff", 6) == 0) {
180  if (!is_hd) {
181  LOGE("aptX HD audio stream");
182  result = false;
183  }
184  } else {
185  if (length >= 4 && memcmp(input_buffer, "\x6b\xbf\x6b\xbf", 4) == 0) {
186  LOGE("standard aptX audio stream - not supported");
187  result = false;
188  } else {
189  LOGE("No aptX nor aptX HD audio stream");
190  result = false;
