109 void writeData(size_t written, bool &is_ok) {
110 if (written > 0) {

112 LOGI("written: %d", written);
113 LOGI("samples: %d", samples);
114 int24_t *p_int24 = (int24_t *)output_buffer.data();
115 int16_t *p_int16 = (int16_t *)output_buffer.data();
116 for (int j = 0; j < samples; j++) {

129 void checkSync(bool synced, bool dropped, bool &is_ok) {
130 /* Check all possible states of synced, syncing and dropped status */
131 if (!synced) {
132 if (!syncing) {
133 LOGE("aptX decoding failed, synchronizing");
134 syncing = true;
135 is_ok = false;

137 if (dropped) {
138 LOGE("aptX synchronization successful, dropped %lu byte%s",
139 (unsigned long)dropped, (dropped != 1) ? "s" : "");
140 syncing = false;
141 is_ok = true;
142 }
143 if (!syncing) {
144 LOGE("aptX decoding failed, synchronizing");
145 syncing = true;
146 is_ok = false;
147 }
148 } else {
149 if (dropped) {
150 if (!syncing) LOGE("aptX decoding failed, synchronizing");
151 LOGE("aptX synchronization successful, dropped %lu byte%s",

153 syncing = false;
154 is_ok = false;
155 } else if (syncing) {
156 LOGI("aptX synchronization successful");
157 syncing = false;

164 bool checkPrefix(const void *input_buffer, size_t length) {
165 bool result = true;
166 if (length >= 4 && memcmp(input_buffer, "\x4b\xbf\x4b\xbf", 4) == 0) {
167 if (is_hd) {
168 LOGE("aptX audio stream (not aptX HD)");
169 result = false;
170 }
171 } else if (length >= 6 &&
172 memcmp(input_buffer, "\x73\xbe\xff\x73\xbe\xff", 6) == 0) {
173 if (!is_hd) {
174 LOGE("aptX HD audio stream");
175 result = false;
176 }
177 } else {
178 if (length >= 4 && memcmp(input_buffer, "\x6b\xbf\x6b\xbf", 4) == 0) {
179 LOGE("standard aptX audio stream - not supported");
180 result = false;
181 } else {
182 LOGE("No aptX nor aptX HD audio stream");
183 result = false;
184 }
185 }
186 return result;
