203  size_t write(const uint8_t *data, size_t len) override {
204  int open = len;
205  size_t result = 0;
206  int retry_count = 0;
207  while (open > 0) {
208  if (available_to_write > 0) {
209  resetAvailableToWrite();
210  size_t send_len = min(open, ESP_NOW_MAX_DATA_LEN);
211  esp_err_t rc = esp_now_send(nullptr, data + result, send_len);
212  // wait for confirmation
213  if (cfg.use_send_ack) {
214  while (available_to_write == 0) {
215  delay(1);
216  }
217  } else {
218  is_write_ok = true;
219  }
220  // check status
221  if (rc == ESP_OK && is_write_ok) {
222  open -= send_len;
223  result += send_len;
224  } else {
225  LOGW("Write failed - retrying again");
226  retry_count++;
227  if (cfg.write_retry_count>0 && retry_count>=cfg.write_retry_count){
228  LOGE("Write error after %d retries", cfg.write_retry_count);
229  // break loop
230  return 0;
231  }
232  }
233  // if we do have no partner to write we stall and retry later
234  } else {
235  delay(cfg.delay_after_write_ms);
236  }
237 
238  // Wait some time before we retry
239  if (!is_write_ok) {
240  delay(cfg.delay_after_failed_write_ms);
241  }
242  }
243  return result;
244  }
