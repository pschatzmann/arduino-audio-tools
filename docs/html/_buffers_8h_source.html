534  filled_buffers[j] = filled_buffers[j + 1];
535  // clear last enty

544  if (write_buffer_count < buffer_count) {
545  filled_buffers[write_buffer_count++] = buffer;

565  LOGI("BufferedArray(%d)", len);

569  // access values, the offset and length are specified in samples of type <T>
570  int16_t *getValues(size_t offset, size_t length) {
571  LOGD("getValues(%d,%d) - max %d", offset, length, array.size());
572  if (offset == 0) {
573  // we restart at the beginning
574  last_end = 0;
575  actual_end = length;
576  } else {
577  // if first position is at end we do not want to read the full buffer
578  last_end = actual_end>=0 ? actual_end : offset;
579  // increase actual end if bigger then old
580  actual_end = offset + length > actual_end ? offset + length : actual_end;
581  }
582  int size = actual_end - last_end;
583  if (size > 0) {
584  LOGD("readBytes(%d,%d)", last_end, size);
585  assert(last_end+size<=array.size());
586  p_stream->readBytes((uint8_t *)(&array[last_end]), size * 2);
587  }
588  assert(offset<actual_end);
589  return &array[offset];
590  }
591 
592  protected:
593  int actual_end = -1;
594  int last_end = 0;
595  Vector<T> array;
596  Stream *p_stream = nullptr;
