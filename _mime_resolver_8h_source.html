100 return custom_mime_table[i].mime;
101 }
102
103 // search in default table
104 for (size_t i = 0; mime_table[i].ext != nullptr; i++) {
105 if (extView.equalsIgnoreCase(mime_table[i].ext))
106 return mime_table[i].mime;

109 }
