230 bool startConfiguredStreams() {
231 if (cfg.rx_tx_mode == TX_MODE) {
232 return i2s_trigger(i2s_dev, I2S_DIR_TX, I2S_TRIGGER_START) == 0;
233 }
234
235 if (cfg.rx_tx_mode == RX_MODE) {
236 return i2s_trigger(i2s_dev, I2S_DIR_RX, I2S_TRIGGER_START) == 0;
237 }
238
239 int result = i2s_trigger(i2s_dev, I2S_DIR_BOTH, I2S_TRIGGER_START);
240 if (result == 0) return true;
241 if (result != -ENOSYS) return false;
242
243 return i2s_trigger(i2s_dev, I2S_DIR_TX, I2S_TRIGGER_START) == 0 &&
244 i2s_trigger(i2s_dev, I2S_DIR_RX, I2S_TRIGGER_START) == 0;
245 }
