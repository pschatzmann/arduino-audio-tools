63class FIR : public Filter<T> {
64 public:
65 template <size_t B>
66 FIR(const T (&b)[B], const T factor = 1.0) : lenB(B), factor(factor) {
67 setValues(b);
68 }
69
70 template <size_t B>
71 void setValues(const T (&b)[B]) {
72 x.resize(lenB);
73 coeff_b.resize(2 * lenB - 1);
74 for (uint16_t i = 0; i < 2 * lenB - 1; i++) {
75 coeff_b[i] = b[(2 * lenB - 1 - i) % lenB];
76 }
77 }
78
79 T process(T value) {
80 x[i_b] = value;
81 T b_terms = 0;
82 T *b_shift = &coeff_b[lenB - i_b - 1];
83 for (uint8_t i = 0; i < lenB; i++) {
84 b_terms += b_shift[i] * x[i];
85 }
86 i_b++;
87 if (i_b == lenB) i_b = 0;
88
89#ifdef USE_TYPETRAITS
90 if (!(std::is_same<T, float>::value || std::is_same<T, float>::value)) {
91 b_terms = b_terms / factor;
92 }
93#else
94 if (factor != 1.0) {
95 b_terms = b_terms / factor;
96 }
97#endif
98 return b_terms;
99 }
100
101 private:
102 const uint8_t lenB;
103 uint8_t i_b = 0;
104 Vector<T> x;
105 Vector<T> coeff_b;

117template <typename T>
118class IIR : public Filter<T> {
119 public:
120 template <size_t B, size_t A>
121 IIR(const T (&b)[B], const T (&_a)[A], T factor = 1.0)
122 : factor(factor), lenB(B), lenA(A - 1) {
123 x.resize(lenB);
124 y.resize(lenA);
125 coeff_b.resize(2 * lenB - 1);
126 coeff_a.resize(2 * lenA - 1);
127 T a0 = _a[0];
128 const T *a = &_a[1];
129 for (uint16_t i = 0; i < 2 * lenB - 1; i++) {
130 coeff_b[i] = b[(2 * lenB - 1 - i) % lenB] / a0;
131 }
132 for (uint16_t i = 0; i < 2 * lenA - 1; i++) {
133 coeff_a[i] = a[(2 * lenA - 2 - i) % lenA] / a0;
134 }
135 }
136
137
138 T process(T value) {
139 x[i_b] = value;
140 T b_terms = 0;
141 T *b_shift = &coeff_b[lenB - i_b - 1];
142
143 T a_terms = 0;
144 T *a_shift = &coeff_a[lenA - i_a - 1];
145
146 for (uint8_t i = 0; i < lenB; i++) {
147 b_terms += x[i] * b_shift[i];
148 }
149 for (uint8_t i = 0; i < lenA; i++) {
150 a_terms += y[i] * a_shift[i];
151 }
152
153 T filtered = b_terms - a_terms;
154 y[i_a] = filtered;
155 i_b++;
156 if (i_b == lenB) i_b = 0;
157 i_a++;
158 if (i_a == lenA) i_a = 0;
159
160#ifdef USE_TYPETRAITS
161 if (!(std::is_same<T, float>::value || std::is_same<T, float>::value)) {
162 filtered = filtered / factor;
163 }
164#else
165 if (factor != 1.0) {
166 filtered = filtered / factor;
167 }
168#endif
169 return filtered;
170 }
171
172 private:
173 T factor;
174 const uint8_t lenB, lenA;
175 uint8_t i_b = 0, i_a = 0;
176 Vector<T> x;
177 Vector<T> y;
178 Vector<T> coeff_b;
179 Vector<T> coeff_a;
180};

444template <typename T>
445class LowShelfFilter : public BiQuadDF2<T> {
446 public:
447 LowShelfFilter() = default;
448 LowShelfFilter(float frequency, float sampleRate, float gain,
449 float slope = 1.0f)
450 : BiQuadDF2<T>() {
451 begin(frequency, sampleRate, gain, slope);
452 }
453
454 void begin(float frequency, float sampleRate, float gain,
455 float slope = 1.0f) {
456 T a = pow(10.0, gain / 40.0f);
457 T w0 = frequency * (2.0f * PI / sampleRate);
458 T sinW0 = sin(w0);
459 // float alpha = (sinW0 * sqrt((a+1/a)*(1/slope-1)+2) ) / 2.0;
460 T cosW0 = cos(w0);
461 // generate three helper-values (intermediate results):
462 T sinsq = sinW0 *
463 sqrt((pow(a, 2.0) + 1.0) * (1.0 / (float)slope - 1.0) + 2.0 * a);
464 T aMinus = (a - 1.0) * cosW0;
465 T aPlus = (a + 1.0) * cosW0;
466 T scale = 1.0 / ((a + 1.0) + aMinus + sinsq);
467 BiQuadDF2<T>::b_0 = a * ((a + 1.0) - aMinus + sinsq) * scale;
468 BiQuadDF2<T>::b_1 = 2.0 * a * ((a - 1.0) - aPlus) * scale;
469 BiQuadDF2<T>::b_2 = a * ((a + 1.0) - aMinus - sinsq) * scale;
470 BiQuadDF2<T>::a_1 = -2.0 * ((a - 1.0) + aPlus) * scale;
471 BiQuadDF2<T>::a_2 = ((a + 1.0) + aMinus - sinsq) * scale;
472 }
473};
