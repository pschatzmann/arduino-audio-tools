21class ParseBuffer {
22public:
23 size_t writeArray(uint8_t *data, size_t len) {
24 int to_write = min(availableToWrite(), (size_t)len);
25 memmove(vector.data() + available_byte_count, data, to_write);
26 available_byte_count += to_write;
27 return to_write;
28 }
29 void consume(int size) {
30 memmove(vector.data(), &vector[size], available_byte_count - size);
31 available_byte_count -= size;
32 }
33 void resize(int size) { vector.resize(size + 4); }
34
35 uint8_t *data() { return vector.data(); }
36
37 size_t availableToWrite() { return size() - available_byte_count; }
38
39 size_t available() { return available_byte_count; }
40
41 void clear() {
42 available_byte_count = 0;
43 memset(vector.data(), 0, vector.size());
44 }
45
46 bool isEmpty() { return available_byte_count == 0; }
47
48 size_t size() { return vector.size(); }
49
50 long indexOf(const char *str) {
51 uint8_t *ptr = (uint8_t *)memmem(vector.data(), available_byte_count, str,
52 strlen(str));
53 return ptr == nullptr ? -1l : ptr - vector.data();
54 }
55
56protected:
57 Vector<uint8_t> vector{0};
58 size_t available_byte_count = 0;
59};

392 // we return true if at least one parse step was successful
393 bool parse() {
394 bool result = true;
395 switch (parse_state) {
396 case ParseHeader: {
397 result = parseHeader();
398 if (result)
399 parse_state = ParseHdrl;
400 } break;
401
402 case ParseHdrl: {
403 ParseObject hdrl = parseList("hdrl");
404 result = hdrl.isValid();
405 if (result) {
406 parse_state = ParseAvih;
407 }
408 } break;
409
410 case ParseAvih: {
411 ParseObject avih = parseChunk("avih");
412 result = avih.isValid();
413 if (result) {
414 main_header = *(avih.asAVIMainHeader(parse_buffer.data()));
415 stream_header.resize(main_header.dwStreams);
416 consume(avih.size());
417 parse_state = ParseStrl;
418 }
419 } break;
420
421 case ParseStrl: {
422 ParseObject strl = parseList("strl");
423 ParseObject strh = parseChunk("strh");
424 stream_header[++stream_header_idx] =
425 *(strh.asAVIStreamHeader(parse_buffer.data()));
426 consume(strh.size());
427 parse_state = ParseStrf;
428 } break;
429
430 case ParseStrf: {
431 ParseObject strf = parseChunk("strf");
432 if (isCurrentStreamAudio()) {
433 audio_info = *(strf.asAVIAudioFormat(parse_buffer.data()));
434 setupAudioInfo();
435 LOGI("audioFormat: %d (%x)", (int)audioFormat(),(int)audioFormat());
436 content_types.push_back(Audio);
437 consume(strf.size());
438 } else if (isCurrentStreamVideo()) {
439 video_info = *(strf.asAVIVideoFormat(parse_buffer.data()));
440 setupVideoInfo();
441 LOGI("videoFormat: %s", videoFormat());
442 content_types.push_back(Video);
443 video_format[4] = 0;
444 consume(strf.size());
445 } else {
446 result = false;
447 }
448 parse_state = AfterStrf;
449 } break;
450
451 case AfterStrf: {
452 // ignore all data until we find a new List
453 int pos = parse_buffer.indexOf("LIST");
454 if (pos >= 0) {
455 consume(pos);
456 ParseObject tmp = tryParseList();
457 if (StrView(tmp.id()).equals("strl")) {
458 parse_state = ParseStrl;
459 } else if (StrView(tmp.id()).equals("movi")) {
460 parse_state = ParseMovi;
461 } else {
462 // e.g. ignore info
463 consume(tmp.size() + LIST_HEADER_SIZE);
464 }
465 } else {
466 // no valid data, so throw it away, we keep the last 4 digits in case
467 // if it contains the beginning of a LIST
468 cleanupStack();
469 consume(parse_buffer.available() - 4);
470 }
471 } break;
472
473 case ParseMovi: {
474 ParseObject movi = tryParseList();
475 if (StrView(movi.id()).equals("movi")) {
476 consume(LIST_HEADER_SIZE);
477 is_metadata_ready = true;
478 if (validation_cb)
479 is_parsing_active = (validation_cb(*this));
480 processStack(movi);
481 movi_end_pos = movi.end_pos;
482 parse_state = SubChunk;
483 // trigger new write
484 result = false;
485 }
486 } break;
487
488 case SubChunk: {
489 // rec is optinal
490 ParseObject hdrl = tryParseList();
491 if (StrView(hdrl.id()).equals("rec")) {
492 consume(CHUNK_HEADER_SIZE);
493 processStack(hdrl);
494 }
495
496 current_stream_data = parseAVIStreamData();
497 parse_state = SubChunkContinue;
498 open_subchunk_len = current_stream_data.open;
499 if (current_stream_data.isVideo()) {
500 LOGI("video:[%d]->[%d]", (int)current_stream_data.start_pos,
501 (int)current_stream_data.end_pos);
502 if (p_output_video != nullptr)
503 p_output_video->beginFrame(current_stream_data.open);
504 } else if (current_stream_data.isAudio()) {
505 LOGI("audio:[%d]->[%d]", (int)current_stream_data.start_pos,
506 (int)current_stream_data.end_pos);
507 } else {
508 LOGW("unknown subchunk at %d", (int)current_pos);
509 }
510
511 } break;
512
513 case SubChunkContinue: {
514 writeData();
515 if (open_subchunk_len == 0) {
516 if (current_stream_data.isVideo() && p_output_video != nullptr) {
517 uint32_t time_used_ms = p_output_video->endFrame();
518 p_synch->delayVideoFrame(main_header.dwMicroSecPerFrame, time_used_ms);
519 }
520 if (tryParseChunk("idx").isValid()) {
521 parse_state = ParseIgnore;
522 } else if (tryParseList("rec").isValid()) {
523 parse_state = ParseRec;
524 } else {
525 if (current_pos >= movi_end_pos) {
526 parse_state = ParseIgnore;
527 } else {
528 parse_state = SubChunk;
529 }
530 }
531 }
532 } break;
533
534 case ParseIgnore: {
535 LOGD("ParseIgnore");
536 parse_buffer.clear();
537 } break;
538
539 default:
540 result = false;
541 break;
542 }
543 return result;
544 }

679 void processStack(ParseObject &result) {
680 cleanupStack();
681 object_stack.push(result);
682 spaces.setChars(' ', object_stack.size());
683 LOGD("%s - %s (%d-%d) size:%d", spaces.c_str(), result.id(),
684 (int)result.start_pos, (int)result.end_pos, (int)result.data_size);
685 }

687 void cleanupStack() {
688 ParseObject current;
689 // make sure that we remove the object from the stack of we past the end
690 object_stack.peek(current);
691 while (current.end_pos <= current_pos) {
692 object_stack.pop(current);
693 object_stack.peek(current);
694 }
695 }
