70/* @brief Addition in Galois Fields
71 * @param x - left operand
72 * @param y - right operand
73 * @return x + y */
74inline uint8_t add(uint8_t x, uint8_t y) {
75 return x^y;
76}

79/* @brief Subtraction in Galois Fields
80 * @param x - left operand
81 * @param y - right operand
82 * @return x - y */
83inline uint8_t sub(uint8_t x, uint8_t y) {
84 return x^y;

87/* @brief Multiplication in Galois Fields
88 * @param x - left operand
89 * @param y - right operand
90 * @return x * y */
91inline uint8_t mul(uint16_t x, uint16_t y){
92 if (x == 0 || y == 0)
93 return 0;
94 return exp[(log[x] + log[y]) % 255];
95}
96
97/* @brief Division in Galois Fields
98 * @param x - dividend
99 * @param y - divisor
100 * @return x / y */
101inline uint8_t div(uint8_t x, uint8_t y){
102 assert(y != 0);
103 if(x == 0) return 0;
104 return exp[(log[x] + 255 - log[y]) % 255];
105}
106
107/* @brief X in power Y w
108 * @param x - operand
109 * @param power - power
110 * @return x^power */
111inline uint8_t pow(uint8_t x, intmax_t power){
112 intmax_t i = log[x];
113 i *= power;
114 i %= 255;
115 if(i < 0) i = i + 255;
116 return exp[i % 255];
117}
118
119/* @brief Inversion in Galois Fields
120 * @param x - number
121 * @return inversion of x */
122inline uint8_t inverse(uint8_t x){
123 return exp[(255 - log[x]) % 255]; /* == div(1, x); */
124}

130/* @brief Multiplication polynomial by scalar
131 * @param &p - source polynomial
132 * @param &newp - destination polynomial
133 * @param x - scalar */
134inline void
135poly_scale(const Poly *p, Poly *newp, uint16_t x) {
136 newp->length = p->length;
137 for(uint16_t i = 0; i < p->length; i++){
138 newp->at(i) = mul(p->at(i), x);
139 }
140}

142/* @brief Addition of two polynomials
143 * @param &p - right operand polynomial
144 * @param &q - left operand polynomial
145 * @param &newp - destination polynomial */
146inline void
147poly_add(const Poly *p, const Poly *q, Poly *newp) {
148 newp->length = poly_max(p->length, q->length);
149 memset(newp->ptr(), 0, newp->length * sizeof(uint8_t));
150
151 for(uint8_t i = 0; i < p->length; i++){
152 newp->at(i + newp->length - p->length) = p->at(i);
153 }
154
155 for(uint8_t i = 0; i < q->length; i++){
156 newp->at(i + newp->length - q->length) ^= q->at(i);
157 }
158}

161/* @brief Multiplication of two polynomials
162 * @param &p - right operand polynomial
163 * @param &q - left operand polynomial
164 * @param &newp - destination polynomial */
165inline void
166poly_mul(const Poly *p, const Poly *q, Poly *newp) {
167 newp->length = p->length + q->length - 1;
168 memset(newp->ptr(), 0, newp->length * sizeof(uint8_t));
169 /* Compute the polynomial multiplication (just like the outer product of two vectors,
170 * we multiply each coefficients of p with all coefficients of q) */
171 for(uint8_t j = 0; j < q->length; j++){
172 for(uint8_t i = 0; i < p->length; i++){
173 newp->at(i+j) ^= mul(p->at(i), q->at(j)); /* == r[i + j] = gf_add(r[i+j], gf_mul(p[i], q[j])) */
174 }
175 }
176}

178/* @brief Division of two polynomials
179 * @param &p - right operand polynomial
180 * @param &q - left operand polynomial
181 * @param &newp - destination polynomial */
182inline void
183poly_div(const Poly *p, const Poly *q, Poly *newp) {
184 if(p->ptr() != newp->ptr()) {
185 memcpy(newp->ptr(), p->ptr(), p->length*sizeof(uint8_t));
186 }
187
188 newp->length = p->length;
189
190 uint8_t coef;
191
192 for(int i = 0; i < (p->length-(q->length-1)); i++){
193 coef = newp->at(i);
194 if(coef != 0){
195 for(uint8_t j = 1; j < q->length; j++){
196 if(q->at(j) != 0)
197 newp->at(i+j) ^= mul(q->at(j), coef);
198 }
199 }
200 }
201
202 size_t sep = p->length-(q->length-1);
203 memmove(newp->ptr(), newp->ptr()+sep, (newp->length-sep) * sizeof(uint8_t));
204 newp->length = newp->length-sep;
205}

207/* @brief Evaluation of polynomial in x
208 * @param &p - polynomial to evaluate
209 * @param x - evaluation point */
210inline int8_t
211poly_eval(const Poly *p, uint16_t x) {
212 uint8_t y = p->at(0);
213 for(uint8_t i = 1; i < p->length; i++){
214 y = mul(y, x) ^ p->at(i);
215 }
216 return y;
217}
