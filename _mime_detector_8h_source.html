78 static bool checkAACExt(uint8_t* start, size_t len) {
79 // checking logic for files
80 if (memcmp(start + 4, "ftypM4A", 7) == 0) {
81 return true;
82 }
83 // check for streaming
84 HeaderParserAAC aac;
85 // it should start with a synch word

87 if (pos == -1) {
88 return false;
89 }
90 // make sure that it is not an mp3
91 if (aac.isValid(start + pos, len - pos)) {
92 return false;
93 }

116 if (len < 4) return false;

125 static bool checkOggFLAC(uint8_t* start, size_t len) {
126 // Check for OGG FLAC - OGG container with FLAC content
127 // OGG starts with "OggS" and may contain FLAC codec
128 if (len >= 32 && memcmp(start, "OggS", 4) == 0) {
129 // Look for FLAC signature within the first OGG page
130 // FLAC in OGG typically has "\x7FFLAC" or "FLAC" as codec identifier
131 for (size_t i = 4; i < len - 4 && i < 64; i++) {
132 if (memcmp(start + i, "FLAC", 4) == 0) {
133 return true;
134 }
135 // Also check for the more specific OGG FLAC header
136 if (i < len - 5 && start[i] == 0x7F && memcmp(start + i + 1, "FLAC", 4) == 0) {
137 return true;
138 }
139 }
140 }
141
142 return false;
143 }

157 static bool checkM4A(uint8_t* header, size_t len) {
158 if (len < 12) return false;
159
160 // prevent false detecton by mp3 files
161 if (memcmp(header, "ID3", 3) == 0) return false;
162
163 // Special hack when we position to start of mdat box
164 if (memcmp(header + 4, "mdat", 4) != 0) return true;
165
166 // Check for "ftyp" at offset 4
167 if (memcmp(header + 4, "ftyp", 4) != 0) return false;
168
169 // Check for "M4A " or similar major brand
170 if (memcmp(header + 8, "M4A ", 4) == 0 ||
171 memcmp(header + 8, "mp42", 4) == 0 ||
172 memcmp(header + 8, "isom", 4) == 0)
173 return true;
174
175 return false;
176 }
