128 static bool checkAACExt(uint8_t* start, size_t len) {
129 // checking logic for files
130 if (memcmp(start + 4, "ftypM4A", 7) == 0) {
131 return true;
132 }
133 // check for streaming
134 HeaderParserAAC aac;
135 // it should start with a synch word
136 int pos = aac.findSyncWord((const uint8_t*)start, len);
137 if (pos == -1) {
138 return false;
139 }
140 // make sure that it is not an mp3
141 if (aac.isValid(start + pos, len - pos)) {
142 return false;
143 }
144 return true;
145 }

175 static bool checkOggFLAC(uint8_t* start, size_t len) {
176 // Check for OGG FLAC - OGG container with FLAC content
177 // OGG starts with "OggS" and may contain FLAC codec
178 if (len >= 32 && memcmp(start, "OggS", 4) == 0) {
179 // Look for FLAC signature within the first OGG page
180 // FLAC in OGG typically has "\x7FFLAC" or "FLAC" as codec identifier
181 for (size_t i = 4; i < len - 4 && i < 64; i++) {
182 if (memcmp(start + i, "FLAC", 4) == 0) {
183 return true;
184 }
185 // Also check for the more specific OGG FLAC header
186 if (i < len - 5 && start[i] == 0x7F &&
187 memcmp(start + i + 1, "FLAC", 4) == 0) {
188 return true;
189 }
190 }
191 }
192
193 return false;
194 }

262 static bool checkM4A(uint8_t* header, size_t len) {
263 if (len < 12) return false;
264
265 // prevent false detecton by mp3 files
266 if (memcmp(header, "ID3", 3) == 0) return false;
267
268 // Special hack when we position to start of mdat box
269 if (memcmp(header + 4, "mdat", 4) != 0) return true;
270
271 // Check for "ftyp" at offset 4
272 if (memcmp(header + 4, "ftyp", 4) != 0) return false;
273
274 // Check for "M4A " or similar major brand
275 if (memcmp(header + 8, "M4A ", 4) == 0 ||
276 memcmp(header + 8, "mp42", 4) == 0 ||
277 memcmp(header + 8, "isom", 4) == 0)
278 return true;

280 return false;

334 const char* lookupMime(uint8_t* data, size_t len) {
335 for (int j = 0; j < checks.size(); j++) {
336 Check l_check = checks[j];
