97  inline size_t copyBytes(size_t bytes){
98  LOGD("copy %d bytes %s", (int) bytes, log_name);
99  if (!active) return 0;
100  // if not initialized we do nothing
101  if (from==nullptr && to==nullptr) return 0;
102 
103  // synchronize AudioInfo
104  syncAudioInfo();
105 
106  // E.g. if we try to write to a server we might not have any output destination yet
107  int to_write = to->availableForWrite();
108  if (check_available_for_write && to_write==0){
109  delay(500);
110  return 0;
111  }
112 
113  // resize copy buffer if necessary
114  if (buffer.size() < bytes){
115  LOGI("Resize to %d", (int) bytes);
116  buffer.resize(bytes);
117  }
118 
119  size_t result = 0;
120  size_t delayCount = 0;
121  size_t len = bytes;
122  if (check_available) {
123  len = available();
124  }
125  size_t bytes_to_read = bytes;
126  size_t bytes_read = 0;
127 
128  if (len > 0){
129  bytes_to_read = min(len, static_cast<size_t>(buffer_size));
130  // don't overflow buffer
131  if (to_write > 0){
132  bytes_to_read = min((int)bytes_to_read, to_write);
133  }
134 
135  // round to full frames
136  int copy_size = minCopySize();
137  if (copy_size > 0){
138  size_t samples = bytes_to_read / minCopySize();
139  bytes_to_read = samples * minCopySize();
140  }
141 
142  // get the data now
143  bytes_read = 0;
144  if (bytes_to_read>0){
145  bytes_read = from->readBytes((uint8_t*)&buffer[0], bytes_to_read);
146  }
147 
148  // determine mime
149  notifyMime(buffer.data(), bytes_to_read);
150 
151  // write data
152  result = write(bytes_read, delayCount);
153 
154  // callback with unconverted data
155  if (onWrite!=nullptr) onWrite(onWriteObj, &buffer[0], result);
156 
157  #ifndef COPY_LOG_OFF
158  LOGI("StreamCopy::copy %s %u -> %u -> %u bytes - in %u hops",log_name, (unsigned int)bytes_to_read,(unsigned int) bytes_read, (unsigned int)result, (unsigned int)delayCount);
159  #endif
160  //TRACED();
161 
162  if (result == 0){
163  TRACED();
164  // give the processor some time
165  delay(delay_on_no_data);
166  }
167 
168  //TRACED();
169  CHECK_MEMORY();
170  } else {
171  // give the processor some time
172  delay(delay_on_no_data);
173  LOGD("no data %s", log_name);
174  }
175  //TRACED();
176  return result;
177  }

382  size_t write(size_t len, size_t &delayCount ){
383  if (!buffer || len==0) return 0;
384  LOGD("write: %d", (int)len);
385  size_t total = 0;
386  long open = len;
387  int retry = 0;
388  while(open > 0){
389  size_t written = to->write((const uint8_t*)buffer.data()+total, open);
390  LOGD("write: %d -> %d", (int) open, (int) written);
391  total += written;
392  open -= written;
393  delayCount++;
394 
395  if (open > 0){
396  // if we still have progress we reset the retry counter
397  if (written>0) retry = 0;
398 
399  // abort if we reached the retry limit
400  if (retry++ > retryLimit){
401  LOGE("write %s to target has failed after %d retries! (%ld bytes)", log_name, retry, open);
402  break;
403  }
404 
405  // wait a bit
406  if (retry>1) {
407  delay(retry_delay);
408  LOGI("try write %s - %d (open %ld bytes) ",log_name, retry, open);
409  }
410  }
411 
412  CHECK_MEMORY();
413  }
414  return total;
415  }

419  if (is_first && len>4) {

421  actual_mime = "audio/basic";

427  actual_mime = "audio/vnd.wave";
