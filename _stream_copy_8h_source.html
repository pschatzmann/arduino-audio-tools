109 inline size_t copyBytes(size_t bytes){
110 LOGD("copy %d bytes %s", (int) bytes, log_name);
111 if (!active) return 0;
112 // if not initialized we do nothing
113 if (from==nullptr && to==nullptr) return 0;
114
115 // synchronize AudioInfo
116 syncAudioInfo();
117
118 // E.g. if we try to write to a server we might not have any output destination yet
119 int to_write = to->availableForWrite();
120 if (check_available_for_write && to_write==0){
121 delay(500);
122 return 0;
123 }
124
125 // resize copy buffer if necessary
126 if (buffer.size() < bytes){
127 LOGI("Resize to %d", (int) bytes);
128 buffer.resize(bytes);
129 }
130
131 size_t result = 0;
132 size_t delayCount = 0;
133 size_t len = bytes;
134 if (check_available) {
135 len = available();
136 }
137 size_t bytes_to_read = bytes;
138 size_t bytes_read = 0;
139
140 if (len > 0){
141 bytes_to_read = min(len, static_cast<size_t>(buffer_size));
142 // don't overflow buffer
143 if (to_write > 0){
144 bytes_to_read = min((int)bytes_to_read, to_write);
145 }
146
147 // round to full frames
148 int copy_size = minCopySize();
149 if (copy_size > 0){
150 size_t samples = bytes_to_read / minCopySize();
151 bytes_to_read = samples * minCopySize();
152 }
153
154 // get the data now
155 bytes_read = 0;
156 if (bytes_to_read>0){
157 bytes_read = from->readBytes((uint8_t*)&buffer[0], bytes_to_read);
158 }
159
160 // determine mime
161 notifyMime(buffer.data(), bytes_to_read);
162
163 // convert data
164 if (p_converter!=nullptr) p_converter->convert((uint8_t*)buffer.data(), bytes_read );
165
166 // write data
167 result = write(bytes_read, delayCount);
168
169 // callback with unconverted data
170 if (onWrite!=nullptr) onWrite(onWriteObj, &buffer[0], result);
171
172 #ifndef COPY_LOG_OFF
173 LOGI("StreamCopy::copy %s %u -> %u -> %u bytes - in %u hops",log_name, (unsigned int)bytes_to_read,(unsigned int) bytes_read, (unsigned int)result, (unsigned int)delayCount);
174 #endif
175 //TRACED();
176
177 if (result == 0){
178 TRACED();
179 // give the processor some time
180 delay(delay_on_no_data);
181 }
182
183 //TRACED();
184 CHECK_MEMORY();
185 } else {
186 // give the processor some time
187 delay(delay_on_no_data);
188 LOGD("no data %s", log_name);
189 }
190 //TRACED();
191 return result;
192 }

212 size_t copyAll(int retryCount=5, int retryWaitMs=200){
213 TRACED();
214 if (!active) return 0;
215 size_t result = 0;
216 int retry = 0;
217
218 if (from==nullptr || to == nullptr)
219 return result;
220
221 // copy while source has data available
222 int count=0;
223 while (true){
224 count = copy();
225 result += count;
226 if (count==0){
227 // wait for more data
228 retry++;
229 delay(retryWaitMs);
230 } else {
231 retry = 0; // after we got new data we restart the counting
232 }
233 // stop the processing if we passed the retry limit
234 if (retry>retryCount){
235 break;
236 }
237 }
238 return result;
239 }

400 size_t write(size_t len, size_t &delayCount ){
401 if (!buffer || len==0) return 0;
402 LOGD("write: %d", (int)len);
403 size_t total = 0;
404 long open = len;
405 int retry = 0;
406 while(open > 0){
407 size_t written = to->write((const uint8_t*)buffer.data()+total, open);
408 LOGD("write: %d -> %d", (int) open, (int) written);
409 total += written;
410 open -= written;
411 delayCount++;
412
413 if (open > 0){
414 // if we still have progress we reset the retry counter
415 if (written>0) retry = 0;
416
417 // abort if we reached the retry limit
418 if (retry++ > retryLimit){
419 LOGE("write %s to target has failed after %d retries! (%ld bytes)", log_name, retry, open);
420 break;
421 }
422
423 // wait a bit
424 if (retry>1) {
425 delay(retry_delay);
426 LOGI("try write %s - %d (open %ld bytes) ",log_name, retry, open);
427 }
428 }
429
430 CHECK_MEMORY();
431 }
432 return total;
433 }

436 void notifyMime(void* data, size_t len){
437 if (is_first && len>4) {
438 const uint8_t *start = (const uint8_t *) data;
439 actual_mime = "audio/basic";
440 if (start[0]==0xFF && start[1]==0xF1){
441 actual_mime = "audio/aac";
442 } else if (memcmp(start,"ID3",3) || start[0]==0xFF || start[0]==0xFE ){
443 actual_mime = "audio/mpeg";
444 } else if (memcmp(start,"RIFF",4)){
445 actual_mime = "audio/vnd.wave";
446 }
447 if (notifyMimeCallback!=nullptr){
448 notifyMimeCallback(actual_mime);
449 }
450 }
451 is_first = false;
452 }
