57 virtual bool begin() {
58 streamer->initAudioSource();

82 protected:
83 int port = 554;
84 typename Platform::TcpServerType* server = nullptr;
85 typename Platform::TcpClientType client;
86 RTSPAudioStreamerBase<Platform>* streamer = nullptr;
87 int client_count = 0;
88 RtspSession<Platform>* rtspSession = nullptr;
89 bool (*onSessionPathCb)(const char*, void*) = nullptr;
90 void* onSessionPathRef = nullptr;
91 unsigned long sessionTimeoutMs = 60000; // 60 seconds
92 unsigned long lastRequestTime = 0;
93
94 // Accept new client if none is active
95 void acceptClient() {
96 if (client_count == 0 && server) {
97 auto newClient = Platform::getAvailableClient(server);
98 if (newClient.connected()) {
99 client = newClient;
100 client_count++;
101 // Create session
102 rtspSession = new RtspSession<Platform>(client, *streamer);
103 if (onSessionPathCb) {
104 rtspSession->setOnSessionPath(onSessionPathCb, onSessionPathRef);
105 }
106 lastRequestTime = millis();
107 }
108 }
109 }
110
111 // Handle session if active
112 void handleSession() {
113 if (client_count > 0 && rtspSession) {
114 uint32_t timeout = 30;
115 bool gotRequest = rtspSession->handleRequests(timeout);
116 if (gotRequest) {
117 lastRequestTime = millis();
118 }
119 // If streaming, check for session timeout
120 if (sessionTimeoutMs > 0 && rtspSession->isStreaming()) {
121 if ((millis() - lastRequestTime) > sessionTimeoutMs) {
122 // Timeout, mark session closed
123 rtspSession->closeSession();
124 /* If you want to trigger cleanup, you could also call:
125 rtspSession->handleRtspTeardown();
126 (if public, or expose a public method for teardown)
127 */
128 }
129 }
130 // Clean up if session closed
131 if (!rtspSession->isSessionOpen()) {
132 delete rtspSession;
133 rtspSession = nullptr;
134 if (client.connected()) {
135 Platform::closeSocket(&client);
136 }
137 client_count--;
138 }
139 }
140 }
