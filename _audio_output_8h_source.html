606 inline void mixSamples(size_t samples) {
607#if defined(USE_ESP32_DSP)
608 // Temporary float buffers for mixing
609 float mix_out[samples] = {0.0f};
610 float temp[samples] = {0.0f};
611
612 for (uint8_t j = 0; j < output_count; j++) {
613 const float factor = weights[j] / total_weights;
614 // Read int16_t samples and convert to float
615 for (uint16_t i = 0; i < samples; i++) {
616 T s = 0;
617 buffers[j]->read(s);
618 temp[i] = static_cast<float>(s) * factor;
619 }
620 // Add to output
621 dsps_add_f32(mix_out, temp, mix_out, samples, 1, 1, 1);
622 }
623 // Convert back to int16_t with clamping
624 output.resize(samples);
625 for (size_t i = 0; i < samples; i++) {
626 float v = mix_out[i];
627 output[i] = static_cast<T>(v);
628 }
629#else
630 // Fallback: original scalar code
631 output.resize(samples);
632 memset(output.data(), 0, samples * sizeof(T));
633 for (int j = 0; j < output_count; j++) {
634 float factor = weights[j] / total_weights;
635 for (int i = 0; i < samples; i++) {
636 T sample = 0;
637 buffers[j]->read(sample);
638 output[i] += static_cast<T>(factor * sample);
639 }
640 }
641#endif
642 }
