198 } else {
199 out_ptr->print(delimiter_str);

201 } else {
202 LOGE("Unsupported size: %d for channels %d and bits: %d", (int)len,

714 size_t write(const uint8_t *data, size_t len) override {
715 switch (cfg.bits_per_sample) {
716 case 16:
717 return writeT<int16_t>(data, len);
718 case 24:
719 return writeT<int24_t>(data, len);
720 case 32:
721 return writeT<int32_t>(data, len);
722 default:
723 return 0;
724 }
725 }
726
727 protected:
728 struct ChannelSelectionOutputDef {
729 Print *p_out = nullptr;
730 int channel;
731 };
732 Vector<ChannelSelectionOutputDef> out_channels;
733
734 template <typename T = int16_t>
735 size_t writeT(const uint8_t *buffer, size_t size) {
736 int sample_count = size / sizeof(T);
737 int result_size = sample_count / cfg.channels;
738 T *data = (T *)buffer;
739 T result[result_size];
740
741 for (int ch = 0; ch < out_channels.size(); ch++) {
742 ChannelSelectionOutputDef &def = out_channels[ch];
743 // extract mono result
744 int i = 0;
745 for (int j = def.channel; j < sample_count; j += cfg.channels) {
746 result[i++] = data[j];
747 }
748 // write mono result
749 size_t written =
750 def.p_out->write((uint8_t *)result, result_size * sizeof(T));
751 if (written != result_size * sizeof(T)) {
752 LOGW("Could not write all samples");
753 }
754 }
755 return size;
756 }
