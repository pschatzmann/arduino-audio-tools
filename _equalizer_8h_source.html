161 void filterSamples(const uint8_t *data, size_t len) {
162 if (state == nullptr){
163 LOGE("You need to call begin() before using the equalizer");
164 return;
165 }
166 switch (p_cfg->bits_per_sample) {
167 case 16: {
168 int16_t *p_dataT = (int16_t *)data;
169 size_t sample_count = len / sizeof(int16_t);
170 for (size_t j = 0; j < sample_count; j += p_cfg->channels) {
171 for (int ch = 0; ch < p_cfg->channels; ch++) {
172 p_dataT[j + ch] = NumberConverter::fromFloat(sample(state[ch], NumberConverter::toFloat(p_dataT[j + ch], 16)), 16);
173 }
174 }
175 } break;
176 case 24: {
177 int24_t *p_dataT = (int24_t *)data;
178 size_t sample_count = len / sizeof(int24_t);
179 for (size_t j = 0; j < sample_count; j += p_cfg->channels) {
180 for (int ch = 0; ch < p_cfg->channels; ch++) {
181 p_dataT[j + ch] = NumberConverter::fromFloat(sample(state[ch], NumberConverter::toFloat(p_dataT[j + ch], 24)), 24);
182 }
183 }
184 } break;
185 case 32: {
186 int32_t *p_dataT = (int32_t *)data;
187 size_t sample_count = len / sizeof(int32_t);
188 for (size_t j = 0; j < sample_count; j += p_cfg->channels) {
189 for (int ch = 0; ch < p_cfg->channels; ch++) {
190 p_dataT[j + ch] = NumberConverter::fromFloat(sample(state[ch], NumberConverter::toFloat(p_dataT[j + ch], 32)), 32);
191 }
192 }
193 } break;
194
195 default:
196 LOGE("Only 16 bits supported: %d", p_cfg->bits_per_sample);
197 break;
198 }
199 }

202 // calculates a single sample using the indicated state
203 float sample(EQSTATE &es, float sample) {
204 // Locals
205 float l, m, h; // Low / Mid / High - Sample Values
206 // Filter #1 (lowpass)
207 es.f1p0 += (es.lf * (sample - es.f1p0)) + vsa;
208 es.f1p1 += (es.lf * (es.f1p0 - es.f1p1));
209 es.f1p2 += (es.lf * (es.f1p1 - es.f1p2));
210 es.f1p3 += (es.lf * (es.f1p2 - es.f1p3));
211
212 l = es.f1p3;
213
214 // Filter #2 (highpass)
215 es.f2p0 += (es.hf * (sample - es.f2p0)) + vsa;
216 es.f2p1 += (es.hf * (es.f2p0 - es.f2p1));
217 es.f2p2 += (es.hf * (es.f2p1 - es.f2p2));
218 es.f2p3 += (es.hf * (es.f2p2 - es.f2p3));
219
220 h = es.sdm3 - es.f2p3;
221 // Calculate midrange (signal - (low + high))
222 m = es.sdm3 - (h + l);
223 // Scale, Combine and store
224 l *= p_cfg->gain_low;
225 m *= p_cfg->gain_medium;
226 h *= p_cfg->gain_high;
227
228 // Shuffle history buffer
229 es.sdm3 = es.sdm2;
230 es.sdm2 = es.sdm1;
231 es.sdm1 = sample;
232
233 // Return result
234 return (l + m + h);
235 }
