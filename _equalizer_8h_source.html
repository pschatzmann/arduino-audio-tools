234 void filterSamples(const uint8_t *data, size_t len) {
235 // no filter if not active
236 if (!is_active) return;
237
238
239 // process samples
240 switch (p_cfg->bits_per_sample) {
241 case 16: {
242 int16_t *p_dataT = (int16_t *)data;
243 size_t sample_count = len / sizeof(int16_t);
244 for (size_t j = 0; j < sample_count; j += p_cfg->channels) {
245 for (int ch = 0; ch < p_cfg->channels; ch++) {
246 p_dataT[j + ch] = NumberConverter::fromFloat(sample(state[ch], NumberConverter::toFloat(p_dataT[j + ch], 16)), 16);
247 }
248 }
249 } break;
250 case 24: {
251 int24_t *p_dataT = (int24_t *)data;
252 size_t sample_count = len / sizeof(int24_t);
253 for (size_t j = 0; j < sample_count; j += p_cfg->channels) {
254 for (int ch = 0; ch < p_cfg->channels; ch++) {
255 p_dataT[j + ch] = NumberConverter::fromFloat(sample(state[ch], NumberConverter::toFloat(p_dataT[j + ch], 24)), 24);
256 }
257 }
258 } break;
259 case 32: {
260 int32_t *p_dataT = (int32_t *)data;
261 size_t sample_count = len / sizeof(int32_t);
262 for (size_t j = 0; j < sample_count; j += p_cfg->channels) {
263 for (int ch = 0; ch < p_cfg->channels; ch++) {
264 p_dataT[j + ch] = NumberConverter::fromFloat(sample(state[ch], NumberConverter::toFloat(p_dataT[j + ch], 32)), 32);
265 }
266 }
267 } break;
268
269 default:
270 LOGE("Only 16 bits supported: %d", p_cfg->bits_per_sample);
271 break;
272 }
273 }

279 float sample(EQSTATE &es, float sample) {
280 // Locals
281 float l, m, h; // Low / Mid / High - Sample Values
282 // Filter #1 (lowpass)
283 es.f1p0 += (es.lf * (sample - es.f1p0)) + vsa;
284 es.f1p1 += (es.lf * (es.f1p0 - es.f1p1));
285 es.f1p2 += (es.lf * (es.f1p1 - es.f1p2));
286 es.f1p3 += (es.lf * (es.f1p2 - es.f1p3));
287
288 l = es.f1p3;
289
290 // Filter #2 (highpass)
291 es.f2p0 += (es.hf * (sample - es.f2p0)) + vsa;
292 es.f2p1 += (es.hf * (es.f2p0 - es.f2p1));
293 es.f2p2 += (es.hf * (es.f2p1 - es.f2p2));
294 es.f2p3 += (es.hf * (es.f2p2 - es.f2p3));
295
296 h = es.sdm3 - es.f2p3;
297 // Calculate midrange (signal - (low + high))
298 m = es.sdm3 - (h + l);
299 // Scale, Combine and store
300 l *= p_cfg->gain_low;
301 m *= p_cfg->gain_medium;
302 h *= p_cfg->gain_high;
303
304 // Shuffle history buffer
305 es.sdm3 = es.sdm2;
306 es.sdm2 = es.sdm1;
307 es.sdm1 = sample;
308
309 // Return result
310 return (l + m + h);
311 }
