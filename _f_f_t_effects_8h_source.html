298 TRACED();
299 FFTBin bin;
300 int max = fft.size();
301
302 if (shift < 0) {
303 // copy bins: left shift
304 for (int n = -shift; n < max; n++) {
305 int to_bin = n + shift;
306 assert(to_bin >= 0);
307 assert(to_bin < max);
308 fft.getBin(n, bin);
309 fft.setBin(to_bin, bin);
310 }
311 // clear tail
312 bin.clear();
313 for (int n = max + shift; n < max; n++) {
314 fft.setBin(n, bin);
315 }
316 } else if (shift > 0) {
317 // copy bins: right shift
318 for (int n = max - shift - 1; n >= 0; n--) {
319 int to_bin = n + shift;
320 assert(to_bin >= 0);
321 assert(to_bin < max);
322 fft.getBin(n, bin);
323 fft.setBin(to_bin, bin);
324 }
325 // clear head
326 bin.clear();
327 for (int n = 0; n < shift; n++) {
328 fft.setBin(n, bin);

448 void setupBands() {
449 float nyquist = audioInfo().sample_rate / 2.0f;
450 float f_min = log10f(20.0f);
451 float f_max = log10f(nyquist);
452 float step = (f_max - f_min) / band_count;
453 int prev_bin = 0;
454 for (int b = 0; b < band_count; b++) {
455 // last band always reaches the last bin, regardless of rounding
456 int edge_bin = (b == band_count - 1)
457 ? p_fft->size()
458 : p_fft->frequencyToBin(
459 (int)powf(10.0f, f_min + step * (b + 1)));
460 band_bin_from[b] = prev_bin;
461 band_bin_to[b] = edge_bin;
462 band_center_freq[b] = powf(10.0f, f_min + step * (b + 0.5f));
463 prev_bin = edge_bin;
464 }
465 }

469 void effect(AudioFFTBase &fft) override {
470 TRACED();
471 FFTBin bin;
472 for (int b = 0; b < band_count; b++) {
473 float gain = gains_linear[b];
474 if (gain == 1.0f) continue; // 0dB: bin is already correct
475 for (int n = band_bin_from[b]; n < band_bin_to[b]; n++) {
476 fft.getBin(n, bin);
477 bin.real *= gain;
478 bin.img *= gain;
479 fft.setBin(n, bin);
480 }
481 }
