256 bool begin(){
257 TRACEI();
258 // 24-bit samples are commonly stored padded to 4 bytes (int24_4bytes_t,
259 // the default int24_t) for alignment/performance, so sizeof(T) won't
260 // equal bits_per_sample/8==3 in that case; accept sizeof(int24_t) too.
261 bool bits_ok = sizeof(T)==info.bits_per_sample/8
262 || (info.bits_per_sample==24 && sizeof(T)==sizeof(int24_t));
263 if (bits_ok){
264 active = true;
265 setupChannels();
266 } else {
267 LOGE("bits_per_sample not consistent: %d",info.bits_per_sample);
268 active = false;
269 }

291 size_t readBytes(uint8_t *data, size_t len) override {
292 if (!active || p_io==nullptr)return 0;
293
294 // read data from source
295 size_t result = p_io->readBytes((uint8_t*)data, len);
296 int frames = result / sizeof(T) / info.channels;
297 T* samples = (T*) data;
298
299 for (int f=0;f<frames;f++){
300 for (int ch=0;ch<info.channels;ch++){
301 T &sample = samples[f*info.channels+ch];
302 AudioEffectCommon &chain = channel_effects[ch];
303 for (int j=0; j<chain.size(); j++){
304 sample = chain[j]->process(sample);
305 }
306 }
307 }
308 return result;

315 size_t write(const uint8_t *data, size_t len) override {
316 if (!active)return 0;
317 // length must be multple of channels
318 assert(len % (sizeof(T)*info.channels)==0);
319 int frames = len / sizeof(T) / info.channels;
320 size_t result_size = 0;
321
322 // process all samples
323 for (int f=0;f<frames;f++){
324 const T* p_buffer = ((const T*)data) + (f*info.channels);
325 for (int ch=0;ch<info.channels;ch++){
326 T sample = p_buffer[ch];
327 AudioEffectCommon &chain = channel_effects[ch];
328 for (int j=0; j<chain.size(); j++){
329 sample = chain[j]->process(sample);
330 }
331
332 if (p_io!=nullptr){
333 p_io->write((uint8_t*)&sample, sizeof(T));
334 } else if (p_print!=nullptr){
335 p_print->write((uint8_t*)&sample, sizeof(T));
336 }
337 result_size += sizeof(T);
338 }
339 }
340 return result_size;
341 }
