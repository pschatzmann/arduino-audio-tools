269 size_t write(const uint8_t* peer, const uint8_t* data, size_t len) {
270 // initialization: setup semaphore
271 setupSemaphore();
272
273 // initialization: if no peers registered and peer is nullptr, add broadcast
274 if (!has_peers && peer == nullptr) {
275 addBroadcastPeer();
276 }
277
278 size_t total_sent = 0;
279 size_t remaining = len;
280
281 while (remaining > 0) {
282 size_t chunk_size = min(remaining, (size_t)MY_ESP_NOW_MAX_LEN);
283 int retry_count = 0;
284
285 bool success =
286 sendPacket(data + total_sent, chunk_size, retry_count, peer);
287
288 if (success) {
289 // Chunk sent successfully
290 total_sent += chunk_size;
291 remaining -= chunk_size;
292 } else {
293 // Max retries exceeded for this chunk
294 LOGE(
295 "write: failed to send chunk after %d attempts (sent %zu/%zu "
296 "bytes)",
297 retry_count, total_sent, len);
298 // Return bytes successfully sent so far (may be 0 if first chunk
299 // failed)
300 return total_sent;
301 }
302 }
303 return total_sent;
304 }

376 bool sendPacket(const uint8_t* data, size_t len, int& retry_count,
377 const uint8_t* destination = nullptr) {
378 TRACED();
379 const uint8_t* target = destination;
380 if (target == nullptr && is_broadcast) {
381 target = BROADCAST_MAC;
382 }
383
384 while (true) {
385 resetAvailableToWrite();
386
387 // Wait for previous send to complete (if using ACKs)
388 if (cfg.use_send_ack) {
389 TickType_t ticks = (cfg.ack_semaphore_timeout_ms == portMAX_DELAY)
390 ? portMAX_DELAY
391 : pdMS_TO_TICKS(cfg.ack_semaphore_timeout_ms);
392 if (xSemaphoreTake(xSemaphore, ticks) != pdTRUE) {
393 // Timeout waiting for previous send - check retry limit BEFORE
394 // incrementing
395 if (cfg.write_retry_count >= 0 &&
396 retry_count >= cfg.write_retry_count) {
397 LOGE("Timeout waiting for ACK semaphore after %d retries",
398 retry_count);
399 return false;
400 }
401 retry_count++;
402 LOGW("ACK semaphore timeout (attempt %d)", retry_count);
403 delay(cfg.delay_after_failed_write_ms);
404 continue;
405 }
406 }
407
408 // Try to queue the packet
409 esp_err_t rc = esp_now_send(target, data, len);
410
411 if (rc == ESP_OK) {
412 // Packet queued - wait for transmission result
413 if (handleTransmissionResult(retry_count)) {
414 return true; // Success
415 }
416 // Transmission failed - check if we've exceeded the limit
417 // handleTransmissionResult returns false both when limit is reached
418 // and when we should retry, so check the limit here
419 if (cfg.write_retry_count >= 0 &&
420 retry_count >= cfg.write_retry_count) {
421 return false; // Give up - limit reached
422 }
423 // Continue to retry
424 } else {
425 // Failed to queue - callback will NOT be called
426 if (cfg.use_send_ack) {
427 xSemaphoreGive(xSemaphore); // Give back semaphore
428 }
429
430 // Check limit BEFORE incrementing
431 if (cfg.write_retry_count >= 0 &&
432 retry_count >= cfg.write_retry_count) {
433 LOGE("esp_now_send queue error (rc=%d/0x%04X) after %d retries", rc,
434 rc, retry_count);
435 return false;
436 }
437
438 retry_count++;
439 LOGW("esp_now_send failed (rc=%d/0x%04X) - retrying (attempt %d)", rc,
440 rc, retry_count);
441 delay(cfg.delay_after_failed_write_ms);
442 }
443 }
444 }
445
447 bool handleTransmissionResult(int& retry_count) {
448 TRACED();
449 if (cfg.use_send_ack) {
450 // Wait for callback to signal result
451 TickType_t ticks = (cfg.ack_semaphore_timeout_ms == portMAX_DELAY)
452 ? portMAX_DELAY
453 : pdMS_TO_TICKS(cfg.ack_semaphore_timeout_ms);
454 if (xSemaphoreTake(xSemaphore, ticks) != pdTRUE) {
455 // Callback never came - check limit BEFORE incrementing
456 if (cfg.write_retry_count >= 0 &&
457 retry_count >= cfg.write_retry_count) {
458 LOGE("Transmission callback timeout after %d retries", retry_count);
459 return false;
460 }
461 retry_count++;
462 LOGW("Transmission callback timeout (attempt %d)", retry_count);
463 delay(cfg.delay_after_failed_write_ms);
464 return false; // Retry
465 }
466
467 // Got callback - check result
468 if (last_send_success) {
469 xSemaphoreGive(xSemaphore); // Release for next send
470 return true;
471 } else {
472 xSemaphoreGive(xSemaphore); // Release for retry
473
474 // Check limit BEFORE incrementing
475 if (cfg.write_retry_count >= 0 &&
476 retry_count >= cfg.write_retry_count) {
477 LOGE("Transmission failed after %d retries", retry_count);
478 return false;
479 }
480
481 retry_count++;
482 LOGI("Transmission failed - retrying (attempt %d)", retry_count);
483 delay(cfg.delay_after_failed_write_ms);
484 return false; // Retry
485 }
486 }
487 return true; // No ACK mode - assume success
488 }
489
491 bool handleQueueError(esp_err_t rc, int& retry_count) {
492 TRACED();
493 // esp_now_send failed to queue - callback will NOT be called
494 // Give back the semaphore we took earlier
495 if (cfg.use_send_ack) {
496 xSemaphoreGive(xSemaphore);
497 }
498
499 retry_count++;
500 LOGW("esp_now_send failed to queue (rc=%d/0x%04X) - retrying (attempt %d)",
501 rc, rc, retry_count);
502
503 if (cfg.write_retry_count >= 0 && retry_count > cfg.write_retry_count) {
504 LOGE("Send queue error after %d retries", retry_count);
505 return false;
506 }
507
508 delay(cfg.delay_after_failed_write_ms);
509 return true; // Continue retrying
510 }

578#if ESP_IDF_VERSION >= ESP_IDF_VERSION_VAL(5, 0, 0)
579 static void default_recv_cb(const esp_now_recv_info* info,
580 const uint8_t* data, int data_len)
581#else
582 static void default_recv_cb(const uint8_t* mac_addr, const uint8_t* data,
583 int data_len)
584#endif
585 {
586 LOGD("rec_cb: %d", data_len);
587 // make sure that the receive buffer is available - moved from begin to make
588 // sure that it is only allocated when needed
589 ESPNowStreamSelf->setupReceiveBuffer();
590
591 // update last io time
592 ESPNowStreamSelf->last_io_success_time = millis();
593
594 // blocking write
595 size_t result = ESPNowStreamSelf->buffer.writeArray(data, data_len);
596 if (result != data_len) {
597 LOGE("writeArray %d -> %d", data_len, result);
598 }
599 // manage ready state
600 if (ESPNowStreamSelf->read_ready == false) {
601 if (ESPNowStreamSelf->cfg.start_read_threshold_percent == 0) {
602 ESPNowStreamSelf->read_ready = true;
603 } else {
604 float percent = ESPNowStreamSelf->getBufferPercent();
605 ESPNowStreamSelf->read_ready =
606 percent >= ESPNowStreamSelf->cfg.start_read_threshold_percent;
607 }
608 }
609 }

611#if ESP_IDF_VERSION >= ESP_IDF_VERSION_VAL(5, 5, 0)
612
613 static void default_send_cb(const wifi_tx_info_t* tx_info,
614 esp_now_send_status_t status) {
615 const uint8_t* mac_addr = tx_info->des_addr;
616#else
617 static void default_send_cb(const uint8_t* mac_addr,
618 esp_now_send_status_t status) {
619#endif
620 static uint8_t first_mac[ESP_NOW_KEY_LEN] = {0};
621 // we use the first confirming mac_addr for further confirmations and
622 // ignore others
623 if (first_mac[0] == 0) {
624 strncpy((char*)first_mac, (char*)mac_addr, ESP_NOW_KEY_LEN);
625 }
626 LOGD("default_send_cb - %s -> %s", ESPNowStreamSelf->mac2str(mac_addr),
627 status == ESP_NOW_SEND_SUCCESS ? "+" : "-");
628
629 // ignore others
630 if (strncmp((char*)mac_addr, (char*)first_mac, ESP_NOW_KEY_LEN) == 0) {
631 ESPNowStreamSelf->available_to_write = ESPNowStreamSelf->cfg.buffer_size;
632
633 // Track send success/failure
634 ESPNowStreamSelf->last_send_success = (status == ESP_NOW_SEND_SUCCESS);
635
636 if (status == ESP_NOW_SEND_SUCCESS) {
637 ESPNowStreamSelf->last_io_success_time = millis();
638 } else {
639 LOGI(
640 "Send Error to %s! Status: %d (Possible causes: out of range, "
641 "receiver busy/offline, channel mismatch, or buffer full)",
642 ESPNowStreamSelf->mac2str(mac_addr), status);
643 }
644
645 // Release semaphore to allow write to check status and retry if needed
646 xSemaphoreGive(ESPNowStreamSelf->xSemaphore);
647 }
