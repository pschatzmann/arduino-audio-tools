307 size_t minuspos = currentLine.find('-', 13);
308
309 firstbyte = std::strtol(currentLine.c_str() + 13, nullptr, 10);
310 lastbyte = std::strtol(currentLine.c_str() + minuspos + 1, nullptr, 10);
311 }
312
313 // if the current line is blank, you got two newline characters in a
314 // row. that's the end of the client HTTP request, so send a response:
315 if (currentLine.length() == 0) {
316 sendReplyHeader();
317 sendReplyContent();
318
319 max_bytes = lastbyte - firstbyte;
320 sent = 0;
321
322 break;
323 } else {

325 }
