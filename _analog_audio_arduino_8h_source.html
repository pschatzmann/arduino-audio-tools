180  static void callback(void *arg) {
181  int16_t value = 0;
182  AnalogAudioArduino *self = (AnalogAudioArduino *)arg;
183  // prevent NPE
184  if (self->buffer == nullptr) return;
185 
186  // Logic for reading audio data
187  if (self->config.rx_tx_mode == RX_MODE) {
188  int channels = self->config.channels;
189  for (int j = 0; j < channels; j++) {
190  // provides value in range 0â€¦4095
191  value = analogRead(self->config.start_pin + j);
192  if (self->config.is_auto_center_read) {
193  self->updateMinMax(value);
194  }
195  value = (value - self->avg_value) * 16;
196  self->buffer->write(value);
197  }
198  // Logic for writing audio data
199  } else if (self->config.rx_tx_mode == TX_MODE) {
200  int channels = self->config.channels;

202  int16_t sample = self->buffer->read();
203  sample = map(sample, -32768, 32767, 0, 255);
