420class RTSPFormatPCM : public RTSPFormatAudioTools {
421 public:
423 const char *format(char *buffer, int len) override {
424 TRACEI();
425 snprintf(buffer, len,
426 "s=%s\r\n" // Stream Name

428 "t=0 0\r\n" // start / stop - 0 -> unbounded and permanent session
429 "m=audio 0 RTP/AVP %d\r\n" // UDP sessions with format 10 or 11
430 "a=rtpmap:%s\r\n"
431 "a=rate:%i\r\n", // provide sample rate

446 int format(int channels) {
447 int result = 0;
448 switch (channels) {
449 case 1:
450 result = 11;
451 break;
452 case 2:
453 result = 10;
454 break;
455 default:
456 LOGE("unsupported audio type");
457 break;
458 }

516class RTSPFormatADPCM : public RTSPFormatAudioTools {
517 public:
518 // Provides the IMA ADPCM format information
519 // See RFC 3551 for details
520 const char *format(char *buffer, int len) override {
521 TRACEI();
522 // Only certain sample rates are valid for DVI4 (IMA ADPCM)
523 int sr = cfg.sample_rate;
524 int payload_type = 5; // default to 8000 Hz
525 switch (sr) {
526 case 8000:
527 payload_type = 5;
528 break;
529 case 16000:
530 payload_type = 6;
531 break;
532 case 11025:
533 payload_type = 16;
534 break;
535 case 22050:
536 payload_type = 17;
537 break;
538 default:
539 LOGE("Unsupported sample rate for IMA ADPCM: %d", sr);
540 sr = 8000;
541 payload_type = 5;
542 break;
543 }
544 snprintf(buffer, len,
545 "s=%s\r\n"
546 "c=IN IP4 0.0.0.0\r\n"
547 "t=0 0\r\n"
548 "m=audio 0 RTP/AVP %d\r\n"
549 "a=rtpmap:%d DVI4/%d\r\n",
550 name(), payload_type, payload_type, sr);
551 return (const char *)buffer;
552 }
553 AudioInfo defaultConfig() {
554 AudioInfo cfg(22050, 1, 16); // 4 bits per sample for IMA ADPCM
555 return cfg;
556 }
