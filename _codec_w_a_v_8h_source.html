62  void parse() {
63  LOGI("WAVHeader::begin: %u", (unsigned)len);
64  this->data_pos = 0l;
65  memset((void *)&headerInfo, 0, sizeof(WAVAudioInfo));
66  while (!eof()) {
67  uint32_t tag, tag2, length;
68  tag = read_tag();
69  if (eof()) break;
70  length = read_int32();
71  if (!length || length >= 0x7fff0000) {
72  headerInfo.is_streamed = true;
73  length = ~0;
74  }
75  if (tag != TAG('R', 'I', 'F', 'F') || length < 4) {
76  seek(length, SEEK_CUR);
77  continue;
78  }
79  headerInfo.file_size = length;
80  tag2 = read_tag();
81  length -= 4;
82  if (tag2 != TAG('W', 'A', 'V', 'E')) {
83  seek(length, SEEK_CUR);
84  continue;
85  }
86  // RIFF chunk found, iterate through it
87  while (length >= 8) {
88  uint32_t subtag, sublength;
89  subtag = read_tag();
90  if (eof()) break;
91  sublength = read_int32();
92  length -= 8;
93  if (length < sublength) break;
94  if (subtag == TAG('f', 'm', 't', ' ')) {
95  if (sublength < 16) {
96  // Insufficient data for 'fmt '
97  break;
98  }
99  headerInfo.format = (AudioFormat)read_int16();
100  headerInfo.channels = read_int16();
101  headerInfo.sample_rate = read_int32();
102  headerInfo.byte_rate = read_int32();
103  headerInfo.block_align = read_int16();
104  headerInfo.bits_per_sample = read_int16();
105  if (headerInfo.format == (AudioFormat) 0xfffe) {
106  if (sublength < 28) {
107  // Insufficient data for waveformatex
108  break;
109  }
110  skip(8);
111  headerInfo.format = (AudioFormat)read_int32();
112  skip(sublength - 28);
113  } else {
114  skip(sublength - 16);
115  }
116  headerInfo.is_valid = true;
117  } else if (subtag == TAG('d', 'a', 't', 'a')) {
118  sound_pos = tell();
119  headerInfo.data_length = sublength;
120  if (!headerInfo.data_length || headerInfo.is_streamed) {
121  headerInfo.is_streamed = true;
122  logInfo();
123  return;
124  }
125  seek(sublength, SEEK_CUR);
126  } else {
127  skip(sublength);
128  }
129  length -= sublength;
130  }
131  if (length > 0) {
132  // Bad chunk?
133  seek(length, SEEK_CUR);
134  }
135  }
136  logInfo();
137  len = 0;
138  }

368  // convert int24 to int32
369  size_t write_out_24(const uint8_t *in_ptr, size_t in_size) {
370  // make sure we can store a frame of 24bit (3bytes)
371  AudioInfo& info = header.audioInfo();
372  // in_size might be not a multiple of 3, so we use a buffer for a single frame
373  buffer24.resize(info.channels*3);
374  int result = 0;
375  int32_t frame[info.channels];
376  uint8_t val24[3]={0};
377 
378  // add all bytes to buffer
379  for (int j=0;j<in_size;j++){
380  buffer24.write(in_ptr[j]);
381  // if buffer is full convert and output
382  if (buffer24.availableForWrite()==0){
383  for (int ch=0;ch<info.channels;ch++){
384  buffer24.readArray((uint8_t*)&val24[0], 3);
385  frame[ch] = interpret24bitAsInt32(val24);
386  //LOGW("%d", frame[ch]);
387  }
388  assert(buffer24.available()==0);
389  buffer24.reset();
390  size_t written = out().write((uint8_t*)frame,sizeof(frame));
391  assert(written==sizeof(frame));
392  result += written;
393  }
394  }

417  size_t len = in_size - written;

419  isFirst = false;

421 
422  LOGI("WAV sample_rate: %d", (int) header.audioInfo().sample_rate);
423  LOGI("WAV data_length: %u", (unsigned)header.audioInfo().data_length);
424  LOGI("WAV is_streamed: %d", header.audioInfo().is_streamed);
425  LOGI("WAV is_valid: %s",
426  header.audioInfo().is_valid ? "true" : "false");
427 
428  // check format

430  isValid = format == decoder_format;
