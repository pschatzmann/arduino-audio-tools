382 // Convert 8-bit PCM to 16-bit PCM and write out
383 size_t write_out_8to16(const uint8_t *in_ptr, size_t in_size) {
384 const size_t batch_size = 256;
385 size_t total_written = 0;
386 size_t samples_remaining = in_size;
387 size_t offset = 0;
388 int16_t out_buf[batch_size];
389 while (samples_remaining > 0) {
390 size_t current_batch =
391 samples_remaining > batch_size ? batch_size : samples_remaining;
392 for (size_t i = 0; i < current_batch; ++i) {
393 out_buf[i] = ((int16_t)in_ptr[offset + i] - 128) << 8;
394 }
395 size_t written =
396 out().write((uint8_t *)out_buf, current_batch * sizeof(int16_t));
397 total_written += written / sizeof(int16_t);
398 offset += current_batch;
399 samples_remaining -= current_batch;
400 }
401 return total_written / 2;
402 }

404 // convert int24 to int32
405 size_t write_out_24(const uint8_t *in_ptr, size_t in_size) {
406 // make sure we can store a frame of 24bit (3bytes)
407 AudioInfo &info = header.audioInfo();
408 // in_size might be not a multiple of 3, so we use a buffer for a single
409 // frame
410 buffer24.resize(info.channels * 3);
411 int result = 0;
412 int32_t frame[info.channels];
413 uint8_t val24[3] = {0};
414
415 // add all bytes to buffer
416 for (int j = 0; j < in_size; j++) {
417 buffer24.write(in_ptr[j]);
418 // if buffer is full convert and output
419 if (buffer24.availableForWrite() == 0) {
420 for (int ch = 0; ch < info.channels; ch++) {
421 buffer24.readArray((uint8_t *)&val24[0], 3);
422 frame[ch] = interpret24bitAsInt32(val24);
423 // LOGW("%d", frame[ch]);
424 }
425 assert(buffer24.available() == 0);
426 buffer24.reset();
427 size_t written = out().write((uint8_t *)frame, sizeof(frame));
428 assert(written == sizeof(frame));
429 result += written;
430 }
431 }
432 return result;
433 }

435 int32_t interpret24bitAsInt32(uint8_t *byteArray) {
436 return ((static_cast<int32_t>(byteArray[2]) << 24) |
437 (static_cast<int32_t>(byteArray[1]) << 16) |
438 (static_cast<int32_t>(byteArray[0]) << 8));
439 }

442 int decodeHeader(uint8_t *in_ptr, size_t in_size) {
443 int result = in_size;
444 // we expect at least the full header
445 int written = header.write(in_ptr, in_size);
446 if (!header.isDataComplete()) {
447 LOGW("WAV header misses 'data' section in len: %d",
448 (int)header.available());
449 header.dumpHeader();
450 return 0;
451 }
452 // parse header
453 if (!header.parse()) {
454 LOGE("WAV header parsing failed");
455 return 0;
456 }
457
458 isFirst = false;
459 isValid = header.audioInfo().is_valid;
460
461 LOGI("WAV sample_rate: %d", (int)header.audioInfo().sample_rate);
462 LOGI("WAV data_length: %u", (unsigned)header.audioInfo().data_length);
463 LOGI("WAV is_streamed: %d", header.audioInfo().is_streamed);
464 LOGI("WAV is_valid: %s", header.audioInfo().is_valid ? "true" : "false");
465
466 // check format
467 AudioFormat format = header.audioInfo().format;
468 isValid = format == decoder_format;
469 if (isValid) {
470 // update blocksize
471 if (p_decoder != nullptr) {
472 int block_size = header.audioInfo().block_align;
473 p_decoder->setBlockSize(block_size);
474 }
475
476 // update sampling rate if the target supports it
477 AudioInfo bi = audioInfo();
478 notifyAudioChange(bi);
479 } else {
480 LOGE("WAV format not supported: %d", (int)format);
481 }
482 return header.getDataPos();
483 }
