86class OpusOggWriter : public OggContainerOutput {
87 protected:
88 OpusOggHeader header;
89 OpusOggCommentHeader comment;
90 ogg_packet oh1;
91 Vector<uint8_t> pendingPacket{0};
92 ogg_int64_t totalGranulepos = 0;
93 ogg_int64_t samplesPerPacket = 960;
94 ogg_int64_t finalPacketSamples = 0;
95 bool hasPendingPacket = false;
96
97 bool writeHeader() override {
98 LOGI("writeHeader");
99 bool result = true;
100 totalGranulepos = 0;
101 finalPacketSamples = 0;
102 hasPendingPacket = false;
103 header.sampleRate = cfg.sample_rate;
104 header.channelCount = cfg.channels;
105 // write header
106 oh.packet = (uint8_t *)&header;
107 oh.bytes = sizeof(header);
108 oh.granulepos = 0;
109 oh.packetno = packetno++;
110 oh.b_o_s = true;
111 oh.e_o_s = false;
112 if (!writePacket(oh)) {
113 result = false;
114 LOGE("writePacket-header");
115 }
116
117 // write comment header
118 oh1.packet = (uint8_t *)&comment;
119 oh1.bytes = sizeof(comment);
120 oh1.granulepos = 0;
121 oh1.packetno = packetno++;
122 oh1.b_o_s = false;
123 oh1.e_o_s = false;
124 if (!writePacket(oh1, OGGZ_FLUSH_AFTER)) {
125 result = false;
126 LOGE("writePacket-header1");
127 }
128 TRACED();
129 return result;
130 }
131
132 bool emitBufferedPacket(ogg_int64_t granulePosition, bool endOfStream) {
133 if (!hasPendingPacket) return true;
134
135 op.packet = pendingPacket.data();
136 op.bytes = pendingPacket.size();
137 op.granulepos = granulePosition;
138 op.b_o_s = false;
139 op.e_o_s = endOfStream;
140 op.packetno = packetno++;
141 is_audio = true;
142 if (!writePacket(op, OGGZ_FLUSH_AFTER)) {
143 return false;
144 }
145
146 while ((oggz_write(p_oggz, op.bytes)) > 0)
147 ;
148
149 hasPendingPacket = false;
150 pendingPacket.resize(0);
151 return true;
152 }
153
154 public:
155 void setPreSkipSamples(uint16_t preSkipSamples) {
156 header.preSkip = preSkipSamples;
157 }
158
159 void setFrameDurationUs(uint32_t frameDurationUs) {
160 ogg_int64_t opusSamples =
161 (ogg_int64_t)(((uint64_t)frameDurationUs * 48000ULL) / 1000000ULL);
162 samplesPerPacket = opusSamples > 0 ? opusSamples : 960;
163 }
164
165 void setFinalPacketSamples(ogg_int64_t samples) {
166 finalPacketSamples = samples;
167 }
168
169 size_t write(const uint8_t *data, size_t len) override {
170 if (data == nullptr) return 0;
171 LOGD("OpusOggWriter::write: %d", (int)len);
172
173 if (hasPendingPacket) {
174 totalGranulepos += samplesPerPacket;
175 if (!emitBufferedPacket(totalGranulepos, false)) {
176 return 0;
177 }
178 }
179
180 pendingPacket.resize(len);
181 memcpy(pendingPacket.data(), data, len);
182 hasPendingPacket = true;
183
184 return len;
185 }
186
187 void end() override {
188 TRACED();
189
190 if (hasPendingPacket) {
191 ogg_int64_t keptSamples =
192 finalPacketSamples > 0 ? finalPacketSamples : samplesPerPacket;
193 if (keptSamples > samplesPerPacket) {
194 keptSamples = samplesPerPacket;
195 }
196 totalGranulepos += keptSamples;
197 emitBufferedPacket(totalGranulepos, true);
198 }
199
200 is_open = false;
201 oggz_close(p_oggz);
202 p_oggz = nullptr;
203 }

229 uint32_t frameDurationUs() override {
230 // Get frame duration from encoder settings
231 int frameDurationMs = config().frame_sizes_ms_x2;
232 uint32_t frameDurationUs = 10000;
233 switch (frameDurationMs) {
234 case OPUS_FRAMESIZE_2_5_MS:
235 frameDurationUs = 2500;
236 break;
237 case OPUS_FRAMESIZE_5_MS:
238 frameDurationUs = 5000;
239 break;
240 case OPUS_FRAMESIZE_10_MS:
241 frameDurationUs = 10000;
242 break;
243 case OPUS_FRAMESIZE_20_MS:
244 frameDurationUs = 20000;
245 break;
246 case OPUS_FRAMESIZE_40_MS:
247 frameDurationUs = 40000;
248 break;
249 case OPUS_FRAMESIZE_60_MS:
250 frameDurationUs = 60000;
251 break;
252 case OPUS_FRAMESIZE_80_MS:
253 frameDurationUs = 80000;
254 break;
255 case OPUS_FRAMESIZE_100_MS:
256 frameDurationUs = 100000;
257 break;
258 case OPUS_FRAMESIZE_120_MS:
259 frameDurationUs = 120000;
260 break;
261 }
262 return frameDurationUs;
263 }
264
265 bool begin(AudioInfo from) override {
266 setAudioInfo(from);
267 return begin();
268 }
269
270 bool begin() override {
271 if (p_codec == nullptr) return false;
272
273 ogg_writer.setFrameDurationUs(frameDurationUs());
274 p_codec->setOutput(*p_ogg);
275 if (!p_codec->begin(p_ogg->audioInfo())) {
276 return false;
277 }
278
279 int lookaheadSamples = enc.lookaheadSamples();
280 if (lookaheadSamples > 0) {
281 uint32_t preSkipSamples =
282 (uint32_t)(((uint64_t)lookaheadSamples * 48000U) /
283 (uint32_t)enc.config().sample_rate);
284 ogg_writer.setPreSkipSamples((uint16_t)preSkipSamples);
285 }
286
287 if (!p_ogg->begin()) {
288 p_codec->end();
289 return false;
290 }
291
292 return true;
293 }
294
295 void end() override {
296 int finalSamples = enc.pendingSamples();
297 if (finalSamples <= 0) {
298 finalSamples = enc.frameSizeSamples();
299 }
300
301 ogg_int64_t finalGranuleSamples =
302 (ogg_int64_t)(((uint64_t)finalSamples * 48000ULL) /
303 (uint32_t)enc.config().sample_rate);
304 ogg_writer.setFinalPacketSamples(finalGranuleSamples);
305
306 OggContainerEncoder::end();
307 }

309 protected:
310 // use custom writer
311 OpusOggWriter ogg_writer;
312 // use opus encoder
313 OpusAudioEncoder enc;
314};
