109 void setupBuffers(size_t len) {

111 LOGD("setupBuffers: %d", (int)len);
112 float byte_factor = p_transform->getByteFactor();

140 void fillResultQueue(size_t len) {
141 if (result_queue.available() >= len) return;
142 LOGD("fillResultQueue: %d", (int)len);
143
144 // Detect misconfigured buffer: if the ring buffer capacity is smaller than
145 // the requested len bytes we can never satisfy the condition and will loop
146 // forever. Issue an error and bail out early.
147 if ((int)len > result_queue_buffer.size()) {
148 LOGE("fillResultQueue: result_queue_buffer too small: %d < %d. "
149 "Increase result_queue_factor or call resizeReadResultQueue().",
150 result_queue_buffer.size(), (int)len);
151 return;
152 }
153
154 Print* tmp = setupOutput();
155 int zero_count = 0;
156 while (result_queue.available() < len) {
157 // Detect buffer-full stall: if we can't write any more data but the
158 // queue is still below len, we must stop to avoid an endless loop.
159 if (result_queue.available() >= result_queue_buffer.size()) {
160 LOGE("fillResultQueue: result_queue full (%d) but target not reached "
161 "(%d/%d). Increase result_queue_factor or call "
162 "resizeReadResultQueue().",
163 result_queue_buffer.size(), result_queue.available(), (int)len);
164 break;
165 }
166 int read_size = buffer.size();
167 int read_eff = p_stream->readBytes(buffer.data(), read_size);
168 LOGD("readBytes from source: %d -> %d", read_size,read_eff);
169 if (read_eff > 0) {
170 zero_count = 0; // reset 0 count
171 if (read_eff != buffer.size()) {
172 LOGD("readBytes %d -> %d", buffer.size(), read_eff);
173 }
174 int write_eff = p_transform->write(buffer.data(), read_eff);
175 if (write_eff != read_eff) {
176 LOGE("TransformationReader::write %d -> %d", read_eff, write_eff);
177 }
178 } else {
179 // limit the number of reads which provide 0;
180 if (++zero_count > MAX_ZERO_READ_COUNT) {
181 is_eof = true;
182 // Flush any buffered/final encoder bytes into result_queue.
183 p_transform->flush();
184 break;
185 }
186 // wait for some more data
187 delay(5);
188 }
189 }
190 LOGD("fillResultQueue available: %d", result_queue.available());
191 restoreOutput(tmp);
192 }
