386 template <typename FromT, typename ToT>
387 static ToT convert(FromT value) {
388 float value1 = value;
389 float minTo = minValueT<ToT>();
390 float maxTo = maxValueT<ToT>();
391 float maxFrom = maxValueT<FromT>();
392 float minFrom = minValueT<FromT>();
393
394 if (maxTo - minTo > 1.0f || maxFrom - minFrom > 1.0f) {
395 return mapT<float>(value1, minFrom, maxFrom, minTo, maxTo);
396 }
397
398 return value1 * maxValueT<ToT>() / maxValueT<FromT>();
399 }

402 template <typename FromT, typename ToT>
403 static void convertArray(FromT* from, ToT* to, int samples,
404 float vol = 1.0f) {
405 // convert() already maps FromT range to ToT range. Apply only the gain
406 // and clip to the valid ToT range to avoid double scaling.
407 for (int j = 0; j < samples; j++) {
408 ToT mapped = convert<FromT, ToT>(from[j]);
409 float scaled = static_cast<float>(mapped) * vol;
410 to[j] = clipT<ToT>(scaled);
411 }
412 }

454T readSample(Stream* p_stream) {
455 T result = 0;
456 uint8_t* p_result = (uint8_t*)&result;
457 int open = sizeof(T);
458 int total = 0;
459 // copy missing data

463 total += read;
464 }

469template <typename T>
470size_t readSamples(Stream* p_stream, T* data, int samples,
471 int retryCount = -1) {
472 uint8_t* p_result = (uint8_t*)data;
473 int open = samples * sizeof(T);
474 int total = 0;
475 int count0 = 0;
476 // copy missing data - we abort after 5 x read returned 0 bytes
477 while (open > 0) {
478 int read = p_stream->readBytes(p_result + total, open);
479 open -= read;
480 total += read;
481 if (read == 0) {
482 count0++;
483 delay(1);
484 } else {
485 count0 = 0;
486 }
487 // abort loop
488 if (retryCount >= 0 && count0 > retryCount) break;
489 }
490 // convert bytes to samples
491 return total / sizeof(T);
492}

495template <typename T, class P>
496size_t writeDataT(P* p_out, T* data, int samples, int maxSamples = 512) {
497 uint8_t* p_result = (uint8_t*)data;
498 int open = samples * sizeof(T);
499 int total = 0;
500 // copy missing data
501 while (open > 0) {
502 int to_write = min(open, static_cast<int>(maxSamples * sizeof(T)));
503 int written = p_out->write(p_result + total, to_write);
504 open -= written;
505 total += written;
506 }
507 // convert bytes to samples
508 return total / sizeof(T);
509}
