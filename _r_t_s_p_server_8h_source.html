178 bool runAsync() {
179 LOGI("Running RTSP server on port %d", port);
180 streamer->initAudioSource();
181 if (server == nullptr) {
182 server = Platform::createServer(port);
183 LOGI("RTSP server started on port %d", port);
184 }
185 if (!serverTask.begin([this]() { serverThreadLoop(); })) {
186 LOGE("Couldn't start server thread");
187 return false;
188 }
189 return true;
190 }

203 void serverThreadLoop() {
204 unsigned long lastCheck = millis();
205
206 LOGD("Server thread listening... (numClients: %d)", client_count);
207
208 // only allow one client at a time
209 if (client_count == 0) {
210 auto newClient = Platform::getAvailableClient(server);
211 if (newClient.connected()) {
212 client = newClient; // copy/move assign
213 LOGI("Client connected");
214 if (!sessionTask.begin([this]() { sessionThreadLoop(); })) {
215 LOGE("Couldn't start sessionThread");
216 Platform::closeSocket(&client);
217 } else {
218 client_count++;
219 LOGI("Number of clients: %d", client_count);
220 }
221 }
222 } else {
223 LOGD("Waiting for current session to end (numClients: %d)", client_count);
224 }
225 int time = 200 - (millis() - lastCheck);
226 if (time < 0) time = 0;
227 delay(time);
228 }
229
233 void stop() {
234 LOGI("Stopping RTSP server");
235
236 // Stop tasks
237 sessionTask.end();
238 serverTask.end();
239
240 // Close sockets
241 if (server) {
242 // server->stop(); // not supported by EthernetServer
243 delete server;
244 server = nullptr;
245 }
246
247 client_count = 0;
248 LOGI("RTSP server stopped");
249 }

264 void sessionThreadLoop() {
265 typename Platform::TcpClientType* s = &client;
266 unsigned long lastCheck = millis();
267
268 // TODO check if everything is ok to go
269 LOGD("RTSP Task running");
270
271 // our threads RTSP session and state
272 RtspSession<Platform>* rtsp = new RtspSession<Platform>(*s, *streamer);
273 if (onSessionPathCb) {
274 rtsp->setOnSessionPath(onSessionPathCb, onSessionPathRef);
275 }
276 assert(rtsp != nullptr);
277
278 LOGI("Session ready");
279
280 while (rtsp->isSessionOpen()) {
281 uint32_t timeout = 30;
282 if (!rtsp->handleRequests(timeout)) {
283 LOGI("Request handling timed out or no data yet");
284 } else {
285 LOGD("Request handling successful");
286 }
287
288 int time = timeout - (millis() - lastCheck);
289 if (time < 0) time = 0;
290 delay(time);
291 }
292
293 LOGI("Session loop exited - session no longer open");
294
295 // cleanup when session ends
296 LOGI("sessionThread stopped, cleaning up");
297 delete rtsp;
298
299 // Clean up client socket
300 if (client.connected()) {
301 Platform::closeSocket(&client);
302 }
303
304 // Add delay to ensure complete cleanup before accepting new connections
305 delay(500);
306
307 client_count--;
308 LOGI("Session cleaned up: (numClients: %d)", client_count);
309
310 // end the task - this should be the last thing we do
311 sessionTask.end();
312 }
