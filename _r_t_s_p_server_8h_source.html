22template <typename Platform>
23class RTSPServer : public RTSPServerBase<Platform> {
24 public:
25 using streamer_t = RTSPAudioStreamerBase<Platform>;
26
27 RTSPServer(streamer_t& streamer, int port = 8554, int core = 1)
28 : RTSPServerBase<Platform>(streamer, port), core(core) {}
29
30 ~RTSPServer() { end(); }
31
32#ifdef ESP32
33
34 bool begin(const char* ssid, const char* password) {
35 return RTSPServerBase<Platform>::begin(ssid, password);
36 }
37#endif

39 bool begin() {
40 if (!RTSPServerBase<Platform>::begin()) return false;
41 if (!serverTask.begin([this]() { serverThreadLoop(); })) {
42 LOGE("Couldn't start server thread");
43 return false;
44 }
45 return true;
46 }
47
48 void end() {
49 sessionTask.end();
50 serverTask.end();
51 RTSPServerBase<Platform>::end();
52 }
53
54 protected:
55 int core = 0;
56 audio_tools::Task serverTask{"RTSPServerThread", 10000, 5, core};
57 audio_tools::Task sessionTask{"RTSPSessionTask", 8000, 8, core};
58
59 void serverThreadLoop() {
60 unsigned long lastCheck = millis();
61 LOGD("Server thread listening... (numClients: %d)", this->client_count);
62 int prevClientCount = this->client_count;
63 this->acceptClient();
64 // If a new client was accepted, start session task
65 if (this->client_count > prevClientCount) {
66 LOGI("Client connected");
67 if (!sessionTask.begin([this]() { sessionThreadLoop(); })) {
68 LOGE("Couldn't start sessionThread");
69 Platform::closeSocket(&this->client);
70 this->client_count--;
71 } else {
72 LOGI("Number of clients: %d", this->client_count);
73 }
74 } else if (this->client_count > 0) {
75 LOGD("Waiting for current session to end (numClients: %d)", this->client_count);
76 }
77 int time = 200 - (millis() - lastCheck);
78 if (time < 0) time = 0;
79 delay(time);
80 }
81
82 void sessionThreadLoop() {
83 LOGD("RTSP Task running");
84 while (this->rtspSession && this->rtspSession->isSessionOpen()) {
85 this->handleSession();
86 delay(30);
87 }
88 delay(500);
89 LOGI("Session cleaned up: (numClients: %d)", this->client_count);
90 sessionTask.end();
91 }
92};
