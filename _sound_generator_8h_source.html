235public:

237 : SineWaveGenerator<T>(amplitude, phase) {

240
241 virtual T readSample() override {
242 float angle =
243 SineWaveGenerator<T>::m_cycles + SineWaveGenerator<T>::m_phase;
244 T result = SineWaveGenerator<T>::m_amplitude * sine(angle);

256 float p = (t - (int)t) - 0.5f; // 0 <= p <= 1

258 return (p - 6.283211f * pp * p + 9.132843f * pp * pp * p) * -6.221086f;

351template <class T> class PinkNoiseGenerator : public SoundGenerator<T> {
352public:
354 PinkNoiseGenerator(T amplitude = 32767) {
355 this->amplitude = amplitude;
356 max_key = 0x1f; // Five bits set
357 key = 0;
358 for (int i = 0; i < 5; i++)
359 white_values[i] = rand() % (amplitude / 5);
360 }
361
363 T readSample() {
364 T last_key = key;
365 unsigned int sum;
366
367 key++;
368 if (key > max_key)
369 key = 0;
370 // Exclusive-Or previous value with current value. This gives
371 // a list of bits that have changed.
372 int diff = last_key ^ key;
373 sum = 0;
374 for (int i = 0; i < 5; i++) {
375 // If bit changed get new random number for corresponding
376 // white_value
377 if (diff & (1 << i))
378 white_values[i] = rand() % (amplitude / 5);
379 sum += white_values[i];
380 }
381 return sum;
382 }
383
384protected:
385 T max_key;
386 T key;
387 unsigned int white_values[5];
388 unsigned int amplitude;
389};

748 T interpolate(float angle) {
749 bool positive = (angle <= 180.0f);
750 float angle_positive = positive ? angle : angle - 180.0f;
751 int angle_int1 = angle_positive;
752 int angle_int2 = angle_int1 + 1;
753 T v1 = values[angle_int1] * amplitude;
754 T v2 = values[angle_int2] * amplitude;
755 T result = v1 < v2 ? map(angle_positive, angle_int1, angle_int2, v1, v2)
